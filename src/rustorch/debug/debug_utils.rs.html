15/// System information for debugging
16#[derive(Debug, Clone)]
17pub struct SystemInfo {
18    pub os: String,
19    pub architecture: String,
20    pub cpu_count: usize,
21    pub available_memory_mb: usize,
22    pub rust_version: String,
23    pub debug_build: bool,
24    pub environment_vars: HashMap<String, String>,
25    pub timestamp: SystemTime,
26}
27
28impl SystemInfo {
29    /// Collect current system information
30    pub fn collect() -> Self {
31        let cpu_count = thread::available_parallelism()
32            .map(|n| n.get())
33            .unwrap_or(1);
34
35        // Collect relevant environment variables
36        let mut env_vars = HashMap::new();
37        for (key, value) in env::vars() {
38            if key.starts_with("RUST")
39                || key.starts_with("CARGO")
40                || key.contains("CUDA")
41                || key.contains("GPU")
42                || key.contains("OPENCL")
43            {
44                env_vars.insert(key, value);
45            }
46        }
47
48        Self {
49            os: env::consts::OS.to_string(),
50            architecture: env::consts::ARCH.to_string(),
51            cpu_count,
52            available_memory_mb: Self::estimate_available_memory(),
53            rust_version: std::env::var("RUSTC_VERSION").unwrap_or_else(|_| "unknown".to_string()),
54            debug_build: cfg!(debug_assertions),
55            environment_vars: env_vars,
56            timestamp: SystemTime::now(),
57        }
58    }
59
60    /// Estimate available system memory (simplified)
61    fn estimate_available_memory() -> usize {
62        // This is a rough estimate - in a real system you'd use platform-specific APIs
63        match std::env::consts::OS {
64            "linux" | "macos" => 8192, // Assume 8GB default
65            "windows" => 16384,        // Assume 16GB default
66            _ => 4096,                 // Conservative default
67        }
68    }
69
70    /// Format system info as string
71    pub fn format_summary(&self) -> String {
72        format!(
73            "System: {} {}, {} CPUs, ~{}MB RAM, Rust {}, Debug: {}",
74            self.os,
75            self.architecture,
76            self.cpu_count,
77            self.available_memory_mb,
78            self.rust_version,
79            self.debug_build
80        )
81    }
82}

84/// Stack trace information (simplified)
85#[derive(Debug, Clone)]
86pub struct StackTrace {
87    pub frames: Vec<String>,
88    pub timestamp: SystemTime,
89    pub thread_name: String,
90}
91
92impl StackTrace {
93    /// Capture current stack trace (simplified implementation)
94    pub fn capture() -> Self {
95        // Note: This is a simplified implementation
96        // A real implementation would use backtrace crate or similar
97        let frames = vec![
98            "rustorch::debug::capture_stack".to_string(),
99            "rustorch::tensor::operation".to_string(),
100            "rustorch::main".to_string(),
101        ];
102
103        let thread_name = thread::current().name().unwrap_or("unnamed").to_string();
104
105        Self {
106            frames,
107            timestamp: SystemTime::now(),
108            thread_name,
109        }
110    }
111
112    /// Format stack trace as string
113    pub fn format_trace(&self) -> String {
114        let mut trace = format!("Stack trace (thread: {}):\n", self.thread_name);
115        for (i, frame) in self.frames.iter().enumerate() {
116            trace.push_str(&format!("  {}: {}\n", i, frame));
117        }
118        trace
119    }

122/// Performance measurement helper
123#[derive(Debug, Clone)]
124pub struct PerfTimer {
125    name: String,
126    start_time: Instant,
127    checkpoints: Vec<(String, Duration)>,
128}
129
130impl PerfTimer {
131    /// Start new performance timer
132    pub fn start(name: String) -> Self {
133        Self {
134            name,
135            start_time: Instant::now(),
136            checkpoints: Vec::new(),
137        }
138    }
139
140    /// Add checkpoint
141    pub fn checkpoint(&mut self, label: &str) {
142        let elapsed = self.start_time.elapsed();
143        self.checkpoints.push((label.to_string(), elapsed));
144    }
145
146    /// Finish timing and get total duration
147    pub fn finish(mut self) -> (Duration, Vec<(String, Duration)>) {
148        let total_duration = self.start_time.elapsed();
149        self.checkpoints.push(("TOTAL".to_string(), total_duration));
150        (total_duration, self.checkpoints)
151    }
152
153    /// Get current elapsed time
154    pub fn elapsed(&self) -> Duration {
155        self.start_time.elapsed()
156    }
157
158    /// Generate timing report
159    pub fn generate_report(&self) -> String {
160        let mut report = format!("Performance Report: {}\n", self.name);
161        report.push_str("====================\n");
162
163        for (label, duration) in &self.checkpoints {
164            report.push_str(&format!(
165                "{}: {:.3}ms\n",
166                label,
167                duration.as_secs_f64() * 1000.0
168            ));
169        }
170
171        let total_elapsed = self.elapsed();
172        report.push_str(&format!(
173            "Current Elapsed: {:.3}ms\n",
174            total_elapsed.as_secs_f64() * 1000.0
175        ));
176
177        report
178    }
179}

181/// Memory usage snapshot
182#[derive(Debug, Clone)]
183pub struct MemorySnapshot {
184    pub timestamp: SystemTime,
185    pub component_usage: HashMap<String, usize>,
186    pub total_allocated: usize,
187    pub estimated_available: usize,
188}
189
190impl MemorySnapshot {
191    /// Take memory snapshot (simplified)
192    pub fn take(component_usage: HashMap<String, usize>) -> Self {
193        let total_allocated: usize = component_usage.values().sum();
194
195        Self {
196            timestamp: SystemTime::now(),
197            component_usage,
198            total_allocated,
199            estimated_available: SystemInfo::estimate_available_memory() * 1024 * 1024, // Convert to bytes
200        }
201    }
202
203    /// Calculate memory utilization percentage
204    pub fn utilization_percent(&self) -> f64 {
205        if self.estimated_available > 0 {
206            (self.total_allocated as f64 / self.estimated_available as f64) * 100.0
207        } else {
208            0.0
209        }
210    }
211
212    /// Format memory snapshot
213    pub fn format_summary(&self) -> String {
214        let total_mb = self.total_allocated as f64 / (1024.0 * 1024.0);
215        let available_mb = self.estimated_available as f64 / (1024.0 * 1024.0);
216
217        format!(
218            "Memory: {:.1}MB / {:.1}MB ({:.1}% used)",
219            total_mb,
220            available_mb,
221            self.utilization_percent()
222        )
223    }
224}

226/// Diagnostic context for debugging
227#[derive(Debug, Clone)]
228pub struct DiagnosticContext {
229    pub operation_name: String,
230    pub parameters: HashMap<String, String>,
231    pub system_info: SystemInfo,
232    pub memory_snapshot: Option<MemorySnapshot>,
233    pub stack_trace: Option<StackTrace>,
234    pub timestamp: SystemTime,
235}
236
237impl DiagnosticContext {
238    /// Create new diagnostic context
239    pub fn new(operation_name: String) -> Self {
240        Self {
241            operation_name,
242            parameters: HashMap::new(),
243            system_info: SystemInfo::collect(),
244            memory_snapshot: None,
245            stack_trace: None,
246            timestamp: SystemTime::now(),
247        }
248    }
249
250    /// Add parameter
251    pub fn add_parameter(&mut self, key: &str, value: &str) {
252        self.parameters.insert(key.to_string(), value.to_string());
253    }
254
255    /// Set memory snapshot
256    pub fn set_memory_snapshot(&mut self, snapshot: MemorySnapshot) {
257        self.memory_snapshot = Some(snapshot);
258    }
259
260    /// Capture stack trace
261    pub fn capture_stack_trace(&mut self) {
262        self.stack_trace = Some(StackTrace::capture());
263    }
264
265    /// Generate comprehensive diagnostic report
266    pub fn generate_diagnostic_report(&self) -> String {
267        let mut report = String::new();
268
269        report.push_str("🔧 Diagnostic Report\n");
270        report.push_str("===================\n\n");
271
272        report.push_str(&format!("Operation: {}\n", self.operation_name));
273        report.push_str(&format!("Timestamp: {:?}\n\n", self.timestamp));
274
275        // System information
276        report.push_str("🖥️ System Information:\n");
277        report.push_str(&format!("  {}\n\n", self.system_info.format_summary()));
278
279        // Parameters
280        if !self.parameters.is_empty() {
281            report.push_str("📋 Parameters:\n");
282            for (key, value) in &self.parameters {
283                report.push_str(&format!("  {}: {}\n", key, value));
284            }
285            report.push('\n');
286        }
287
288        // Memory snapshot
289        if let Some(snapshot) = &self.memory_snapshot {
290            report.push_str("🧠 Memory Status:\n");
291            report.push_str(&format!("  {}\n", snapshot.format_summary()));
292
293            if !snapshot.component_usage.is_empty() {
294                report.push_str("  Component Usage:\n");
295                let mut components: Vec<_> = snapshot.component_usage.iter().collect();
296                components.sort_by(|a, b| b.1.cmp(a.1));
297
298                for (component, usage) in components.iter().take(5) {
299                    let mb = **usage as f64 / (1024.0 * 1024.0);
300                    report.push_str(&format!("    {}: {:.1}MB\n", component, mb));
301                }
302            }
303            report.push('\n');
304        }
305
306        // Stack trace
307        if let Some(trace) = &self.stack_trace {
308            report.push_str("📚 Stack Trace:\n");
309            report.push_str(&trace.format_trace());
310            report.push('\n');
311        }
312
313        // Environment variables
314        if !self.system_info.environment_vars.is_empty() {
315            report.push_str("🌍 Relevant Environment:\n");
316            for (key, value) in &self.system_info.environment_vars {
317                report.push_str(&format!("  {}: {}\n", key, value));
318            }
319        }
320
321        report
322    }
323}

354    /// Format duration for human reading
355    pub fn format_duration(duration: Duration) -> String {
356        let total_ms = duration.as_secs_f64() * 1000.0;
357
358        if total_ms < 1.0 {
359            format!("{:.3}μs", total_ms * 1000.0)
360        } else if total_ms < 1000.0 {
361            format!("{:.3}ms", total_ms)
362        } else if total_ms < 60000.0 {
363            format!("{:.2}s", total_ms / 1000.0)
364        } else {
365            let minutes = (total_ms / 60000.0) as u32;
366            let seconds = (total_ms % 60000.0) / 1000.0;
367            format!("{}m {:.1}s", minutes, seconds)
368        }
369    }
370
371    /// Format bytes for human reading
372    pub fn format_bytes(bytes: usize) -> String {
373        const UNITS: &[&str] = &["B", "KB", "MB", "GB", "TB"];
374        let mut size = bytes as f64;
375        let mut unit_index = 0;
376
377        while size >= 1024.0 && unit_index < UNITS.len() - 1 {
378            size /= 1024.0;
379            unit_index += 1;
380        }
381
382        if unit_index == 0 {
383            format!("{}B", bytes)
384        } else {
385            format!("{:.2}{}", size, UNITS[unit_index])
386        }
387    }

430    /// Simple assertion with diagnostic context
431    pub fn debug_assert_with_context<F>(condition: bool, context_fn: F, message: &str)
432    where
433        F: FnOnce() -> DiagnosticContext,
434    {
435        if cfg!(debug_assertions) && !condition {
436            let context = context_fn();
437            eprintln!("🚨 Debug Assertion Failed: {}", message);
438            eprintln!("{}", context.generate_diagnostic_report());
439            panic!("Debug assertion failed: {}", message);
440        }
441    }

443    /// Conditional debugging output
444    pub fn debug_print<T: fmt::Display>(value: &T, enabled: bool) {
445        if enabled && cfg!(debug_assertions) {
446            eprintln!("🐛 DEBUG: {}", value);
447        }
448    }

450    /// Time a code block and return result with timing
451    pub fn time_block<T, F>(name: &str, block: F) -> (T, Duration)
452    where
453        F: FnOnce() -> T,
454    {
455        let start = Instant::now();
456        let result = block();
457        let duration = start.elapsed();
458
459        if cfg!(debug_assertions) {
460            eprintln!("⏱️ {}: {}", name, Self::format_duration(duration));
461        }
462
463        (result, duration)
464    }

467/// Macro for easy diagnostic context creation
468#[macro_export]
469macro_rules! diagnostic_context {
470    ($operation:expr) => {
471        $crate::debug::DebugUtils::create_diagnostic_context($operation)
472    };
473    ($operation:expr, $($key:expr => $value:expr),*) => {{
474        let mut context = $crate::debug::DebugUtils::create_diagnostic_context($operation);
475        $(
476            context.add_parameter($key, &$value.to_string());
477        )*
478        context
479    }};
480}
