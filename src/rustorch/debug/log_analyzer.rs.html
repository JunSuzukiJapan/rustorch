14/// Log pattern for detection
15#[derive(Debug, Clone)]
16pub struct LogPattern {
17    pub name: String,
18    pub pattern: Regex,
19    pub severity: LogLevel,
20    pub description: String,
21    pub action_required: bool,
22}
23
24impl LogPattern {
25    /// Create new log pattern
26    pub fn new(
27        name: String,
28        pattern_str: &str,
29        severity: LogLevel,
30        description: String,
31        action_required: bool,
32    ) -> RusTorchResult<Self> {
33        let pattern = Regex::new(pattern_str).map_err(|e| RusTorchError::Debug {
34            message: format!("Invalid regex pattern '{}': {}", pattern_str, e),
35        })?;
36
37        Ok(Self {
38            name,
39            pattern,
40            severity,
41            description,
42            action_required,
43        })
44    }
45
46    /// Check if log entry matches this pattern
47    pub fn matches(&self, log_message: &str) -> bool {
48        self.pattern.is_match(log_message)
49    }
50
51    /// Extract capture groups from log message
52    pub fn extract_captures(&self, log_message: &str) -> Option<Vec<String>> {
53        self.pattern.captures(log_message).map(|caps| {
54            caps.iter()
55                .skip(1) // Skip the full match
56                .filter_map(|m| m.map(|m| m.as_str().to_string()))
57                .collect()
58        })
59    }
60}

62/// Alert rule for automated notifications
63#[derive(Debug, Clone)]
64pub struct AlertRule {
65    pub name: String,
66    pub pattern_name: String,
67    pub threshold_count: usize,
68    pub time_window_seconds: u64,
69    pub enabled: bool,
70    pub last_triggered: Option<SystemTime>,
71    pub cooldown_seconds: u64,
72}
73
74impl AlertRule {
75    /// Create new alert rule
76    pub fn new(
77        name: String,
78        pattern_name: String,
79        threshold_count: usize,
80        time_window_seconds: u64,
81        cooldown_seconds: u64,
82    ) -> Self {
83        Self {
84            name,
85            pattern_name,
86            threshold_count,
87            time_window_seconds,
88            enabled: true,
89            last_triggered: None,
90            cooldown_seconds,
91        }
92    }
93
94    /// Check if rule can be triggered (not in cooldown)
95    pub fn can_trigger(&self) -> bool {
96        if !self.enabled {
97            return false;
98        }
99
100        if let Some(last_triggered) = self.last_triggered {
101            let elapsed = SystemTime::now()
102                .duration_since(last_triggered)
103                .unwrap_or_default();
104            elapsed.as_secs() >= self.cooldown_seconds
105        } else {
106            true
107        }
108    }
109
110    /// Mark rule as triggered
111    pub fn mark_triggered(&mut self) {
112        self.last_triggered = Some(SystemTime::now());
113    }
114}

339    /// Analyze log entry for patterns
340    pub fn analyze_log_entry(
341        &mut self,
342        level: LogLevel,
343        message: &str,
344        metadata: &HashMap<String, String>,
345    ) -> RusTorchResult<Vec<PatternMatch>> {
346        if !self.enabled {
347            return Ok(Vec::new());
348        }
349
350        self.total_logs_analyzed += 1;
351        let mut matches = Vec::new();
352
353        // Check against all patterns
354        for pattern in self.patterns.values() {
355            if pattern.matches(message) {
356                let captures = pattern.extract_captures(message).unwrap_or_default();
357
358                let pattern_match = PatternMatch {
359                    pattern_name: pattern.name.clone(),
360                    message: message.to_string(),
361                    timestamp: SystemTime::now(),
362                    level,
363                    captures,
364                    metadata: metadata.clone(),
365                };
366
367                matches.push(pattern_match.clone());
368                self.pattern_matches.push_back(pattern_match);
369                self.patterns_detected += 1;
370
371                // Track error patterns
372                if level >= LogLevel::Error {
373                    *self.error_patterns.entry(pattern.name.clone()).or_insert(0) += 1;
374                    self.recent_errors.push_back(message.to_string());
375
376                    // Maintain recent errors window
377                    if self.recent_errors.len() > 50 {
378                        self.recent_errors.pop_front();
379                    }
380                }
381
382                // Track performance patterns
383                if self
384                    .bottleneck_keywords
385                    .iter()
386                    .any(|keyword| message.to_lowercase().contains(keyword))
387                {
388                    self.performance_patterns.push(pattern.name.clone());
389                }
390            }
391        }
392
393        // Maintain pattern matches window
394        while self.pattern_matches.len() > self.window_size {
395            self.pattern_matches.pop_front();
396        }
397
398        // Check for triggered alerts
399        self.check_alert_rules(&matches)?;
400
401        Ok(matches)
402    }

404    /// Check alert rules and generate notifications
405    fn check_alert_rules(&mut self, new_matches: &[PatternMatch]) -> RusTorchResult<()> {
406        let now = SystemTime::now();
407
408        for rule in self.alert_rules.values_mut() {
409            if !rule.can_trigger() {
410                continue;
411            }
412
413            // Count matching patterns within time window
414            let window_start = now - Duration::from_secs(rule.time_window_seconds);
415            let matching_patterns: Vec<&PatternMatch> = self
416                .pattern_matches
417                .iter()
418                .filter(|m| m.pattern_name == rule.pattern_name && m.timestamp >= window_start)
419                .collect();
420
421            if matching_patterns.len() >= rule.threshold_count {
422                // Create alert notification
423                let recent_messages: Vec<String> = matching_patterns
424                    .iter()
425                    .rev()
426                    .take(5)
427                    .map(|m| m.message.clone())
428                    .collect();
429
430                let severity = matching_patterns
431                    .iter()
432                    .map(|m| m.level)
433                    .max()
434                    .unwrap_or(LogLevel::Warning);
435
436                let alert = AlertNotification {
437                    rule_name: rule.name.clone(),
438                    pattern_name: rule.pattern_name.clone(),
439                    message: format!("Alert triggered for pattern '{}'", rule.pattern_name),
440                    severity,
441                    match_count: matching_patterns.len(),
442                    time_window: Duration::from_secs(rule.time_window_seconds),
443                    timestamp: now,
444                    recent_matches: recent_messages,
445                };
446
447                self.triggered_alerts.push(alert);
448                rule.mark_triggered();
449                self.alerts_triggered += 1;
450
451                // In a real system, this would send notifications
452                eprintln!("ðŸš¨ {}", self.triggered_alerts.last().unwrap());
453            }
454        }
455
456        Ok(())
457    }

459    /// Get analysis summary
460    pub fn get_analysis_summary(&self) -> AnalysisSummary {
461        let most_common_errors: Vec<(String, usize)> = {
462            let mut errors: Vec<(String, usize)> = self
463                .error_patterns
464                .iter()
465                .map(|(k, v)| (k.clone(), *v))
466                .collect();
467            errors.sort_by(|a, b| b.1.cmp(&a.1));
468            errors.into_iter().take(5).collect()
469        };
470
471        let performance_bottlenecks: Vec<String> = {
472            let mut bottlenecks = self.performance_patterns.clone();
473            bottlenecks.sort();
474            bottlenecks.dedup();
475            bottlenecks
476        };
477
478        AnalysisSummary {
479            patterns_detected: self.patterns_detected,
480            alerts_triggered: self.alerts_triggered,
481            most_common_errors,
482            performance_bottlenecks,
483        }
484    }

551        let summary = self.get_analysis_summary();

553        let mut report = String::new();

555        report.push_str("=====================\n\n");

564            "  Alerts Triggered: {}\n",

568        report.push_str(&format!("  Alert Rules: {}\n\n", self.alert_rules.len()));
569
570        if !summary.most_common_errors.is_empty() {
571            report.push_str("ðŸš¨ Most Common Error Patterns:\n");
572            for (i, (pattern, count)) in summary.most_common_errors.iter().enumerate() {

575                    i + 1,
576                    pattern,
577                    count
578                ));
579            }
580            report.push('\n');
581        }
582
583        if !summary.performance_bottlenecks.is_empty() {
584            report.push_str("âš¡ Performance Issues Detected:\n");
585            for (i, bottleneck) in summary.performance_bottlenecks.iter().enumerate() {

588            report.push('\n');
589        }
590
591        if !self.triggered_alerts.is_empty() {
592            report.push_str("ðŸ”” Recent Alerts:\n");

595                    "  {}. {} ({})\n",
596                    i + 1,
597                    alert.rule_name,
598                    alert.severity
599                ));
600            }
601        }
