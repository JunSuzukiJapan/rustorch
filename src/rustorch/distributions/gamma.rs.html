3/// Gamma Distribution - torch.distributions.Gamma compatible
4/// ガンマ分布 - torch.distributions.Gamma互換
5///
6/// Implements the Gamma distribution parameterized by shape (α) and rate (β)
7/// or shape (α) and scale (1/β).
8/// 形状パラメータ（α）と率パラメータ（β）、または形状（α）とスケール（1/β）で
9/// パラメータ化されたガンマ分布を実装

14/// Gamma Distribution
15/// ガンマ分布
16///
17/// The Gamma distribution is defined by:
18/// - Shape parameter α > 0 (concentration)
19/// - Rate parameter β > 0, or Scale parameter θ = 1/β > 0
20///
21/// PDF: f(x; α, β) = (β^α / Γ(α)) * x^(α-1) * exp(-βx) for x > 0
22#[derive(Debug, Clone)]
23pub struct Gamma<T: Float> {
24    /// Shape parameter (α, concentration)
25    /// 形状パラメータ（α、集中度）
26    pub concentration: Tensor<T>,
27
28    /// Rate parameter (β) - optional if scale is provided
29    /// 率パラメータ（β）- スケールが提供されている場合はオプション
30    pub rate: Option<Tensor<T>>,
31
32    /// Scale parameter (θ = 1/β) - optional if rate is provided  
33    /// スケールパラメータ（θ = 1/β）- 率が提供されている場合はオプション
34    pub scale: Option<Tensor<T>>,
35
36    /// Base distribution properties
37    /// 基本分布特性
38    pub base: Distribution,
39
40    /// Phantom data for type parameter
41    /// 型パラメータ用のファントムデータ
42    _phantom: PhantomData<T>,
43}
44
45impl<T: Float + 'static> Gamma<T>
46where
47    T: rand::distributions::uniform::SampleUniform + num_traits::FromPrimitive + std::fmt::Display,
48{
49    /// Create a new Gamma distribution with concentration and rate
50    /// 集中度と率でガンマ分布を作成
51    ///
52    /// # Arguments
53    /// * `concentration` - Shape parameter (α > 0)
54    /// * `rate` - Rate parameter (β > 0)
55    /// * `validate_args` - Whether to validate parameters
56    pub fn from_concentration_rate(
57        concentration: Tensor<T>,
58        rate: Tensor<T>,
59        validate_args: bool,
60    ) -> RusTorchResult<Self> {
61        if validate_args {
62            DistributionUtils::validate_positive(&concentration, "concentration")?;
63            DistributionUtils::validate_positive(&rate, "rate")?;
64        }
65
66        let batch_shape = Distribution::broadcast_shapes(concentration.shape(), rate.shape())?;
67        let event_shape = vec![]; // Gamma is univariate
68
69        Ok(Self {
70            concentration,
71            rate: Some(rate),
72            scale: None,
73            base: Distribution::new(batch_shape, event_shape, validate_args),
74            _phantom: PhantomData,
75        })
76    }
77
78    /// Create a new Gamma distribution with concentration and scale
79    /// 集中度とスケールでガンマ分布を作成
80    ///
81    /// # Arguments
82    /// * `concentration` - Shape parameter (α > 0)
83    /// * `scale` - Scale parameter (θ > 0)  
84    /// * `validate_args` - Whether to validate parameters
85    pub fn from_concentration_scale(
86        concentration: Tensor<T>,
87        scale: Tensor<T>,
88        validate_args: bool,
89    ) -> RusTorchResult<Self> {
90        if validate_args {
91            DistributionUtils::validate_positive(&concentration, "concentration")?;
92            DistributionUtils::validate_positive(&scale, "scale")?;
93        }
94
95        let batch_shape = Distribution::broadcast_shapes(concentration.shape(), scale.shape())?;
96        let event_shape = vec![];
97
98        Ok(Self {
99            concentration,
100            rate: None,
101            scale: Some(scale),
102            base: Distribution::new(batch_shape, event_shape, validate_args),
103            _phantom: PhantomData,
104        })
105    }

107    /// Create standard Gamma distribution (shape=1, scale=1) - exponential
108    /// 標準ガンマ分布（形状=1、スケール=1）- 指数分布
109    pub fn exponential(rate: T, validate_args: bool) -> RusTorchResult<Self> {
110        let concentration = Tensor::from_vec(vec![T::one()], vec![]);
111        let rate_tensor = Tensor::from_vec(vec![rate], vec![]);
112        Self::from_concentration_rate(concentration, rate_tensor, validate_args)
113    }
114
115    /// Get rate parameter (convert from scale if necessary)
116    /// 率パラメータを取得（必要に応じてスケールから変換）
117    pub fn get_rate(&self) -> RusTorchResult<Tensor<T>> {
118        match (&self.rate, &self.scale) {
119            (Some(rate), _) => Ok(rate.clone()),
120            (None, Some(scale)) => {
121                // rate = 1 / scale
122                let scale_data = scale.data.as_slice().unwrap();
123                let rate_data: Vec<T> = scale_data.iter().map(|&s| T::one() / s).collect();
124                Ok(Tensor::from_vec(rate_data, scale.shape().to_vec()))
125            }
126            _ => Err(RusTorchError::invalid_parameter(
127                "Either rate or scale must be specified",
128            )),
129        }
130    }
131
132    /// Get scale parameter (convert from rate if necessary)
133    /// スケールパラメータを取得（必要に応じて率から変換）
134    pub fn get_scale(&self) -> RusTorchResult<Tensor<T>> {
135        match (&self.scale, &self.rate) {
136            (Some(scale), _) => Ok(scale.clone()),
137            (None, Some(rate)) => {
138                // scale = 1 / rate
139                let rate_data = rate.data.as_slice().unwrap();
140                let scale_data: Vec<T> = rate_data.iter().map(|&r| T::one() / r).collect();
141                Ok(Tensor::from_vec(scale_data, rate.shape().to_vec()))
142            }
143            _ => Err(RusTorchError::invalid_parameter(
144                "Either rate or scale must be specified",
145            )),
146        }
147    }

149    /// Log Gamma function approximation using Stirling's formula
150    /// スターリングの公式を使ったログガンマ関数の近似
151    fn log_gamma_approx(x: T) -> T {
152        // Stirling's approximation: ln(Γ(x)) ≈ (x-0.5)ln(x) - x + 0.5*ln(2π)
153        let half = T::from(0.5).unwrap();
154        let pi = T::from(std::f64::consts::PI).unwrap();
155        let two = T::from(2.0).unwrap();
156
157        if x < T::one() {
158            // Use Γ(x+1) = x*Γ(x) for small x
159            let gamma_x_plus_1 = Self::log_gamma_approx(x + T::one());
160            gamma_x_plus_1 - x.ln()
161        } else {
162            (x - half) * x.ln() - x + half * (two * pi).ln()
163        }
164    }
165
166    /// Gamma function PDF normalization constant
167    /// ガンマ関数PDF正規化定数
168    fn log_normalizing_constant(&self) -> RusTorchResult<Tensor<T>> {
169        let concentration_data = self.concentration.data.as_slice().unwrap();
170        let rate = self.get_rate()?;
171        let rate_data = rate.data.as_slice().unwrap();
172
173        // log(β^α / Γ(α)) = α*log(β) - log(Γ(α))
174        let result_data: Vec<T> = concentration_data
175            .iter()
176            .zip(rate_data.iter().cycle())
177            .map(|(&alpha, &beta)| alpha * beta.ln() - Self::log_gamma_approx(alpha))
178            .collect();
179
180        Ok(Tensor::from_vec(
181            result_data,
182            self.concentration.shape().to_vec(),
183        ))
184    }
185
186    /// Marsaglia and Tsang's method for sampling Gamma distribution
187    /// ガンマ分布サンプリングのためのMarsaglia-Tsangの手法
188    fn sample_gamma(shape: T) -> T
189    where
190        T: rand::distributions::uniform::SampleUniform,
191    {
192        use rand::{thread_rng, Rng};
193        let mut rng = thread_rng();
194
195        if shape >= T::one() {
196            // Use Marsaglia-Tsang method for α >= 1
197            #[allow(clippy::many_single_char_names)]
198            // Mathematical variables d, c, x, v, u are conventional in gamma distribution algorithm
199            {
200                let d = shape - T::from(1.0 / 3.0).unwrap();
201                let c = T::one() / (T::from(9.0).unwrap() * d).sqrt();
202
203                loop {
204                    let x: T = rng.gen_range(T::from(-4.0).unwrap()..T::from(4.0).unwrap());
205                    let v = (T::one() + c * x).powi(3);
206
207                    if v > T::zero() {
208                        let u: T = rng.gen_range(T::zero()..T::one());
209                        let x_squared = x * x;
210
211                        if u < T::one() - T::from(0.0331).unwrap() * x_squared * x_squared {
212                            return d * v;
213                        }
214
215                        if u.ln() < T::from(0.5).unwrap() * x_squared + d * (T::one() - v + v.ln())
216                        {
217                            return d * v;
218                        }
219                    }
220                }
221            }
222        } else {
223            // For α < 1, use Γ(α) = Γ(α+1) / α and recursion
224            let gamma_alpha_plus_1 = Self::sample_gamma(shape + T::one());
225            let u: T = rng.gen_range(T::from(1e-10).unwrap()..T::one());
226            gamma_alpha_plus_1 * u.powf(T::one() / shape)
227        }
228    }
229}

231impl<T: Float + 'static> DistributionTrait<T> for Gamma<T>
232where
233    T: rand::distributions::uniform::SampleUniform + num_traits::FromPrimitive + std::fmt::Display,
234{
235    fn sample(&self, shape: Option<&[usize]>) -> RusTorchResult<Tensor<T>> {
236        let sample_shape = self.base.expand_shape(shape);
237        let concentration_data = self.concentration.data.as_slice().unwrap();
238        let scale = self.get_scale()?;
239        let scale_data = scale.data.as_slice().unwrap();
240
241        let sample_size: usize = sample_shape.iter().product();
242        let mut result_data = Vec::with_capacity(sample_size);
243
244        for i in 0..sample_size {
245            let batch_idx = i % concentration_data.len();
246            let alpha = concentration_data[batch_idx];
247            let theta = scale_data[batch_idx % scale_data.len()];
248
249            // Sample from Gamma(α, 1) and scale by θ
250            let gamma_sample = Self::sample_gamma(alpha);
251            result_data.push(gamma_sample * theta);
252        }
253
254        Ok(Tensor::from_vec(result_data, sample_shape))
255    }
256
257    fn log_prob(&self, value: &Tensor<T>) -> RusTorchResult<Tensor<T>> {
258        // log p(x) = (α-1)*log(x) - βx + α*log(β) - log(Γ(α))
259        let concentration_data = self.concentration.data.as_slice().unwrap();
260        let rate = self.get_rate()?;
261        let rate_data = rate.data.as_slice().unwrap();
262        let value_data = value.data.as_slice().unwrap();
263        let log_norm = self.log_normalizing_constant()?;
264        let log_norm_data = log_norm.data.as_slice().unwrap();
265
266        let result_data: Vec<T> = (0..value_data.len())
267            .map(|i| {
268                let x = value_data[i];
269                let alpha = concentration_data[i % concentration_data.len()];
270                let beta = rate_data[i % rate_data.len()];
271                let ln_norm = log_norm_data[i % log_norm_data.len()];
272
273                if x <= T::zero() {
274                    T::from(-1e10).unwrap() // Very negative for x <= 0
275                } else {
276                    (alpha - T::one()) * x.ln() - beta * x + ln_norm
277                }
278            })
279            .collect();
280
281        Ok(Tensor::from_vec(result_data, value.shape().to_vec()))
282    }

284    fn cdf(&self, _value: &Tensor<T>) -> RusTorchResult<Tensor<T>> {
285        // Gamma CDF requires incomplete gamma function - complex to implement
286        Err(RusTorchError::UnsupportedOperation(
287            "Gamma CDF requires incomplete gamma function (not implemented)",
288        ))
289    }

291    fn icdf(&self, _value: &Tensor<T>) -> RusTorchResult<Tensor<T>> {
292        // Gamma inverse CDF is very complex
293        Err(RusTorchError::UnsupportedOperation(
294            "Gamma inverse CDF not implemented",
295        ))
296    }
297
298    fn mean(&self) -> RusTorchResult<Tensor<T>> {
299        // Mean of Gamma(α, β) = α/β = α*θ
300        let scale = self.get_scale()?;
301        let concentration_data = self.concentration.data.as_slice().unwrap();
302        let scale_data = scale.data.as_slice().unwrap();
303
304        let result_data: Vec<T> = concentration_data
305            .iter()
306            .zip(scale_data.iter().cycle())
307            .map(|(&alpha, &theta)| alpha * theta)
308            .collect();
309
310        Ok(Tensor::from_vec(
311            result_data,
312            self.concentration.shape().to_vec(),
313        ))
314    }
315
316    fn variance(&self) -> RusTorchResult<Tensor<T>> {
317        // Variance of Gamma(α, β) = α/β² = α*θ²
318        let scale = self.get_scale()?;
319        let concentration_data = self.concentration.data.as_slice().unwrap();
320        let scale_data = scale.data.as_slice().unwrap();
321
322        let result_data: Vec<T> = concentration_data
323            .iter()
324            .zip(scale_data.iter().cycle())
325            .map(|(&alpha, &theta)| alpha * theta * theta)
326            .collect();
327
328        Ok(Tensor::from_vec(
329            result_data,
330            self.concentration.shape().to_vec(),
331        ))
332    }
333
334    fn entropy(&self) -> RusTorchResult<Tensor<T>> {
335        // Entropy = α - log(β) + log(Γ(α)) + (1-α)*ψ(α)
336        // Simplified without digamma function: α + log(θ) + log(Γ(α))
337        let concentration_data = self.concentration.data.as_slice().unwrap();
338        let scale = self.get_scale()?;
339        let scale_data = scale.data.as_slice().unwrap();
340
341        let result_data: Vec<T> = concentration_data
342            .iter()
343            .zip(scale_data.iter().cycle())
344            .map(|(&alpha, &theta)| alpha + theta.ln() + Self::log_gamma_approx(alpha))
345            .collect();
346
347        Ok(Tensor::from_vec(
348            result_data,
349            self.concentration.shape().to_vec(),
350        ))
351    }

458        // Test some known values
459        // Γ(1) = 1, so log(Γ(1)) = 0
460        let log_gamma_1 = Gamma::<f32>::log_gamma_approx(1.0);
461        assert_abs_diff_eq!(log_gamma_1, 0.0, epsilon = 0.1);
462
463        // Γ(2) = 1, so log(Γ(2)) = 0
464        let log_gamma_2 = Gamma::<f32>::log_gamma_approx(2.0);
465        assert_abs_diff_eq!(log_gamma_2, 0.0, epsilon = 0.1);
466
467        // Γ(3) = 2, so log(Γ(3)) = log(2) ≈ 0.693
468        let log_gamma_3 = Gamma::<f32>::log_gamma_approx(3.0);
469        assert_abs_diff_eq!(log_gamma_3, 2.0f32.ln(), epsilon = 0.1);
