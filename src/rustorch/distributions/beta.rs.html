12/// Beta Distribution
13/// ベータ分布
14///
15/// The Beta distribution is defined by:
16/// - concentration1 (α): first shape parameter (must be positive)
17/// - concentration0 (β): second shape parameter (must be positive)
18///
19/// PDF: f(x) = (Γ(α+β) / (Γ(α)Γ(β))) * x^(α-1) * (1-x)^(β-1) for x ∈ \[0,1\]
20#[derive(Debug, Clone)]
21pub struct Beta<T: Float> {
22    /// First concentration parameter (α)
23    /// 第一濃度パラメータ (α)
24    pub concentration1: Tensor<T>,
25
26    /// Second concentration parameter (β)
27    /// 第二濃度パラメータ (β)
28    pub concentration0: Tensor<T>,
29
30    /// Base distribution properties
31    /// 基本分布特性
32    pub base: Distribution,
33
34    /// Phantom data for type parameter
35    /// 型パラメータ用のファントムデータ
36    _phantom: PhantomData<T>,
37}

39impl<T: Float + 'static> Beta<T>
40where
41    T: rand::distributions::uniform::SampleUniform + num_traits::FromPrimitive + std::fmt::Display,
42{
43    /// Create a new Beta distribution
44    /// 新しいベータ分布を作成
45    ///
46    /// # Arguments
47    /// * `concentration1` - First concentration (α) parameter tensor
48    /// * `concentration0` - Second concentration (β) parameter tensor
49    /// * `validate_args` - Whether to validate parameters
50    pub fn new(
51        concentration1: Tensor<T>,
52        concentration0: Tensor<T>,
53        validate_args: bool,
54    ) -> RusTorchResult<Self> {
55        if validate_args {
56            DistributionUtils::validate_positive(&concentration1, "concentration1")?;
57            DistributionUtils::validate_positive(&concentration0, "concentration0")?;
58        }
59
60        // Determine batch shape from broadcasting concentration parameters
61        let batch_shape =
62            Distribution::broadcast_shapes(concentration1.shape(), concentration0.shape())?;
63        let event_shape = vec![]; // Beta is a univariate distribution
64
65        Ok(Self {
66            concentration1,
67            concentration0,
68            base: Distribution::new(batch_shape, event_shape, validate_args),
69            _phantom: PhantomData,
70        })
71    }
72
73    /// Create Beta distribution with scalar parameters
74    /// スカラーパラメータでベータ分布を作成
75    pub fn from_scalars(alpha: T, beta: T, validate_args: bool) -> RusTorchResult<Self> {
76        let alpha_tensor = Tensor::from_vec(vec![alpha], vec![]);
77        let beta_tensor = Tensor::from_vec(vec![beta], vec![]);
78        Self::new(alpha_tensor, beta_tensor, validate_args)
79    }
80
81    /// Uniform Beta distribution (α=1, β=1) - equivalent to Uniform(0,1)
82    /// 一様ベータ分布 (α=1, β=1) - Uniform(0,1)と等価
83    pub fn uniform(validate_args: bool) -> RusTorchResult<Self> {
84        Self::from_scalars(T::one(), T::one(), validate_args)
85    }

87    /// Symmetric Beta distribution with equal parameters (α=β)
88    /// 等しいパラメータを持つ対称ベータ分布 (α=β)
89    pub fn symmetric(concentration: T, validate_args: bool) -> RusTorchResult<Self> {
90        Self::from_scalars(concentration, concentration, validate_args)
91    }

93    /// Log-beta function: log(B(α, β)) = log(Γ(α)) + log(Γ(β)) - log(Γ(α+β))
94    /// ログベータ関数: log(B(α, β)) = log(Γ(α)) + log(Γ(β)) - log(Γ(α+β))
95    fn log_beta(&self) -> RusTorchResult<Tensor<T>> {
96        let alpha_data = self.concentration1.data.as_slice().unwrap();
97        let beta_data = self.concentration0.data.as_slice().unwrap();
98
99        let result_data: Vec<T> = alpha_data
100            .iter()
101            .zip(beta_data.iter().cycle())
102            .map(|(&a, &b)| {
103                // Use Stirling's approximation for log-gamma
104                Self::log_gamma_stirling(a) + Self::log_gamma_stirling(b)
105                    - Self::log_gamma_stirling(a + b)
106            })
107            .collect();
108
109        Ok(Tensor::from_vec(
110            result_data,
111            self.concentration1.shape().to_vec(),
112        ))
113    }
114
115    /// Log-gamma function using Stirling's approximation
116    /// スターリング近似を使ったログガンマ関数
117    fn log_gamma_stirling(x: T) -> T {
118        if x < T::one() {
119            // For x < 1, use recurrence relation: Γ(x) = Γ(x+1) / x
120            Self::log_gamma_stirling(x + T::one()) - x.ln()
121        } else if x < T::from(12.0).unwrap() {
122            // Use series expansion for better accuracy
123            let mut result = T::zero();
124            let mut term = x;
125
126            // Use recurrence to get to x >= 12
127            while term < T::from(12.0).unwrap() {
128                result = result - term.ln();
129                term = term + T::one();
130            }
131
132            result + Self::stirling_approx(term)
133        } else {
134            Self::stirling_approx(x)
135        }
136    }
137
138    /// Stirling's approximation for log(Γ(x))
139    /// log(Γ(x))のスターリング近似
140    fn stirling_approx(x: T) -> T {
141        let half = T::from(0.5).unwrap();
142        let two = T::from(2.0).unwrap();
143        let pi = T::from(std::f64::consts::PI).unwrap();
144        let twelve = T::from(12.0).unwrap();
145
146        // Stirling: log(Γ(x)) ≈ (x - 0.5) * log(x) - x + 0.5 * log(2π) + 1/(12x)
147        (x - half) * x.ln() - x + half * (two * pi).ln() + T::one() / (twelve * x)
148    }
149
150    /// Digamma function (ψ) - derivative of log-gamma
151    /// ディガンマ関数 (ψ) - ログガンマの導関数
152    fn digamma(x: T) -> T {
153        if x < T::from(6.0).unwrap() {
154            // For small x, use recurrence relation
155            Self::digamma(x + T::one()) - T::one() / x
156        } else {
157            // Asymptotic expansion for large x
158            let inv_x = T::one() / x;
159            let inv_x_sq = inv_x * inv_x;
160
161            x.ln() - T::from(0.5).unwrap() * inv_x - inv_x_sq / T::from(12.0).unwrap()
162                + inv_x_sq * inv_x_sq / T::from(120.0).unwrap()
163        }
164    }
165
166    /// Sample using rejection sampling (for general case)
167    /// 棄却サンプリングを使用してサンプル生成（一般的なケース）
168    fn sample_rejection(&self, shape: &[usize]) -> RusTorchResult<Tensor<T>> {
169        let total_size: usize = shape.iter().product();
170        let mut samples = Vec::with_capacity(total_size);
171
172        let alpha_data = self.concentration1.data.as_slice().unwrap();
173        let beta_data = self.concentration0.data.as_slice().unwrap();
174
175        for i in 0..total_size {
176            let alpha = alpha_data[i % alpha_data.len()];
177            let beta = beta_data[i % beta_data.len()];
178
179            // For beta distribution, we can use gamma sampling: X = G1/(G1+G2)
180            // where G1 ~ Gamma(α,1) and G2 ~ Gamma(β,1)
181            let g1 = Self::sample_gamma_marsaglia(alpha);
182            let g2 = Self::sample_gamma_marsaglia(beta);
183
184            samples.push(g1 / (g1 + g2));
185        }
186
187        Ok(Tensor::from_vec(samples, shape.to_vec()))
188    }
189
190    /// Sample from Gamma distribution using Marsaglia-Tsang method
191    /// マルサグリア・ツァング法を使ってガンマ分布からサンプル
192    fn sample_gamma_marsaglia(shape: T) -> T {
193        if shape < T::one() {
194            // For shape < 1, use transformation: if Y ~ Gamma(shape+1), then Y * U^(1/shape) ~ Gamma(shape)
195            let y = Self::sample_gamma_marsaglia(shape + T::one());
196            let u = DistributionUtils::random_uniform_scalar::<T>();
197            y * u.powf(T::one() / shape)
198        } else {
199            // Marsaglia-Tsang algorithm for shape >= 1
200            #[allow(clippy::many_single_char_names)]
201            // Mathematical variables d, c, x, v, u are conventional in statistical algorithms
202            {
203                let d = shape - T::from(1.0 / 3.0).unwrap();
204                let c = T::one() / (T::from(9.0).unwrap() * d).sqrt();
205
206                loop {
207                    let mut x = DistributionUtils::random_normal_scalar::<T>();
208                    let v = (T::one() + c * x).powi(3);
209
210                    if v <= T::zero() {
211                        continue;
212                    }
213
214                    x = x * x;
215                    let u = DistributionUtils::random_uniform_scalar::<T>();
216
217                    if u < T::one() - T::from(0.0331).unwrap() * x * x {
218                        return d * v;
219                    }
220
221                    if u.ln() < T::from(0.5).unwrap() * x + d * (T::one() - v + v.ln()) {
222                        return d * v;
223                    }
224                }
225            }
226        }
227    }

239    fn log_prob(&self, value: &Tensor<T>) -> RusTorchResult<Tensor<T>> {
240        let value_data = value.data.as_slice().unwrap();
241        let alpha_data = self.concentration1.data.as_slice().unwrap();
242        let beta_data = self.concentration0.data.as_slice().unwrap();
243
244        let log_beta_vals = self.log_beta()?;
245        let log_beta_data = log_beta_vals.data.as_slice().unwrap();
246
247        let neg_inf = T::neg_infinity();
248        let result_data: Vec<T> = value_data
249            .iter()
250            .zip(alpha_data.iter().cycle())
251            .zip(beta_data.iter().cycle())
252            .zip(log_beta_data.iter().cycle())
253            .map(|(((&v, &a), &b), &lb)| {
254                if v <= T::zero() || v >= T::one() {
255                    neg_inf
256                } else {
257                    // log p(x) = (α-1)*log(x) + (β-1)*log(1-x) - log(B(α,β))
258                    (a - T::one()) * v.ln() + (b - T::one()) * (T::one() - v).ln() - lb
259                }
260            })
261            .collect();
262
263        Ok(Tensor::from_vec(result_data, value.shape().to_vec()))
264    }
265
266    fn cdf(&self, value: &Tensor<T>) -> RusTorchResult<Tensor<T>> {
267        let value_data = value.data.as_slice().unwrap();
268        let alpha_data = self.concentration1.data.as_slice().unwrap();
269        let beta_data = self.concentration0.data.as_slice().unwrap();
270
271        let result_data: Vec<T> = value_data
272            .iter()
273            .zip(alpha_data.iter().cycle())
274            .zip(beta_data.iter().cycle())
275            .map(|((&v, &a), &b)| {
276                if v <= T::zero() {
277                    T::zero()
278                } else if v >= T::one() {
279                    T::one()
280                } else {
281                    // Use incomplete beta function approximation
282                    Self::incomplete_beta_approx(v, a, b)
283                }
284            })
285            .collect();
286
287        Ok(Tensor::from_vec(result_data, value.shape().to_vec()))
288    }

297    fn mean(&self) -> RusTorchResult<Tensor<T>> {
298        // Mean = α / (α + β)
299        let alpha_data = self.concentration1.data.as_slice().unwrap();
300        let beta_data = self.concentration0.data.as_slice().unwrap();
301
302        let mean_data: Vec<T> = alpha_data
303            .iter()
304            .zip(beta_data.iter().cycle())
305            .map(|(&a, &b)| a / (a + b))
306            .collect();
307
308        Ok(Tensor::from_vec(
309            mean_data,
310            self.concentration1.shape().to_vec(),
311        ))
312    }
313
314    fn variance(&self) -> RusTorchResult<Tensor<T>> {
315        // Variance = (α*β) / ((α+β)²*(α+β+1))
316        let alpha_data = self.concentration1.data.as_slice().unwrap();
317        let beta_data = self.concentration0.data.as_slice().unwrap();
318
319        let var_data: Vec<T> = alpha_data
320            .iter()
321            .zip(beta_data.iter().cycle())
322            .map(|(&a, &b)| {
323                let sum = a + b;
324                (a * b) / (sum * sum * (sum + T::one()))
325            })
326            .collect();
327
328        Ok(Tensor::from_vec(
329            var_data,
330            self.concentration1.shape().to_vec(),
331        ))
332    }
333
334    fn entropy(&self) -> RusTorchResult<Tensor<T>> {
335        let alpha_data = self.concentration1.data.as_slice().unwrap();
336        let beta_data = self.concentration0.data.as_slice().unwrap();
337        let log_beta_vals = self.log_beta()?;
338        let log_beta_data = log_beta_vals.data.as_slice().unwrap();
339
340        let result_data: Vec<T> = alpha_data
341            .iter()
342            .zip(beta_data.iter().cycle())
343            .zip(log_beta_data.iter().cycle())
344            .map(|((&a, &b), &lb)| {
345                let sum = a + b;
346                // Entropy = log(B(α,β)) - (α-1)*ψ(α) - (β-1)*ψ(β) + (α+β-2)*ψ(α+β)
347                lb - (a - T::one()) * Self::digamma(a) - (b - T::one()) * Self::digamma(b)
348                    + (sum - T::from(2.0).unwrap()) * Self::digamma(sum)
349            })
350            .collect();
351
352        Ok(Tensor::from_vec(
353            result_data,
354            self.concentration1.shape().to_vec(),
355        ))
356    }
357}

359impl<T: Float + 'static> Beta<T>
360where
361    T: rand::distributions::uniform::SampleUniform + num_traits::FromPrimitive + std::fmt::Display,
362{
363    /// Incomplete beta function approximation
364    /// 不完全ベータ関数近似
365    fn incomplete_beta_approx(x: T, a: T, b: T) -> T {
366        // Simple continued fraction approximation for incomplete beta
367        let mut cf = T::one();
368        let mut d = T::one() / (T::one() - (a + b) * x / (a + T::one()));
369        cf = cf * d;
370
371        for m in 1..50 {
372            let m_f = T::from(m).unwrap();
373            let two_m = T::from(2 * m).unwrap();
374
375            let num = m_f * (b - m_f) * x / ((a + two_m - T::one()) * (a + two_m));
376            d = T::one() / (T::one() + num * d);
377            cf = cf * d;
378
379            let num2 = -(a + m_f) * (a + b + m_f) * x / ((a + two_m) * (a + two_m + T::one()));
380            d = T::one() / (T::one() + num2 * d);
381            cf = cf * d;
382        }
383
384        let beta_approx =
385            x.powf(a) * (T::one() - x).powf(b) / (a * Self::beta_function_approx(a, b));
386        beta_approx * cf
387    }
388
389    /// Beta function approximation
390    /// ベータ関数近似
391    fn beta_function_approx(a: T, b: T) -> T {
392        (Self::log_gamma_stirling(a) + Self::log_gamma_stirling(b)
393            - Self::log_gamma_stirling(a + b))
394        .exp()
395    }
396}
