37/// Distribution trait for all statistical distributions
38/// 全ての統計分布のための分布トレイト
39pub trait DistributionTrait<T: Float + 'static> {
40    /// Sample from the distribution
41    /// 分布からサンプルを生成
42    fn sample(&self, shape: Option<&[usize]>) -> RusTorchResult<Tensor<T>>;
43
44    /// Sample with replacement
45    /// 復元抽出でサンプル
46    fn sample_n(&self, n: usize) -> RusTorchResult<Tensor<T>> {
47        self.sample(Some(&[n]))
48    }
49
50    /// Compute log probability density/mass function
51    /// 対数確率密度/質量関数を計算
52    fn log_prob(&self, value: &Tensor<T>) -> RusTorchResult<Tensor<T>>;
53
54    /// Compute probability density/mass function
55    /// 確率密度/質量関数を計算
56    fn prob(&self, value: &Tensor<T>) -> RusTorchResult<Tensor<T>> {
57        let log_p = self.log_prob(value)?;
58        // Apply exp to convert log prob to prob
59        let data = log_p.data.as_slice().unwrap();
60        let prob_data: Vec<T> = data.iter().map(|&x| x.exp()).collect();
61        Ok(Tensor::from_vec(prob_data, log_p.shape().to_vec()))
62    }
63
64    /// Compute cumulative distribution function
65    /// 累積分布関数を計算
66    fn cdf(&self, value: &Tensor<T>) -> RusTorchResult<Tensor<T>>;
67
68    /// Compute inverse cumulative distribution function (quantile function)
69    /// 逆累積分布関数（分位点関数）を計算
70    fn icdf(&self, value: &Tensor<T>) -> RusTorchResult<Tensor<T>>;
71
72    /// Get distribution mean
73    /// 分布の平均を取得
74    fn mean(&self) -> RusTorchResult<Tensor<T>>;
75
76    /// Get distribution variance
77    /// 分布の分散を取得
78    fn variance(&self) -> RusTorchResult<Tensor<T>>;
79
80    /// Get distribution standard deviation
81    /// 分布の標準偏差を取得
82    fn stddev(&self) -> RusTorchResult<Tensor<T>> {
83        let var = self.variance()?;
84        let data = var.data.as_slice().unwrap();
85        let stddev_data: Vec<T> = data.iter().map(|&x| x.sqrt()).collect();
86        Ok(Tensor::from_vec(stddev_data, var.shape().to_vec()))
87    }
88
89    /// Get distribution entropy
90    /// 分布のエントロピーを取得
91    fn entropy(&self) -> RusTorchResult<Tensor<T>>;
92}

98impl DistributionUtils {
99    /// Validate probability parameter (0 <= p <= 1)
100    /// 確率パラメータの検証 (0 <= p <= 1)
101    pub fn validate_probability<T: Float + std::fmt::Display>(p: &Tensor<T>) -> RusTorchResult<()> {
102        let data = p.data.as_slice().unwrap();
103        for &val in data {
104            if val < T::zero() || val > T::one() {
105                return Err(RusTorchError::invalid_parameter(&format!(
106                    "Probability must be in [0, 1], got {}",
107                    val
108                )));
109            }
110        }
111        Ok(())
112    }
113
114    /// Validate positive parameter (> 0)
115    /// 正のパラメータの検証 (> 0)
116    pub fn validate_positive<T: Float + std::fmt::Display>(
117        param: &Tensor<T>,
118        name: &str,
119    ) -> RusTorchResult<()> {
120        let data = param.data.as_slice().unwrap();
121        for &val in data {
122            if val <= T::zero() {
123                return Err(RusTorchError::invalid_parameter(&format!(
124                    "{} must be positive, got {}",
125                    name, val
126                )));
127            }
128        }
129        Ok(())
130    }
131
132    /// Validate non-negative parameter (>= 0)
133    /// 非負パラメータの検証 (>= 0)
134    pub fn validate_non_negative<T: Float + std::fmt::Display>(
135        param: &Tensor<T>,
136        name: &str,
137    ) -> RusTorchResult<()> {
138        let data = param.data.as_slice().unwrap();
139        for &val in data {
140            if val < T::zero() {
141                return Err(RusTorchError::invalid_parameter(&format!(
142                    "{} must be non-negative, got {}",
143                    name, val
144                )));
145            }
146        }
147        Ok(())
148    }

150    /// Generate random uniform samples
151    /// ランダムな一様サンプルを生成
152    pub fn random_uniform<T>(shape: &[usize]) -> Tensor<T>
153    where
154        T: Float + 'static + rand::distributions::uniform::SampleUniform,
155    {
156        use rand::{thread_rng, Rng};
157        let mut rng = thread_rng();
158        let size: usize = shape.iter().product();
159        let data: Vec<T> = (0..size)
160            .map(|_| rng.gen_range(T::zero()..T::one()))
161            .collect();
162        Tensor::from_vec(data, shape.to_vec())
163    }
164
165    /// Generate random normal samples using Box-Muller transform
166    /// Box-Muller変換を使用してランダムな正規サンプルを生成
167    pub fn random_normal<T>(shape: &[usize]) -> Tensor<T>
168    where
169        T: Float + 'static + rand::distributions::uniform::SampleUniform,
170    {
171        use rand::{thread_rng, Rng};
172        let mut rng = thread_rng();
173        let size: usize = shape.iter().product();
174        let mut data = Vec::with_capacity(size);
175
176        // Box-Muller transform for normal distribution
177        let pi = T::from(std::f64::consts::PI).unwrap();
178        let two = T::from(2.0).unwrap();
179
180        for _ in 0..size.div_ceil(2) {
181            let u1: T = rng.gen_range(T::from(1e-10).unwrap()..T::one());
182            let u2: T = rng.gen_range(T::zero()..T::one());
183
184            let z0 = (-two * u1.ln()).sqrt() * (two * pi * u2).cos();
185            let z1 = (-two * u1.ln()).sqrt() * (two * pi * u2).sin();
186
187            data.push(z0);
188            if data.len() < size {
189                data.push(z1);
190            }
191        }
192
193        data.truncate(size);
194        Tensor::from_vec(data, shape.to_vec())
195    }

197    /// Generate single random uniform scalar
198    /// 単一のランダムな一様スカラーを生成
199    pub fn random_uniform_scalar<T>() -> T
200    where
201        T: Float + 'static + rand::distributions::uniform::SampleUniform,
202    {
203        use rand::{thread_rng, Rng};
204        let mut rng = thread_rng();
205        rng.gen_range(T::zero()..T::one())
206    }

208    /// Generate single random normal scalar
209    /// 単一のランダムな正規スカラーを生成
210    pub fn random_normal_scalar<T>() -> T
211    where
212        T: Float + 'static + rand::distributions::uniform::SampleUniform,
213    {
214        use rand::{thread_rng, Rng};
215        let mut rng = thread_rng();
216
217        // Box-Muller transform
218        let pi = T::from(std::f64::consts::PI).unwrap();
219        let two = T::from(2.0).unwrap();
220
221        let u1: T = rng.gen_range(T::from(1e-10).unwrap()..T::one());
222        let u2: T = rng.gen_range(T::zero()..T::one());
223
224        (-two * u1.ln()).sqrt() * (two * pi * u2).cos()
225    }

232    #[test]
233    fn test_validation_utils() {
234        // Test probability validation
235        let valid_prob = Tensor::from_vec([0.0f32, 0.5, 1.0].to_vec(), vec![3]);
236        assert!(DistributionUtils::validate_probability(&valid_prob).is_ok());
237
238        let invalid_prob = Tensor::from_vec([0.0f32, 0.5, 1.5].to_vec(), vec![3]);
239        assert!(DistributionUtils::validate_probability(&invalid_prob).is_err());
240
241        // Test positive validation
242        let positive = Tensor::from_vec([0.1f32, 1.0, 2.0].to_vec(), vec![3]);
243        assert!(DistributionUtils::validate_positive(&positive, "param").is_ok());
244
245        let non_positive = Tensor::from_vec([0.0f32, 1.0, 2.0].to_vec(), vec![3]);
246        assert!(DistributionUtils::validate_positive(&non_positive, "param").is_err());
247    }

249    #[test]
250    fn test_random_generators() {
251        // Test uniform random generation
252        let uniform = DistributionUtils::random_uniform::<f32>(&[10, 5]);
253        assert_eq!(uniform.shape(), &[10, 5]);
254
255        let data = uniform.data.as_slice().unwrap();
256        for &val in data {
257            assert!(val >= 0.0 && val <= 1.0);
258        }
259
260        // Test normal random generation
261        let normal = DistributionUtils::random_normal::<f32>(&[100]);
262        assert_eq!(normal.shape(), &[100]);
263
264        // Basic statistical test - mean should be close to 0
265        let data = normal.data.as_slice().unwrap();
266        let mean: f32 = data.iter().sum::<f32>() / data.len() as f32;
267        assert!(mean.abs() < 0.5); // Should be close to 0 for large sample
268    }
