4/// 正規（ガウス）分布 - torch.distributions.Normal互換

6/// Implements the normal distribution with mean μ and standard deviation σ.
7/// 平均μと標準偏差σを持つ正規分布を実装
8use crate::tensor::Tensor;
9use num_traits::Float;

12/// Normal Distribution
13/// 正規分布
14///
15/// The normal distribution is defined by:
16/// - Mean (μ): location parameter
17/// - Standard deviation (σ): scale parameter (must be positive)
18///
19/// PDF: f(x) = (1 / (σ√(2π))) * exp(-0.5 * ((x - μ) / σ)²)
20#[derive(Debug, Clone)]
21pub struct Normal<T: Float> {
22    /// Mean parameter (μ)
23    /// 平均パラメータ (μ)
24    pub loc: Tensor<T>,
25
26    /// Standard deviation parameter (σ)  
27    /// 標準偏差パラメータ (σ)
28    pub scale: Tensor<T>,
29
30    /// Base distribution properties
31    /// 基本分布特性
32    pub base: Distribution,
33
34    /// Phantom data for type parameter
35    /// 型パラメータ用のファントムデータ
36    _phantom: PhantomData<T>,
37}
38
39impl<T: Float + 'static> Normal<T>
40where
41    T: rand::distributions::uniform::SampleUniform + num_traits::FromPrimitive + std::fmt::Display,
42{
43    /// Create a new Normal distribution
44    /// 新しい正規分布を作成
45    ///
46    /// # Arguments
47    /// * `loc` - Mean (μ) parameter tensor
48    /// * `scale` - Standard deviation (σ) parameter tensor  
49    /// * `validate_args` - Whether to validate parameters
50    pub fn new(loc: Tensor<T>, scale: Tensor<T>, validate_args: bool) -> RusTorchResult<Self> {
51        if validate_args {
52            DistributionUtils::validate_positive(&scale, "scale")?;
53        }
54
55        // Determine batch shape from broadcasting loc and scale
56        let batch_shape = Distribution::broadcast_shapes(loc.shape(), scale.shape())?;
57        let event_shape = vec![]; // Normal is a univariate distribution
58
59        Ok(Self {
60            loc,
61            scale,
62            base: Distribution::new(batch_shape, event_shape, validate_args),
63            _phantom: PhantomData,
64        })
65    }
66
67    /// Create Normal distribution with scalar parameters
68    /// スカラーパラメータで正規分布を作成
69    pub fn from_scalars(loc: T, scale: T, validate_args: bool) -> RusTorchResult<Self> {
70        let loc_tensor = Tensor::from_vec(vec![loc], vec![]);
71        let scale_tensor = Tensor::from_vec(vec![scale], vec![]);
72        Self::new(loc_tensor, scale_tensor, validate_args)
73    }
74
75    /// Standard normal distribution (μ=0, σ=1)
76    /// 標準正規分布 (μ=0, σ=1)
77    pub fn standard(validate_args: bool) -> RusTorchResult<Self> {
78        Self::from_scalars(T::zero(), T::one(), validate_args)
79    }
80
81    /// Compute the log of the normalization constant
82    /// 正規化定数の対数を計算
83    fn log_normalizing_constant(&self) -> RusTorchResult<Tensor<T>> {
84        let pi = T::from(std::f64::consts::PI).unwrap();
85        let two = T::from(2.0).unwrap();
86
87        // log(σ√(2π)) = log(σ) + 0.5 * log(2π)
88        let scale_data = self.scale.data.as_slice().unwrap();
89        let log_scale: Vec<T> = scale_data.iter().map(|&s| s.ln()).collect();
90
91        let half_log_2pi = T::from(0.5).unwrap() * (two * pi).ln();
92        let result_data: Vec<T> = log_scale.iter().map(|&ls| ls + half_log_2pi).collect();
93
94        Ok(Tensor::from_vec(result_data, self.scale.shape().to_vec()))
95    }
96
97    /// Compute standardized values (z-scores)
98    /// 標準化された値（zスコア）を計算
99    fn standardize(&self, value: &Tensor<T>) -> RusTorchResult<Tensor<T>> {
100        // z = (x - μ) / σ
101        let loc_data = self.loc.data.as_slice().unwrap();
102        let scale_data = self.scale.data.as_slice().unwrap();
103        let value_data = value.data.as_slice().unwrap();
104
105        let result_data: Vec<T> = value_data
106            .iter()
107            .zip(loc_data.iter().cycle())
108            .zip(scale_data.iter().cycle())
109            .map(|((&v, &l), &s)| (v - l) / s)
110            .collect();
111
112        Ok(Tensor::from_vec(result_data, value.shape().to_vec()))
113    }
114
115    /// Error function approximation for CDF calculation
116    /// CDF計算のための誤差関数近似
117    fn erf_approx(x: T) -> T {
118        // Abramowitz and Stegun approximation
119        let a1 = T::from(0.254_829_592).unwrap();
120        let a2 = T::from(-0.284_496_736).unwrap();
121        let a3 = T::from(1.421_413_741).unwrap();
122        let a4 = T::from(-1.453_152_027).unwrap();
123        let a5 = T::from(1.061_405_429).unwrap();
124        let p = T::from(0.327_591_1).unwrap();
125
126        let sign = if x >= T::zero() {
127            T::one()
128        } else {
129            T::from(-1.0).unwrap()
130        };
131        let x_abs = x.abs();
132
133        // A&S formula 7.1.26
134        let t = T::one() / (T::one() + p * x_abs);
135        let y = T::one()
136            - (((((a5 * t + a4) * t) + a3) * t + a2) * t + a1) * t * (-x_abs * x_abs).exp();
137
138        sign * y
139    }
140
141    /// Inverse error function approximation
142    /// 逆誤差関数近似
143    fn erf_inv_approx(x: T) -> T {
144        // Beasley-Springer-Moro approximation for inverse erf
145        let a = T::from(0.147).unwrap();
146        let two = T::from(2.0).unwrap();
147        let pi = T::from(std::f64::consts::PI).unwrap();
148
149        let ln_term = (T::one() - x * x).ln();
150        let first_term = two / (pi * a) + ln_term / two;
151
152        let sqrt_term = (first_term * first_term - ln_term / a).sqrt();
153        let sign = if x >= T::zero() {
154            T::one()
155        } else {
156            T::from(-1.0).unwrap()
157        };
158
159        sign * (sqrt_term - first_term).sqrt()
160    }
161}

163impl<T: Float + 'static> DistributionTrait<T> for Normal<T>
164where
165    T: rand::distributions::uniform::SampleUniform + num_traits::FromPrimitive + std::fmt::Display,
166{
167    fn sample(&self, shape: Option<&[usize]>) -> RusTorchResult<Tensor<T>> {
168        let sample_shape = self.base.expand_shape(shape);
169
170        // Generate standard normal samples using Box-Muller
171        let standard_normal = DistributionUtils::random_normal::<T>(&sample_shape);
172
173        // Transform to desired distribution: X = μ + σ * Z
174        let standard_data = standard_normal.data.as_slice().unwrap();
175        let loc_data = self.loc.data.as_slice().unwrap();
176        let scale_data = self.scale.data.as_slice().unwrap();
177
178        let result_data: Vec<T> = standard_data
179            .iter()
180            .zip(loc_data.iter().cycle())
181            .zip(scale_data.iter().cycle())
182            .map(|((&z, &l), &s)| l + s * z)
183            .collect();
184
185        Ok(Tensor::from_vec(result_data, sample_shape))
186    }
187
188    fn log_prob(&self, value: &Tensor<T>) -> RusTorchResult<Tensor<T>> {
189        // log p(x) = -0.5 * ((x - μ) / σ)² - log(σ√(2π))
190        let standardized = self.standardize(value)?;
191        let log_norm = self.log_normalizing_constant()?;
192
193        let std_data = standardized.data.as_slice().unwrap();
194        let log_norm_data = log_norm.data.as_slice().unwrap();
195
196        let half = T::from(-0.5).unwrap();
197        let result_data: Vec<T> = std_data
198            .iter()
199            .zip(log_norm_data.iter().cycle())
200            .map(|(&z, &ln)| half * z * z - ln)
201            .collect();
202
203        Ok(Tensor::from_vec(result_data, value.shape().to_vec()))
204    }
205
206    fn cdf(&self, value: &Tensor<T>) -> RusTorchResult<Tensor<T>> {
207        // CDF = 0.5 * (1 + erf((x - μ) / (σ√2)))
208        let sqrt_2 = T::from(2.0_f64.sqrt()).unwrap();
209        let half = T::from(0.5).unwrap();
210
211        let standardized = self.standardize(value)?;
212        let std_data = standardized.data.as_slice().unwrap();
213
214        let result_data: Vec<T> = std_data
215            .iter()
216            .map(|&z| half * (T::one() + Self::erf_approx(z / sqrt_2)))
217            .collect();
218
219        Ok(Tensor::from_vec(result_data, value.shape().to_vec()))
220    }
221
222    fn icdf(&self, value: &Tensor<T>) -> RusTorchResult<Tensor<T>> {
223        // ICDF = μ + σ * √2 * erf⁻¹(2p - 1)
224        let sqrt_2 = T::from(2.0_f64.sqrt()).unwrap();
225        let two = T::from(2.0).unwrap();
226
227        let value_data = value.data.as_slice().unwrap();
228        let loc_data = self.loc.data.as_slice().unwrap();
229        let scale_data = self.scale.data.as_slice().unwrap();
230
231        let result_data: Vec<T> = value_data
232            .iter()
233            .zip(loc_data.iter().cycle())
234            .zip(scale_data.iter().cycle())
235            .map(|((&p, &l), &s)| {
236                let erf_input = two * p - T::one();
237                l + s * sqrt_2 * Self::erf_inv_approx(erf_input)
238            })
239            .collect();
240
241        Ok(Tensor::from_vec(result_data, value.shape().to_vec()))
242    }
243
244    fn mean(&self) -> RusTorchResult<Tensor<T>> {
245        Ok(self.loc.clone())
246    }
247
248    fn variance(&self) -> RusTorchResult<Tensor<T>> {
249        let scale_data = self.scale.data.as_slice().unwrap();
250        let var_data: Vec<T> = scale_data.iter().map(|&s| s * s).collect();
251        Ok(Tensor::from_vec(var_data, self.scale.shape().to_vec()))
252    }
253
254    fn entropy(&self) -> RusTorchResult<Tensor<T>> {
255        // Entropy = 0.5 * log(2πe * σ²) = log(σ) + 0.5 * log(2πe)
256        let pi = T::from(std::f64::consts::PI).unwrap();
257        let e = T::from(std::f64::consts::E).unwrap();
258        let two = T::from(2.0).unwrap();
259        let half = T::from(0.5).unwrap();
260
261        let scale_data = self.scale.data.as_slice().unwrap();
262        let constant = half * (two * pi * e).ln();
263
264        let result_data: Vec<T> = scale_data.iter().map(|&s| s.ln() + constant).collect();
265
266        Ok(Tensor::from_vec(result_data, self.scale.shape().to_vec()))
267    }

295    #[test]
296    fn test_normal_sampling() {
297        let normal = Normal::<f32>::standard(true).unwrap();
298        let samples = normal.sample(Some(&[1000])).unwrap();
299
300        assert_eq!(samples.shape(), &[1000]);
301
302        // Basic statistical test
303        let data = samples.data.as_slice().unwrap();
304        let sample_mean: f32 = data.iter().sum::<f32>() / data.len() as f32;
305        let sample_var: f32 =
306            data.iter().map(|&x| (x - sample_mean).powi(2)).sum::<f32>() / (data.len() - 1) as f32;
307
308        // Should be approximately 0 and 1 for large samples
309        // Use more relaxed tolerances to account for statistical variance
310        assert_abs_diff_eq!(sample_mean, 0.0, epsilon = 0.15);
311        assert_abs_diff_eq!(sample_var, 1.0, epsilon = 0.2);
312    }

314    #[test]
315    fn test_normal_log_prob() {
316        let normal = Normal::<f32>::standard(true).unwrap();
317        let values = Tensor::from_vec(vec![0.0f32, 1.0, -1.0], vec![3]);
318
319        let log_probs = normal.log_prob(&values).unwrap();
320        let log_prob_data = log_probs.data.as_slice().unwrap();
321
322        // For standard normal, log p(0) = -0.5 * log(2π) ≈ -0.9189
323        assert_abs_diff_eq!(log_prob_data[0], -0.9189f32, epsilon = 0.01);
324
325        // Symmetric around 0
326        assert_abs_diff_eq!(log_prob_data[1], log_prob_data[2], epsilon = 1e-6);
327    }

341        assert_abs_diff_eq!(cdf_data[1], 0.975, epsilon = 0.01);

343        // Symmetric around 0.5

345    }
