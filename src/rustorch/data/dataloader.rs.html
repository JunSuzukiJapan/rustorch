15/// Main DataLoader implementation using Phase 5 Dataset trait
16/// フェーズ5 Datasetトレイトを使用するメインDataLoader実装
17pub struct DataLoader<'a, T: Float, D: Dataset<T>> {
18    dataset: &'a D,
19    sampler: Box<dyn Sampler + Send + Sync>,
20    batch_size: usize,
21    drop_last: bool,
22    num_workers: usize,
23    _phantom: std::marker::PhantomData<T>,
24}
25
26impl<'a, T: Float, D: Dataset<T>> DataLoader<'a, T, D> {
27    /// Create new DataLoader
28    /// 新しいDataLoaderを作成
29    pub fn new(dataset: &'a D, sampler: Box<dyn Sampler + Send + Sync>, batch_size: usize) -> Self {
30        Self {
31            dataset,
32            sampler,
33            batch_size,
34            drop_last: false,
35            num_workers: 1,
36            _phantom: std::marker::PhantomData,
37        }
38    }
39
40    /// Create DataLoader with all options
41    /// 全オプション付きDataLoaderを作成
42    pub fn with_options(
43        dataset: &'a D,
44        sampler: Box<dyn Sampler + Send + Sync>,
45        batch_size: usize,
46        drop_last: bool,
47        num_workers: usize,
48    ) -> Self {
49        Self {
50            dataset,
51            sampler,
52            batch_size,
53            drop_last,
54            num_workers,
55            _phantom: std::marker::PhantomData,
56        }
57    }
58
59    /// Get next batch
60    /// 次のバッチを取得
61    pub fn next_batch(&mut self) -> Option<Vec<T>> {
62        let mut indices = Vec::new();
63
64        for _ in 0..self.batch_size {
65            if let Some(idx) = self.sampler.sample() {
66                indices.push(idx);
67            } else {
68                break;
69            }
70        }
71
72        if indices.is_empty() {
73            return None;
74        }
75
76        if self.drop_last && indices.len() < self.batch_size {
77            return None;
78        }
79
80        // Collect batch items
81        let mut batch = Vec::new();
82        for idx in indices {
83            if let Ok(item) = self.dataset.get_item(idx) {
84                batch.push(item);
85            }
86        }
87
88        if batch.is_empty() {
89            None
90        } else {
91            Some(batch)
92        }
93    }
94
95    /// Reset the sampler
96    /// サンプラーをリセット
97    pub fn reset(&mut self) {
98        self.sampler.reset();
99    }
100
101    /// Check if exhausted
102    /// 枯渇したかチェック
103    pub fn is_empty(&self) -> bool {
104        self.sampler.is_empty()
105    }
106
107    /// Get batch size
108    /// バッチサイズを取得
109    pub fn batch_size(&self) -> usize {
110        self.batch_size
111    }
112
113    /// Get number of workers
114    /// ワーカー数を取得
115    pub fn num_workers(&self) -> usize {
116        self.num_workers
117    }
118}
