12/// Word Embedding layer
13/// 単語埋め込み層
14///
15/// Maps discrete tokens to dense vectors of fixed size.
16/// 離散的なトークンを固定サイズの密ベクトルにマッピングします。
17#[derive(Debug)]
18pub struct Embedding<
19    T: Float + Send + Sync + 'static + ndarray::ScalarOperand + num_traits::FromPrimitive,
20> {
21    /// Embedding weight matrix of shape (vocab_size, embedding_dim)
22    /// 埋め込み重み行列 (語彙サイズ, 埋め込み次元)
23    weight: Variable<T>,
24
25    /// Vocabulary size
26    /// 語彙サイズ
27    vocab_size: usize,
28
29    /// Embedding dimension
30    /// 埋め込み次元
31    embedding_dim: usize,
32
33    /// Padding index (tokens at this index will have zero gradients)
34    /// パディングインデックス（このインデックスのトークンは勾配がゼロになる）
35    padding_idx: Option<usize>,
36
37    /// Maximum norm for embedding vectors
38    /// 埋め込みベクトルの最大ノルム
39    max_norm: Option<T>,
40
41    /// Whether to freeze the embedding weights
42    /// 埋め込み重みを凍結するかどうか
43    freeze: bool,
44}
45
46impl<T> Embedding<T>
47where
48    T: Float
49        + Debug
50        + Default
51        + FromPrimitive
52        + ToPrimitive
53        + Zero
54        + 'static
55        + Send
56        + Sync
57        + Copy
58        + ndarray::ScalarOperand
59        + num_traits::FromPrimitive,
60{
61    /// Creates a new Embedding layer
62    /// 新しいEmbedding層を作成します
63    pub fn new(
64        vocab_size: usize,
65        embedding_dim: usize,
66        padding_idx: Option<usize>,
67        max_norm: Option<T>,
68        freeze: Option<bool>,
69    ) -> Self {
70        assert!(vocab_size > 0, "vocab_size must be greater than 0");
71        assert!(embedding_dim > 0, "embedding_dim must be greater than 0");
72
73        let freeze = freeze.unwrap_or(false);
74
75        // Initialize embedding weights with normal distribution
76        let std = T::from_f32(1.0 / (embedding_dim as f32).sqrt()).unwrap();
77        let normal = Normal::new(0.0, std.to_f64().unwrap()).unwrap();
78        let mut rng = rand::thread_rng();
79
80        let weight_data: Vec<T> = (0..vocab_size * embedding_dim)
81            .map(|_| T::from_f32(normal.sample(&mut rng) as f32).unwrap())
82            .collect();
83
84        let weight = Variable::new(
85            Tensor::from_vec(weight_data, vec![vocab_size, embedding_dim]),
86            !freeze, // requires_grad is opposite of freeze
87        );
88
89        // Zero out padding index if specified
90        if let Some(pad_idx) = padding_idx {
91            if pad_idx < vocab_size {
92                let weight_binding = weight.data();
93                let mut weight_data = weight_binding.write().unwrap();
94                let weight_array = weight_data.as_array_mut();
95
96                // Zero out the padding row
97                for j in 0..embedding_dim {
98                    weight_array[[pad_idx, j]] = T::zero();
99                }
100            }
101        }
102
103        Embedding {
104            weight,
105            vocab_size,
106            embedding_dim,
107            padding_idx,
108            max_norm,
109            freeze,
110        }
111    }
112
113    /// Forward pass of the Embedding layer
114    /// Embedding層の順伝播
115    pub fn forward(&self, input: &Variable<T>) -> Variable<T> {
116        let input_binding = input.data();
117        let input_data = input_binding.read().unwrap();
118        let input_shape = input_data.shape();
119
120        // Input should contain integer indices
121        // Convert to usize indices for lookup
122        let indices = self.extract_indices(&input_data);
123
124        // Perform embedding lookup
125        let output_data = self.embedding_lookup(&indices, input_shape);
126
127        let requires_grad = !self.freeze && self.weight.requires_grad();
128        Variable::new(output_data, requires_grad)
129    }
130
131    /// Extract indices from input tensor
132    /// 入力テンソルからインデックスを抽出
133    fn extract_indices(&self, input: &Tensor<T>) -> Vec<usize> {
134        input
135            .as_array()
136            .iter()
137            .map(|&x| {
138                let idx = x.to_usize().unwrap_or(0);
139                if idx >= self.vocab_size {
140                    panic!(
141                        "Index {} is out of bounds for vocab_size {}",
142                        idx, self.vocab_size
143                    );
144                }
145                idx
146            })
147            .collect()
148    }
149
150    /// Perform embedding lookup
151    /// 埋め込み検索を実行
152    fn embedding_lookup(&self, indices: &[usize], input_shape: &[usize]) -> Tensor<T> {
153        let weight_binding = self.weight.data();
154        let weight_data = weight_binding.read().unwrap();
155        let weight_array = weight_data.as_array();
156
157        // Create output shape: input_shape + [embedding_dim]
158        let mut output_shape = input_shape.to_vec();
159        output_shape.push(self.embedding_dim);
160
161        // Lookup embeddings
162        let mut output_data = Vec::with_capacity(indices.len() * self.embedding_dim);
163
164        for &idx in indices {
165            for j in 0..self.embedding_dim {
166                let embedding_val = weight_array[[idx, j]];
167
168                // Apply max_norm constraint if specified
169                let final_val = if let Some(max_norm) = self.max_norm {
170                    // Simplified norm constraint (would need proper vector norm in full implementation)
171                    if embedding_val.abs() > max_norm {
172                        if embedding_val > T::zero() {
173                            max_norm
174                        } else {
175                            -max_norm
176                        }
177                    } else {
178                        embedding_val
179                    }
180                } else {
181                    embedding_val
182                };
183
184                output_data.push(final_val);
185            }
186        }
187
188        Tensor::from_vec(output_data, output_shape)
189    }
190
191    /// Returns the vocabulary size
192    /// 語彙サイズを返します
193    pub fn vocab_size(&self) -> usize {
194        self.vocab_size
195    }
196
197    /// Returns the embedding dimension
198    /// 埋め込み次元を返します
199    pub fn embedding_dim(&self) -> usize {
200        self.embedding_dim
201    }
202
203    /// Returns the padding index
204    /// パディングインデックスを返します
205    pub fn padding_idx(&self) -> Option<usize> {
206        self.padding_idx
207    }
208
209    /// Freezes or unfreezes the embedding weights
210    /// 埋め込み重みを凍結または解凍します
211    pub fn set_freeze(&mut self, freeze: bool) {
212        self.freeze = freeze;
213        // Update requires_grad of weight
214        let weight_data = self.weight.data();
215        let _weight_lock = weight_data.write().unwrap();
216        // Note: In a full implementation, we'd update the requires_grad flag
217    }
218
219    /// Returns the parameters of the layer
220    /// レイヤーのパラメータを返します
221    pub fn parameters(&self) -> Vec<Variable<T>> {
222        if self.freeze {
223            vec![]
224        } else {
225            vec![self.weight.clone()]
226        }
227    }

258/// Positional Embedding layer
259/// 位置埋め込み層
260///
261/// Adds positional information to input embeddings using learnable parameters.
262/// 学習可能なパラメータを使用して入力埋め込みに位置情報を追加します。
263#[derive(Debug)]
264pub struct PositionalEmbedding<
265    T: Float + Send + Sync + 'static + ndarray::ScalarOperand + num_traits::FromPrimitive,
266> {
267    /// Positional embedding weight matrix of shape (max_length, embedding_dim)
268    /// 位置埋め込み重み行列 (最大長, 埋め込み次元)
269    weight: Variable<T>,
270
271    /// Maximum sequence length
272    /// 最大シーケンス長
273    max_length: usize,
274
275    /// Embedding dimension
276    /// 埋め込み次元
277    embedding_dim: usize,
278}
279
280impl<T> PositionalEmbedding<T>
281where
282    T: Float
283        + Debug
284        + Default
285        + FromPrimitive
286        + ToPrimitive
287        + Zero
288        + 'static
289        + Send
290        + Sync
291        + Copy
292        + ndarray::ScalarOperand
293        + num_traits::FromPrimitive,
294{
295    /// Creates a new PositionalEmbedding layer
296    /// 新しいPositionalEmbedding層を作成します
297    pub fn new(max_length: usize, embedding_dim: usize) -> Self {
298        assert!(max_length > 0, "max_length must be greater than 0");
299        assert!(embedding_dim > 0, "embedding_dim must be greater than 0");
300
301        // Initialize with small random values
302        let std = T::from_f32(0.02).unwrap();
303        let normal = Normal::new(0.0, std.to_f64().unwrap()).unwrap();
304        let mut rng = rand::thread_rng();
305
306        let weight_data: Vec<T> = (0..max_length * embedding_dim)
307            .map(|_| T::from_f32(normal.sample(&mut rng) as f32).unwrap())
308            .collect();
309
310        let weight = Variable::new(
311            Tensor::from_vec(weight_data, vec![max_length, embedding_dim]),
312            true,
313        );
314
315        PositionalEmbedding {
316            weight,
317            max_length,
318            embedding_dim,
319        }
320    }
321
322    /// Forward pass of the PositionalEmbedding layer
323    /// PositionalEmbedding層の順伝播
324    pub fn forward(&self, input: &Variable<T>) -> Variable<T> {
325        let input_binding = input.data();
326        let input_data = input_binding.read().unwrap();
327        let input_shape = input_data.shape();
328
329        // Input shape should be (batch_size, seq_length, embedding_dim)
330        if input_shape.len() != 3 {
331            panic!("PositionalEmbedding expects 3D input (batch_size, seq_length, embedding_dim), got {:?}", input_shape);
332        }
333
334        let batch_size = input_shape[0];
335        let seq_length = input_shape[1];
336        let embed_dim = input_shape[2];
337
338        if embed_dim != self.embedding_dim {
339            panic!(
340                "Input embedding_dim {} doesn't match layer embedding_dim {}",
341                embed_dim, self.embedding_dim
342            );
343        }
344
345        if seq_length > self.max_length {
346            panic!(
347                "Sequence length {} exceeds max_length {}",
348                seq_length, self.max_length
349            );
350        }
351
352        // Get positional embeddings for the sequence length
353        let pos_embeddings = self.get_positional_embeddings(seq_length);
354
355        // Add positional embeddings to input
356        let output_data =
357            self.add_positional_embeddings(&input_data, &pos_embeddings, batch_size, seq_length);
358
359        let requires_grad = input.requires_grad() || self.weight.requires_grad();
360        Variable::new(output_data, requires_grad)
361    }
362
363    /// Get positional embeddings for given sequence length
364    /// 指定されたシーケンス長の位置埋め込みを取得
365    fn get_positional_embeddings(&self, seq_length: usize) -> Tensor<T> {
366        let weight_binding = self.weight.data();
367        let weight_data = weight_binding.read().unwrap();
368        let weight_array = weight_data.as_array();
369
370        // Extract first seq_length rows
371        let mut pos_data = Vec::with_capacity(seq_length * self.embedding_dim);
372
373        for i in 0..seq_length {
374            for j in 0..self.embedding_dim {
375                pos_data.push(weight_array[[i, j]]);
376            }
377        }
378
379        Tensor::from_vec(pos_data, vec![seq_length, self.embedding_dim])
380    }
381
382    /// Add positional embeddings to input
383    /// 入力に位置埋め込みを追加
384    fn add_positional_embeddings(
385        &self,
386        input: &Tensor<T>,
387        pos_embeddings: &Tensor<T>,
388        batch_size: usize,
389        seq_length: usize,
390    ) -> Tensor<T> {
391        let input_array = input.as_array();
392        let pos_array = pos_embeddings.as_array();
393
394        let mut output_data = Vec::with_capacity(batch_size * seq_length * self.embedding_dim);
395
396        for b in 0..batch_size {
397            for s in 0..seq_length {
398                for d in 0..self.embedding_dim {
399                    let input_val = input_array[[b, s, d]];
400                    let pos_val = pos_array[[s, d]];
401                    output_data.push(input_val + pos_val);
402                }
403            }
404        }
405
406        Tensor::from_vec(
407            output_data,
408            vec![batch_size, seq_length, self.embedding_dim],
409        )
410    }
411
412    /// Returns the maximum sequence length
413    /// 最大シーケンス長を返します
414    pub fn max_length(&self) -> usize {
415        self.max_length
416    }
417
418    /// Returns the embedding dimension
419    /// 埋め込み次元を返します
420    pub fn embedding_dim(&self) -> usize {
421        self.embedding_dim
422    }
423
424    /// Returns the parameters of the layer
425    /// レイヤーのパラメータを返します
426    pub fn parameters(&self) -> Vec<Variable<T>> {
427        vec![self.weight.clone()]
428    }
429}

459/// Sinusoidal Positional Encoding (fixed, non-learnable)
460/// 正弦波位置エンコーディング（固定、学習不可）
461///
462/// Uses sinusoidal functions to encode positional information.
463/// 正弦波関数を使用して位置情報をエンコードします。
464#[derive(Debug)]
465pub struct SinusoidalPositionalEncoding<
466    T: Float + Send + Sync + 'static + ndarray::ScalarOperand + num_traits::FromPrimitive,
467> {
468    /// Maximum sequence length
469    /// 最大シーケンス長
470    max_length: usize,
471
472    /// Embedding dimension
473    /// 埋め込み次元
474    embedding_dim: usize,
475
476    /// Precomputed positional encodings
477    /// 事前計算された位置エンコーディング
478    encodings: Tensor<T>,
479}
480
481impl<T> SinusoidalPositionalEncoding<T>
482where
483    T: Float
484        + Debug
485        + Default
486        + FromPrimitive
487        + ToPrimitive
488        + Zero
489        + 'static
490        + Send
491        + Sync
492        + Copy
493        + ndarray::ScalarOperand
494        + num_traits::FromPrimitive,
495{
496    /// Creates a new SinusoidalPositionalEncoding layer
497    /// 新しいSinusoidalPositionalEncoding層を作成します
498    pub fn new(max_length: usize, embedding_dim: usize) -> Self {
499        assert!(max_length > 0, "max_length must be greater than 0");
500        assert!(embedding_dim > 0, "embedding_dim must be greater than 0");
501
502        // Precompute sinusoidal encodings
503        let encodings = Self::create_sinusoidal_encodings(max_length, embedding_dim);
504
505        SinusoidalPositionalEncoding {
506            max_length,
507            embedding_dim,
508            encodings,
509        }
510    }
511
512    /// Create sinusoidal positional encodings
513    /// 正弦波位置エンコーディングを作成
514    fn create_sinusoidal_encodings(max_length: usize, embedding_dim: usize) -> Tensor<T> {
515        let mut encodings = Vec::with_capacity(max_length * embedding_dim);
516
517        for pos in 0..max_length {
518            for i in 0..embedding_dim {
519                let pos_f = pos as f32;
520                let i_f = i as f32;
521                let dim_f = embedding_dim as f32;
522
523                let angle = pos_f / (10000.0_f32).powf(2.0 * (i_f / 2.0).floor() / dim_f);
524
525                let encoding = if i % 2 == 0 { angle.sin() } else { angle.cos() };
526
527                encodings.push(T::from_f32(encoding).unwrap());
528            }
529        }
530
531        Tensor::from_vec(encodings, vec![max_length, embedding_dim])
532    }
533
534    /// Forward pass of the SinusoidalPositionalEncoding layer
535    /// SinusoidalPositionalEncoding層の順伝播
536    pub fn forward(&self, input: &Variable<T>) -> Variable<T> {
537        let input_binding = input.data();
538        let input_data = input_binding.read().unwrap();
539        let input_shape = input_data.shape();
540
541        // Input shape should be (batch_size, seq_length, embedding_dim)
542        if input_shape.len() != 3 {
543            panic!("SinusoidalPositionalEncoding expects 3D input (batch_size, seq_length, embedding_dim), got {:?}", input_shape);
544        }
545
546        let batch_size = input_shape[0];
547        let seq_length = input_shape[1];
548        let embed_dim = input_shape[2];
549
550        if embed_dim != self.embedding_dim {
551            panic!(
552                "Input embedding_dim {} doesn't match layer embedding_dim {}",
553                embed_dim, self.embedding_dim
554            );
555        }
556
557        if seq_length > self.max_length {
558            panic!(
559                "Sequence length {} exceeds max_length {}",
560                seq_length, self.max_length
561            );
562        }
563
564        // Add positional encodings to input
565        let output_data = self.add_sinusoidal_encodings(&input_data, batch_size, seq_length);
566
567        Variable::new(output_data, input.requires_grad())
568    }
569
570    /// Add sinusoidal encodings to input
571    /// 入力に正弦波エンコーディングを追加
572    fn add_sinusoidal_encodings(
573        &self,
574        input: &Tensor<T>,
575        batch_size: usize,
576        seq_length: usize,
577    ) -> Tensor<T> {
578        let input_array = input.as_array();
579        let encodings_array = self.encodings.as_array();
580
581        let mut output_data = Vec::with_capacity(batch_size * seq_length * self.embedding_dim);
582
583        for b in 0..batch_size {
584            for s in 0..seq_length {
585                for d in 0..self.embedding_dim {
586                    let input_val = input_array[[b, s, d]];
587                    let encoding_val = encodings_array[[s, d]];
588                    output_data.push(input_val + encoding_val);
589                }
590            }
591        }
592
593        Tensor::from_vec(
594            output_data,
595            vec![batch_size, seq_length, self.embedding_dim],
596        )
597    }
598
599    /// Returns the maximum sequence length
600    /// 最大シーケンス長を返します
601    pub fn max_length(&self) -> usize {
602        self.max_length
603    }
604
605    /// Returns the embedding dimension
606    /// 埋め込み次元を返します
607    pub fn embedding_dim(&self) -> usize {
608        self.embedding_dim
609    }
610}
