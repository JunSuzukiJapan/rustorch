11/// Safe tensor operations with proper error handling
12/// 適切なエラーハンドリングを持つ安全なテンソル操作
13pub struct SafeOps;
14
15impl SafeOps {
16    /// Safely create a variable with validation
17    /// 検証付きで安全にVariableを作成
18    pub fn create_variable<T>(
19        data: Vec<T>,
20        shape: Vec<usize>,
21        requires_grad: bool,
22    ) -> NNResult<Variable<T>>
23    where
24        T: Float
25            + Send
26            + Sync
27            + 'static
28            + Debug
29            + ndarray::ScalarOperand
30            + num_traits::FromPrimitive,
31    {
32        // Validate shape consistency
33        let expected_size: usize = shape.iter().product();
34        if data.len() != expected_size {
35            return Err(RusTorchError::InvalidDimensions(format!(
36                "Data length {} does not match shape {:?} (expected {})",
37                data.len(),
38                shape,
39                expected_size
40            )));
41        }
42
43        // Check for empty dimensions
44        if shape.iter().any(|&dim| dim == 0) {
45            return Err(RusTorchError::InvalidDimensions(
46                "Shape dimensions must be positive".to_string(),
47            ));
48        }
49
50        let tensor = Tensor::from_vec(data, shape);
51        Ok(Variable::new(tensor, requires_grad))
52    }
53
54    /// Safely reshape a variable
55    /// 安全にVariableをリシェイプ
56    pub fn reshape<T>(variable: &Variable<T>, new_shape: Vec<usize>) -> NNResult<Variable<T>>
57    where
58        T: Float
59            + Send
60            + Sync
61            + 'static
62            + Debug
63            + Clone
64            + ndarray::ScalarOperand
65            + num_traits::FromPrimitive,
66    {
67        let binding = variable.data();
68        let data_guard = binding
69            .read()
70            .map_err(|_| RusTorchError::MemoryError("Failed to acquire data lock".to_string()))?;
71
72        let current_size = data_guard.shape().iter().product::<usize>();
73        let new_size: usize = new_shape.iter().product();
74
75        if current_size != new_size {
76            return Err(RusTorchError::InvalidDimensions(format!(
77                "Cannot reshape tensor of size {} to size {}",
78                current_size, new_size
79            )));
80        }
81
82        // Create new tensor with same data but new shape
83        let data_vec = data_guard.as_array().iter().cloned().collect();
84        let new_tensor = Tensor::from_vec(data_vec, new_shape);
85
86        Ok(Variable::new(new_tensor, variable.requires_grad()))
87    }
88
89    /// Safely apply a function to tensor data
90    /// テンソルデータに安全に関数を適用
91    pub fn apply_function<T, F>(variable: &Variable<T>, f: F) -> NNResult<Variable<T>>
92    where
93        T: Float
94            + Send
95            + Sync
96            + 'static
97            + Debug
98            + Clone
99            + ndarray::ScalarOperand
100            + num_traits::FromPrimitive,
101        F: Fn(T) -> T,
102    {
103        let binding = variable.data();
104        let data_guard = binding
105            .read()
106            .map_err(|_| RusTorchError::MemoryError("Failed to acquire data lock".to_string()))?;
107
108        let new_data: Vec<T> = data_guard.as_array().iter().map(|&x| f(x)).collect();
109
110        let new_tensor = Tensor::from_vec(new_data, data_guard.shape().to_vec());
111        Ok(Variable::new(new_tensor, variable.requires_grad()))
112    }

114    /// Safely get tensor statistics
115    /// 安全にテンソル統計を取得
116    pub fn get_stats<T>(variable: &Variable<T>) -> NNResult<TensorStats<T>>
117    where
118        T: Float
119            + Send
120            + Sync
121            + 'static
122            + Debug
123            + Clone
124            + PartialOrd
125            + ndarray::ScalarOperand
126            + num_traits::FromPrimitive,
127    {
128        let binding = variable.data();
129        let data_guard = binding
130            .read()
131            .map_err(|_| RusTorchError::MemoryError("Failed to acquire data lock".to_string()))?;
132
133        let data = data_guard.as_array();
134
135        if data.is_empty() {
136            return Err(RusTorchError::InvalidDimensions("Empty tensor"));
137        }
138
139        let first = *data.iter().next().unwrap();
140        let min = data
141            .iter()
142            .fold(first, |acc, &x| if x < acc { x } else { acc });
143        let max = data
144            .iter()
145            .fold(first, |acc, &x| if x > acc { x } else { acc });
146
147        let sum = data.iter().fold(T::zero(), |acc, &x| acc + x);
148        let mean = sum / T::from(data.len()).unwrap();
149
150        // Compute variance
151        let variance = data
152            .iter()
153            .map(|&x| {
154                let diff = x - mean;
155                diff * diff
156            })
157            .fold(T::zero(), |acc, x| acc + x)
158            / T::from(data.len()).unwrap();
159
160        Ok(TensorStats {
161            min,
162            max,
163            mean,
164            variance,
165            count: data.len(),
166        })
167    }

169    /// Safely validate tensor for NaN or infinity
170    /// NaNや無限大のテンソルを安全に検証
171    pub fn validate_finite<T>(variable: &Variable<T>) -> NNResult<()>
172    where
173        T: Float
174            + Send
175            + Sync
176            + 'static
177            + Debug
178            + ndarray::ScalarOperand
179            + num_traits::FromPrimitive,
180    {
181        let binding = variable.data();
182        let data_guard = binding
183            .read()
184            .map_err(|_| RusTorchError::MemoryError("Failed to acquire data lock".to_string()))?;
185
186        for (i, &value) in data_guard.as_array().iter().enumerate() {
187            if value.is_nan() {
188                return Err(RusTorchError::ComputationError(format!(
189                    "NaN detected at index {}",
190                    i
191                )));
192            }
193            if value.is_infinite() {
194                return Err(RusTorchError::ComputationError(format!(
195                    "Infinity detected at index {}",
196                    i
197                )));
198            }
199        }
200
201        Ok(())
202    }

204    /// Apply ReLU activation function: max(0, x)
205    /// ReLU活性化関数を適用: max(0, x)
206    pub fn relu<T>(variable: &Variable<T>) -> NNResult<Variable<T>>
207    where
208        T: Float
209            + Send
210            + Sync
211            + 'static
212            + Debug
213            + Clone
214            + ndarray::ScalarOperand
215            + num_traits::FromPrimitive,
216    {
217        let binding = variable.data();
218        let data_guard = binding
219            .read()
220            .map_err(|_| RusTorchError::MemoryError("Failed to acquire data lock".to_string()))?;
221
222        let new_data: Vec<T> = data_guard
223            .as_array()
224            .iter()
225            .map(|&x| if x < T::zero() { T::zero() } else { x })
226            .collect();
227
228        let new_tensor = Tensor::from_vec(new_data, data_guard.shape().to_vec());
229        Ok(Variable::new(new_tensor, variable.requires_grad()))
230    }

233/// Statistics about a tensor
234/// テンソルの統計情報
235#[derive(Debug, Clone)]
236pub struct TensorStats<T> {
237    /// Minimum value
238    /// 最小値
239    pub min: T,
240    /// Maximum value
241    /// 最大値
242    pub max: T,
243    /// Mean value
244    /// 平均値
245    pub mean: T,
246    /// Variance
247    /// 分散
248    pub variance: T,
249    /// Number of elements
250    /// 要素数
251    pub count: usize,
252}
253
254impl<T: Float + ndarray::ScalarOperand + num_traits::FromPrimitive> TensorStats<T> {
255    /// Get standard deviation
256    /// 標準偏差を取得
257    pub fn std_dev(&self) -> T {
258        self.variance.sqrt()
259    }
260
261    /// Check if values are within reasonable range
262    /// 値が妥当な範囲内かチェック
263    pub fn is_reasonable(&self) -> bool {
264        !self.min.is_infinite()
265            && !self.max.is_infinite()
266            && !self.mean.is_nan()
267            && !self.variance.is_nan()
268            && self.variance >= T::zero()
269    }
270}
