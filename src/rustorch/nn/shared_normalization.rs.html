6/// Common normalization trait for both regular and WASM implementations
7/// 通常実装とWASM実装の両方用の共通正規化トレイト
8pub trait NormalizationFunction<T> {
9    type Error;
10    type Input;
11    type Output;
12
13    /// Batch normalization
14    /// バッチ正規化
15    fn batch_norm(
16        &self,
17        input: Self::Input,
18        gamma: &[T],
19        beta: &[T],
20        running_mean: &mut [T],
21        running_var: &mut [T],
22        momentum: T,
23        epsilon: T,
24        training: bool,
25    ) -> Result<Self::Output, Self::Error>;
26
27    /// Layer normalization
28    /// レイヤー正規化
29    fn layer_norm(
30        &self,
31        input: Self::Input,
32        gamma: &[T],
33        beta: &[T],
34        epsilon: T,
35    ) -> Result<Self::Output, Self::Error>;
36
37    /// Group normalization
38    /// グループ正規化
39    fn group_norm(
40        &self,
41        input: Self::Input,
42        gamma: &[T],
43        beta: &[T],
44        num_groups: usize,
45        epsilon: T,
46    ) -> Result<Self::Output, Self::Error>;
47}

49/// Shared normalization implementations
50/// 共通正規化実装
51pub mod shared_normalizations {
52    use super::*;
53
54    /// Compute mean for normalization
55    /// 正規化用の平均を計算
56    pub fn compute_mean<T: Float>(data: &[T]) -> T {
57        if data.is_empty() {
58            return T::zero();
59        }
60
61        data.iter().fold(T::zero(), |acc, &x| acc + x) / T::from(data.len()).unwrap_or(T::one())
62    }
63
64    /// Compute variance for normalization
65    /// 正規化用の分散を計算
66    pub fn compute_variance<T: Float>(data: &[T], mean: T) -> T {
67        if data.is_empty() {
68            return T::zero();
69        }
70
71        let sum_sq_diff = data
72            .iter()
73            .map(|&x| (x - mean) * (x - mean))
74            .fold(T::zero(), |acc, x| acc + x);
75
76        sum_sq_diff / T::from(data.len()).unwrap_or(T::one())
77    }
78
79    /// Apply normalization transformation: (x - mean) / sqrt(var + epsilon) * gamma + beta
80    /// 正規化変換を適用: (x - mean) / sqrt(var + epsilon) * gamma + beta
81    pub fn apply_normalization<T: Float>(
82        data: &[T],
83        mean: T,
84        variance: T,
85        gamma: &[T],
86        beta: &[T],
87        epsilon: T,
88    ) -> Result<Vec<T>, &'static str> {
89        if gamma.len() != beta.len() {
90            return Err("Gamma and beta must have same length");
91        }
92
93        let inv_std = T::one() / (variance + epsilon).sqrt();
94        let mut result = Vec::with_capacity(data.len());
95
96        for (i, &x) in data.iter().enumerate() {
97            let gamma_idx = i % gamma.len();
98            let beta_idx = i % beta.len();
99
100            let normalized = (x - mean) * inv_std;
101            let transformed = normalized * gamma[gamma_idx] + beta[beta_idx];
102            result.push(transformed);
103        }
104
105        Ok(result)
106    }
107
108    /// Simple batch normalization for 1D data
109    /// 1Dデータ用の簡単なバッチ正規化
110    pub fn batch_norm_1d<T: Float>(
111        input: &[T],
112        gamma: &[T],
113        beta: &[T],
114        running_mean: &mut [T],
115        running_var: &mut [T],
116        momentum: T,
117        epsilon: T,
118        training: bool,
119    ) -> Result<Vec<T>, &'static str> {
120        if training {
121            // Training mode: compute statistics from current batch
122            let batch_mean = compute_mean(input);
123            let batch_var = compute_variance(input, batch_mean);
124
125            // Update running statistics
126            for (rm, bm) in running_mean.iter_mut().zip(std::iter::repeat(batch_mean)) {
127                *rm = (T::one() - momentum) * (*rm) + momentum * bm;
128            }
129            for (rv, bv) in running_var.iter_mut().zip(std::iter::repeat(batch_var)) {
130                *rv = (T::one() - momentum) * (*rv) + momentum * bv;
131            }
132
133            apply_normalization(input, batch_mean, batch_var, gamma, beta, epsilon)
134        } else {
135            // Inference mode: use running statistics
136            let mean = running_mean.get(0).copied().unwrap_or(T::zero());
137            let var = running_var.get(0).copied().unwrap_or(T::one());
138
139            apply_normalization(input, mean, var, gamma, beta, epsilon)
140        }
141    }
142
143    /// Layer normalization implementation
144    /// レイヤー正規化実装
145    pub fn layer_norm_1d<T: Float>(
146        input: &[T],
147        gamma: &[T],
148        beta: &[T],
149        epsilon: T,
150    ) -> Result<Vec<T>, &'static str> {
151        let mean = compute_mean(input);
152        let variance = compute_variance(input, mean);
153
154        apply_normalization(input, mean, variance, gamma, beta, epsilon)
155    }
