12/// 3D Convolution layer for volumetric data
13/// ボリュメトリックデータ用の3D畳み込み層
14///
15/// This layer applies 3D convolution over an input signal composed of several input planes.
16/// 複数の入力プレーンからなる入力信号に対して3次元畳み込みを適用します。
17///
18/// Input shape: (batch_size, in_channels, depth, height, width)
19/// Output shape: (batch_size, out_channels, out_depth, out_height, out_width)
20#[derive(Debug)]
21pub struct Conv3d<T: Float + Send + Sync + ndarray::ScalarOperand + num_traits::FromPrimitive> {
22    /// Weight tensor of shape (out_channels, in_channels/groups, kernel_d, kernel_h, kernel_w)
23    /// 重みテンソル (出力チャンネル, 入力チャンネル/グループ, カーネル深度, カーネル高さ, カーネル幅)
24    weight: Variable<T>,
25
26    /// Bias tensor of shape (out_channels,)
27    /// バイアステンソル (出力チャンネル,)
28    bias: Option<Variable<T>>,
29
30    /// Input channels
31    /// 入力チャンネル数
32    in_channels: usize,
33
34    /// Output channels
35    /// 出力チャンネル数
36    out_channels: usize,
37
38    /// Kernel size (depth, height, width)
39    /// カーネルサイズ (深度, 高さ, 幅)
40    kernel_size: (usize, usize, usize),
41
42    /// Stride (depth, height, width)
43    /// ストライド (深度, 高さ, 幅)
44    stride: (usize, usize, usize),
45
46    /// Padding (depth, height, width)
47    /// パディング (深度, 高さ, 幅)
48    padding: (usize, usize, usize),
49
50    /// Dilation (depth, height, width)
51    /// 膨張 (深度, 高さ, 幅)
52    dilation: (usize, usize, usize),
53
54    /// Groups for grouped convolution
55    /// グループ畳み込み用のグループ数
56    groups: usize,
57}

72    /// Create a new Conv3d layer
73    /// 新しいConv3d層を作成
74    pub fn new(
75        in_channels: usize,
76        out_channels: usize,
77        kernel_size: (usize, usize, usize),
78        stride: Option<(usize, usize, usize)>,
79        padding: Option<(usize, usize, usize)>,
80        dilation: Option<(usize, usize, usize)>,
81        groups: Option<usize>,
82        bias: Option<bool>,
83    ) -> Self {
84        let stride = stride.unwrap_or((1, 1, 1));
85        let padding = padding.unwrap_or((0, 0, 0));
86        let dilation = dilation.unwrap_or((1, 1, 1));
87        let groups = groups.unwrap_or(1);
88        let use_bias = bias.unwrap_or(true);
89
90        // Validate parameters
91        assert!(
92            in_channels % groups == 0,
93            "in_channels must be divisible by groups"
94        );
95        assert!(
96            out_channels % groups == 0,
97            "out_channels must be divisible by groups"
98        );
99        assert!(
100            kernel_size.0 > 0 && kernel_size.1 > 0 && kernel_size.2 > 0,
101            "kernel_size must be positive"
102        );
103        assert!(
104            stride.0 > 0 && stride.1 > 0 && stride.2 > 0,
105            "stride must be positive"
106        );
107        assert!(
108            dilation.0 > 0 && dilation.1 > 0 && dilation.2 > 0,
109            "dilation must be positive"
110        );
111        assert!(groups > 0, "groups must be positive");
112
113        // Initialize weight tensor with shape [out_channels, in_channels/groups, kernel_d, kernel_h, kernel_w]
114        let weight_shape = vec![
115            out_channels,
116            in_channels / groups,
117            kernel_size.0,
118            kernel_size.1,
119            kernel_size.2,
120        ];
121        let weight_size = weight_shape.iter().product::<usize>();
122
123        // Initialize with Kaiming uniform (suitable for ReLU)
124        let fan_in = (in_channels / groups) * kernel_size.0 * kernel_size.1 * kernel_size.2;
125        let bound = (6.0 / fan_in as f32).sqrt();
126
127        let mut rng = rand::thread_rng();
128        let normal = Normal::new(0.0, bound).unwrap();
129        let weight_data: Vec<T> = (0..weight_size)
130            .map(|_| <T as From<f32>>::from(normal.sample(&mut rng)))
131            .collect();
132
133        let weight_tensor = Tensor::from_vec(weight_data, weight_shape);
134        let weight = Variable::new(weight_tensor, true);
135
136        // Initialize bias tensor if needed
137        let bias = if use_bias {
138            let bias_data = vec![T::default(); out_channels];
139            let bias_tensor = Tensor::from_vec(bias_data, vec![out_channels]);
140            Some(Variable::new(bias_tensor, true))
141        } else {
142            None
143        };
144
145        Self {
146            weight,
147            bias,
148            in_channels,
149            out_channels,
150            kernel_size,
151            stride,
152            padding,
153            dilation,
154            groups,
155        }
156    }

167    /// Get layer parameters
168    /// レイヤーのパラメータを取得
169    pub fn parameters(&self) -> Vec<Variable<T>> {
170        let mut params = vec![self.weight.clone()];
171        if let Some(ref bias) = self.bias {
172            params.push(bias.clone());
173        }
174        params
175    }

177    /// Calculate output size for 3D convolution
178    /// 3D畳み込みの出力サイズを計算
179    pub fn calculate_output_size(
180        &self,
181        input_size: (usize, usize, usize),
182    ) -> (usize, usize, usize) {
183        let (input_d, input_h, input_w) = input_size;
184        let (kernel_d, kernel_h, kernel_w) = self.kernel_size;
185        let (stride_d, stride_h, stride_w) = self.stride;
186        let (pad_d, pad_h, pad_w) = self.padding;
187        let (dil_d, dil_h, dil_w) = self.dilation;
188
189        let out_d = (input_d + 2 * pad_d - dil_d * (kernel_d - 1) - 1) / stride_d + 1;
190        let out_h = (input_h + 2 * pad_h - dil_h * (kernel_h - 1) - 1) / stride_h + 1;
191        let out_w = (input_w + 2 * pad_w - dil_w * (kernel_w - 1) - 1) / stride_w + 1;
192
193        (out_d, out_h, out_w)
194    }

196    /// Create a Conv3d for video processing (common parameters)
197    /// ビデオ処理用のConv3d作成（一般的なパラメータ）
198    pub fn for_video_processing(

215    /// Create a Conv3d with temporal kernel (3x1x1) for temporal modeling
216    /// 時系列モデリング用の時間的カーネル（3x1x1）のConv3d作成
217    pub fn temporal_conv(
218        in_channels: usize,
219        out_channels: usize,
220        temporal_kernel_size: usize,
221    ) -> Self {
222        Self::new(
223            in_channels,
224            out_channels,
225            (temporal_kernel_size, 1, 1),
226            Some((1, 1, 1)),                        // stride
227            Some((temporal_kernel_size / 2, 0, 0)), // padding to maintain spatial size
228            None,                                   // dilation
229            None,                                   // groups
230            Some(true),                             // bias
231        )
232    }

234    /// Create a Conv3d with spatial kernel (1x3x3) for spatial modeling
235    /// 空間モデリング用の空間的カーネル（1x3x3）のConv3d作成
236    pub fn spatial_conv(
237        in_channels: usize,
238        out_channels: usize,
239        spatial_kernel_size: usize,
240    ) -> Self {
241        Self::new(
242            in_channels,
243            out_channels,
244            (1, spatial_kernel_size, spatial_kernel_size),
245            Some((1, 1, 1)),                                             // stride
246            Some((0, spatial_kernel_size / 2, spatial_kernel_size / 2)), // padding
247            None,                                                        // dilation
248            None,                                                        // groups
249            Some(true),                                                  // bias
250        )
251    }
252
253    /// Get number of parameters
254    /// パラメータ数を取得
255    pub fn num_parameters(&self) -> usize {
256        let weight_params = self.out_channels
257            * (self.in_channels / self.groups)
258            * self.kernel_size.0
259            * self.kernel_size.1
260            * self.kernel_size.2;
261        let bias_params = if self.bias.is_some() {
262            self.out_channels
263        } else {
264            0
265        };
266        weight_params + bias_params
267    }
268
269    /// Get receptive field size
270    /// 受容野サイズを取得
271    pub fn receptive_field(&self) -> (usize, usize, usize) {
272        let field_d = self.dilation.0 * (self.kernel_size.0 - 1) + 1;
273        let field_h = self.dilation.1 * (self.kernel_size.1 - 1) + 1;
274        let field_w = self.dilation.2 * (self.kernel_size.2 - 1) + 1;
275        (field_d, field_h, field_w)
276    }
