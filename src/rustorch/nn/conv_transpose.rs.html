12/// 2D Transposed Convolution layer for generative models
13/// 生成モデル用の2D転置畳み込み層
14///
15/// This layer applies 2D transposed convolution over an input signal composed of several input planes.
16/// 複数の入力プレーンからなる入力信号に対して2次元転置畳み込みを適用します。
17///
18/// Input shape: (batch_size, in_channels, height, width)
19/// Output shape: (batch_size, out_channels, out_height, out_width)
20#[derive(Debug)]
21pub struct ConvTranspose2d<
22    T: Float + Send + Sync + ndarray::ScalarOperand + num_traits::FromPrimitive,
23> {
24    /// Weight tensor of shape (in_channels, out_channels/groups, kernel_height, kernel_width)
25    /// 重みテンソル (入力チャンネル, 出力チャンネル/グループ, カーネル高さ, カーネル幅)
26    weight: Variable<T>,
27
28    /// Bias tensor of shape (out_channels,)
29    /// バイアステンソル (出力チャンネル,)
30    bias: Option<Variable<T>>,
31
32    /// Input channels
33    /// 入力チャンネル数
34    in_channels: usize,
35
36    /// Output channels
37    /// 出力チャンネル数
38    out_channels: usize,
39
40    /// Kernel size (height, width)
41    /// カーネルサイズ (高さ, 幅)
42    kernel_size: (usize, usize),
43
44    /// Stride (height, width)
45    /// ストライド (高さ, 幅)
46    stride: (usize, usize),
47
48    /// Padding (height, width)
49    /// パディング (高さ, 幅)
50    padding: (usize, usize),
51
52    /// Output padding (height, width)
53    /// 出力パディング (高さ, 幅)
54    output_padding: (usize, usize),
55
56    /// Dilation (height, width)
57    /// 膨張 (高さ, 幅)
58    dilation: (usize, usize),
59
60    /// Groups for grouped convolution
61    /// グループ畳み込み用のグループ数
62    groups: usize,
63}
64
65impl<T> ConvTranspose2d<T>
66where
67    T: Float
68        + Debug
69        + Default
70        + From<f32>
71        + 'static
72        + Send
73        + Sync
74        + Copy
75        + ndarray::ScalarOperand
76        + num_traits::FromPrimitive,

78    /// Create a new ConvTranspose2d layer
79    /// 新しいConvTranspose2d層を作成
80    #[allow(clippy::too_many_arguments)]
81    pub fn new(
82        in_channels: usize,
83        out_channels: usize,
84        kernel_size: (usize, usize),
85        stride: Option<(usize, usize)>,
86        padding: Option<(usize, usize)>,
87        output_padding: Option<(usize, usize)>,
88        dilation: Option<(usize, usize)>,
89        groups: Option<usize>,
90        bias: Option<bool>,
91    ) -> Self {
92        let stride = stride.unwrap_or((1, 1));
93        let padding = padding.unwrap_or((0, 0));
94        let output_padding = output_padding.unwrap_or((0, 0));
95        let dilation = dilation.unwrap_or((1, 1));
96        let groups = groups.unwrap_or(1);
97        let use_bias = bias.unwrap_or(true);
98
99        // Validate parameters
100        assert!(
101            in_channels % groups == 0,
102            "in_channels must be divisible by groups"
103        );
104        assert!(
105            out_channels % groups == 0,
106            "out_channels must be divisible by groups"
107        );
108        assert!(
109            output_padding.0 < stride.0,
110            "output_padding.0 must be less than stride.0"
111        );
112        assert!(
113            output_padding.1 < stride.1,
114            "output_padding.1 must be less than stride.1"
115        );
116
117        // Initialize weight tensor with shape [in_channels, out_channels/groups, kernel_h, kernel_w]
118        let weight_shape = vec![
119            in_channels,
120            out_channels / groups,
121            kernel_size.0,
122            kernel_size.1,
123        ];
124        let weight_size = weight_shape.iter().product::<usize>();
125
126        // Initialize with Kaiming uniform (suitable for ReLU)
127        let fan_in = (out_channels / groups) * kernel_size.0 * kernel_size.1;
128        let bound = (6.0 / fan_in as f32).sqrt();
129
130        let mut rng = rand::thread_rng();
131        let normal = Normal::new(0.0, bound).unwrap();
132        let weight_data: Vec<T> = (0..weight_size)
133            .map(|_| <T as From<f32>>::from(normal.sample(&mut rng)))
134            .collect();
135
136        let weight_tensor = Tensor::from_vec(weight_data, weight_shape);
137        let weight = Variable::new(weight_tensor, true);
138
139        // Initialize bias tensor if needed
140        let bias = if use_bias {
141            let bias_data = vec![T::default(); out_channels];
142            let bias_tensor = Tensor::from_vec(bias_data, vec![out_channels]);
143            Some(Variable::new(bias_tensor, true))
144        } else {
145            None
146        };
147
148        Self {
149            weight,
150            bias,
151            in_channels,
152            out_channels,
153            kernel_size,
154            stride,
155            padding,
156            output_padding,
157            dilation,
158            groups,
159        }
160    }
161
162    /// Perform forward pass
163    /// 順伝搬を実行
164    pub fn forward(&self, input: &Variable<T>) -> Variable<T> {
165        let input_tensor = input.data();
166        let input_guard = input_tensor.read().unwrap();
167        let input_shape = input_guard.shape();
168
169        // Input validation
170        assert!(
171            input_shape.len() == 4,
172            "Input must be 4D tensor (batch, channels, height, width)"
173        );
174        assert_eq!(input_shape[1], self.in_channels, "Input channels mismatch");
175
176        let batch_size = input_shape[0];
177        let input_h = input_shape[2];
178        let input_w = input_shape[3];
179
180        // Calculate output dimensions
181        let (output_h, output_w) = self.calculate_output_size((input_h, input_w));
182        let output_shape = vec![batch_size, self.out_channels, output_h, output_w];
183
184        // Initialize output tensor
185        let mut output_data = vec![T::default(); output_shape.iter().product()];
186
187        // Perform transposed convolution
188        self.transposed_conv_2d(
189            input_guard.as_slice().unwrap(),
190            &mut output_data,
191            input_shape,
192            &output_shape,
193        );
194
195        // Add bias if present
196        if let Some(ref bias) = self.bias {
197            let bias_data_arc = bias.data();
198            let bias_guard = bias_data_arc.read().unwrap();
199            self.add_bias(
200                &mut output_data,
201                &output_shape,
202                bias_guard.as_slice().unwrap(),
203            );
204        }
205
206        let output_tensor = Tensor::from_vec(output_data, output_shape);
207        Variable::new(output_tensor, input.requires_grad())
208    }

210    /// Get layer parameters
211    /// レイヤーのパラメータを取得
212    pub fn parameters(&self) -> Vec<Variable<T>> {
213        let mut params = vec![self.weight.clone()];
214        if let Some(ref bias) = self.bias {
215            params.push(bias.clone());
216        }
217        params
218    }

220    /// Calculate output size for transposed convolution
221    /// 転置畳み込みの出力サイズを計算
222    pub fn calculate_output_size(&self, input_size: (usize, usize)) -> (usize, usize) {
223        let (input_h, input_w) = input_size;
224        let (kernel_h, kernel_w) = self.kernel_size;
225        let (stride_h, stride_w) = self.stride;
226        let (pad_h, pad_w) = self.padding;
227        let (out_pad_h, out_pad_w) = self.output_padding;
228        let (dil_h, dil_w) = self.dilation;
229
230        let out_h = (input_h - 1) * stride_h - 2 * pad_h + dil_h * (kernel_h - 1) + out_pad_h + 1;
231        let out_w = (input_w - 1) * stride_w - 2 * pad_w + dil_w * (kernel_w - 1) + out_pad_w + 1;
232
233        (out_h, out_w)
234    }

247            None,         // groups

268    /// Get number of parameters
269    /// パラメータ数を取得
270    pub fn num_parameters(&self) -> usize {
271        let weight_params = self.in_channels
272            * (self.out_channels / self.groups)
273            * self.kernel_size.0
274            * self.kernel_size.1;
275        let bias_params = if self.bias.is_some() {
276            self.out_channels
277        } else {
278            0
279        };
280        weight_params + bias_params
281    }

283    /// Core transposed convolution operation
284    /// 転置畳み込みのコア演算
285    fn transposed_conv_2d(
286        &self,
287        input: &[T],
288        output: &mut [T],
289        input_shape: &[usize],
290        output_shape: &[usize],
291    ) {
292        let batch_size = input_shape[0];
293        let input_h = input_shape[2];
294        let input_w = input_shape[3];
295        let output_h = output_shape[2];
296        let output_w = output_shape[3];
297
298        let weight_data_arc = self.weight.data();
299        let weight_guard = weight_data_arc.read().unwrap();
300        let weight_data = weight_guard.as_slice().unwrap();
301        let (kernel_h, kernel_w) = self.kernel_size;
302        let (stride_h, stride_w) = self.stride;
303        let (pad_h, pad_w) = self.padding;
304        let (out_pad_h, out_pad_w) = self.output_padding;
305
306        // Process each batch sequentially for memory safety
307        for b in 0..batch_size {
308            let input_batch_offset = b * self.in_channels * input_h * input_w;
309            let output_batch_offset = b * self.out_channels * output_h * output_w;
310
311            // Process each input channel
312            for in_ch in 0..self.in_channels {
313                let input_ch_offset = input_batch_offset + in_ch * input_h * input_w;
314
315                // Process each output channel in this group
316                let group_id = in_ch / (self.in_channels / self.groups);
317                let out_ch_start = group_id * (self.out_channels / self.groups);
318                let out_ch_end = out_ch_start + (self.out_channels / self.groups);
319
320                for out_ch in out_ch_start..out_ch_end {
321                    let out_ch_in_group = out_ch - out_ch_start;
322                    let output_ch_offset = output_batch_offset + out_ch * output_h * output_w;
323                    let weight_offset =
324                        in_ch * (self.out_channels / self.groups) * kernel_h * kernel_w
325                            + out_ch_in_group * kernel_h * kernel_w;
326
327                    // Transposed convolution: for each input pixel, spread to output region
328                    for i_h in 0..input_h {
329                        for i_w in 0..input_w {
330                            let input_idx = input_ch_offset + i_h * input_w + i_w;
331                            let input_val = input[input_idx];
332
333                            // Map input position to output region
334                            let out_h_start = i_h * stride_h;
335                            let out_w_start = i_w * stride_w;
336
337                            // Apply kernel
338                            for k_h in 0..kernel_h {
339                                for k_w in 0..kernel_w {
340                                    let out_h = out_h_start + k_h;
341                                    let out_w = out_w_start + k_w;
342
343                                    // Check bounds with padding
344                                    if out_h >= pad_h
345                                        && out_w >= pad_w
346                                        && out_h < output_h + pad_h - out_pad_h
347                                        && out_w < output_w + pad_w - out_pad_w
348                                    {
349                                        let adj_out_h = out_h - pad_h;
350                                        let adj_out_w = out_w - pad_w;
351
352                                        if adj_out_h < output_h && adj_out_w < output_w {
353                                            let output_idx =
354                                                output_ch_offset + adj_out_h * output_w + adj_out_w;
355                                            let weight_idx = weight_offset + k_h * kernel_w + k_w;
356
357                                            // Accumulate the contribution
358                                            unsafe {
359                                                let output_ptr =
360                                                    output.as_mut_ptr().add(output_idx);
361                                                *output_ptr = *output_ptr
362                                                    + input_val * weight_data[weight_idx];
363                                            }
364                                        }
365                                    }
366                                }
367                            }
368                        }
369                    }
370                }
371            }
372        }
373    }

375    /// Add bias to output tensor
376    /// 出力テンソルにバイアスを追加
377    fn add_bias(&self, output: &mut [T], output_shape: &[usize], bias: &[T]) {
378        let batch_size = output_shape[0];
379        let output_h = output_shape[2];
380        let output_w = output_shape[3];
381
382        for b in 0..batch_size {
383            for ch in 0..self.out_channels {
384                let ch_offset =
385                    b * self.out_channels * output_h * output_w + ch * output_h * output_w;
386                let bias_val = bias[ch];
387
388                for i in 0..(output_h * output_w) {
389                    output[ch_offset + i] = output[ch_offset + i] + bias_val;
390                }
391            }
392        }
393    }
