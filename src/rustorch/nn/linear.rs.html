15/// A linear (fully connected) layer.
16/// 線形（全結合）レイヤー
17///
18/// This layer applies a linear transformation to the incoming data: `y = xW^T + b`
19pub struct Linear<
20    T: Float + Send + Sync + 'static + ndarray::ScalarOperand + num_traits::FromPrimitive,
21> {
22    /// The weight matrix of shape (output_features, input_features)
23    /// 重み行列 (出力特徴量, 入力特徴量)
24    weight: Variable<T>,
25
26    /// The bias vector of shape (output_features,)
27    /// バイアスベクトル (出力特徴量,)
28    bias: Option<Variable<T>>,
29
30    /// Input size (number of input features)
31    /// 入力サイズ（入力特徴量の数）
32    input_size: usize,
33
34    /// Output size (number of output features)
35    /// 出力サイズ（出力特徴量の数）
36    output_size: usize,
37}
38
39impl<T: Float + Send + Sync + 'static + ndarray::ScalarOperand + num_traits::FromPrimitive>
40    std::fmt::Debug for Linear<T>
41{
42    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
43        f.debug_struct("Linear")
44            .field("input_size", &self.input_size)
45            .field("output_size", &self.output_size)
46            .field("has_bias", &self.bias.is_some())
47            .finish()
48    }
49}

68    /// Creates a new linear layer with the given input and output sizes.
69    /// 入力サイズと出力サイズを指定して新しい線形レイヤーを作成します。
70    pub fn new(input_size: usize, output_size: usize) -> Self {
71        // Initialize weights using Kaiming initialization
72        let k = (2.0 / input_size as f32).sqrt();
73        let normal = Normal::new(0.0, k as f64).unwrap();
74
75        // Initialize weights
76        let weight_data: Vec<T> = (0..input_size * output_size)
77            .map(|_| {
78                num_traits::cast(normal.sample(&mut rand::thread_rng()) as f64).unwrap_or(T::zero())
79            })
80            .collect();
81
82        let weight = Variable::new(
83            Tensor::new(
84                Array::from_shape_vec((output_size, input_size), weight_data)
85                    .unwrap()
86                    .into_dyn(),
87            ),
88            true,
89        );
90
91        // Initialize bias
92        let bias_data: Vec<T> = (0..output_size)
93            .map(|_| {
94                num_traits::cast(normal.sample(&mut rand::thread_rng()) as f64).unwrap_or(T::zero())
95            })
96            .collect();
97
98        let bias = Variable::new(
99            Tensor::new(
100                Array::from_shape_vec((output_size,), bias_data)
101                    .unwrap()
102                    .into_dyn(),
103            ),
104            true,
105        );
106
107        Linear {
108            weight,
109            bias: Some(bias),
110            input_size,
111            output_size,
112        }
113    }
114
115    /// Creates a new linear layer without a bias term.
116    /// バイアス項を持たない線形レイヤーを作成します。
117    pub fn new_no_bias(input_size: usize, output_size: usize) -> Self {
118        // Initialize weights using Kaiming initialization
119        let k = (2.0 / input_size as f32).sqrt();
120        let normal = Normal::new(0.0, k as f64).unwrap();
121
122        // Initialize weights
123        let weight_data: Vec<T> = (0..input_size * output_size)
124            .map(|_| {
125                num_traits::cast(normal.sample(&mut rand::thread_rng()) as f64).unwrap_or(T::zero())
126            })
127            .collect();
128
129        let weight = Variable::new(
130            Tensor::new(
131                Array::from_shape_vec((output_size, input_size), weight_data)
132                    .unwrap()
133                    .into_dyn(),
134            ),
135            true,
136        );
137
138        Linear {
139            weight,
140            bias: None,
141            input_size,
142            output_size,
143        }
144    }

146    /// Performs the forward pass of the linear layer.
147    /// 線形レイヤーの順伝搬を実行します。
148    pub fn forward(&self, input: &Variable<T>) -> Variable<T> {
149        let input_binding = input.data();
150        let input_data = input_binding.read().unwrap();
151        let weight_binding = self.weight.data();
152        let weight_data = weight_binding.read().unwrap();
153
154        // Matrix multiplication: input @ weight.T
155        // For batch processing: (batch_size, input_features) @ (input_features, output_features)
156        let weight_t = weight_data.transpose().expect("Transpose failed");
157        let mut output_data = input_data.matmul(&weight_t).expect("MatMul failed");
158
159        // Add bias if present
160        if let Some(ref bias) = self.bias {
161            let bias_binding = bias.data();
162            let bias_data = bias_binding.read().unwrap();
163
164            // Broadcast bias to match output shape
165            let output_shape = output_data.shape();
166            if output_shape.len() == 2 {
167                // Batch processing: add bias to each sample
168                let _batch_size = output_shape[0];
169                let output_features = output_shape[1];
170
171                // Create bias tensor with shape (1, output_features) for broadcasting
172                let bias_expanded = bias_data
173                    .as_array()
174                    .clone()
175                    .into_shape_with_order((1, output_features))
176                    .unwrap();
177                let bias_tensor = Tensor::new(bias_expanded.into_dyn());
178
179                output_data = &output_data + &bias_tensor;
180            } else if output_shape.len() == 1 {
181                // Single sample: direct addition
182                output_data = &output_data + &*bias_data;
183            }
184        }
185
186        let requires_grad = input.requires_grad()
187            || self.weight.requires_grad()
188            || self.bias.as_ref().map_or(false, |b| b.requires_grad());
189
190        if requires_grad {
191            // Create gradient function for backpropagation
192            use crate::autograd::linear_grad_fn::LinearBackward;
193            use std::sync::{Arc, RwLock};
194
195            let grad_fn = Arc::new(LinearBackward {
196                input: Arc::new(RwLock::new(input_data.clone())),
197                weight: Arc::new(RwLock::new(weight_data.clone())),
198                input_var: input.clone(),
199                weight_var: self.weight.clone(),
200                bias_var: self.bias.clone(),
201            });
202
203            Variable::new_with_grad_fn(output_data, true, Some(grad_fn))
204        } else {
205            Variable::new(output_data, false)
206        }
207    }

243    fn parameters(&self) -> Vec<Variable<T>> {
244        match &self.bias {
245            Some(bias) => vec![self.weight.clone(), bias.clone()],
246            None => vec![self.weight.clone()],
247        }

259    #[test]
260    fn test_linear_forward() {
261        // Test with bias
262        let linear = Linear::<f32>::new(3, 2);
263        let input = Variable::new(Tensor::from_vec(vec![1.0, 2.0, 3.0], vec![1, 3]), false);
264        let output = linear.forward(&input);
265
266        // Check that we get some output (shape might vary based on implementation)
267        let output_binding = output.data();
268        let output_data = output_binding.read().unwrap();
269        assert!(!output_data.is_empty());
270
271        // Test without bias
272        let linear = Linear::<f32>::new_no_bias(3, 2);
273        let input = Variable::new(Tensor::from_vec(vec![1.0, 2.0, 3.0], vec![1, 3]), false);
274        let output = linear.forward(&input);
275        let output_binding = output.data();
276        let output_data = output_binding.read().unwrap();
277        assert!(!output_data.is_empty());
278    }
279
280    #[test]
281    fn test_linear_parameters() {
282        let linear = Linear::<f32>::new(3, 2);
283        let params = linear.parameters();
284        assert_eq!(params.len(), 2); // weight and bias
285
286        let linear = Linear::<f32>::new_no_bias(3, 2);
287        let params = linear.parameters();
288        assert_eq!(params.len(), 1); // only weight
289    }
