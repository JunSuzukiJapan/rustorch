141    /// Forward pass of the BatchNorm1d layer
142    /// BatchNorm1dレイヤーの順伝播
143    pub fn forward(&self, input: &Variable<T>) -> Variable<T> {
144        let input_binding = input.data();
145        let input_data = input_binding.read().unwrap();
146        let input_shape = input_data.shape();
147
148        if input_shape.len() != 2 || input_shape[1] != self.num_features {
149            panic!(
150                "Expected 2D input with {} features, got shape {:?}",
151                self.num_features, input_shape
152            );
153        }
154
155        // Simple implementation using tensor operations
156        let normalized_tensor = if self.is_training() {
157            // Training mode: normalize using batch statistics
158            self.normalize_training(&input_data)
159        } else {
160            // Evaluation mode: normalize using running statistics
161            self.normalize_eval(&input_data)
162        };
163
164        let requires_grad =
165            input.requires_grad() || self.weight.requires_grad() || self.bias.requires_grad();
166        Variable::new(normalized_tensor, requires_grad)
167    }

169    /// Normalize using batch statistics (training mode)
170    /// バッチ統計を使用した正規化（訓練モード）
171    fn normalize_training(&self, input: &Tensor<T>) -> Tensor<T> {
172        let input_shape = input.shape();
173        let batch_size = input_shape[0];
174        let num_features = input_shape[1];
175
176        // Compute batch statistics using numerically stable Welford's algorithm
177        let (batch_mean, batch_var) =
178            self.compute_batch_statistics(input, batch_size, num_features);
179
180        // Update running statistics with exponential moving average
181        self.update_running_statistics(&batch_mean, &batch_var);
182
183        // Apply normalization using batch statistics
184        self.apply_normalization(input, &batch_mean, &batch_var)
185    }

196    /// Apply normalization: (x - mean) / sqrt(var + eps) * weight + bias
197    /// 正規化を適用: (x - mean) / sqrt(var + eps) * weight + bias
198    fn apply_normalization(
199        &self,
200        input: &Tensor<T>,
201        mean: &Tensor<T>,
202        var: &Tensor<T>,
203    ) -> Tensor<T> {
204        let weight_binding = self.weight.data();
205        let weight_data = weight_binding.read().unwrap();
206        let bias_binding = self.bias.data();
207        let bias_data = bias_binding.read().unwrap();
208
209        let input_array = input.as_array();
210        let mean_array = mean.as_array();
211        let var_array = var.as_array();
212        let weight_array = weight_data.as_array();
213        let bias_array = bias_data.as_array();
214
215        let input_shape = input.shape();
216        let batch_size = input_shape[0];
217        let num_features = input_shape[1];
218
219        let mut output_data = Vec::with_capacity(batch_size * num_features);
220
221        // Apply normalization with improved numerical stability
222        for b in 0..batch_size {
223            for f in 0..num_features {
224                let x = input_array[[b, f]];
225                let mu = mean_array[f];
226                let sigma2 = var_array[f];
227
228                // Improved numerical stability: check for very small variance
229                let ten = T::from_f32(10.0).unwrap();
230                let eps_adjusted = if sigma2 < self.eps * ten {
231                    self.eps * ten
232                } else {
233                    self.eps
234                };
235
236                // Normalize: (x - μ) / √(σ² + ε)
237                let normalized = (x - mu) / (sigma2 + eps_adjusted).sqrt();
238
239                // Scale and shift: γ * normalized + β
240                let output_val = weight_array[f] * normalized + bias_array[f];
241                output_data.push(output_val);
242            }
243        }
244
245        Tensor::from_vec(output_data, input_shape.to_vec())
246    }

248    /// Compute batch statistics using Welford's algorithm for numerical stability
249    /// 数値安定性のためWelfordアルゴリズムを使用してバッチ統計を計算
250    fn compute_batch_statistics(
251        &self,
252        input: &Tensor<T>,
253        batch_size: usize,
254        num_features: usize,
255    ) -> (Tensor<T>, Tensor<T>) {
256        let input_array = input.as_array();
257
258        let mut mean_vec = vec![T::zero(); num_features];
259        let mut var_vec = vec![T::zero(); num_features];
260
261        // Compute mean and variance per feature using Welford's algorithm
262        for f in 0..num_features {
263            let mut mean = T::zero();
264            let mut m2 = T::zero();
265
266            // Welford's online algorithm for mean and variance
267            for b in 0..batch_size {
268                let x = input_array[[b, f]];
269                let delta = x - mean;
270                mean = mean + delta / T::from_usize(b + 1).unwrap();
271                let delta2 = x - mean;
272                m2 = m2 + delta * delta2;
273            }
274
275            mean_vec[f] = mean;
276
277            // Compute sample variance (divide by N-1, but use N for batch norm)
278            let variance = if batch_size > 1 {
279                m2 / T::from_usize(batch_size).unwrap()
280            } else {
281                T::one() // Fallback for single sample
282            };
283
284            // Apply bias correction and ensure minimum variance
285            let bias_corrected_var = if batch_size > 1 {
286                variance * T::from_usize(batch_size).unwrap()
287                    / T::from_usize(batch_size - 1).unwrap()
288            } else {
289                variance
290            };
291
292            let min_var_threshold = self.eps * T::from_f32(0.1).unwrap();
293            var_vec[f] = bias_corrected_var.max(min_var_threshold);
294        }
295
296        let mean_tensor = Tensor::from_vec(mean_vec, vec![num_features]);
297        let var_tensor = Tensor::from_vec(var_vec, vec![num_features]);
298
299        (mean_tensor, var_tensor)
300    }

302    /// Update running statistics with exponential moving average
303    /// 指数移動平均で移動統計を更新
304    fn update_running_statistics(&self, batch_mean: &Tensor<T>, batch_var: &Tensor<T>) {
305        if let (Ok(mut running_mean), Ok(mut running_var)) =
306            (self.running_mean.write(), self.running_var.write())
307        {
308            let batch_mean_array = batch_mean.as_array();
309            let batch_var_array = batch_var.as_array();
310
311            let running_mean_array = running_mean.as_array_mut();
312            let running_var_array = running_var.as_array_mut();
313
314            let momentum = self.momentum;
315            let one_minus_momentum = T::one() - momentum;
316
317            // Update running mean: running_mean = (1 - momentum) * running_mean + momentum * batch_mean
318            for i in 0..self.num_features {
319                running_mean_array[i] =
320                    one_minus_momentum * running_mean_array[i] + momentum * batch_mean_array[i];
321                running_var_array[i] =
322                    one_minus_momentum * running_var_array[i] + momentum * batch_var_array[i];
323            }
324        }
325    }

574    /// Compute channel-wise statistics for 2D batch normalization
575    /// 2Dバッチ正規化のためのチャンネル別統計を計算
576    fn compute_channel_statistics(
577        &self,
578        input: &Tensor<T>,
579        batch_size: usize,
580        channels: usize,
581        height: usize,
582        width: usize,
583    ) -> (Tensor<T>, Tensor<T>) {
584        let input_array = input.as_array();
585        let spatial_size = height * width;
586        let total_elements_per_channel = batch_size * spatial_size;
587
588        let mut mean_vec = vec![T::zero(); channels];
589        let mut var_vec = vec![T::zero(); channels];
590
591        // Compute statistics per channel using Welford's algorithm
592        for c in 0..channels {
593            let mut mean = T::zero();
594            let mut m2 = T::zero();
595            let mut count = 0;
596
597            // Iterate over all spatial locations and batch elements for this channel
598            for b in 0..batch_size {
599                for h in 0..height {
600                    for w in 0..width {
601                        let x = input_array[[b, c, h, w]];
602                        count += 1;
603
604                        let delta = x - mean;
605                        mean = mean + delta / T::from_usize(count).unwrap();
606                        let delta2 = x - mean;
607                        m2 = m2 + delta * delta2;
608                    }
609                }
610            }
611
612            mean_vec[c] = mean;
613
614            // Compute variance
615            let variance = if total_elements_per_channel > 1 {
616                m2 / T::from_usize(total_elements_per_channel).unwrap()
617            } else {
618                T::one()
619            };
620
621            // Apply bias correction for small batches
622            let bias_corrected_var = if total_elements_per_channel > 1 {
623                variance * T::from_usize(total_elements_per_channel).unwrap()
624                    / T::from_usize(total_elements_per_channel - 1).unwrap()
625            } else {
626                variance
627            };
628
629            let min_var_threshold = self.eps * T::from_f32(0.1).unwrap();
630            var_vec[c] = bias_corrected_var.max(min_var_threshold);
631        }
632
633        let mean_tensor = Tensor::from_vec(mean_vec, vec![channels]);
634        let var_tensor = Tensor::from_vec(var_vec, vec![channels]);
635
636        (mean_tensor, var_tensor)
637    }

639    /// Apply channel-wise normalization with improved numerical stability
640    /// 数値安定性を改善したチャンネル別正規化を適用
641    fn apply_channel_normalization_2d(
642        &self,
643        input: &Tensor<T>,
644        mean: &Tensor<T>,
645        var: &Tensor<T>,
646    ) -> Tensor<T> {
647        let weight_binding = self.weight.data();
648        let weight_data = weight_binding.read().unwrap();
649        let bias_binding = self.bias.data();
650        let bias_data = bias_binding.read().unwrap();
651
652        let input_array = input.as_array();
653        let mean_array = mean.as_array();
654        let var_array = var.as_array();
655        let weight_array = weight_data.as_array();
656        let bias_array = bias_data.as_array();
657
658        let input_shape = input.shape();
659        let batch_size = input_shape[0];
660        let channels = input_shape[1];
661        let height = input_shape[2];
662        let width = input_shape[3];
663
664        let mut output_data = Vec::with_capacity(batch_size * channels * height * width);
665
666        // Apply normalization with improved numerical stability
667        for b in 0..batch_size {
668            for c in 0..channels {
669                let mu = mean_array[c];
670                let sigma2 = var_array[c];
671                let gamma = weight_array[c];
672                let beta = bias_array[c];
673
674                // Improved numerical stability: check for very small variance
675                let ten = T::from_f32(10.0).unwrap();
676                let eps_adjusted = if sigma2 < self.eps * ten {
677                    self.eps * ten
678                } else {
679                    self.eps
680                };
681
682                let inv_std = T::one() / (sigma2 + eps_adjusted).sqrt();
683
684                for h in 0..height {
685                    for w in 0..width {
686                        let x = input_array[[b, c, h, w]];
687
688                        // Normalize: (x - μ) / √(σ² + ε)
689                        let normalized = (x - mu) * inv_std;
690
691                        // Scale and shift: γ * normalized + β
692                        let output_val = gamma * normalized + beta;
693                        output_data.push(output_val);
694                    }
695                }
696            }
697        }
698
699        Tensor::from_vec(output_data, input_shape.to_vec())
700    }
701
702    /// Update running statistics for 2D batch normalization
703    /// 2Dバッチ正規化のための移動統計を更新
704    fn update_running_statistics_2d(&self, batch_mean: &Tensor<T>, batch_var: &Tensor<T>) {
705        if let (Ok(mut running_mean), Ok(mut running_var)) =
706            (self.running_mean.write(), self.running_var.write())
707        {
708            let batch_mean_array = batch_mean.as_array();
709            let batch_var_array = batch_var.as_array();
710
711            let running_mean_array = running_mean.as_array_mut();
712            let running_var_array = running_var.as_array_mut();
713
714            let momentum = self.momentum;
715            let one_minus_momentum = T::one() - momentum;
716
717            // Update running statistics: running_stat = (1 - momentum) * running_stat + momentum * batch_stat
718            for i in 0..self.num_features {
719                running_mean_array[i] =
720                    one_minus_momentum * running_mean_array[i] + momentum * batch_mean_array[i];
721                running_var_array[i] =
722                    one_minus_momentum * running_var_array[i] + momentum * batch_var_array[i];
723            }
724        }
725    }

731    }
