13/// LSTM cell implementation
14/// LSTMセルの実装
15///
16/// An LSTM cell computes:
17/// LSTMセルは以下を計算します：
18///
19/// i_t = σ(W_ii @ x_t + b_ii + W_hi @ h_{t-1} + b_hi)  (input gate)
20/// f_t = σ(W_if @ x_t + b_if + W_hf @ h_{t-1} + b_hf)  (forget gate)
21/// g_t = tanh(W_ig @ x_t + b_ig + W_hg @ h_{t-1} + b_hg)  (cell gate)
22/// o_t = σ(W_io @ x_t + b_io + W_ho @ h_{t-1} + b_ho)  (output gate)
23/// c_t = f_t ⊙ c_{t-1} + i_t ⊙ g_t  (cell state)
24/// h_t = o_t ⊙ tanh(c_t)  (hidden state)
25#[derive(Debug)]
26pub struct LSTMCell<T: Float + Send + Sync + ndarray::ScalarOperand + num_traits::FromPrimitive> {
27    /// Input-to-hidden weights for all gates [input, forget, cell, output]
28    /// 全ゲート用の入力から隠れ状態への重み [input, forget, cell, output]
29    weight_ih: Variable<T>,
30
31    /// Hidden-to-hidden weights for all gates [input, forget, cell, output]
32    /// 全ゲート用の隠れ状態から隠れ状態への重み [input, forget, cell, output]
33    weight_hh: Variable<T>,
34
35    /// Input-to-hidden bias for all gates
36    /// 全ゲート用の入力から隠れ状態へのバイアス
37    bias_ih: Option<Variable<T>>,
38
39    /// Hidden-to-hidden bias for all gates
40    /// 全ゲート用の隠れ状態から隠れ状態へのバイアス
41    bias_hh: Option<Variable<T>>,
42
43    /// Input size
44    /// 入力サイズ
45    input_size: usize,
46
47    /// Hidden size
48    /// 隠れ状態サイズ
49    hidden_size: usize,
50
51    /// Training mode flag
52    /// 学習モードフラグ
53    training: bool,
54}
55
56impl<T> LSTMCell<T>
57where
58    T: Float + Send + Sync + Debug + 'static + ndarray::ScalarOperand + num_traits::FromPrimitive,
59{
60    /// Create a new LSTM cell
61    /// 新しいLSTMセルを作成
62    pub fn new(input_size: usize, hidden_size: usize, bias: bool) -> Self {
63        // Use common recurrent operations for weight initialization
64        let (weight_ih, weight_hh) = RecurrentOps::init_weights(input_size, hidden_size, 4);
65        let (bias_ih, bias_hh) = if bias {
66            RecurrentOps::init_bias(hidden_size, 4)
67        } else {
68            (None, None)
69        };
70
71        LSTMCell {
72            weight_ih,
73            weight_hh,
74            bias_ih,
75            bias_hh,
76            input_size,
77            hidden_size,
78            training: true,
79        }
80    }
81
82    /// Forward pass through the LSTM cell
83    /// LSTMセルの順伝播
84    pub fn forward(
85        &self,
86        input: &Variable<T>,
87        hidden: Option<(&Variable<T>, &Variable<T>)>,
88    ) -> (Variable<T>, Variable<T>) {
89        let input_binding = input.data();
90        let input_data = input_binding.read().unwrap();
91        let batch_size = input_data.shape()[0];
92
93        let (h_prev, c_prev) = match hidden {
94            Some((h, c)) => (h.clone(), c.clone()),
95            None => {
96                let h = Variable::new(Tensor::zeros(&[batch_size, self.hidden_size]), false);
97                let c = Variable::new(Tensor::zeros(&[batch_size, self.hidden_size]), false);
98                (h, c)
99            }
100        };
101
102        // Compute gate values
103        let gi = RecurrentOps::linear_transform(input, &self.weight_ih, self.bias_ih.as_ref());
104        let gh = RecurrentOps::linear_transform(&h_prev, &self.weight_hh, self.bias_hh.as_ref());
105
106        let gates = RecurrentOps::add_variables(&gi, &gh);
107
108        // Split gates into [input, forget, cell, output]
109        let input_gate =
110            RecurrentOps::sigmoid(&RecurrentOps::slice_gates(&gates, 0, self.hidden_size));
111        let forget_gate =
112            RecurrentOps::sigmoid(&RecurrentOps::slice_gates(&gates, 1, self.hidden_size));
113        let cell_gate = RecurrentOps::tanh(&RecurrentOps::slice_gates(&gates, 2, self.hidden_size));
114        let output_gate =
115            RecurrentOps::sigmoid(&RecurrentOps::slice_gates(&gates, 3, self.hidden_size));
116
117        // Update cell state: c_t = f_t * c_{t-1} + i_t * g_t
118        let forget_term = RecurrentOps::multiply_variables(&forget_gate, &c_prev);
119        let input_term = RecurrentOps::multiply_variables(&input_gate, &cell_gate);
120        let new_cell = RecurrentOps::add_variables(&forget_term, &input_term);
121
122        // Update hidden state: h_t = o_t * tanh(c_t)
123        let cell_tanh = RecurrentOps::tanh(&new_cell);
124        let new_hidden = RecurrentOps::multiply_variables(&output_gate, &cell_tanh);
125
126        (new_hidden, new_cell)
127    }

129    /// Get input size
130    /// 入力サイズを取得
131    pub fn input_size(&self) -> usize {
132        self.input_size
133    }
134
135    /// Get hidden size
136    /// 隠れ状態サイズを取得
137    pub fn hidden_size(&self) -> usize {
138        self.hidden_size
139    }
140
141    /// Set training mode
142    /// 学習モードを設定
143    pub fn set_training(&mut self, training: bool) {
144        self.training = training;
145    }
146
147    /// Check if in training mode
148    /// 学習モードかどうかをチェック
149    pub fn is_training(&self) -> bool {
150        self.training
151    }
152}

154impl<T> Module<T> for LSTMCell<T>
155where
156    T: Float + Send + Sync + Debug + 'static + ndarray::ScalarOperand + num_traits::FromPrimitive,
157{
158    fn forward(&self, input: &Variable<T>) -> Variable<T> {
159        // For Module trait compatibility, return only hidden state
160        let (hidden, _cell) = self.forward(input, None);
161        hidden
162    }
163
164    fn parameters(&self) -> Vec<Variable<T>> {
165        collect_recurrent_parameters(
166            &self.weight_ih,
167            &self.weight_hh,
168            &self.bias_ih,
169            &self.bias_hh,
170        )
171    }
172
173    fn as_any(&self) -> &dyn std::any::Any {
174        self
175    }
176
177    fn train(&mut self) {
178        self.training = true;
179    }
180
181    fn eval(&mut self) {
182        self.training = false;
183    }
