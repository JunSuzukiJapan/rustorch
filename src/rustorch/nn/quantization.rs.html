19/// Quantization scheme types
20/// 量子化スキーム型
21#[derive(Debug, Clone, Copy, PartialEq)]
22pub enum QuantizationType {
23    /// 8-bit integer quantization
24    /// 8ビット整数量子化
25    Int8,
26    /// 4-bit integer quantization (for extreme compression)
27    /// 4ビット整数量子化（極限圧縮用）
28    Int4,
29    /// 16-bit floating point quantization
30    /// 16ビット浮動小数点量子化
31    Float16,
32    /// Dynamic range quantization
33    /// 動的範囲量子化
34    Dynamic,
35}
36
37/// Quantization calibration mode
38/// 量子化キャリブレーションモード
39#[derive(Debug, Clone, Copy, PartialEq)]
40pub enum CalibrationMode {
41    /// Use min-max values for range determination
42    /// 範囲決定に最小最大値を使用
43    MinMax,
44    /// Use percentile values for outlier-robust quantization
45    /// 外れ値に頑健な量子化のためにパーセンタイル値を使用
46    Percentile(f32),
47    /// Use entropy-based optimization (KL divergence)
48    /// エントロピーベース最適化（KLダイバージェンス）を使用
49    Entropy,
50}

52/// Quantization parameters for a tensor
53/// テンソル用量子化パラメータ
54#[derive(Debug, Clone)]
55pub struct QuantizationParams<T: Float> {
56    /// Scale factor for dequantization
57    /// 逆量子化用スケール係数
58    pub scale: T,
59    /// Zero point for asymmetric quantization
60    /// 非対称量子化用ゼロ点
61    pub zero_point: i32,
62    /// Minimum value in the quantized range
63    /// 量子化範囲の最小値
64    pub qmin: i32,
65    /// Maximum value in the quantized range
66    /// 量子化範囲の最大値
67    pub qmax: i32,
68    /// Quantization type
69    /// 量子化タイプ
70    pub qtype: QuantizationType,
71}

73impl<T: Float + FromPrimitive> QuantizationParams<T> {
74    /// Create quantization parameters for Int8 symmetric quantization
75    /// Int8対称量子化用の量子化パラメータを作成
76    pub fn int8_symmetric(scale: T) -> Self {
77        QuantizationParams {
78            scale,
79            zero_point: 0,
80            qmin: -128,
81            qmax: 127,
82            qtype: QuantizationType::Int8,
83        }
84    }
85
86    /// Create quantization parameters for Int8 asymmetric quantization
87    /// Int8非対称量子化用の量子化パラメータを作成
88    pub fn int8_asymmetric(scale: T, zero_point: i32) -> Self {
89        QuantizationParams {
90            scale,
91            zero_point,
92            qmin: -128,
93            qmax: 127,
94            qtype: QuantizationType::Int8,
95        }
96    }
97
98    /// Create quantization parameters for Int4 quantization
99    /// Int4量子化用の量子化パラメータを作成
100    pub fn int4_symmetric(scale: T) -> Self {
101        QuantizationParams {
102            scale,
103            zero_point: 0,
104            qmin: -8,
105            qmax: 7,
106            qtype: QuantizationType::Int4,
107        }
108    }
109}

111/// Quantized tensor representation
112/// 量子化テンソル表現
113#[derive(Debug, Clone)]
114pub struct QuantizedTensor<T: Float> {
115    /// Quantized integer values
116    /// 量子化整数値
117    pub data: Vec<i8>,
118    /// Original tensor shape
119    /// 元のテンソル形状
120    pub shape: Vec<usize>,
121    /// Quantization parameters
122    /// 量子化パラメータ
123    pub params: QuantizationParams<T>,
124}
125
126impl<T> QuantizedTensor<T>
127where
128    T: Float
129        + FromPrimitive
130        + ToPrimitive
131        + Debug
132        + Default
133        + Zero
134        + One
135        + Send
136        + Sync
137        + Copy
138        + ScalarOperand
139        + Sum
140        + Signed,
141{
142    /// Create a new quantized tensor
143    /// 新しい量子化テンソルを作成
144    pub fn new(data: Vec<i8>, shape: Vec<usize>, params: QuantizationParams<T>) -> Self {
145        QuantizedTensor {
146            data,
147            shape,
148            params,
149        }
150    }
151
152    /// Dequantize back to floating point tensor
153    /// 浮動小数点テンソルに逆量子化
154    pub fn dequantize(&self) -> Tensor<T> {
155        let mut float_data = Vec::with_capacity(self.data.len());
156
157        for &qval in &self.data {
158            let qval_adjusted = i32::from(qval) - self.params.zero_point;
159            let float_val = T::from_i32(qval_adjusted).unwrap() * self.params.scale;
160            float_data.push(float_val);
161        }
162
163        Tensor::from_vec(float_data, self.shape.clone())
164    }
165
166    /// Get the compression ratio compared to full precision
167    /// 全精度と比較した圧縮比を取得
168    pub fn compression_ratio(&self) -> f32 {
169        match self.params.qtype {
170            QuantizationType::Int8 => 4.0,    // 32-bit float to 8-bit int
171            QuantizationType::Int4 => 8.0,    // 32-bit float to 4-bit int
172            QuantizationType::Float16 => 2.0, // 32-bit float to 16-bit float
173            QuantizationType::Dynamic => 3.0, // Average compression
174        }
175    }
176
177    /// Get memory usage in bytes
178    /// メモリ使用量をバイト単位で取得
179    pub fn memory_bytes(&self) -> usize {
180        match self.params.qtype {
181            QuantizationType::Int8 => self.data.len(),
182            QuantizationType::Int4 => self.data.len().div_ceil(2), // 4 bits per element
183            QuantizationType::Float16 => self.data.len() * 2,
184            QuantizationType::Dynamic => self.data.len(),
185        }
186    }
187}

189/// Quantizer for neural network layers and tensors
190/// ニューラルネットワーク層とテンソル用量子化器
191#[derive(Debug)]
192pub struct Quantizer<T: Float> {
193    /// Calibration mode for range determination
194    /// 範囲決定用キャリブレーションモード
195    calibration_mode: CalibrationMode,
196    /// Cache of quantization parameters per layer
197    /// 層ごとの量子化パラメータキャッシュ
198    param_cache: HashMap<String, QuantizationParams<T>>,
199    /// Whether to use symmetric quantization
200    /// 対称量子化を使用するかどうか
201    symmetric: bool,
202}
203
204impl<T> Quantizer<T>
205where
206    T: Float
207        + FromPrimitive
208        + ToPrimitive
209        + Debug
210        + Default
211        + Zero
212        + One
213        + Send
214        + Sync
215        + Copy
216        + ScalarOperand
217        + Sum
218        + Signed,
219{
220    /// Create a new quantizer
221    /// 新しい量子化器を作成
222    pub fn new(calibration_mode: CalibrationMode, symmetric: bool) -> Self {
223        Quantizer {
224            calibration_mode,
225            param_cache: HashMap::new(),
226            symmetric,
227        }
228    }
229
230    /// Quantize a tensor using the specified quantization type
231    /// 指定された量子化タイプを使用してテンソルを量子化
232    pub fn quantize_tensor(
233        &mut self,
234        tensor: &Tensor<T>,
235        qtype: QuantizationType,
236        layer_name: Option<&str>,
237    ) -> QuantizedTensor<T> {
238        // Compute or retrieve quantization parameters
239        let params = if let Some(name) = layer_name {
240            if let Some(cached_params) = self.param_cache.get(name) {
241                cached_params.clone()
242            } else {
243                let params = self.compute_quantization_params(tensor, qtype);
244                self.param_cache.insert(name.to_string(), params.clone());
245                params
246            }
247        } else {
248            self.compute_quantization_params(tensor, qtype)
249        };
250
251        // Quantize the tensor data
252        let quantized_data = self.quantize_data(tensor, &params);
253
254        QuantizedTensor::new(quantized_data, tensor.shape().to_vec(), params)
255    }
256
257    /// Compute quantization parameters for a tensor
258    /// テンソル用の量子化パラメータを計算
259    fn compute_quantization_params(
260        &self,
261        tensor: &Tensor<T>,
262        qtype: QuantizationType,
263    ) -> QuantizationParams<T> {
264        let tensor_array = tensor.as_array();
265        let tensor_slice = tensor_array
266            .as_slice()
267            .unwrap_or_else(|| panic!("Cannot get slice from tensor"));
268
269        let (min_val, max_val) = match self.calibration_mode {
270            CalibrationMode::MinMax => {
271                let min = tensor_slice
272                    .iter()
273                    .fold(T::infinity(), |a, &b| if a < b { a } else { b });
274                let max = tensor_slice
275                    .iter()
276                    .fold(T::neg_infinity(), |a, &b| if a > b { a } else { b });
277                (min, max)
278            }
279            CalibrationMode::Percentile(p) => {
280                let mut sorted = tensor_slice.to_vec();
281                sorted.sort_by(|a, b| a.partial_cmp(b).unwrap());
282                let len = sorted.len();
283                let low_idx = ((1.0 - p) * 0.5 * len as f32) as usize;
284                let high_idx = len - 1 - low_idx;
285                (sorted[low_idx], sorted[high_idx])
286            }
287            CalibrationMode::Entropy => {
288                // Simplified entropy-based calibration (would need full KL divergence implementation)
289                let min = tensor_slice
290                    .iter()
291                    .fold(T::infinity(), |a, &b| if a < b { a } else { b });
292                let max = tensor_slice
293                    .iter()
294                    .fold(T::neg_infinity(), |a, &b| if a > b { a } else { b });
295                (min, max)
296            }
297        };
298
299        match qtype {
300            QuantizationType::Int8 => {
301                if self.symmetric {
302                    let abs_max = min_val.abs().max(max_val.abs());
303                    let scale = abs_max / T::from_i32(127).unwrap();
304                    QuantizationParams::int8_symmetric(scale)
305                } else {
306                    let range = max_val - min_val;
307                    let scale = range / T::from_i32(255).unwrap();
308                    let zero_point = -(min_val / scale).round().to_i32().unwrap_or(0) - 128;
309                    QuantizationParams::int8_asymmetric(scale, zero_point)
310                }
311            }
312            QuantizationType::Int4 => {
313                let abs_max = min_val.abs().max(max_val.abs());
314                let scale = abs_max / T::from_i32(7).unwrap();
315                QuantizationParams::int4_symmetric(scale)
316            }
317            QuantizationType::Float16 => {
318                // For float16, we just use direct conversion (simplified)
319                QuantizationParams::int8_symmetric(T::one())
320            }
321            QuantizationType::Dynamic => {
322                // Dynamic quantization uses different scales per channel/group
323                let abs_max = min_val.abs().max(max_val.abs());
324                let scale = abs_max / T::from_i32(127).unwrap();
325                QuantizationParams::int8_symmetric(scale)
326            }
327        }
328    }
329
330    /// Quantize tensor data using the given parameters
331    /// 与えられたパラメータを使用してテンソルデータを量子化
332    fn quantize_data(&self, tensor: &Tensor<T>, params: &QuantizationParams<T>) -> Vec<i8> {
333        let tensor_array = tensor.as_array();
334        let tensor_slice = tensor_array.as_slice().unwrap();
335
336        tensor_slice
337            .iter()
338            .map(|&val| {
339                let scaled = val / params.scale;
340                let quantized = scaled.round().to_i32().unwrap_or(0) + params.zero_point;
341                let clamped = quantized.max(params.qmin).min(params.qmax);
342                clamped as i8
343            })
344            .collect()
345    }
346
347    /// Apply post-training quantization to a module
348    /// モジュールにポストトレーニング量子化を適用
349    pub fn quantize_module<M: Module<T>>(
350        &mut self,
351        module: &M,
352        qtype: QuantizationType,
353        layer_name: &str,
354    ) -> Vec<QuantizedTensor<T>> {
355        let parameters = module.parameters();
356        let mut quantized_params = Vec::new();
357
358        for (i, param) in parameters.iter().enumerate() {
359            let param_name = format!("{}_{}", layer_name, i);
360            let param_tensor = param.data();
361            let param_data = param_tensor.read().unwrap();
362
363            let quantized = self.quantize_tensor(&*param_data, qtype, Some(&param_name));
364            quantized_params.push(quantized);
365        }
366
367        quantized_params
368    }
369
370    /// Clear the parameter cache
371    /// パラメータキャッシュをクリア
372    pub fn clear_cache(&mut self) {
373        self.param_cache.clear();
374    }
375
376    /// Get calibration statistics
377    /// キャリブレーション統計を取得
378    pub fn get_statistics(&self) -> HashMap<String, QuantizationParams<T>> {
379        self.param_cache.clone()
380    }
381}

383/// Quantization-aware training (QAT) wrapper
384/// 量子化対応訓練（QAT）ラッパー
385#[derive(Debug)]
386pub struct QuantizationAwareModule<
387    T: Float + Send + Sync + 'static + ndarray::ScalarOperand + num_traits::FromPrimitive,
388    M: Module<T> + 'static,
389> {
390    /// Underlying module
391    /// 基底モジュール
392    module: M,
393    /// Quantization parameters for each layer
394    /// 各層の量子化パラメータ
395    qparams: HashMap<String, QuantizationParams<T>>,
396    /// Whether QAT is enabled
397    /// QATが有効かどうか
398    qat_enabled: bool,
399    /// Fake quantization during training
400    /// 訓練中の擬似量子化
401    fake_quantize: bool,
402}
403
404impl<T, M> QuantizationAwareModule<T, M>
405where
406    T: Float
407        + FromPrimitive
408        + ToPrimitive
409        + Debug
410        + Default
411        + Zero
412        + One
413        + Send
414        + Sync
415        + Copy
416        + ScalarOperand
417        + Sum
418        + Signed
419        + 'static,
420    M: Module<T> + 'static,
421{
422    /// Create a new QAT module
423    /// 新しいQATモジュールを作成
424    pub fn new(module: M) -> Self {
425        QuantizationAwareModule {
426            module,
427            qparams: HashMap::new(),
428            qat_enabled: false,
429            fake_quantize: true,
430        }
431    }
432
433    /// Enable quantization-aware training
434    /// 量子化対応訓練を有効化
435    pub fn enable_qat(&mut self) {
436        self.qat_enabled = true;
437    }
438
439    /// Disable quantization-aware training
440    /// 量子化対応訓練を無効化
441    pub fn disable_qat(&mut self) {
442        self.qat_enabled = false;
443    }
444
445    /// Set fake quantization mode
446    /// 擬似量子化モードを設定
447    pub fn set_fake_quantize(&mut self, enabled: bool) {
448        self.fake_quantize = enabled;
449    }
450
451    /// Apply fake quantization to simulate quantization effects during training
452    /// 訓練中に量子化効果をシミュレートするための擬似量子化を適用
453    fn apply_fake_quantization(&self, input: &Variable<T>, layer_name: &str) -> Variable<T> {
454        if !self.qat_enabled || !self.fake_quantize {
455            return input.clone();
456        }
457
458        if let Some(params) = self.qparams.get(layer_name) {
459            let input_binding = input.data();
460            let input_data = input_binding.read().unwrap();
461
462            // Simulate quantization by quantizing and dequantizing
463            let quantized_data: Vec<T> = input_data
464                .as_array()
465                .iter()
466                .map(|&val| {
467                    let scaled = val / params.scale;
468                    let quantized = scaled.round().to_i32().unwrap_or(0) + params.zero_point;
469                    let clamped = quantized.max(params.qmin).min(params.qmax);
470                    let qval_adjusted = clamped - params.zero_point;
471                    T::from_i32(qval_adjusted).unwrap() * params.scale
472                })
473                .collect();
474
475            let fake_quantized_tensor =
476                Tensor::from_vec(quantized_data, input_data.shape().to_vec());
477            Variable::new(fake_quantized_tensor, input.requires_grad())
478        } else {
479            input.clone()
480        }
481    }
482}

484impl<T, M> Module<T> for QuantizationAwareModule<T, M>
485where
486    T: Float
487        + FromPrimitive
488        + ToPrimitive
489        + Debug
490        + Default
491        + Zero
492        + One
493        + Send
494        + Sync
495        + Copy
496        + ScalarOperand
497        + Sum
498        + Signed
499        + 'static,
500    M: Module<T> + 'static,
501{
502    fn forward(&self, input: &Variable<T>) -> Variable<T> {
503        let fake_quantized_input = self.apply_fake_quantization(input, "input");
504        self.module.forward(&fake_quantized_input)
505    }
506
507    fn parameters(&self) -> Vec<Variable<T>> {
508        self.module.parameters()
509    }
510
511    fn as_any(&self) -> &dyn std::any::Any {
512        self
513    }
514}

520    #[test]
521    fn test_quantization_params_creation() {
522        let params = QuantizationParams::<f32>::int8_symmetric(0.1);
523        assert_eq!(params.qmin, -128);
524        assert_eq!(params.qmax, 127);
525        assert_eq!(params.zero_point, 0);
526        assert_eq!(params.qtype, QuantizationType::Int8);
527    }
528
529    #[test]
530    fn test_quantizer_creation() {
531        let quantizer = Quantizer::<f32>::new(CalibrationMode::MinMax, true);
532        assert_eq!(quantizer.calibration_mode, CalibrationMode::MinMax);
533        assert!(quantizer.symmetric);
534    }
535
536    #[test]
537    fn test_tensor_quantization() {
538        let mut quantizer = Quantizer::<f32>::new(CalibrationMode::MinMax, true);
539
540        // Create a simple tensor
541        let data = vec![1.0, 2.0, 3.0, -1.0, -2.0, -3.0];
542        let tensor = Tensor::from_vec(data, vec![2, 3]);
543
544        let quantized = quantizer.quantize_tensor(&tensor, QuantizationType::Int8, None);
545
546        assert_eq!(quantized.shape, vec![2, 3]);
547        assert_eq!(quantized.data.len(), 6);
548        assert_eq!(quantized.params.qtype, QuantizationType::Int8);
549    }
550
551    #[test]
552    fn test_quantized_tensor_dequantization() {
553        let params = QuantizationParams::<f32>::int8_symmetric(0.1);
554        let data = vec![10, 20, 30, -10, -20, -30];
555        let shape = vec![2, 3];
556
557        let quantized = QuantizedTensor::new(data, shape, params);
558        let dequantized = quantized.dequantize();
559
560        assert_eq!(dequantized.shape(), &[2, 3]);
561
562        // Check that dequantization is approximately correct
563        let dequant_array = dequantized.as_array();
564        let expected = [1.0, 2.0, 3.0, -1.0, -2.0, -3.0];
565        for (i, &val) in dequant_array.iter().enumerate() {
566            assert!((val - expected[i]).abs() < 0.01);
567        }
568    }
569
570    #[test]
571    fn test_compression_ratio() {
572        let params = QuantizationParams::<f32>::int8_symmetric(0.1);
573        let quantized = QuantizedTensor::new(vec![1, 2, 3], vec![3], params);
574
575        assert_eq!(quantized.compression_ratio(), 4.0);
576    }
