1//! # Parallel Tensor Operations Traits
2//! 並列テンソル操作のトレイト定義
3//!
4//! This module provides a unified trait-based system for parallel tensor operations,
5//! enabling high-performance computation across multiple threads with automatic
6//! SIMD acceleration and intelligent scheduling.
7//!
8//! ## Overview
9//!
10//! The parallel traits system is designed around five core traits:
11//!
12//! - [`ParallelOp`]: Base trait for all parallel operations
13//! - [`BatchParallelOp`]: Batch processing operations
14//! - [`MatrixParallelOp`]: Matrix-specific parallel operations
15//! - [`ReductionParallelOp`]: Parallel reduction operations
16//! - [`SimdParallelOp`]: SIMD-optimized parallel operations (f32 specialized)
17//!
18//! ## Key Features
19//!
20//! - **Automatic Parallelization**: Operations automatically parallelize for large tensors
21//! - **SIMD Integration**: Specialized f32 operations with AVX2/SSE4.1 acceleration
22//! - **Configurable Strategies**: Choose between automatic, forced parallel, or sequential execution
23//! - **Memory Safety**: Arc-based shared ownership ensures thread safety
24//! - **Error Handling**: Structured error types with detailed context information
25//!
26//! ## Usage Examples
27//!
28//! ### Basic Parallel Operations
29//!
30//! ```rust
31//! use rustorch::tensor::Tensor;
32//!
33//! let tensor1 = Tensor::<f32>::ones(&[4, 4]);
34//! let tensor2 = Tensor::<f32>::ones(&[4, 4]);
35//!
36//! // Basic tensor operations
37//! let result = &tensor1 + &tensor2;
38//! let matmul_result = tensor1.matmul(&tensor2);
39//!
40//! # assert_eq!(result.shape(), &[4, 4]);
41//! # assert_eq!(matmul_result.unwrap().shape(), &[4, 4]);
42//! ```
43//!
44//! ### SIMD-Optimized Operations
45//!
46//! ```rust
47//! use rustorch::tensor::{Tensor, parallel_traits::*};
48//!
49//! let tensor1 = Tensor::<f32>::ones(&[10000]);
50//! let tensor2 = Tensor::<f32>::ones(&[10000]);
51//!
52//! // Basic parallel operations
53//! let result = &tensor1 + &tensor2; // Element-wise addition
54//!
55//! // Matrix multiplication
56//! let matrix1 = Tensor::<f32>::ones(&[100, 100]);
57//! let matrix2 = Tensor::<f32>::ones(&[100, 100]);
58//! let matmul_result = matrix1.matmul(&matrix2);
59//! # Ok::<(), Box<dyn std::error::Error>>(())
60//! ```
61//!
62//! ### Configurable Parallel Execution
63//!
64//! ```rust
65//! use rustorch::tensor::parallel_traits::*;
66//!
67//! // Configure parallel execution strategy
68//! let config = ParallelConfig {
69//!     strategy: ParallelStrategy::ForceParallel,
70//!     chunk_size: 2048,
71//!     num_threads: Some(4),
72//! };
73//!
74//! let context = ParallelContext::new(config);
75//!
76//! // Use context to determine execution strategy
77//! let should_parallelize = context.should_parallelize(1000);
78//! println!("Should parallelize: {}", should_parallelize);
79//! ```
80//!
81//! ## Performance Characteristics
82//!
83//! - **Threshold-based**: Operations automatically parallelize when tensor size exceeds 1000 elements
84//! - **SIMD Acceleration**: f32 operations benefit from AVX2/SSE4.1 vectorization
85//! - **Memory Efficient**: Zero-copy operations where possible, minimal memory allocation
86//! - **Thread Safety**: All operations are thread-safe with proper synchronization
87//!
88//! ## Error Handling
89//!
90//! All parallel operations return [`ParallelResult<T>`](Result)
91//! with detailed error information for debugging and error recovery.
92//!
93//! See [`parallel_errors`](crate::tensor::parallel_errors) for complete error type documentation.
94

101/// 並列操作の基本トレイト
102/// Base trait for parallel operations
103pub trait ParallelOp<T: Float + Send + Sync + Clone + 'static> {
104    /// 並列実行の最小サイズ閾値
105    /// Minimum size threshold for parallel execution
106    const MIN_PARALLEL_SIZE: usize = 1000;
107
108    /// 並列実行が効率的かどうかを判定
109    /// Determine if parallel execution is efficient
110    fn should_parallelize(&self, size: usize) -> bool {
111        size >= Self::MIN_PARALLEL_SIZE
112    }
113}

115/// バッチ処理操作のトレイト
116/// Trait for batch processing operations
117pub trait BatchParallelOp<T: Float + Send + Sync + Clone + 'static>: ParallelOp<T> {
118    /// 並列バッチ要素ごと演算
119    /// Parallel batch element-wise operations
120    fn batch_elementwise_op<F>(&self, other: &Tensor<T>, op: F) -> ParallelResult<Tensor<T>>
121    where
122        F: Fn(T, T) -> T + Send + Sync;
123
124    /// 並列バッチスカラー演算
125    /// Parallel batch scalar operations
126    fn batch_scalar_op<F>(&self, scalar: T, op: F) -> Tensor<T>
127    where
128        F: Fn(T, T) -> T + Send + Sync;
129
130    /// 並列バッチ正規化
131    /// Parallel batch normalization
132    fn batch_normalize(&self, epsilon: T) -> Tensor<T>;
133}

135/// 行列演算の並列化トレイト
136/// Trait for parallel matrix operations
137pub trait MatrixParallelOp<T: Float + Send + Sync + Clone + 'static>: ParallelOp<T> {
138    /// 並列バッチ行列乗算

143    /// Parallel convolution operation
144    fn batch_conv2d(
145        &self,

147        stride: usize,
148        padding: usize,
149    ) -> ParallelResult<Tensor<T>>;
150}

152/// リダクション操作の並列化トレイト
153/// Trait for parallel reduction operations
154pub trait ReductionParallelOp<T: Float + Send + Sync + Clone + 'static>: ParallelOp<T> {
155    /// 並列リダクション演算
156    /// Parallel reduction operation
157    fn parallel_reduce<F, R>(&self, dim: usize, init: R, op: F) -> ParallelResult<Tensor<T>>
158    where
159        F: Fn(R, T) -> R + Send + Sync + Clone,
160        R: Send + Sync + Clone + Into<T>;
161
162    /// 並列合計
163    /// Parallel sum
164    fn parallel_sum(&self, dim: usize) -> ParallelResult<Tensor<T>> {
165        self.parallel_reduce(dim, T::zero(), |acc, x| acc + x)
166    }
167
168    /// 並列平均
169    /// Parallel mean
170    fn parallel_mean(&self, dim: usize) -> ParallelResult<Tensor<T>>;
171}

189/// 並列操作の実行戦略
190/// Execution strategy for parallel operations
191#[derive(Debug, Clone, Copy)]
192pub enum ParallelStrategy {
193    /// 自動選択（サイズに基づく）
194    /// Automatic selection (size-based)
195    Auto,
196    /// 強制的に並列実行
197    /// Force parallel execution
198    ForceParallel,
199    /// 強制的に逐次実行
200    /// Force sequential execution
201    ForceSequential,
202}
203
204/// 並列操作の設定
205/// Configuration for parallel operations
206#[derive(Debug, Clone)]
207pub struct ParallelConfig {
208    /// 実行戦略
209    /// Execution strategy
210    pub strategy: ParallelStrategy,
211    /// チャンクサイズ（SIMD最適化用）
212    /// Chunk size (for SIMD optimization)
213    pub chunk_size: usize,
214    /// 並列度（スレッド数）
215    /// Parallelism level (number of threads)
216    pub num_threads: Option<usize>,
217}
218
219impl Default for ParallelConfig {
220    fn default() -> Self {
221        Self {
222            strategy: ParallelStrategy::Auto,
223            chunk_size: 1024,
224            num_threads: None,
225        }
226    }
227}

229/// 並列操作のコンテキスト
230/// Context for parallel operations
231pub struct ParallelContext {
232    /// 設定
233    /// Configuration
234    pub config: ParallelConfig,
235}
236
237impl ParallelContext {
238    /// 新しいコンテキストを作成
239    /// Create a new context
240    pub fn new(config: ParallelConfig) -> Self {
241        Self { config }
242    }
243
244    /// デフォルトコンテキストを作成
245    /// Create a default context
246    pub fn default() -> Self {
247        Self::new(ParallelConfig::default())
248    }
249
250    /// 並列実行すべきかを判定
251    /// Determine if should execute in parallel
252    pub fn should_parallelize(&self, size: usize) -> bool {
253        match self.config.strategy {
254            ParallelStrategy::Auto => size >= 1000,
255            ParallelStrategy::ForceParallel => true,
256            ParallelStrategy::ForceSequential => false,
257        }
258    }
259}

261/// 並列操作のユーティリティ関数
262/// Utility functions for parallel operations
263pub mod parallel_utils {
264    use super::*;
265
266    /// 安全な並列インデックス計算
267    /// Safe parallel index calculation
268    pub fn safe_parallel_index(
269        total_size: usize,
270        chunk_size: usize,
271        chunk_idx: usize,
272    ) -> (usize, usize) {
273        let start = chunk_idx * chunk_size;
274        let end = std::cmp::min(start + chunk_size, total_size);
275        (start, end)
276    }
277
278    /// 並列チャンク処理
279    /// Parallel chunk processing
280    pub fn parallel_chunks<T, F, R>(data: &[T], chunk_size: usize, process_chunk: F) -> Vec<R>
281    where
282        T: Send + Sync,
283        F: Fn(&[T]) -> R + Send + Sync,
284        R: Send,
285    {
286        data.par_chunks(chunk_size).map(process_chunk).collect()
287    }
288
289    /// バッチ次元での並列処理
290    /// Parallel processing over batch dimension
291    pub fn parallel_batch_process<T, F, R>(batch_size: usize, process_batch: F) -> Vec<R>
292    where
293        F: Fn(usize) -> R + Send + Sync,
294        R: Send,
295    {
296        (0..batch_size).into_par_iter().map(process_batch).collect()
297    }
298}
