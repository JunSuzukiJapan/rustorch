91/// Memory block metadata
92/// メモリブロックメタデータ
93#[derive(Debug)]
94struct MemoryBlock {
95    ptr: NonNull<u8>,
96    size: usize,
97    alignment: usize,
98    last_accessed: Instant,
99    access_count: u64,
100    is_huge_page: bool,
101}
102
103impl MemoryBlock {
104    fn new(size: usize, alignment: usize, is_huge_page: bool) -> ParallelResult<Self> {
105        let layout = Layout::from_size_align(size, alignment)
106            .map_err(|e| RusTorchError::memory(format!("Invalid layout: {}", e)))?;
107
108        let ptr = unsafe {
109            let raw_ptr = if is_huge_page {
110                Self::alloc_huge_page(size, alignment)?
111            } else {
112                alloc_zeroed(layout)
113            };
114
115            if raw_ptr.is_null() {
116                return Err(RusTorchError::memory("Allocation failed"));
117            }
118
119            NonNull::new_unchecked(raw_ptr)
120        };
121
122        let now = Instant::now();
123        Ok(Self {
124            ptr,
125            size,
126            alignment,
127            last_accessed: now,
128            access_count: 0,
129            is_huge_page,
130        })
131    }
132
133    #[cfg(target_os = "linux")]
134    fn alloc_huge_page(size: usize, alignment: usize) -> ParallelResult<*mut u8> {
135        use std::fs::OpenOptions;
136        use std::os::unix::io::AsRawFd;
137
138        // Try to allocate using mmap with MAP_HUGETLB
139        // MAP_HUGETLBを使用してmmapで割り当てを試行
140        let fd = OpenOptions::new()
141            .read(true)
142            .write(true)
143            .open("/dev/zero")
144            .map_err(|e| RusTorchError::IO(e))?;
145
146        unsafe {
147            let ptr = libc::mmap(
148                std::ptr::null_mut(),
149                size,
150                libc::PROT_READ | libc::PROT_WRITE,
151                libc::MAP_PRIVATE | libc::MAP_HUGETLB,
152                fd.as_raw_fd(),
153                0,
154            );
155
156            if ptr == libc::MAP_FAILED {
157                // Fallback to regular allocation
158                // 通常の割り当てにフォールバック
159                let layout = Layout::from_size_align(size, alignment)
160                    .map_err(|e| RusTorchError::memory(format!("Invalid layout: {}", e)))?;
161                Ok(alloc_zeroed(layout))
162            } else {
163                Ok(ptr as *mut u8)
164            }
165        }
166    }
167
168    #[cfg(not(target_os = "linux"))]
169    fn alloc_huge_page(size: usize, alignment: usize) -> ParallelResult<*mut u8> {
170        // Fallback to regular allocation on non-Linux systems
171        // Linux以外のシステムでは通常の割り当てにフォールバック
172        let layout = Layout::from_size_align(size, alignment)
173            .map_err(|e| RusTorchError::memory(format!("Invalid layout: {}", e)))?;
174        Ok(unsafe { alloc_zeroed(layout) })
175    }
176
177    fn update_access(&mut self) {
178        self.last_accessed = Instant::now();
179        self.access_count += 1;
180    }
181
182    fn idle_time(&self) -> Duration {
183        self.last_accessed.elapsed()
184    }
185}
186
187impl Drop for MemoryBlock {
188    fn drop(&mut self) {
189        unsafe {
190            if self.is_huge_page {
191                #[cfg(target_os = "linux")]
192                {
193                    libc::munmap(self.ptr.as_ptr() as *mut libc::c_void, self.size);
194                }
195                #[cfg(not(target_os = "linux"))]
196                {
197                    let layout = Layout::from_size_align_unchecked(self.size, self.alignment);
198                    dealloc(self.ptr.as_ptr(), layout);
199                }
200            } else {
201                let layout = Layout::from_size_align_unchecked(self.size, self.alignment);
202                dealloc(self.ptr.as_ptr(), layout);
203            }
204        }
205    }
206}

263    /// Allocate memory with specified strategy
264    /// 指定された戦略でメモリを割り当て
265    pub fn allocate<T: Float + 'static>(
266        &self,
267        size: usize,
268        strategy: AllocationStrategy,
269    ) -> ParallelResult<NonNull<T>> {
270        let alignment = self.get_alignment_for_strategy(strategy);
271        let actual_size = self.round_up_size(size * std::mem::size_of::<T>(), alignment);
272
273        // Try to reuse existing block
274        // 既存ブロックの再利用を試行
275        if let Some(block) = self.try_reuse_block(actual_size, alignment)? {
276            let ptr = unsafe { NonNull::new_unchecked(block.ptr.as_ptr() as *mut T) };
277            self.update_stats_on_allocation(actual_size, true);
278            return Ok(ptr);
279        }
280
281        // Allocate new block
282        // 新しいブロックを割り当て
283        let use_huge_pages = strategy == AllocationStrategy::HugePage
284            || (actual_size >= HUGE_PAGE_SIZE && self.config.enable_huge_pages);
285
286        let mut block = MemoryBlock::new(actual_size, alignment, use_huge_pages)?;
287
288        // Prefault pages if enabled
289        // 有効な場合はページをプリフォルト
290        if self.config.enable_prefaulting {
291            self.prefault_pages(&mut block)?;
292        }
293
294        let ptr = unsafe { NonNull::new_unchecked(block.ptr.as_ptr() as *mut T) };
295
296        // Track allocated block
297        // 割り当てブロックを追跡
298        {
299            let mut allocated = self.allocated_blocks.write().unwrap();
300            allocated.insert(block.ptr.as_ptr(), block);
301        }
302
303        self.update_stats_on_allocation(actual_size, false);
304        Ok(ptr)
305    }
306
307    /// Deallocate memory
308    /// メモリを解放
309    pub fn deallocate<T>(&self, ptr: NonNull<T>) -> ParallelResult<()> {
310        let raw_ptr = ptr.as_ptr() as *mut u8;
311
312        let block = {
313            let mut allocated = self.allocated_blocks.write().unwrap();
314            allocated
315                .remove(&raw_ptr)
316                .ok_or_else(|| RusTorchError::memory("Pointer not found in allocated blocks"))?
317        };
318
319        let size = block.size;
320
321        // Return block to free pool if it's worth keeping
322        // 保持する価値がある場合はフリープールに返却
323        if self.should_keep_block(&block) {
324            let mut free_blocks = self.free_blocks.write().unwrap();
325            free_blocks.entry(size).or_default().push_back(block);
326        }
327        // Otherwise, block will be dropped and memory freed
328        // そうでなければ、ブロックはドロップされメモリが解放される
329
330        self.update_stats_on_deallocation(size);
331        Ok(())
332    }

341    /// Perform garbage collection
342    /// ガベージコレクションを実行
343    pub fn garbage_collect(&self) -> ParallelResult<usize> {
344        let mut freed_memory = 0;
345        let _now = Instant::now();
346        let max_idle_time = Duration::from_secs(300); // 5 minutes
347
348        let mut free_blocks = self.free_blocks.write().unwrap();
349
350        for (size, blocks) in free_blocks.iter_mut() {
351            blocks.retain(|block| {
352                if block.idle_time() > max_idle_time {
353                    freed_memory += size;
354                    false
355                } else {
356                    true
357                }
358            });
359        }
360
361        // Remove empty size buckets
362        // 空のサイズバケットを削除
363        free_blocks.retain(|_, blocks| !blocks.is_empty());
364
365        Ok(freed_memory)
366    }

368    /// Optimize memory layout for NUMA
369    /// NUMA用のメモリレイアウト最適化
370    pub fn optimize_for_numa(&self) -> ParallelResult<()> {
371        if let Some(node) = self.numa_node {
372            // Bind memory allocations to specific NUMA node
373            // メモリ割り当てを特定のNUMAノードにバインド
374            #[cfg(target_os = "linux")]
375            {
376                self.set_numa_policy(node)?;
377            }
378            #[cfg(not(target_os = "linux"))]
379            {
380                let _ = node; // Suppress unused variable warning on non-Linux platforms
381            }
382        }
383        Ok(())
384    }

389    fn get_alignment_for_strategy(&self, strategy: AllocationStrategy) -> usize {
390        match strategy {
391            AllocationStrategy::Standard => std::mem::align_of::<f64>(),
392            AllocationStrategy::CacheAligned => CACHE_LINE_SIZE,
393            AllocationStrategy::PageAligned => PAGE_SIZE,
394            AllocationStrategy::HugePage => HUGE_PAGE_SIZE,
395            AllocationStrategy::Pooled => self.config.alignment,
396            AllocationStrategy::NumaAware => CACHE_LINE_SIZE,
397        }
398    }

400    fn round_up_size(&self, size: usize, alignment: usize) -> usize {
401        (size + alignment - 1) & !(alignment - 1)
402    }

404    fn try_reuse_block(
405        &self,
406        size: usize,
407        alignment: usize,
408    ) -> ParallelResult<Option<MemoryBlock>> {
409        let mut free_blocks = self.free_blocks.write().unwrap();
410
411        // Look for exact size match first
412        // まず正確なサイズマッチを探す
413        if let Some(blocks) = free_blocks.get_mut(&size) {
414            if let Some(mut block) = blocks.pop_front() {
415                if block.alignment >= alignment {
416                    block.update_access();
417                    return Ok(Some(block));
418                } else {
419                    blocks.push_back(block);
420                }
421            }
422        }
423
424        // Look for larger blocks that can be split
425        // 分割可能な大きなブロックを探す
426        for (&block_size, blocks) in free_blocks.iter_mut() {
427            if block_size >= size && !blocks.is_empty() {
428                if let Some(mut block) = blocks.pop_front() {
429                    if block.alignment >= alignment {
430                        block.update_access();
431                        return Ok(Some(block));
432                    } else {
433                        blocks.push_back(block);
434                    }
435                }
436            }
437        }
438
439        Ok(None)
440    }

442    fn should_keep_block(&self, block: &MemoryBlock) -> bool {
443        let current_total = *self.total_allocated.lock().unwrap();
444        let would_exceed_max = current_total + block.size > self.config.max_size;
445
446        !would_exceed_max && block.access_count > 1
447    }

449    fn prefault_pages(&self, block: &mut MemoryBlock) -> ParallelResult<()> {
450        unsafe {
451            let ptr = block.ptr.as_ptr();
452            let size = block.size;
453
454            // Touch each page to prefault
455            // 各ページにタッチしてプリフォルト
456            for offset in (0..size).step_by(PAGE_SIZE) {
457                let page_ptr = ptr.add(offset);
458                std::ptr::write_volatile(page_ptr, 0);
459            }
460        }
461        Ok(())
462    }

464    fn update_stats_on_allocation(&self, size: usize, cache_hit: bool) {
465        let mut stats = self.allocation_stats.lock().unwrap();
466        let mut total = self.total_allocated.lock().unwrap();
467
468        stats.total_allocations += 1;
469        *total += size;
470        stats.current_memory_usage = *total;
471
472        if *total > stats.peak_memory_usage {
473            stats.peak_memory_usage = *total;
474        }
475
476        if cache_hit {
477            stats.cache_hits += 1;
478        } else {
479            stats.cache_misses += 1;
480        }
481    }
482
483    fn update_stats_on_deallocation(&self, size: usize) {
484        let mut stats = self.allocation_stats.lock().unwrap();
485        let mut total = self.total_allocated.lock().unwrap();
486
487        stats.total_deallocations += 1;
488        *total -= size;
489        stats.current_memory_usage = *total;
490    }

525        Self {

530    /// Create tensor with optimized memory allocation
531    /// 最適化メモリ割り当てでテンソルを作成
532    pub fn create_tensor<T: Float + 'static>(
533        &self,
534        shape: &[usize],
535        strategy: AllocationStrategy,
536    ) -> ParallelResult<Tensor<T>> {
537        let total_elements: usize = shape.iter().product();
538        let ptr = self.memory_pool.allocate(total_elements, strategy)?;
539
540        // Create tensor with custom memory
541        // カスタムメモリでテンソルを作成
542        unsafe {
543            let data = std::slice::from_raw_parts_mut(ptr.as_ptr(), total_elements);
544            data.fill(T::zero());
545            Ok(Tensor::from_raw_parts(data, shape))
546        }
547    }
548
549    /// Perform in-place operations to minimize memory allocation
550    /// メモリ割り当てを最小化するインプレース演算
551    pub fn add_inplace<T: Float + 'static>(
552        &self,
553        a: &mut Tensor<T>,
554        b: &Tensor<T>,
555    ) -> ParallelResult<()> {
556        if a.shape() != b.shape() {
557            return Err(RusTorchError::shape_mismatch(a.shape(), b.shape()));
558        }
559
560        // Vectorized in-place addition
561        // ベクトル化インプレース加算
562        let a_slice = a.as_slice_mut().unwrap();
563        let b_slice = b.as_slice().unwrap();
564        for i in 0..a_slice.len() {
565            a_slice[i] = a_slice[i] + b_slice[i];
566        }
567
568        Ok(())
569    }

574        self.memory_pool.get_stats()
575    }

578    /// メモリ最適化を実行
579    pub fn optimize_memory(&self) -> ParallelResult<usize> {
580        self.memory_pool.garbage_collect()
581    }

662            pool.deallocate(ptr).unwrap();
663        }
664
665        let freed = pool.garbage_collect().unwrap();
666        // Some memory should be freed during GC
667        // GC中にいくらかのメモリが解放されるはず
668        // freed is usize, always >= 0
669        assert!(freed == freed); // Keep the variable used
670    }
