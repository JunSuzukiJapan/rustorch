26    /// Check if two shapes can be broadcast together (optimized version)
27    pub(super) fn can_broadcast(shape1: &[usize], shape2: &[usize]) -> bool {
28        let len1 = shape1.len();
29        let len2 = shape2.len();
30        let max_len = len1.max(len2);
31
32        for i in 0..max_len {
33            let dim1 = shape1
34                .get(len1.saturating_sub(max_len - i))
35                .copied()
36                .unwrap_or(1);
37            let dim2 = shape2
38                .get(len2.saturating_sub(max_len - i))
39                .copied()
40                .unwrap_or(1);
41
42            if dim1 != dim2 && dim1 != 1 && dim2 != 1 {
43                return false;
44            }
45        }
46        true
47    }
48
49    /// Calculate the output shape after broadcasting (optimized version)
50    pub(super) fn broadcast_shape(shape1: &[usize], shape2: &[usize]) -> RusTorchResult<Shape> {
51        let len1 = shape1.len();
52        let len2 = shape2.len();
53        let max_len = len1.max(len2);
54        let mut result = Vec::with_capacity(max_len);
55
56        for i in 0..max_len {
57            let dim1 = shape1
58                .get(len1.saturating_sub(max_len - i))
59                .copied()
60                .unwrap_or(1);
61            let dim2 = shape2
62                .get(len2.saturating_sub(max_len - i))
63                .copied()
64                .unwrap_or(1);
65
66            if dim1 != dim2 && dim1 != 1 && dim2 != 1 {
67                return Err(RusTorchError::shape_mismatch(shape1, shape2));
68            }
69
70            result.push(dim1.max(dim2));
71        }
72
73        Ok(result)
74    }

76    /// Calculate broadcast indices with better cache locality
77    pub(super) fn broadcast_index(
78        flat_idx: usize,
79        original_shape: &[usize],
80        target_shape: &[usize],
81    ) -> usize {
82        if original_shape == target_shape {
83            return flat_idx;
84        }
85
86        let mut coords = Vec::with_capacity(target_shape.len());
87        let mut remaining = flat_idx;
88
89        // Convert flat index to coordinates
90        for &dim_size in target_shape.iter().rev() {
91            coords.push(remaining % dim_size);
92            remaining /= dim_size;
93        }
94        coords.reverse();
95
96        // Map to original tensor coordinates
97        let mut result_idx = 0;
98        let mut stride = 1;
99
100        for i in (0..original_shape.len()).rev() {
101            let coord_idx = coords.len().saturating_sub(original_shape.len() - i);
102            let coord = coords.get(coord_idx).copied().unwrap_or(0);
103            let mapped_coord = if original_shape[i] == 1 { 0 } else { coord };
104
105            result_idx += mapped_coord * stride;
106            stride *= original_shape[i];
107        }
108
109        result_idx
110    }

113/// Memory-efficient stride calculations
114/// メモリ効率的なストライド計算
115mod stride_calc {
116    /// Calculate strides for n-dimensional array with better cache efficiency
117    pub(super) fn calculate_strides(shape: &[usize]) -> Vec<usize> {
118        let mut strides = vec![1; shape.len()];
119
120        for i in (0..shape.len().saturating_sub(1)).rev() {
121            strides[i] = strides[i + 1] * shape[i + 1];
122        }
123
124        strides
125    }
126
127    /// Convert flat index to multi-dimensional coordinates
128    pub(super) fn flat_to_coords(flat_idx: usize, strides: &[usize]) -> Vec<usize> {
129        let mut coords = Vec::with_capacity(strides.len());
130        let mut remaining = flat_idx;
131
132        for &stride in strides.iter() {
133            coords.push(remaining / stride);
134            remaining %= stride;
135        }
136
137        coords
138    }
139
140    /// Convert multi-dimensional coordinates to flat index
141    pub(super) fn coords_to_flat(coords: &[usize], strides: &[usize]) -> usize {
142        coords
143            .iter()
144            .zip(strides.iter())
145            .map(|(&coord, &stride)| coord * stride)
146            .sum()
147    }
148}

156    /// Select elements from x or y based on condition (optimized version)
157    /// 条件に基づいてxまたはyから要素を選択（最適化版）
158    ///
159    /// # Arguments
160    /// * `condition` - Boolean mask for selection
161    /// * `x` - Tensor to select from when condition is true
162    /// * `y` - Tensor to select from when condition is false
163    ///
164    /// # Returns
165    /// New tensor with selected elements
166    ///
167    /// # Performance
168    /// Optimized broadcasting with better cache locality and reduced memory allocations
169    pub fn where_<T: Float + 'static>(
170        condition: &BoolMask,
171        x: &Tensor<T>,
172        y: &Tensor<T>,
173    ) -> RusTorchResult<Tensor<T>> {
174        // Validate all shapes can be broadcast together
175        let shapes = [condition.shape(), x.shape(), y.shape()];
176        for i in 0..shapes.len() {
177            for j in (i + 1)..shapes.len() {
178                if !can_broadcast(shapes[i], shapes[j]) {
179                    return Err(RusTorchError::shape_mismatch(shapes[i], shapes[j]));
180                }
181            }
182        }
183
184        // Calculate final output shape
185        let output_shape =
186            broadcast_shape(&broadcast_shape(condition.shape(), x.shape())?, y.shape())?;
187        let total_elements: usize = output_shape.iter().product();
188
189        // Get data slices with proper error handling
190        let condition_data = condition
191            .as_slice()
192            .ok_or_else(|| RusTorchError::tensor_op("Condition mask data not accessible"))?;
193        let x_data = x
194            .data
195            .as_slice()
196            .ok_or_else(|| RusTorchError::tensor_op("X tensor data not accessible"))?;
197        let y_data = y
198            .data
199            .as_slice()
200            .ok_or_else(|| RusTorchError::tensor_op("Y tensor data not accessible"))?;
201
202        // Pre-allocate result with exact capacity
203        let mut result_data = Vec::with_capacity(total_elements);
204
205        // Optimized element selection with better broadcasting
206        for i in 0..total_elements {
207            let cond_idx = broadcast_index(i, condition.shape(), &output_shape);
208            let x_idx = broadcast_index(i, x.shape(), &output_shape);
209            let y_idx = broadcast_index(i, y.shape(), &output_shape);
210
211            let value = if cond_idx < condition_data.len() && condition_data[cond_idx] {
212                if x_idx < x_data.len() {
213                    x_data[x_idx]
214                } else {
215                    T::zero()
216                }
217            } else {
218                if y_idx < y_data.len() {
219                    y_data[y_idx]
220                } else {
221                    T::zero()
222                }
223            };
224
225            result_data.push(value);
226        }
227
228        Ok(Tensor::from_vec(result_data, output_shape))
229    }

231    /// Select elements from input tensor where mask is true (optimized version)
232    /// マスクがtrueの位置から要素を選択（最適化版）
233    ///
234    /// # Arguments
235    /// * `input` - Input tensor
236    /// * `mask` - Boolean mask array
237    ///
238    /// # Returns
239    /// 1D tensor containing selected elements
240    ///
241    /// # Performance
242    /// Uses iterator chaining for better performance and reduced allocations
243    pub fn masked_select<T: Float + 'static>(
244        input: &Tensor<T>,
245        mask: &BoolMask,
246    ) -> RusTorchResult<Tensor<T>> {
247        if input.shape() != mask.shape() {
248            return Err(RusTorchError::shape_mismatch(input.shape(), mask.shape()));
249        }
250
251        let input_data = input
252            .data
253            .as_slice()
254            .ok_or_else(|| RusTorchError::tensor_op("Input tensor data not accessible"))?;
255        let mask_data = mask
256            .as_slice()
257            .ok_or_else(|| RusTorchError::tensor_op("Mask array data not accessible"))?;
258
259        // Count true values first to pre-allocate correctly
260        let true_count = mask_data.iter().filter(|&&val| val).count();
261        let mut selected = Vec::with_capacity(true_count);
262
263        // Efficient selection without intermediate collections
264        for (value, mask_val) in input_data.iter().zip(mask_data.iter()) {
265            if *mask_val {
266                selected.push(*value);
267            }
268        }
269
270        let len = selected.len();
271        Ok(Tensor::from_vec(selected, vec![len]))
272    }

274    /// Fill elements in input tensor where mask is true with specified value (optimized)
275    /// マスクがtrueの位置を指定値で埋める（最適化版）
276    ///
277    /// # Arguments
278    /// * `input` - Mutable input tensor
279    /// * `mask` - Boolean mask array
280    /// * `value` - Value to fill with
281    ///
282    /// # Performance
283    /// Direct iterator-based filling for better cache performance
284    pub fn masked_fill_<T: Float + 'static>(
285        input: &mut Tensor<T>,
286        mask: &BoolMask,
287        value: T,
288    ) -> RusTorchResult<()> {
289        if input.shape() != mask.shape() {
290            return Err(RusTorchError::shape_mismatch(input.shape(), mask.shape()));
291        }
292
293        let input_data = input.data.as_slice_mut().ok_or_else(|| {
294            RusTorchError::tensor_op("Input tensor data not accessible for mutation")
295        })?;
296        let mask_data = mask
297            .as_slice()
298            .ok_or_else(|| RusTorchError::tensor_op("Mask array data not accessible"))?;
299
300        // Direct iterator-based filling for optimal performance
301        input_data
302            .iter_mut()
303            .zip(mask_data.iter())
304            .filter(|(_, &mask_val)| mask_val)
305            .for_each(|(elem, _)| *elem = value);
306
307        Ok(())
308    }

310    /// Non-mutable version of masked_fill that returns a new tensor
311    /// 新しいテンサーを返すmasked_fillの非破壊版
312    ///
313    /// # Performance
314    /// Optimized version that avoids unnecessary cloning when possible
315    pub fn masked_fill<T: Float + 'static>(
316        input: &Tensor<T>,
317        mask: &BoolMask,
318        value: T,
319    ) -> RusTorchResult<Tensor<T>> {
320        // Check for early optimization opportunities
321        if mask
322            .as_slice()
323            .map_or(false, |data| data.iter().all(|&x| !x))
324        {
325            // No elements to fill, return clone
326            return Ok(input.clone());
327        }
328
329        let mut result = input.clone();
330        masked_fill_(&mut result, mask, value)?;
331        Ok(result)
332    }

341    /// Gather values along an axis specified by index (optimized version)
342    /// インデックスで指定された軸に沿って値を収集（最適化版）
343    ///
344    /// # Arguments
345    /// * `input` - Input tensor
346    /// * `dim` - Dimension to gather along
347    /// * `index` - Index array with indices to gather
348    ///
349    /// # Returns
350    /// Gathered tensor with same shape as index
351    ///
352    /// # Performance
353    /// Optimized stride calculations and bounds checking for better performance
354    pub fn gather<T: Float + 'static>(
355        input: &Tensor<T>,
356        dim: usize,
357        index: &IndexArray,
358    ) -> RusTorchResult<Tensor<T>> {
359        let input_shape = input.shape();
360
361        if dim >= input_shape.len() {
362            return Err(RusTorchError::invalid_dimension(
363                dim,
364                input_shape.len().saturating_sub(1),
365            ));
366        }
367
368        let input_data = input
369            .data
370            .as_slice()
371            .ok_or_else(|| RusTorchError::tensor_op("Input tensor data not accessible"))?;
372        let index_data = index
373            .as_slice()
374            .ok_or_else(|| RusTorchError::tensor_op("Index array data not accessible"))?;
375
376        let index_shape = index.shape();
377        let mut result_data = Vec::with_capacity(index_data.len());
378
379        // Pre-calculate strides for better performance
380        let input_strides = calculate_strides(input_shape);
381        let index_strides = calculate_strides(index_shape);
382
383        // Batch bounds checking for better performance
384        let dim_size = input_shape[dim];
385        for &idx in index_data.iter() {
386            if idx < 0 || idx as usize >= dim_size {
387                return Err(RusTorchError::tensor_op(format!(
388                    "Index {} out of bounds for dimension {} with size {}",
389                    idx, dim, dim_size
390                )));
391            }
392        }
393
394        // Optimized gathering with better memory access patterns
395        for flat_idx in 0..index_data.len() {
396            let index_coords = flat_to_coords(flat_idx, &index_strides);
397            let gather_idx = index_data[flat_idx] as usize;
398
399            let mut input_coords = index_coords;
400            // Ensure we have enough dimensions
401            input_coords.resize(input_shape.len(), 0);
402            if dim < input_coords.len() {
403                input_coords[dim] = gather_idx;
404            }
405
406            let input_flat_idx = coords_to_flat(&input_coords, &input_strides);
407
408            if input_flat_idx < input_data.len() {
409                result_data.push(input_data[input_flat_idx]);
410            } else {
411                return Err(RusTorchError::tensor_op(
412                    "gather: Calculated index exceeds tensor bounds",
413                ));
414            }
415        }
416
417        Ok(Tensor::from_vec(result_data, index_shape.to_vec()))
418    }

614    /// Find top-k largest elements along specified dimension (optimized version)

617    /// # Arguments

621    /// * `largest` - If true, return largest elements; if false, return smallest

623    ///

626    ///

630        input: &Tensor<T>,
631        k: usize,
632        dim: usize,
633        largest: bool,
634        sorted: bool,

655            return Ok((
656                Tensor::from_vec(Vec::new(), output_shape.clone()),
657                ArrayD::from_shape_vec(output_shape, Vec::new()).map_err(|_| {
658                    RusTorchError::tensor_op("Failed to create empty indices array")
659                })?,
660            ));
661        }
662
663        let input_data = input
664            .data
665            .as_slice()
666            .ok_or_else(|| RusTorchError::tensor_op("Input tensor data not accessible"))?;
667
668        let outer_size: usize = input_shape[..dim].iter().product();
669        let dim_size = input_shape[dim];
670        let inner_size: usize = input_shape[dim + 1..].iter().product();
671
672        let mut output_shape = input_shape.to_vec();
673        output_shape[dim] = k;
674        let output_size: usize = output_shape.iter().product();
675
676        let mut values = Vec::with_capacity(output_size);
677        let mut indices = Vec::with_capacity(output_size);
678
679        // Optimized sorting with better memory access patterns
680        for outer_idx in 0..outer_size {
681            for inner_idx in 0..inner_size {
682                // Collect values and indices for this slice
683                let mut slice_data = Vec::with_capacity(dim_size);
684
685                for dim_idx in 0..dim_size {
686                    let flat_idx =
687                        outer_idx * dim_size * inner_size + dim_idx * inner_size + inner_idx;
688                    if flat_idx < input_data.len() {
689                        slice_data.push((input_data[flat_idx], dim_idx));
690                    }
691                }
692
693                // Sort slice data efficiently
694                if largest {
695                    slice_data.sort_by(|a, b| b.0.partial_cmp(&a.0).unwrap_or(Ordering::Equal));
696                } else {
697                    slice_data.sort_by(|a, b| a.0.partial_cmp(&b.0).unwrap_or(Ordering::Equal));
698                }
699
700                // Take top-k elements with bounds checking
701                let take_count = k.min(slice_data.len());
702                for i in 0..take_count {
703                    values.push(slice_data[i].0);
704                    indices.push(slice_data[i].1 as i64);
705                }
706
707                // Handle edge case: if not sorted and largest, reverse for PyTorch compatibility
708                if !sorted && largest && take_count > 0 {
709                    let start_idx = values.len() - take_count;
710                    values[start_idx..].reverse();
711                    indices[start_idx..].reverse();
712                }
713
714                // Pad if necessary (shouldn't happen with correct validation)
715                while values.len() % k != 0 && (outer_idx + 1) * (inner_idx + 1) * k <= output_size
716                {
717                    values.push(T::zero());
718                    indices.push(0);
719                }
720            }
721        }
722
723        Ok((
724            Tensor::from_vec(values, output_shape.clone()),
725            ArrayD::from_shape_vec(output_shape, indices)
726                .map_err(|_| RusTorchError::tensor_op("Failed to create indices array"))?,
727        ))
728    }

730    /// Find k-th smallest element along specified dimension (optimized version)
731    /// 指定次元に沿ってk番目に小さい要素を検索（最適化版）
732    ///
733    /// # Arguments
734    /// * `input` - Input tensor
735    /// * `k` - Position to find (0-indexed)
736    /// * `dim` - Dimension to search along
737    /// * `keepdim` - Whether to keep the dimension
738    ///
739    /// # Returns
740    /// Tuple of (values, indices) tensors
741    ///
742    /// # Performance
743    /// Optimized sorting and error handling
744    pub fn kthvalue<T: Float + 'static>(
745        input: &Tensor<T>,
746        k: usize,
747        dim: usize,
748        keepdim: bool,
749    ) -> RusTorchResult<(Tensor<T>, IndexArray)> {
750        let input_shape = input.shape();
751
752        if dim >= input_shape.len() {
753            return Err(RusTorchError::invalid_dimension(
754                dim,
755                input_shape.len().saturating_sub(1),
756            ));
757        }
758
759        if k >= input_shape[dim] {
760            return Err(RusTorchError::tensor_op(format!(
761                "k ({}) must be less than dimension size ({})",
762                k, input_shape[dim]
763            )));
764        }
765
766        let input_data = input
767            .data
768            .as_slice()
769            .ok_or_else(|| RusTorchError::tensor_op("Input tensor data not accessible"))?;
770
771        let outer_size: usize = input_shape[..dim].iter().product();
772        let dim_size = input_shape[dim];
773        let inner_size: usize = input_shape[dim + 1..].iter().product();
774
775        let mut output_shape = input_shape.to_vec();
776        if keepdim {
777            output_shape[dim] = 1;
778        } else {
779            output_shape.remove(dim);
780        }
781
782        let output_size: usize = output_shape.iter().product();
783        let mut values = Vec::with_capacity(output_size);
784        let mut indices = Vec::with_capacity(output_size);
785
786        for outer_idx in 0..outer_size {
787            for inner_idx in 0..inner_size {
788                // Collect values and indices for this slice
789                let mut slice_data = Vec::with_capacity(dim_size);
790
791                for dim_idx in 0..dim_size {
792                    let flat_idx =
793                        outer_idx * dim_size * inner_size + dim_idx * inner_size + inner_idx;
794                    if flat_idx < input_data.len() {
795                        slice_data.push((input_data[flat_idx], dim_idx));
796                    }
797                }
798
799                if slice_data.is_empty() {
800                    return Err(RusTorchError::tensor_op(
801                        "kthvalue: No elements found in slice",
802                    ));
803                }
804
805                // Sort to find k-th value
806                slice_data.sort_by(|a, b| a.0.partial_cmp(&b.0).unwrap_or(Ordering::Equal));
807
808                if k < slice_data.len() {
809                    values.push(slice_data[k].0);
810                    indices.push(slice_data[k].1 as i64);
811                } else {
812                    return Err(RusTorchError::tensor_op(
813                        "kthvalue: k index exceeds available elements",
814                    ));
815                }
816            }
817        }
818
819        Ok((
820            Tensor::from_vec(values, output_shape.clone()),
821            ArrayD::from_shape_vec(output_shape, indices)
822                .map_err(|_| RusTorchError::tensor_op("Failed to create indices array"))?,
823        ))
824    }

826    /// Compute quantiles along specified dimension
827    /// 指定次元に沿って分位数を計算
828    ///
829    /// # Arguments
830    /// * `input` - Input tensor
831    /// * `q` - Quantile values tensor (between 0 and 1)
832    /// * `dim` - Dimension to compute quantiles along
833    /// * `keepdim` - Whether to keep the dimension
834    ///
835    /// # Returns
836    /// Tensor with computed quantiles
837    pub fn quantile_util<T: Float + 'static + std::fmt::Display>(
838        input: &Tensor<T>,
839        q: &Tensor<T>,
840        dim: Option<usize>,
841        keepdim: bool,
842    ) -> RusTorchResult<Tensor<T>> {
843        let q_data = q
844            .data
845            .as_slice()
846            .ok_or_else(|| RusTorchError::tensor_op("Quantile tensor data not accessible"))?;
847
848        // Validate quantile values are in [0, 1]
849        for &q_val in q_data.iter() {
850            if q_val < T::zero() || q_val > T::one() {
851                return Err(RusTorchError::tensor_op(format!(
852                    "Quantile values must be in [0, 1], got {}",
853                    q_val
854                )));
855            }
856        }
857
858        let input_data = input
859            .data
860            .as_slice()
861            .ok_or_else(|| RusTorchError::tensor_op("Input tensor data not accessible"))?;
862
863        match dim {
864            Some(dim_val) => {
865                if dim_val >= input.shape().len() {
866                    return Err(RusTorchError::invalid_dimension(
867                        dim_val,
868                        input.shape().len().saturating_sub(1),
869                    ));
870                }
871
872                let input_shape = input.shape();
873                let outer_size: usize = input_shape[..dim_val].iter().product();
874                let dim_size = input_shape[dim_val];
875                let inner_size: usize = input_shape[dim_val + 1..].iter().product();
876
877                let mut output_shape = input_shape.to_vec();
878                if keepdim {
879                    output_shape[dim_val] = q_data.len();
880                } else {
881                    output_shape.remove(dim_val);
882                    output_shape.insert(dim_val, q_data.len());
883                }
884
885                let mut result_data = Vec::new();
886
887                for outer_idx in 0..outer_size {
888                    for inner_idx in 0..inner_size {
889                        // Collect values for this slice
890                        let mut slice_values = Vec::with_capacity(dim_size);
891
892                        for dim_idx in 0..dim_size {
893                            let flat_idx = outer_idx * dim_size * inner_size
894                                + dim_idx * inner_size
895                                + inner_idx;
896                            if flat_idx < input_data.len() {
897                                slice_values.push(input_data[flat_idx]);
898                            }
899                        }
900
901                        // Sort values for quantile calculation
902                        slice_values.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
903
904                        // Compute quantiles for this slice
905                        for &q_val in q_data.iter() {
906                            let quantile_val = compute_quantile(&slice_values, q_val);
907                            result_data.push(quantile_val);
908                        }
909                    }
910                }
911
912                Ok(Tensor::from_vec(result_data, output_shape))
913            }
914            None => {
915                // Flatten and compute quantiles over entire tensor
916                let mut sorted_data = input_data.to_vec();
917                sorted_data.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
918
919                let mut result_data = Vec::with_capacity(q_data.len());
920                for &q_val in q_data.iter() {
921                    let quantile_val = compute_quantile(&sorted_data, q_val);
922                    result_data.push(quantile_val);
923                }
924
925                Ok(Tensor::from_vec(result_data, vec![q_data.len()]))
926            }
927        }
928    }
929
930    /// Helper function to compute quantile from sorted data
931    /// ソート済みデータから分位数を計算するヘルパー関数
932    fn compute_quantile<T: Float>(sorted_data: &[T], q: T) -> T {
933        if sorted_data.is_empty() {
934            return T::zero();
935        }
936
937        if sorted_data.len() == 1 {
938            return sorted_data[0];
939        }
940
941        let n = sorted_data.len();
942        let index = q * T::from(n - 1).unwrap();
943        let lower_idx = index.floor().to_usize().unwrap_or(0).min(n - 1);
944        let upper_idx = index.ceil().to_usize().unwrap_or(0).min(n - 1);
945
946        if lower_idx == upper_idx {
947            sorted_data[lower_idx]
948        } else {
949            let fraction = index - T::from(lower_idx).unwrap();
950            let lower_val = sorted_data[lower_idx];
951            let upper_val = sorted_data[upper_idx];
952            lower_val + fraction * (upper_val - lower_val)
953        }
954    }

962    /// Compute unique elements and their inverse indices
963    /// 一意要素とその逆インデックスを計算
964    pub fn unique<T: Float + 'static>(
965        input: &Tensor<T>,
966        sorted: bool,
967        return_inverse: bool,
968        return_counts: bool,
969        dim: Option<usize>,
970    ) -> RusTorchResult<(Tensor<T>, Option<ArrayD<i64>>, Option<ArrayD<i64>>)> {
971        let input_data = input
972            .data
973            .as_slice()
974            .ok_or_else(|| RusTorchError::invalid_parameter("Input tensor data not accessible"))?;
975
976        match dim {
977            None => {
978                // Flatten and find unique elements
979                let mut indexed_values: Vec<(T, usize)> = input_data
980                    .iter()
981                    .enumerate()
982                    .map(|(i, &val)| (val, i))
983                    .collect();
984
985                if sorted {
986                    indexed_values.sort_by(|a, b| a.0.partial_cmp(&b.0).unwrap_or(Ordering::Equal));
987                }
988
989                let mut unique_values = Vec::new();
990                let mut inverse_indices = vec![0i64; input_data.len()];
991                let mut counts = Vec::new();
992
993                if !indexed_values.is_empty() {
994                    let mut current_value = indexed_values[0].0;
995                    let mut current_count = 1;
996                    unique_values.push(current_value);
997
998                    let mut unique_idx = 0;
999                    inverse_indices[indexed_values[0].1] = unique_idx;
1000
1001                    for i in 1..indexed_values.len() {
1002                        let (value, original_idx) = indexed_values[i];
1003
1004                        if (value - current_value).abs() < T::from(1e-7).unwrap() {
1005                            current_count += 1;
1006                            inverse_indices[original_idx] = unique_idx;
1007                        } else {
1008                            counts.push(current_count as i64);
1009                            current_value = value;
1010                            current_count = 1;
1011                            unique_idx += 1;
1012                            unique_values.push(current_value);
1013                            inverse_indices[original_idx] = unique_idx;
1014                        }
1015                    }
1016                    counts.push(current_count as i64);
1017                }
1018
1019                let inverse_tensor = if return_inverse {
1020                    Some(
1021                        ArrayD::from_shape_vec(input.shape().to_vec(), inverse_indices).map_err(
1022                            |_| {
1023                                RusTorchError::invalid_parameter(
1024                                    "Invalid shape for inverse indices".to_string(),
1025                                )
1026                            },
1027                        )?,
1028                    )
1029                } else {
1030                    None
1031                };
1032
1033                let counts_tensor = if return_counts {
1034                    Some(
1035                        ArrayD::from_shape_vec(vec![unique_values.len()], counts).map_err(
1036                            |_| {
1037                                RusTorchError::invalid_parameter(
1038                                    "Invalid shape for counts".to_string(),
1039                                )
1040                            },
1041                        )?,
1042                    )
1043                } else {
1044                    None
1045                };
1046
1047                let unique_len = unique_values.len();
1048                Ok((
1049                    Tensor::from_vec(unique_values, vec![unique_len]),
1050                    inverse_tensor,
1051                    counts_tensor,
1052                ))
1053            }
1054            Some(_dim) => {
1055                // For now, return error for dimension-specific unique
1056                Err(RusTorchError::UnsupportedOperation(
1057                    "Unique along specific dimension not yet implemented",
1058                ))
1059            }
1060        }
1061    }

1063    /// Compute histogram of tensor values
1064    /// テンソル値のヒストグラムを計算
1065    pub fn histogram<T: Float + 'static>(
1066        input: &Tensor<T>,
1067        bins: usize,
1068        range: Option<(T, T)>,
1069        density: bool,
1070    ) -> RusTorchResult<(ArrayD<i64>, Tensor<T>)> {
1071        let input_data = input
1072            .data
1073            .as_slice()
1074            .ok_or_else(|| RusTorchError::invalid_parameter("Input tensor data not accessible"))?;
1075
1076        if input_data.is_empty() {
1077            return Err(RusTorchError::invalid_parameter(
1078                "Cannot compute histogram of empty tensor",
1079            ));
1080        }
1081
1082        if bins == 0 {
1083            return Err(RusTorchError::invalid_parameter(
1084                "Number of bins must be positive",
1085            ));
1086        }
1087
1088        // Determine range
1089        let (min_val, max_val) = match range {
1090            Some((min, max)) => {
1091                if min >= max {
1092                    return Err(RusTorchError::invalid_parameter(
1093                        "Range min must be less than max",
1094                    ));
1095                }
1096                (min, max)
1097            }
1098            None => {
1099                let min_val = input_data
1100                    .iter()
1101                    .min_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal))
1102                    .copied()
1103                    .unwrap();
1104                let max_val = input_data
1105                    .iter()
1106                    .max_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal))
1107                    .copied()
1108                    .unwrap();
1109                (min_val, max_val)
1110            }
1111        };
1112
1113        // Create bins
1114        let bin_width = (max_val - min_val) / T::from(bins).unwrap();
1115        let mut bin_edges = Vec::with_capacity(bins + 1);
1116        let mut bin_counts = vec![0i64; bins];
1117
1118        for i in 0..=bins {
1119            bin_edges.push(min_val + T::from(i).unwrap() * bin_width);
1120        }
1121
1122        // Count values in each bin
1123        for &value in input_data.iter() {
1124            if value >= min_val && value <= max_val {
1125                let bin_idx = if value == max_val {
1126                    bins - 1 // Handle edge case where value equals max
1127                } else {
1128                    let idx = ((value - min_val) / bin_width)
1129                        .floor()
1130                        .to_usize()
1131                        .unwrap_or(0);
1132                    idx.min(bins - 1)
1133                };
1134                bin_counts[bin_idx] += 1;
1135            }
1136        }
1137
1138        // Note: Density conversion would need separate tensor for float results
1139        // For now, return raw counts as i64 and let user convert if needed
1140        if density {
1141            return Err(RusTorchError::tensor_op(
1142                "histogram: Density mode not supported with integer counts, use raw counts and normalize separately"
1143            ));
1144        }
1145
1146        Ok((
1147            ArrayD::from_shape_vec(vec![bins], bin_counts).map_err(|_| {
1148                RusTorchError::invalid_parameter("Invalid shape for histogram counts".to_string())
1149            })?,
1150            Tensor::from_vec(bin_edges, vec![bins + 1]),
1151        ))
1152    }
1153}
