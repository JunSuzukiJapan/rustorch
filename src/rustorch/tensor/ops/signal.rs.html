21    /// Fast Fourier Transform (basic implementation)
22    /// 高速フーリエ変換（基本実装）
23    pub fn fft(
24        &self,
25        n: Option<usize>,
26        dim: Option<isize>,
27        norm: Option<&str>,
28    ) -> Result<(Self, Self), String> {
29        let actual_dim = self.resolve_dim(dim.unwrap_or(-1))?;
30        let input_len = self.shape()[actual_dim];
31        let fft_len = n.unwrap_or(input_len);
32
33        // For 1D case on the last dimension
34        if self.shape().len() == 1 && actual_dim == 0 {
35            return self.fft_1d_basic(fft_len, norm, false);
36        }
37
38        Err("Multi-dimensional FFT not implemented in this version".to_string())
39    }

41    /// Inverse Fast Fourier Transform (basic implementation)
42    /// 逆高速フーリエ変換（基本実装）
43    pub fn ifft(
44        &self,
45        real_part: &Self,
46        imag_part: &Self,
47        n: Option<usize>,
48        dim: Option<isize>,
49        norm: Option<&str>,
50    ) -> Result<(Self, Self), String> {
51        if real_part.shape() != imag_part.shape() {
52            return Err("Real and imaginary parts must have the same shape".to_string());
53        }
54
55        let actual_dim = self.resolve_dim(dim.unwrap_or(-1))?;
56        let input_len = real_part.shape()[actual_dim];
57        let fft_len = n.unwrap_or(input_len);
58
59        // For 1D case on the last dimension
60        if real_part.shape().len() == 1 && actual_dim == 0 {
61            return self.ifft_1d_basic(real_part, imag_part, fft_len, norm);
62        }
63
64        Err("Multi-dimensional IFFT not implemented in this version".to_string())
65    }

67    /// Real Fast Fourier Transform (basic implementation)
68    /// 実数高速フーリエ変換（基本実装）
69    pub fn rfft(
70        &self,
71        n: Option<usize>,
72        dim: Option<isize>,
73        norm: Option<&str>,
74    ) -> Result<(Self, Self), String> {
75        // RFFT returns only the first N/2+1 frequencies due to Hermitian symmetry
76        let (real_full, imag_full) = self.fft(n, dim, norm)?;
77        let full_len = real_full.shape()[0];
78        let rfft_len = full_len / 2 + 1;
79
80        // Extract first half + 1
81        let real_data = real_full.as_slice().unwrap();
82        let imag_data = imag_full.as_slice().unwrap();
83
84        let real_rfft: Vec<T> = real_data[0..rfft_len].to_vec();
85        let imag_rfft: Vec<T> = imag_data[0..rfft_len].to_vec();
86
87        Ok((
88            Tensor::from_vec(real_rfft, vec![rfft_len]),
89            Tensor::from_vec(imag_rfft, vec![rfft_len]),
90        ))
91    }

93    /// Shift zero-frequency component to center of spectrum
94    /// ゼロ周波数成分をスペクトラムの中央にシフト
95    pub fn fftshift(&self, dim: Option<&[isize]>) -> Result<Self, String> {
96        let shape = self.shape();
97        let ndim = shape.len() as isize;
98
99        // Determine which dimensions to shift
100        let dims_to_shift: Vec<usize> = if let Some(dims) = dim {
101            dims.iter()
102                .map(|&d| {
103                    let adjusted = if d < 0 {
104                        (ndim + d) as usize
105                    } else {
106                        d as usize
107                    };
108                    if adjusted >= shape.len() {
109                        return Err(format!("Dimension {} is out of bounds", d));
110                    }
111                    Ok(adjusted)
112                })
113                .collect::<Result<Vec<_>, _>>()?
114        } else {
115            // If no dimensions specified, shift all dimensions
116            (0..shape.len()).collect()
117        };
118
119        let mut result = self.clone();
120
121        // Apply shift to each specified dimension
122        for &dim_idx in &dims_to_shift {
123            let dim_size = shape[dim_idx];
124            let shift_amount = dim_size / 2;
125
126            if shift_amount == 0 {
127                continue; // No shift needed for size 1 dimensions
128            }
129
130            // For simplicity, this is a basic implementation
131            // A more efficient implementation would use ndarray's axis operations
132            // 簡単のため、これは基本的な実装です
133            // より効率的な実装では、ndarrayの軸操作を使用します
134            result = Self::_shift_along_axis(&result, dim_idx, shift_amount)?;
135        }
136
137        Ok(result)
138    }

140    /// Inverse of fftshift
141    /// fftshiftの逆
142    pub fn ifftshift(&self, dim: Option<&[isize]>) -> Result<Self, String> {
143        let shape = self.shape();
144        let ndim = shape.len() as isize;
145
146        let dims_to_shift: Vec<usize> = if let Some(dims) = dim {
147            dims.iter()
148                .map(|&d| {
149                    let adjusted = if d < 0 {
150                        (ndim + d) as usize
151                    } else {
152                        d as usize
153                    };
154                    if adjusted >= shape.len() {
155                        return Err(format!("Dimension {} is out of bounds", d));
156                    }
157                    Ok(adjusted)
158                })
159                .collect::<Result<Vec<_>, _>>()?
160        } else {
161            (0..shape.len()).collect()
162        };
163
164        let mut result = self.clone();
165
166        for &dim_idx in &dims_to_shift {
167            let dim_size = shape[dim_idx];
168            let shift_amount = (dim_size + 1) / 2; // Ceiling division for ifftshift
169
170            if shift_amount == 0 || shift_amount == dim_size {
171                continue;
172            }
173
174            result = Self::_shift_along_axis(&result, dim_idx, shift_amount)?;
175        }
176
177        Ok(result)
178    }

180    /// Helper function to shift data along a specific axis
181    /// 特定の軸に沿ってデータをシフトするヘルパー関数
182    fn _shift_along_axis(tensor: &Self, axis: usize, shift: usize) -> Result<Self, String> {
183        let shape = tensor.shape();
184        let axis_size = shape[axis];
185
186        if shift == 0 || shift >= axis_size {
187            return Ok(tensor.clone());
188        }
189
190        // For simplicity, we'll flatten, rearrange, and reshape
191        // This is not the most efficient but works for demonstration
192        // 簡単のため、平坦化、再配置、再形状化を行います
193        // これは最も効率的ではありませんが、デモンストレーション用には動作します
194        let data = tensor.as_slice().unwrap();
195        let mut result_data = Vec::with_capacity(data.len());
196
197        // Calculate strides for the given axis
198        let mut strides = vec![1; shape.len()];
199        for i in (0..shape.len() - 1).rev() {
200            strides[i] = strides[i + 1] * shape[i + 1];
201        }
202
203        let axis_stride = strides[axis];
204        let outer_size = if axis == 0 {
205            1
206        } else {
207            shape[..axis].iter().product()
208        };
209        let inner_size = if axis == shape.len() - 1 {
210            1
211        } else {
212            shape[axis + 1..].iter().product()
213        };
214
215        for outer in 0..outer_size {
216            for inner in 0..inner_size {
217                // Shift the data along the specified axis
218                for i in 0..axis_size {
219                    let src_idx = (i + shift) % axis_size;
220                    let linear_idx =
221                        outer * (axis_size * inner_size) + src_idx * inner_size + inner;
222                    result_data.push(data[linear_idx]);
223                }
224            }
225        }
226
227        Ok(Tensor::from_vec(result_data, shape.to_vec()))
228    }

230    /// Apply window function to the tensor
231    /// テンソルに窓関数を適用
232    pub fn apply_window(
233        &self,
234        window_type: WindowType,
235        axis: Option<usize>,
236    ) -> Result<Self, String> {
237        let target_axis = axis.unwrap_or(self.shape().len() - 1);
238
239        if target_axis >= self.shape().len() {
240            return Err(format!("Axis {} is out of bounds", target_axis));
241        }
242
243        let window_size = self.shape()[target_axis];
244        let window = Self::create_window(window_type, window_size)?;
245
246        // Apply window along the specified axis
247        // 指定軸に沿って窓関数を適用
248        self.apply_window_along_axis(&window, target_axis)
249    }

251    /// Create a window function
252    /// 窓関数を作成
253    fn create_window(window_type: WindowType, size: usize) -> Result<Self, String> {
254        let mut window_data = Vec::with_capacity(size);
255
256        match window_type {
257            WindowType::Hann => {
258                for i in 0..size {
259                    let val = T::from(0.5).unwrap()
260                        * (T::one()
261                            - T::cos(
262                                T::from(2.0 * std::f64::consts::PI * i as f64 / (size - 1) as f64)
263                                    .unwrap(),
264                            ));
265                    window_data.push(val);
266                }
267            }
268            WindowType::Hamming => {
269                for i in 0..size {
270                    let val = T::from(0.54).unwrap()
271                        - T::from(0.46).unwrap()
272                            * T::cos(
273                                T::from(2.0 * std::f64::consts::PI * i as f64 / (size - 1) as f64)
274                                    .unwrap(),
275                            );
276                    window_data.push(val);
277                }
278            }
279            WindowType::Blackman => {
280                for i in 0..size {
281                    let cos1 = T::cos(
282                        T::from(2.0 * std::f64::consts::PI * i as f64 / (size - 1) as f64).unwrap(),
283                    );
284                    let cos2 = T::cos(
285                        T::from(4.0 * std::f64::consts::PI * i as f64 / (size - 1) as f64).unwrap(),
286                    );
287                    let val = T::from(0.42).unwrap() - T::from(0.5).unwrap() * cos1
288                        + T::from(0.08).unwrap() * cos2;
289                    window_data.push(val);
290                }
291            }
292            WindowType::Rectangular => {
293                for _ in 0..size {
294                    window_data.push(T::one());
295                }
296            }
297        }
298
299        Ok(Tensor::from_vec(window_data, vec![size]))
300    }

302    /// Apply window along specified axis
303    /// 指定軸に沿って窓関数を適用
304    fn apply_window_along_axis(&self, window: &Self, axis: usize) -> Result<Self, String> {
305        let shape = self.shape();
306        let window_size = window.numel();
307
308        if shape[axis] != window_size {
309            return Err(format!(
310                "Window size {} doesn't match axis size {}",
311                window_size, shape[axis]
312            ));
313        }
314
315        let data = self.as_slice().unwrap();
316        let window_data = window.as_slice().unwrap();
317        let mut result_data = Vec::with_capacity(data.len());
318
319        // Calculate strides
320        let mut strides = vec![1; shape.len()];
321        for i in (0..shape.len() - 1).rev() {
322            strides[i] = strides[i + 1] * shape[i + 1];
323        }
324
325        let axis_stride = strides[axis];
326        let outer_size = if axis == 0 {
327            1
328        } else {
329            shape[..axis].iter().product()
330        };
331        let inner_size = if axis == shape.len() - 1 {
332            1
333        } else {
334            shape[axis + 1..].iter().product()
335        };
336
337        for outer in 0..outer_size {
338            for i in 0..shape[axis] {
339                for inner in 0..inner_size {
340                    let linear_idx = outer * (shape[axis] * inner_size) + i * inner_size + inner;
341                    result_data.push(data[linear_idx] * window_data[i]);
342                }
343            }
344        }
345
346        Ok(Tensor::from_vec(result_data, shape.to_vec()))
347    }

349    /// Basic 1D FFT implementation (simple DFT for now)
350    /// 基本的な1D FFT実装（今のところはシンプルなDFT）
351    fn fft_1d_basic(
352        &self,
353        n: usize,
354        norm: Option<&str>,
355        inverse: bool,
356    ) -> Result<(Self, Self), String> {
357        let input_data = self.as_slice().unwrap();
358        let input_len = input_data.len();
359
360        // Pad or truncate to desired length
361        let real_input: Vec<T> = if n != input_len {
362            if n > input_len {
363                // Zero-pad
364                let mut padded = input_data.to_vec();
365                padded.resize(n, T::zero());
366                padded
367            } else {
368                // Truncate
369                input_data[0..n].to_vec()
370            }
371        } else {
372            input_data.to_vec()
373        };
374
375        // Convert to complex representation
376        let complex_data: Vec<Complex<T>> = real_input
377            .iter()
378            .map(|&x| Complex::new(x, T::zero()))
379            .collect();
380
381        // Simple DFT implementation (O(n²))
382        let mut result = Vec::with_capacity(n);
383        for k in 0..n {
384            let mut sum = Complex::new(T::zero(), T::zero());
385            for (j, &x) in complex_data.iter().enumerate() {
386                let angle =
387                    T::from(-2.0 * std::f64::consts::PI * j as f64 * k as f64 / n as f64).unwrap();
388                let angle = if inverse { -angle } else { angle };
389                let twiddle = Complex::new(angle.cos(), angle.sin());
390                sum = sum + x * twiddle;
391            }
392            if inverse {
393                sum = Complex::new(sum.re / T::from(n).unwrap(), sum.im / T::from(n).unwrap());
394            }
395            result.push(sum);
396        }
397
398        // Apply normalization based on norm parameter
399        let normalized_result = match norm {
400            Some("ortho") => {
401                let scale = T::from(1.0 / (n as f64).sqrt()).unwrap();
402                result
403                    .iter()
404                    .map(|c| Complex::new(c.re * scale, c.im * scale))
405                    .collect()
406            }
407            Some("forward") => {
408                if !inverse {
409                    let scale = T::from(1.0 / n as f64).unwrap();
410                    result
411                        .iter()
412                        .map(|c| Complex::new(c.re * scale, c.im * scale))
413                        .collect()
414                } else {
415                    result
416                }
417            }
418            _ => result, // "backward" is default
419        };
420
421        // Extract real and imaginary parts
422        let real_part: Vec<T> = normalized_result.iter().map(|c| c.re).collect();
423        let imag_part: Vec<T> = normalized_result.iter().map(|c| c.im).collect();
424
425        Ok((
426            Tensor::from_vec(real_part, vec![n]),
427            Tensor::from_vec(imag_part, vec![n]),
428        ))
429    }

431    /// Basic 1D IFFT implementation
432    /// 基本的な1D IFFT実装
433    fn ifft_1d_basic(
434        &self,
435        real_part: &Self,
436        imag_part: &Self,
437        n: usize,
438        norm: Option<&str>,
439    ) -> Result<(Self, Self), String> {
440        // Reconstruct complex data from separate real and imaginary tensors
441        let real_data = real_part.as_slice().unwrap();
442        let imag_data = imag_part.as_slice().unwrap();
443
444        // Use the real part tensor to create a temporary combined tensor for IFFT
445        // Since we need to call fft_1d_basic, we need to create a combined representation
446        // For simplicity, we'll perform IFFT using the same DFT approach but with inverse=true
447
448        let input_len = real_data.len();
449        let complex_input: Vec<T> = real_data.to_vec();
450        let temp_tensor = Tensor::from_vec(complex_input, vec![input_len]);
451
452        // Call fft_1d_basic with inverse=true
453        temp_tensor.fft_1d_basic(n, norm, true)
454    }
