79    /// Apply two-argument arctangent element-wise
80    /// 要素ごとに二引数アークタンジェントを適用
81    pub fn atan2(&self, other: &Tensor<T>) -> RusTorchResult<Self> {
82        if self.shape() != other.shape() && !self.can_broadcast_with(other) {
83            return Err(RusTorchError::shape_mismatch(self.shape(), other.shape()));
84        }
85
86        if self.shape() == other.shape() {
87            let result_data: Vec<T> = self
88                .data
89                .iter()
90                .zip(other.data.iter())
91                .map(|(&y, &x)| y.atan2(x))
92                .collect();
93            Ok(Tensor::from_vec(result_data, self.shape().to_vec()))
94        } else {
95            let (broadcasted_self, broadcasted_other) = self.broadcast_with(other)?;
96            let result_data: Vec<T> = broadcasted_self
97                .data
98                .iter()
99                .zip(broadcasted_other.data.iter())
100                .map(|(&y, &x)| y.atan2(x))
101                .collect();
102            Ok(Tensor::from_vec(
103                result_data,
104                broadcasted_self.shape().to_vec(),
105            ))
106        }
107    }

146    /// Raise elements to powers from another tensor
147    /// 他のテンソルの要素をべき指数として使用
148    pub fn pow_tensor(&self, exponent: &Tensor<T>) -> RusTorchResult<Self> {
149        if self.shape() != exponent.shape() && !self.can_broadcast_with(exponent) {
150            return Err(RusTorchError::shape_mismatch(
151                self.shape(),
152                exponent.shape(),
153            ));
154        }
155
156        if self.shape() == exponent.shape() {
157            let result_data: Vec<T> = self
158                .data
159                .iter()
160                .zip(exponent.data.iter())
161                .map(|(&base, &exp)| base.powf(exp))
162                .collect();
163            Ok(Tensor::from_vec(result_data, self.shape().to_vec()))
164        } else {
165            let (broadcasted_self, broadcasted_exp) = self.broadcast_with(exponent)?;
166            let result_data: Vec<T> = broadcasted_self
167                .data
168                .iter()
169                .zip(broadcasted_exp.data.iter())
170                .map(|(&base, &exp)| base.powf(exp))
171                .collect();
172            Ok(Tensor::from_vec(
173                result_data,
174                broadcasted_self.shape().to_vec(),
175            ))
176        }
177    }

220    /// Clamp values between min and max
221    /// 値をminとmaxの間にクランプ
222    pub fn clamp(&self, min_val: T, max_val: T) -> Self {
223        let result_data: Vec<T> = self
224            .data
225            .iter()
226            .map(|&x| {
227                if x < min_val {
228                    min_val
229                } else if x > max_val {
230                    max_val
231                } else {
232                    x
233                }
234            })
235            .collect();
236        Tensor::from_vec(result_data, self.shape().to_vec())
237    }

239    /// Sign function: returns -1, 0, or 1
240    /// 符号関数: -1, 0, 1を返す
241    pub fn sign(&self) -> Self {
242        let result_data: Vec<T> = self
243            .data
244            .iter()
245            .map(|&x| {
246                if x > T::zero() {
247                    T::one()
248                } else if x < T::zero() {
249                    -T::one()
250                } else {
251                    T::zero()
252                }
253            })
254            .collect();
255        Tensor::from_vec(result_data, self.shape().to_vec())
256    }

258    /// Linear interpolation between two tensors
259    /// 2つのテンソル間の線形補間
260    pub fn lerp(&self, other: &Tensor<T>, weight: T) -> RusTorchResult<Self> {
261        if self.shape() != other.shape() && !self.can_broadcast_with(other) {
262            return Err(RusTorchError::shape_mismatch(self.shape(), other.shape()));
263        }
264
265        if self.shape() == other.shape() {
266            let result_data: Vec<T> = self
267                .data
268                .iter()
269                .zip(other.data.iter())
270                .map(|(&a, &b)| a + weight * (b - a))
271                .collect();
272            Ok(Tensor::from_vec(result_data, self.shape().to_vec()))
273        } else {
274            let (broadcasted_self, broadcasted_other) = self.broadcast_with(other)?;
275            let result_data: Vec<T> = broadcasted_self
276                .data
277                .iter()
278                .zip(broadcasted_other.data.iter())
279                .map(|(&a, &b)| a + weight * (b - a))
280                .collect();
281            Ok(Tensor::from_vec(
282                result_data,
283                broadcasted_self.shape().to_vec(),
284            ))
285        }
286    }

338    #[test]
339    fn test_hyperbolic_functions() {
340        let tensor = Tensor::from_vec(vec![0.0, 1.0, -1.0], vec![3]);
341
342        // Test sinh, cosh, tanh
343        let sinh_result = tensor.sinh();
344        let cosh_result = tensor.cosh();
345        let tanh_result = tensor.tanh();
346
347        assert_eq!(sinh_result.as_slice().unwrap()[0], 0.0);
348        assert_eq!(cosh_result.as_slice().unwrap()[0], 1.0);
349        assert_eq!(tanh_result.as_slice().unwrap()[0], 0.0);
350
351        // Test that sinh(-x) = -sinh(x)
352        assert!(
353            (sinh_result.as_slice().unwrap()[1] + sinh_result.as_slice().unwrap()[2]).abs() < 1e-10
354        );
355    }
