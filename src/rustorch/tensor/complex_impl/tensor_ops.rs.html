9/// Complex tensor creation and manipulation functions
10impl<T: Float + 'static> Complex<T> {
11    /// Create a complex tensor from real and imaginary parts
12    /// 実部と虚部からの複素テンソルを作成
13    pub fn from_tensors(real: &Tensor<T>, imag: &Tensor<T>) -> Result<Tensor<Complex<T>>, String> {
14        if real.shape() != imag.shape() {
15            return Err("Real and imaginary tensors must have the same shape".to_string());
16        }
17
18        let mut complex_data = Vec::with_capacity(real.data.len());
19        for (r, i) in real.data.iter().zip(imag.data.iter()) {
20            complex_data.push(Complex::new(*r, *i));
21        }
22
23        Ok(Tensor::from_vec(complex_data, real.shape().to_vec()))
24    }
25
26    /// Extract real parts from complex tensor
27    /// 複素テンソルから実部を抽出
28    pub fn tensor_real_part(tensor: &Tensor<Complex<T>>) -> Tensor<T> {
29        let real_data: Vec<T> = tensor.data.iter().map(|z| z.real()).collect();
30        Tensor::from_vec(real_data, tensor.shape().to_vec())
31    }
32
33    /// Extract imaginary parts from complex tensor
34    /// 複素テンソルから虚部を抽出
35    pub fn tensor_imag_part(tensor: &Tensor<Complex<T>>) -> Tensor<T> {
36        let imag_data: Vec<T> = tensor.data.iter().map(|z| z.imag()).collect();
37        Tensor::from_vec(imag_data, tensor.shape().to_vec())
38    }
39
40    /// Extract magnitude from complex tensor
41    /// 複素テンソルから絶対値を抽出
42    pub fn tensor_abs(tensor: &Tensor<Complex<T>>) -> Tensor<T> {
43        let abs_data: Vec<T> = tensor.data.iter().map(|z| z.abs()).collect();
44        Tensor::from_vec(abs_data, tensor.shape().to_vec())
45    }
46
47    /// Extract phase from complex tensor
48    /// 複素テンソルから位相角を抽出
49    pub fn tensor_arg(tensor: &Tensor<Complex<T>>) -> Tensor<T> {
50        let arg_data: Vec<T> = tensor.data.iter().map(|z| z.arg()).collect();
51        Tensor::from_vec(arg_data, tensor.shape().to_vec())
52    }
53
54    /// Complex conjugate of tensor
55    /// テンソルの複素共役
56    pub fn tensor_conj(tensor: &Tensor<Complex<T>>) -> Tensor<Complex<T>> {
57        let conj_data: Vec<Complex<T>> = tensor.data.iter().map(|z| z.conj()).collect();
58        Tensor::from_vec(conj_data, tensor.shape().to_vec())
59    }

62/// Complex tensor factory functions
63impl<T: Float + 'static> Tensor<Complex<T>> {
64    /// Create a complex tensor with all zeros
65    /// すべてゼロの複素テンソルを作成
66    pub fn complex_zeros(shape: &[usize]) -> Self {
67        let total_size = shape.iter().product();
68        let data = vec![Complex::zero(); total_size];
69        Tensor::from_vec(data, shape.to_vec())
70    }
71
72    /// Create a complex tensor with all ones
73    /// すべて1の複素テンソルを作成
74    pub fn complex_ones(shape: &[usize]) -> Self {
75        let total_size = shape.iter().product();
76        let data = vec![Complex::one(); total_size];
77        Tensor::from_vec(data, shape.to_vec())
78    }
79
80    /// Create complex tensor filled with imaginary unit
81    /// 虚数単位で満たされた複素テンソルを作成
82    pub fn complex_i(shape: &[usize]) -> Self {
83        let total_size = shape.iter().product();
84        let data = vec![Complex::i(); total_size];
85        Tensor::from_vec(data, shape.to_vec())
86    }
87
88    /// Create complex tensor from polar coordinates
89    /// 極座標から複素テンソルを作成
90    pub fn from_polar(magnitude: &Tensor<T>, phase: &Tensor<T>) -> Result<Self, String> {
91        if magnitude.shape() != phase.shape() {
92            return Err("Magnitude and phase tensors must have the same shape".to_string());
93        }
94
95        let mut complex_data = Vec::with_capacity(magnitude.data.len());
96        for (mag, ph) in magnitude.data.iter().zip(phase.data.iter()) {
97            complex_data.push(Complex::from_polar(*mag, *ph));
98        }
99
100        Ok(Tensor::from_vec(complex_data, magnitude.shape().to_vec()))
101    }
102
103    /// Create complex tensor from real tensor (imaginary part = 0)
104    /// 実テンソルから複素テンソルを作成（虚部は0）
105    pub fn from_real(real: &Tensor<T>) -> Self {
106        let complex_data: Vec<Complex<T>> =
107            real.data.iter().map(|&r| Complex::from_real(r)).collect();
108        Tensor::from_vec(complex_data, real.shape().to_vec())
109    }
110
111    /// Create complex tensor from imaginary tensor (real part = 0)
112    /// 虚テンソルから複素テンソルを作成（実部は0）
113    pub fn from_imag(imag: &Tensor<T>) -> Self {
114        let complex_data: Vec<Complex<T>> =
115            imag.data.iter().map(|&i| Complex::from_imag(i)).collect();
116        Tensor::from_vec(complex_data, imag.shape().to_vec())
117    }
118}

120/// Complex mathematical operations for tensors
121impl<T: Float + 'static> Tensor<Complex<T>> {
122    /// Element-wise exponential e^z for complex tensor
123    /// 複素テンソルの要素ごとの指数関数 e^z
124    pub fn exp(&self) -> Self {
125        let exp_data: Vec<Complex<T>> = self.data.iter().map(|z| z.exp()).collect();
126        Tensor::from_vec(exp_data, self.shape().to_vec())
127    }

129    /// Element-wise natural logarithm ln(z) for complex tensor
130    /// 複素テンソルの要素ごとの自然対数 ln(z)
131    pub fn ln(&self) -> Self {
132        let ln_data: Vec<Complex<T>> = self.data.iter().map(|z| z.ln()).collect();
133        Tensor::from_vec(ln_data, self.shape().to_vec())
134    }
135
136    /// Element-wise power z^w for complex tensor
137    /// 複素テンソルの要素ごとのべき乗 z^w
138    pub fn pow(&self, exponent: &Self) -> Result<Self, String> {
139        if self.shape() != exponent.shape() {
140            return Err("Shape mismatch for power operation".to_string());
141        }
142
143        let pow_data: Vec<Complex<T>> = self
144            .data
145            .iter()
146            .zip(exponent.data.iter())
147            .map(|(z, exp)| z.pow(*exp))
148            .collect();
149
150        Ok(Tensor::from_vec(pow_data, self.shape().to_vec()))
151    }
152
153    /// Element-wise power with scalar exponent z^s
