493    fn analyze_operation_trend(&self, operation_name: &str) -> RusTorchResult<PerformanceTrend> {
494        let measurements = self
495            .performance_history
496            .get(operation_name)
497            .ok_or_else(|| RusTorchError::Profiling {
498                message: format!("No measurements for operation: {}", operation_name),
499            })?;
500
501        if measurements.len() < self.config.min_measurements {
502            return Ok(PerformanceTrend {
503                operation_name: operation_name.to_string(),
504                direction: TrendDirection::Unknown,
505                change_rate_percent: 0.0,
506                confidence: 0.0,
507                measurements: measurements.clone(),
508                projection: None,
509            });
510        }
511
512        // Calculate trend using linear regression
513        let (slope, confidence) = self.calculate_linear_trend(measurements)?;
514
515        let direction = if slope.abs() < self.config.significance_threshold {
516            TrendDirection::Stable
517        } else if slope < 0.0 {
518            TrendDirection::Improving // Negative slope = decreasing time = better performance
519        } else {
520            TrendDirection::Degrading
521        };
522
523        let change_rate_percent = slope * 100.0;
524
525        // Generate projection if trend is significant
526        let projection = if confidence > 0.5 {
527            Some(self.generate_projection(measurements, slope, confidence)?)
528        } else {
529            None
530        };
531
532        Ok(PerformanceTrend {
533            operation_name: operation_name.to_string(),
534            direction,
535            change_rate_percent,
536            confidence,
537            measurements: measurements.clone(),
538            projection,
539        })
540    }

542    fn calculate_linear_trend(
543        &self,
544        measurements: &[TrendMeasurement],
545    ) -> RusTorchResult<(f64, f64)> {
546        if measurements.len() < 2 {
547            return Ok((0.0, 0.0));
548        }
549
550        // Convert timestamps to seconds from first measurement
551        let first_time = measurements[0].timestamp;
552        let x_values: Vec<f64> = measurements
553            .iter()
554            .map(|m| m.timestamp.duration_since(first_time).as_secs_f64())
555            .collect();
556
557        let y_values: Vec<f64> = measurements.iter().map(|m| m.value).collect();
558
559        // Calculate linear regression
560        let n = measurements.len() as f64;
561        let sum_x: f64 = x_values.iter().sum();
562        let sum_y: f64 = y_values.iter().sum();
563        let sum_xy: f64 = x_values.iter().zip(&y_values).map(|(x, y)| x * y).sum();
564        let sum_x2: f64 = x_values.iter().map(|x| x * x).sum();
565
566        let slope = (n * sum_xy - sum_x * sum_y) / (n * sum_x2 - sum_x * sum_x);
567
568        // Calculate R-squared for confidence
569        let mean_y = sum_y / n;
570        let ss_tot: f64 = y_values.iter().map(|y| (y - mean_y).powi(2)).sum();
571        let intercept = (sum_y - slope * sum_x) / n;
572        let ss_res: f64 = x_values
573            .iter()
574            .zip(&y_values)
575            .map(|(x, y)| (y - (slope * x + intercept)).powi(2))
576            .sum();
577
578        let r_squared = if ss_tot > 0.0 {
579            1.0 - (ss_res / ss_tot)
580        } else {
581            0.0
582        };
583        let confidence = r_squared.max(0.0).min(1.0);
584
585        Ok((slope, confidence))
586    }

588    fn generate_projection(
589        &self,
590        measurements: &[TrendMeasurement],
591        slope: f64,
592        confidence: f64,
593    ) -> RusTorchResult<PerformanceProjection> {
594        let last_measurement = &measurements[measurements.len() - 1];
595        let time_horizon = Duration::from_secs(3600); // 1 hour projection
596
597        let next_value = last_measurement.value + slope * time_horizon.as_secs_f64();
598
599        // Calculate uncertainty range based on confidence
600        let uncertainty_factor = (1.0 - confidence) * 0.2; // Max 20% uncertainty
601        let range_width = next_value * uncertainty_factor;
602
603        Ok(PerformanceProjection {
604            next_value,
605            confidence,
606            time_horizon,
607            value_range: (next_value - range_width, next_value + range_width),
608        })
609    }

611    fn generate_recommendations(
612        &mut self,
613        trends: &HashMap<String, PerformanceTrend>,
614    ) -> RusTorchResult<()> {
615        self.recommendations.clear();
616
617        for (operation_name, trend) in trends {
618            match trend.direction {
619                TrendDirection::Degrading
620                    if trend.confidence > self.config.min_recommendation_confidence =>
621                {
622                    let recommendation =
623                        self.create_degradation_recommendation(operation_name, trend)?;
624                    self.recommendations.push(recommendation);
625                }
626                TrendDirection::Stable => {
627                    // Look for optimization opportunities even in stable performance
628                    if let Some(recommendation) =
629                        self.create_optimization_recommendation(operation_name, trend)?
630                    {
631                        self.recommendations.push(recommendation);
632                    }
633                }
634                _ => {}
635            }
636        }
637
638        // Sort recommendations by priority
639        self.recommendations
640            .sort_by(|a, b| a.priority.cmp(&b.priority));
641
642        Ok(())
643    }
