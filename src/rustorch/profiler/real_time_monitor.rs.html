9/// Monitoring configuration
10/// 監視設定
11#[derive(Debug, Clone)]
12pub struct MonitorConfig {
13    /// Sampling interval
14    /// サンプリング間隔
15    pub sampling_interval: Duration,
16    /// Alert thresholds
17    /// アラート閾値
18    pub alert_thresholds: AlertThresholds,
19    /// Enable system monitoring
20    /// システム監視を有効化
21    pub enable_system_monitoring: bool,
22}
23
24impl Default for MonitorConfig {
25    fn default() -> Self {
26        Self {
27            sampling_interval: Duration::from_millis(100),
28            alert_thresholds: AlertThresholds::default(),
29            enable_system_monitoring: true,
30        }
31    }
32}

101/// Real-time monitor
102/// リアルタイム監視
103pub struct RealTimeMonitor {
104    /// Configuration
105    /// 設定
106    config: MonitorConfig,
107    /// Running state
108    /// 実行状態
109    is_running: Arc<Mutex<bool>>,
110    /// Collected alerts
111    /// 収集されたアラート
112    alerts: Arc<Mutex<Vec<SystemAlert>>>,
113}
114
115impl RealTimeMonitor {
116    /// Create new real-time monitor
117    /// 新しいリアルタイム監視を作成
118    pub fn new(config: MonitorConfig) -> Self {
119        Self {
120            config,
121            is_running: Arc::new(Mutex::new(false)),
122            alerts: Arc::new(Mutex::new(Vec::new())),
123        }
124    }
125
126    /// Start monitoring
127    /// 監視開始
128    pub fn start(&self) -> RusTorchResult<()> {
129        let mut running = self
130            .is_running
131            .lock()
132            .map_err(|_| RusTorchError::Profiling {
133                message: "Failed to acquire running lock".to_string(),
134            })?;
135
136        if *running {
137            return Err(RusTorchError::Profiling {
138                message: "Monitor already running".to_string(),
139            });
140        }
141
142        *running = true;
143        println!("📡 Real-time monitoring started");
144        Ok(())
145    }
146
147    /// Stop monitoring
148    /// 監視停止
149    pub fn stop(&self) -> RusTorchResult<()> {
150        let mut running = self
151            .is_running
152            .lock()
153            .map_err(|_| RusTorchError::Profiling {
154                message: "Failed to acquire running lock".to_string(),
155            })?;
156
157        *running = false;
158        println!("📡 Real-time monitoring stopped");
159        Ok(())
160    }
161
162    /// Get current alerts
163    /// 現在のアラートを取得
164    pub fn get_alerts(&self) -> RusTorchResult<Vec<SystemAlert>> {
165        let alerts = self.alerts.lock().map_err(|_| RusTorchError::Profiling {
166            message: "Failed to acquire alerts lock".to_string(),
167        })?;
168
169        Ok(alerts.clone())
170    }
171
172    /// Clear alerts
173    /// アラートをクリア
174    pub fn clear_alerts(&self) -> RusTorchResult<()> {
175        let mut alerts = self.alerts.lock().map_err(|_| RusTorchError::Profiling {
176            message: "Failed to acquire alerts lock".to_string(),
177        })?;
178
179        alerts.clear();
180        Ok(())
181    }
182}
183
184impl Default for RealTimeMonitor {
185    fn default() -> Self {
186        Self::new(MonitorConfig::default())
187    }
