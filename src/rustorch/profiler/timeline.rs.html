8/// Timeline for tracking profiling events
9/// プロファイリングイベント追跡用タイムライン
10#[derive(Debug, Clone)]
11pub struct Timeline {
12    /// Timeline events
13    /// タイムラインイベント
14    events: Vec<TimelineEvent>,
15    /// Start time of profiling
16    /// プロファイリング開始時刻
17    start_time: Option<Instant>,
18    /// Event stack for nested operations
19    /// ネストされた操作用イベントスタック
20    event_stack: Vec<usize>,
21}
22
23/// Individual timeline event
24/// 個別タイムラインイベント
25#[derive(Debug, Clone)]
26pub struct TimelineEvent {
27    /// Event name
28    pub name: String,
29    /// Event category
30    pub category: EventCategory,
31    /// Start time
32    pub start_time: Instant,
33    /// End time
34    pub end_time: Option<Instant>,
35    /// Duration
36    pub duration: Option<Duration>,
37    /// Thread ID
38    pub thread_id: std::thread::ThreadId,
39    /// Process ID
40    pub process_id: u32,
41    /// Additional metadata
42    pub metadata: HashMap<String, String>,
43    /// Parent event index
44    pub parent_idx: Option<usize>,
45}
46
47/// Event categories for grouping
48/// グループ化用イベントカテゴリ
49#[derive(Debug, Clone, Serialize, Deserialize)]
50pub enum EventCategory {
51    /// CPU operation
52    Cpu,
53    /// GPU kernel
54    Gpu,
55    /// Memory operation
56    Memory,
57    /// Data transfer
58    DataTransfer,
59    /// Synchronization
60    Synchronization,
61    /// User annotation
62    UserAnnotation,
63}

65impl Timeline {
66    /// Create new timeline
67    pub fn new() -> Self {
68        Self {
69            events: Vec::new(),
70            start_time: None,
71            event_stack: Vec::new(),
72        }
73    }
74
75    /// Add event to timeline
76    pub fn add_event(&mut self, name: &str, start_time: Instant, category: Option<EventCategory>) {
77        if self.start_time.is_none() {
78            self.start_time = Some(start_time);
79        }
80
81        let parent_idx = self.event_stack.last().cloned();
82
83        let event = TimelineEvent {
84            name: name.to_string(),
85            category: category.unwrap_or(EventCategory::Cpu),
86            start_time,
87            end_time: None,
88            duration: None,
89            thread_id: std::thread::current().id(),
90            process_id: std::process::id(),
91            metadata: HashMap::new(),
92            parent_idx,
93        };
94
95        self.events.push(event);
96        self.event_stack.push(self.events.len() - 1);
97    }
98
99    /// End an event
100    pub fn end_event(&mut self, name: &str, end_time: Instant) {
101        // Find the most recent matching event in the stack
102        if let Some(idx) = self
103            .event_stack
104            .iter()
105            .rposition(|&i| self.events.get(i).map_or(false, |e| e.name == name))
106        {
107            let event_idx = self.event_stack.remove(idx);
108            if let Some(event) = self.events.get_mut(event_idx) {
109                event.end_time = Some(end_time);
110                event.duration = Some(end_time.duration_since(event.start_time));
111            }
112        }
113    }
114
115    /// Add metadata to the last event
116    pub fn add_metadata(&mut self, key: String, value: String) {
117        if let Some(event) = self.events.last_mut() {
118            event.metadata.insert(key, value);
119        }
120    }
121
122    /// Clear timeline
123    pub fn clear(&mut self) {
124        self.events.clear();
125        self.start_time = None;
126        self.event_stack.clear();
127    }
128
129    /// Export timeline to Chrome tracing format
130    pub fn export_chrome_trace(&self) -> String {
131        let mut trace_events = Vec::new();
132        let base_time = self.start_time.unwrap_or_else(Instant::now);
133
134        for event in &self.events {
135            let start_us = event.start_time.duration_since(base_time).as_micros() as f64;
136            let duration_us = event.duration.map(|d| d.as_micros() as f64).unwrap_or(0.0);
137
138            let trace_event = ChromeTraceEvent {
139                name: event.name.clone(),
140                cat: format!("{:?}", event.category),
141                ph: "X".to_string(), // Complete event
142                ts: start_us,
143                dur: duration_us,
144                pid: event.process_id,
145                tid: format!("{:?}", event.thread_id),
146                args: event.metadata.clone(),
147            };
148
149            trace_events.push(trace_event);
150        }
151
152        let trace = ChromeTrace {
153            traceEvents: trace_events,
154            displayTimeUnit: "ms".to_string(),
155        };
156
157        serde_json::to_string_pretty(&trace).unwrap_or_else(|_| "{}".to_string())
158    }
159
160    /// Get timeline summary
161    pub fn get_summary(&self) -> TimelineSummary {
162        let total_events = self.events.len();
163        let completed_events = self.events.iter().filter(|e| e.end_time.is_some()).count();
164
165        let total_duration =
166            if let (Some(start), Some(last_event)) = (self.start_time, self.events.last()) {
167                last_event
168                    .end_time
169                    .or(Some(Instant::now()))
170                    .map(|end| end.duration_since(start))
171            } else {
172                None
173            };
174
175        let events_by_category = self.events.iter().fold(HashMap::new(), |mut acc, event| {
176            *acc.entry(format!("{:?}", event.category)).or_insert(0) += 1;
177            acc
178        });
179
180        TimelineSummary {
181            total_events,
182            completed_events,
183            total_duration,
184            events_by_category,
185        }
186    }
187}

226/// Flame graph data structure
227/// フレームグラフデータ構造
228#[derive(Debug, Clone)]
229pub struct FlameGraphData {
230    /// Root node
231    pub root: FlameGraphNode,
232    /// Total time
233    pub total_time: Duration,
234}
235
236/// Flame graph node
237/// フレームグラフノード
238#[derive(Debug, Clone)]
239pub struct FlameGraphNode {
240    /// Node name
241    pub name: String,
242    /// Self time
243    pub self_time: Duration,
244    /// Total time including children
245    pub total_time: Duration,
246    /// Child nodes
247    pub children: Vec<FlameGraphNode>,
248}
249
250impl Timeline {
251    /// Generate flame graph data
252    pub fn generate_flame_graph(&self) -> Option<FlameGraphData> {
253        if self.events.is_empty() {
254            return None;
255        }
256
257        // Build tree structure
258        let mut root = FlameGraphNode {
259            name: "root".to_string(),
260            self_time: Duration::ZERO,
261            total_time: Duration::ZERO,
262            children: Vec::new(),
263        };
264
265        // Find root events (no parent)
266        let root_events: Vec<_> = self
267            .events
268            .iter()
269            .enumerate()
270            .filter(|(_, e)| e.parent_idx.is_none())
271            .collect();
272
273        for (idx, event) in root_events {
274            if let Some(duration) = event.duration {
275                let node = self.build_flame_graph_node(idx, &self.events);
276                root.children.push(node);
277                root.total_time += duration;
278            }
279        }
280
281        let total_time = root.total_time;
282        Some(FlameGraphData { root, total_time })
283    }
284
285    fn build_flame_graph_node(&self, event_idx: usize, events: &[TimelineEvent]) -> FlameGraphNode {
286        let event = &events[event_idx];
287        let mut node = FlameGraphNode {
288            name: event.name.clone(),
289            self_time: event.duration.unwrap_or(Duration::ZERO),
290            total_time: event.duration.unwrap_or(Duration::ZERO),
291            children: Vec::new(),
292        };
293
294        // Find child events
295        for (idx, child_event) in events.iter().enumerate() {
296            if child_event.parent_idx == Some(event_idx) {
297                let child_node = self.build_flame_graph_node(idx, events);
298                node.self_time = node.self_time.saturating_sub(child_node.total_time);
299                node.children.push(child_node);
300            }
301        }
302
303        node
304    }
305}

313    fn test_timeline_basic() {
314        let mut timeline = Timeline::new();
315
316        let start = Instant::now();
317        timeline.add_event("operation1", start, Some(EventCategory::Cpu));
