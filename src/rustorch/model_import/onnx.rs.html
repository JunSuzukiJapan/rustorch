66impl OnnxDataType {
67    /// Convert ONNX data type to RusTorch DType
68    /// ONNXデータ型をRusTorch DTypeに変換
69    pub fn to_dtype(self) -> DType {
70        match self {
71            OnnxDataType::Float => DType::Float32,
72            OnnxDataType::Double => DType::Float64,
73            OnnxDataType::Float16 => DType::Float16,
74            OnnxDataType::Int8 => DType::Int8,
75            OnnxDataType::Uint8 => DType::UInt8,
76            OnnxDataType::Int16 => DType::Int16,
77            OnnxDataType::Uint16 => DType::UInt16,
78            OnnxDataType::Int32 => DType::Int32,
79            OnnxDataType::Uint32 => DType::UInt32,
80            OnnxDataType::Int64 => DType::Int64,
81            OnnxDataType::Uint64 => DType::UInt64,
82            OnnxDataType::Bool => DType::Bool,
83            OnnxDataType::Complex64 => DType::Complex64,
84            OnnxDataType::Complex128 => DType::Complex128,
85            _ => DType::Float32, // Default fallback
86        }
87    }

167
168    // Parse ONNX model (mock implementation)
169    let onnx_model = parse_onnx_data(&onnx_data)?;
170
171    // Convert to RusTorch format
172    let metadata = create_metadata(&onnx_model);
173    let weights = extract_weights(&onnx_model)?;
174    let architecture = create_architecture(&onnx_model)?;
175
176    Ok(ImportedModel {

179        architecture,
180    })

278    let mut weights = HashMap::new();
279
280    for initializer in &onnx_model.graph.initializers {

288/// Convert ONNX tensor to RusTorch tensor
289/// ONNXテンソルをRusTorchテンソルに変換
290fn convert_onnx_tensor_to_rustorch(onnx_tensor: &OnnxTensorInfo) -> RusTorchResult<Tensor<f32>> {
291    // Convert shape from i64 to usize, handling dynamic dimensions
292    let shape: Vec<usize> = onnx_tensor
293        .shape
294        .iter()
295        .map(|&dim| if dim < 0 { 1 } else { dim as usize })
296        .collect();
297
298    match onnx_tensor.data_type {
299        OnnxDataType::Float => {
300            // Convert bytes to f32 values
301            let float_data: Vec<f32> = onnx_tensor
302                .data
303                .chunks_exact(4)
304                .map(|chunk| {
305                    let bytes = [chunk[0], chunk[1], chunk[2], chunk[3]];
306                    f32::from_le_bytes(bytes)
307                })
308                .collect();
309
310            if float_data.is_empty() {
311                // Create zero tensor for empty data
312                Ok(Tensor::zeros(&shape))
313            } else {
314                Ok(Tensor::from_vec(float_data, shape))
315            }
316        }
317        _ => {
318            // For other data types, create zero tensor for now
319            // In a real implementation, this would handle type conversion
320            Ok(Tensor::zeros(&shape))
321        }
322    }
323}

325/// Create model architecture from ONNX model
326/// ONNXモデルからモデルアーキテクチャを作成
327fn create_architecture(onnx_model: &OnnxModel) -> RusTorchResult<ModelArchitecture> {
328    let inputs = onnx_model
329        .graph
330        .inputs
331        .iter()
332        .map(create_tensor_spec)
333        .collect();
334
335    let outputs = onnx_model
336        .graph
337        .outputs
338        .iter()
339        .map(create_tensor_spec)
340        .collect();
341
342    let layer_descs: Vec<LayerDescription> = onnx_model
343        .graph
344        .nodes
345        .iter()
346        .map(create_layer_info)
347        .collect();
348
349    let layers = layer_descs
350        .into_iter()
351        .map(|desc| LayerInfo {
352            name: desc.name,
353            layer_type: desc.layer_type,
354            input_shape: desc
355                .input_shape
356                .iter()
357                .map(|&x| if x == 0 { None } else { Some(x) })
358                .collect(),
359            output_shape: desc
360                .output_shape
361                .iter()
362                .map(|&x| if x == 0 { None } else { Some(x) })
363                .collect(),
364            params: 0, // Would need to calculate from initializers
365            attributes: std::collections::HashMap::new(),
366        })
367        .collect();
368
369    let parameter_count = onnx_model
370        .graph
371        .initializers
372        .iter()
373        .map(|init| {
374            init.shape
375                .iter()
376                .map(|&dim| dim.max(1) as usize)
377                .product::<usize>()
378        })
379        .sum();
380
381    let model_size = onnx_model
382        .graph
383        .initializers
384        .iter()
385        .map(|init| init.data.len())
386        .sum();
387
388    Ok(ModelArchitecture {
389        inputs,
390        outputs,
391        layers,
392        parameter_count,
393        model_size,
394    })
395}

425/// Map ONNX operation type to RusTorch layer type
426/// ONNX操作タイプをRusTorchレイヤータイプにマップ
427fn map_onnx_op_to_layer_type(op_type: &str) -> String {
428    match op_type {
429        "MatMul" => "Linear".to_string(),
430        "Conv" => "Conv2d".to_string(),
431        "MaxPool" => "MaxPool2d".to_string(),
432        "AveragePool" => "AvgPool2d".to_string(),
433        "BatchNormalization" => "BatchNorm2d".to_string(),
434        "Relu" => "ReLU".to_string(),
435        "Sigmoid" => "Sigmoid".to_string(),
436        "Tanh" => "Tanh".to_string(),
437        "Softmax" => "Softmax".to_string(),
438        "Add" => "Add".to_string(),
439        "Mul" => "Mul".to_string(),
440        "Reshape" => "Reshape".to_string(),
441        "Transpose" => "Transpose".to_string(),
442        _ => op_type.to_string(),
443    }
