7/// Platform optimization levels
8/// プラットフォーム最適化レベル
9#[derive(Debug, Clone, Copy, PartialEq)]
10pub enum OptimizationLevel {
11    /// No optimization (O0)
12    None,
13    /// Basic optimization (O1)
14    Basic,
15    /// Standard optimization (O2)
16    Standard,
17    /// Aggressive optimization (O3)
18    Aggressive,
19}
20
21/// Platform-specific features
22/// プラットフォーム特化機能
23#[derive(Debug, Clone)]
24pub struct PlatformFeatures {
25    /// Operating system
26    pub os: String,
27    /// Architecture
28    pub arch: String,
29    /// Number of CPU cores
30    pub cpu_cores: usize,
31    /// Total system memory (bytes)
32    pub total_memory: usize,
33    /// Cache line size
34    pub cache_line_size: usize,
35    /// Page size
36    pub page_size: usize,
37    /// Supports huge pages
38    pub supports_huge_pages: bool,
39    /// Supports prefetching
40    pub supports_prefetch: bool,
41}
42
43/// Platform-specific optimizer
44/// プラットフォーム特化最適化器
45pub struct PlatformOptimizer {
46    features: PlatformFeatures,
47    optimization_level: OptimizationLevel,
48    thread_pool_size: usize,
49}
50
51impl PlatformOptimizer {
52    /// Create new platform optimizer
53    /// 新規プラットフォーム最適化器作成
54    pub fn new() -> Self {
55        let features = Self::detect_features();
56        let optimization_level = OptimizationLevel::Standard;
57        let thread_pool_size = Self::calculate_optimal_threads(&features);
58
59        PlatformOptimizer {
60            features,
61            optimization_level,
62            thread_pool_size,
63        }
64    }
65
66    /// Detect platform features
67    /// プラットフォーム機能検出
68    fn detect_features() -> PlatformFeatures {
69        PlatformFeatures {
70            os: env::consts::OS.to_string(),
71            arch: env::consts::ARCH.to_string(),
72            cpu_cores: num_cpus::get(),
73            total_memory: Self::get_total_memory(),
74            cache_line_size: Self::detect_cache_line_size(),
75            page_size: Self::get_page_size(),
76            supports_huge_pages: Self::check_huge_pages_support(),
77            supports_prefetch: Self::check_prefetch_support(),
78        }
79    }
80
81    /// Get total system memory
82    /// 総システムメモリ取得
83    fn get_total_memory() -> usize {
84        #[cfg(target_os = "linux")]
85        {
86            // Linux memory detection would require sys-info crate
87            8 * 1024 * 1024 * 1024
88        }
89        #[cfg(target_os = "macos")]
90        {
91            unsafe {
92                let mut size: usize = std::mem::size_of::<i64>();
93                let mut total_mem: i64 = 0;
94                let mut mib = [6i32, 0i32]; // CTL_HW, HW_MEMSIZE
95
96                libc::sysctl(
97                    mib.as_mut_ptr(),
98                    2,
99                    &mut total_mem as *mut _ as *mut _,
100                    &mut size,
101                    std::ptr::null_mut(),
102                    0,
103                );
104
105                if total_mem > 0 {
106                    total_mem as usize
107                } else {
108                    8_usize
109                        .saturating_mul(1024)
110                        .saturating_mul(1024)
111                        .saturating_mul(1024)
112                }
113            }
114        }
115        #[cfg(not(any(target_os = "linux", target_os = "macos")))]
116        {
117            // Default fallback: 8GB for other platforms
118            8_usize
119                .saturating_mul(1024)
120                .saturating_mul(1024)
121                .saturating_mul(1024)
122        }
123    }
124
125    /// Detect cache line size
126    /// キャッシュラインサイズ検出
127    fn detect_cache_line_size() -> usize {
128        #[cfg(any(target_arch = "x86", target_arch = "x86_64"))]
129        {
130            // Most x86/x86_64 processors have 64-byte cache lines
131            64
132        }
133
134        #[cfg(target_arch = "aarch64")]
135        {
136            // Most ARM64 processors have 64 or 128-byte cache lines
137            64
138        }
139
140        #[cfg(not(any(target_arch = "x86", target_arch = "x86_64", target_arch = "aarch64")))]
141        {
142            // Conservative default
143            64
144        }
145    }
146
147    /// Get system page size
148    /// システムページサイズ取得
149    fn get_page_size() -> usize {
150        #[cfg(unix)]
151        {
152            unsafe { libc::sysconf(libc::_SC_PAGESIZE) as usize }
153        }
154
155        #[cfg(target_os = "windows")]
156        {
157            // Windows support would require winapi crate
158            4096
159        }
160
161        #[cfg(not(any(unix, windows)))]
162        {
163            // Default page size
164            4096
165        }
166    }
167
168    /// Check huge pages support
169    /// Huge Pagesサポート確認
170    fn check_huge_pages_support() -> bool {
171        #[cfg(target_os = "linux")]
172        {
173            std::path::Path::new("/sys/kernel/mm/transparent_hugepage/enabled").exists()
174        }
175
176        #[cfg(not(target_os = "linux"))]
177        {
178            false
179        }
180    }
181
182    /// Check prefetch support
183    /// プリフェッチサポート確認
184    fn check_prefetch_support() -> bool {
185        #[cfg(any(target_arch = "x86", target_arch = "x86_64"))]
186        {
187            true // x86/x86_64 generally supports prefetch
188        }
189
190        #[cfg(target_arch = "aarch64")]
191        {
192            true // ARM64 supports prefetch
193        }
194
195        #[cfg(not(any(target_arch = "x86", target_arch = "x86_64", target_arch = "aarch64")))]
196        {
197            false
198        }
199    }

201    /// Calculate optimal thread pool size
202    /// 最適スレッドプールサイズ計算
203    fn calculate_optimal_threads(features: &PlatformFeatures) -> usize {
204        // Use 75% of available cores for compute, leave some for system
205        let compute_cores = (features.cpu_cores as f32 * 0.75) as usize;
206        compute_cores.max(1)
207    }
208
209    /// Apply platform-specific memory alignment
210    /// プラットフォーム特化メモリアライメント適用
211    pub fn align_memory(&self, size: usize) -> usize {
212        let alignment = self.features.cache_line_size;
213        (size + alignment - 1) & !(alignment - 1)
214    }
215
216    /// Allocate aligned memory
217    /// アライメント済みメモリ割り当て
218    pub fn allocate_aligned<T>(&self, count: usize) -> RusTorchResult<Vec<T>>
219    where
220        T: Default + Clone,
221    {
222        let size = count * std::mem::size_of::<T>();
223        let aligned_size = self.align_memory(size);
224        let aligned_count = aligned_size / std::mem::size_of::<T>();
225
226        Ok(vec![T::default(); aligned_count])
227    }

229    /// Apply prefetch hints for memory access
230    /// メモリアクセスのプリフェッチヒント適用
231    #[cfg(any(target_arch = "x86", target_arch = "x86_64"))]
232    pub unsafe fn prefetch_read<T>(ptr: *const T) {
233        #[cfg(target_arch = "x86_64")]
234        {
235            use std::arch::x86_64::_mm_prefetch;
236            _mm_prefetch(ptr as *const i8, 0); // _MM_HINT_T0
237        }
238    }
239
240    #[cfg(not(any(target_arch = "x86", target_arch = "x86_64")))]
241    pub unsafe fn prefetch_read<T>(_ptr: *const T) {
242        // No-op on unsupported platforms
243    }

262        &self.features
263    }
264
265    /// Get optimization level
266    /// 最適化レベル取得
267    pub fn optimization_level(&self) -> OptimizationLevel {
268        self.optimization_level
269    }
270
271    /// Set optimization level
272    /// 最適化レベル設定
273    pub fn set_optimization_level(&mut self, level: OptimizationLevel) {
274        self.optimization_level = level;
275        // Adjust thread pool size based on optimization level
276        match level {
277            OptimizationLevel::None => self.thread_pool_size = 1,
278            OptimizationLevel::Basic => self.thread_pool_size = self.features.cpu_cores / 2,
279            OptimizationLevel::Standard => {
280                self.thread_pool_size = Self::calculate_optimal_threads(&self.features)
281            }
282            OptimizationLevel::Aggressive => self.thread_pool_size = self.features.cpu_cores,
283        }
284    }
285
286    /// Get optimal thread pool size
287    /// 最適スレッドプールサイズ取得
288    pub fn thread_pool_size(&self) -> usize {
289        self.thread_pool_size
290    }

297    #[test]
298    fn test_platform_detection() {
299        let optimizer = PlatformOptimizer::new();
300        let features = optimizer.features();
301
302        println!("Platform features:");
303        println!("  OS: {}", features.os);
304        println!("  Architecture: {}", features.arch);
305        println!("  CPU cores: {}", features.cpu_cores);
306        println!(
307            "  Total memory: {} GB",
308            features.total_memory / (1024 * 1024 * 1024)
309        );
310        println!("  Cache line size: {} bytes", features.cache_line_size);
311        println!("  Page size: {} bytes", features.page_size);
312        println!("  Huge pages: {}", features.supports_huge_pages);
313        println!("  Prefetch: {}", features.supports_prefetch);
314
315        assert!(features.cpu_cores > 0);
316        assert!(features.cache_line_size > 0);
317        assert!(features.page_size > 0);
318    }

320    #[test]
321    fn test_memory_alignment() {
322        let optimizer = PlatformOptimizer::new();
323
324        let unaligned = 100;
325        let aligned = optimizer.align_memory(unaligned);
326
327        assert!(aligned >= unaligned);
328        assert_eq!(aligned % optimizer.features().cache_line_size, 0);
329    }

331    #[test]
332    fn test_optimization_levels() {
333        let mut optimizer = PlatformOptimizer::new();
334
335        optimizer.set_optimization_level(OptimizationLevel::None);
336        assert_eq!(optimizer.thread_pool_size(), 1);
337
338        optimizer.set_optimization_level(OptimizationLevel::Aggressive);
339        assert_eq!(optimizer.thread_pool_size(), optimizer.features().cpu_cores);
