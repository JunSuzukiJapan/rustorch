12/// Pruning strategy enumeration
13/// プルーニング戦略列挙
14#[derive(Debug, Clone, Copy, PartialEq)]
15pub enum PruningStrategy {
16    /// Magnitude-based pruning - remove smallest weights
17    /// マグニチュードベースプルーニング - 最小重みを削除
18    Magnitude,
19    /// Random pruning for baseline comparison
20    /// ベースライン比較用ランダムプルーニング
21    Random,
22    /// Structured pruning - remove entire neurons/channels
23    /// 構造化プルーニング - ニューロン/チャンネル全体を削除
24    Structured,
25    /// Gradient-based pruning using importance scores
26    /// 重要度スコアを使用した勾配ベースプルーニング
27    GradientBased,
28    /// SNIP (Single-shot Network Pruning)
29    /// SNIP（シングルショットネットワークプルーニング）
30    SNIP,
31}

33/// Pruning configuration and parameters
34/// プルーニング設定とパラメータ
35#[derive(Debug, Clone)]
36pub struct PruningConfig {
37    /// Target sparsity ratio (0.0 = dense, 1.0 = completely sparse)
38    /// 目標スパース率（0.0 = 密, 1.0 = 完全スパース）
39    pub target_sparsity: f32,
40    /// Pruning strategy to use
41    /// 使用するプルーニング戦略
42    pub strategy: PruningStrategy,
43    /// Whether to prune structured (entire neurons) or unstructured (individual weights)
44    /// 構造化（ニューロン全体）または非構造化（個別重み）プルーニング
45    pub structured: bool,
46    /// Gradual pruning schedule (None for one-shot pruning)
47    /// 段階的プルーニングスケジュール（ワンショットプルーニングの場合はNone）
48    pub schedule: Option<PruningSchedule>,
49}

51/// Gradual pruning schedule
52/// 段階的プルーニングスケジュール
53#[derive(Debug, Clone)]
54pub struct PruningSchedule {
55    /// Initial sparsity
56    /// 初期スパース率
57    pub initial_sparsity: f32,
58    /// Final sparsity
59    /// 最終スパース率
60    pub final_sparsity: f32,
61    /// Number of pruning steps
62    /// プルーニングステップ数
63    pub num_steps: usize,
64    /// Current step
65    /// 現在のステップ
66    pub current_step: usize,
67}
68
69impl PruningSchedule {
70    /// Create a new gradual pruning schedule
71    /// 新しい段階的プルーニングスケジュールを作成
72    pub fn new(initial_sparsity: f32, final_sparsity: f32, num_steps: usize) -> Self {
73        Self {
74            initial_sparsity,
75            final_sparsity,
76            num_steps,
77            current_step: 0,
78        }
79    }
80
81    /// Get current target sparsity based on schedule
82    /// スケジュールに基づく現在の目標スパース率を取得
83    pub fn current_sparsity(&self) -> f32 {
84        if self.current_step >= self.num_steps {
85            return self.final_sparsity;
86        }
87
88        let progress = self.current_step as f32 / self.num_steps as f32;
89        self.initial_sparsity + progress * (self.final_sparsity - self.initial_sparsity)
90    }
91
92    /// Advance to next pruning step
93    /// 次のプルーニングステップに進む
94    pub fn step(&mut self) {
95        self.current_step = (self.current_step + 1).min(self.num_steps);
96    }
97}

99/// Model pruner for applying sparsification algorithms
100/// スパース化アルゴリズム適用用モデルプルーナー
101pub struct ModelPruner<T: Float> {
102    /// Pruning configuration
103    /// プルーニング設定
104    pub config: PruningConfig,
105    /// Importance scores for gradient-based pruning
106    /// 勾配ベースプルーニング用重要度スコア
107    pub importance_scores: HashMap<String, Array1<T>>,
108}
109
110use std::collections::HashMap;
111
112impl<
113        T: Float
114            + PartialOrd
115            + Copy
116            + Send
117            + Sync
118            + ndarray::ScalarOperand
119            + FromPrimitive
120            + std::ops::AddAssign,
121    > ModelPruner<T>
122{
123    /// Create a new model pruner
124    /// 新しいモデルプルーナーを作成
125    pub fn new(config: PruningConfig) -> Self {
126        Self {
127            config,
128            importance_scores: HashMap::new(),
129        }
130    }
131
132    /// Prune a tensor based on the configured strategy
133    /// 設定された戦略に基づいてテンソルをプルーニング
134    pub fn prune_tensor(&self, tensor: &ArrayD<T>) -> RusTorchResult<SparseTensor<T>> {
135        match self.config.strategy {
136            PruningStrategy::Magnitude => self.magnitude_pruning(tensor),
137            PruningStrategy::Random => self.random_pruning(tensor),
138            PruningStrategy::Structured => self.structured_pruning(tensor),
139            PruningStrategy::GradientBased => self.gradient_based_pruning(tensor),
140            PruningStrategy::SNIP => self.snip_pruning(tensor),
141        }
142    }
143
144    /// Magnitude-based pruning - keep largest magnitude weights
145    /// マグニチュードベースプルーニング - 最大マグニチュード重みを保持
146    fn magnitude_pruning(&self, tensor: &ArrayD<T>) -> RusTorchResult<SparseTensor<T>> {
147        let target_sparsity = self.get_current_sparsity();
148        let total_elements = tensor.len();
149        let elements_to_keep = ((1.0 - target_sparsity) * total_elements as f32) as usize;
150
151        // Calculate magnitudes and sort indices
152        let mut magnitude_indices: Vec<(usize, T)> = tensor
153            .iter()
154            .enumerate()
155            .map(|(i, &val)| (i, val.abs()))
156            .collect();
157
158        magnitude_indices.sort_by(|a, b| b.1.partial_cmp(&a.1).unwrap_or(Ordering::Equal));
159
160        // Keep top elements_to_keep weights
161        let kept_indices: Vec<usize> = magnitude_indices
162            .iter()
163            .take(elements_to_keep)
164            .map(|(idx, _)| *idx)
165            .collect();
166
167        self.create_sparse_from_indices(tensor, &kept_indices)
168    }
169
170    /// Random pruning for baseline comparison
171    /// ベースライン比較用ランダムプルーニング
172    fn random_pruning(&self, tensor: &ArrayD<T>) -> RusTorchResult<SparseTensor<T>> {
173        let target_sparsity = self.get_current_sparsity();
174        let total_elements = tensor.len();
175        let elements_to_keep = ((1.0 - target_sparsity) * total_elements as f32) as usize;
176
177        // Generate random indices to keep
178        use rand::seq::SliceRandom;
179        let mut rng = rand::thread_rng();
180        let mut all_indices: Vec<usize> = (0..total_elements).collect();
181        all_indices.shuffle(&mut rng);
182
183        let kept_indices = &all_indices[..elements_to_keep];
184        self.create_sparse_from_indices(tensor, kept_indices)
185    }
186
187    /// Structured pruning - remove entire neurons or channels
188    /// 構造化プルーニング - ニューロンまたはチャンネル全体を削除
189    fn structured_pruning(&self, tensor: &ArrayD<T>) -> RusTorchResult<SparseTensor<T>> {
190        if tensor.ndim() < 2 {
191            return Err(RusTorchError::InvalidOperation {
192                operation: "structured_pruning".to_string(),
193                message: "Structured pruning requires at least 2D tensors".to_string(),
194            });
195        }
196
197        let target_sparsity = self.get_current_sparsity();
198
199        // For 2D tensors, prune entire rows (neurons)
200        if tensor.ndim() == 2 {
201            let rows = tensor.shape()[0];
202            let rows_to_keep = ((1.0 - target_sparsity) * rows as f32) as usize;
203
204            // Calculate L2 norm for each row
205            let mut row_norms: Vec<(usize, T)> = (0..rows)
206                .map(|i| {
207                    let row = tensor.slice(s![i, ..]);
208                    let norm_sq = row.iter().map(|&x| x * x).fold(T::zero(), |a, b| a + b);
209                    (i, norm_sq.sqrt())
210                })
211                .collect();
212
213            // Sort by norm (descending)
214            row_norms.sort_by(|a, b| b.1.partial_cmp(&a.1).unwrap_or(Ordering::Equal));
215
216            // Keep top rows
217            let kept_rows: Vec<usize> = row_norms
218                .iter()
219                .take(rows_to_keep)
220                .map(|(idx, _)| *idx)
221                .collect();
222
223            // Create indices for kept elements
224            let mut kept_indices = Vec::new();
225            let cols = tensor.shape()[1];
226
227            for &row in &kept_rows {
228                for col in 0..cols {
229                    kept_indices.push(row * cols + col);
230                }
231            }
232
233            return self.create_sparse_from_indices(tensor, &kept_indices);
234        }
235
236        // For higher-dimensional tensors, fall back to magnitude pruning
237        self.magnitude_pruning(tensor)
238    }
239
240    /// Gradient-based pruning using importance scores
241    /// 重要度スコアを使用した勾配ベースプルーニング
242    fn gradient_based_pruning(&self, tensor: &ArrayD<T>) -> RusTorchResult<SparseTensor<T>> {
243        // This is a simplified implementation
244        // In practice, would use accumulated gradient information
245        self.magnitude_pruning(tensor)
246    }
247
248    /// SNIP (Single-shot Network Pruning) based on connection sensitivity
249    /// 接続感度に基づくSNIP（シングルショットネットワークプルーニング）
250    fn snip_pruning(&self, tensor: &ArrayD<T>) -> RusTorchResult<SparseTensor<T>> {
251        // SNIP uses gradient information at initialization
252        // For now, fall back to magnitude-based pruning
253        self.magnitude_pruning(tensor)
254    }
255
256    /// Helper function to create sparse tensor from kept indices
257    /// 保持インデックスからスパーステンソルを作成するヘルパー関数
258    fn create_sparse_from_indices(
259        &self,
260        tensor: &ArrayD<T>,
261        kept_indices: &[usize],
262    ) -> RusTorchResult<SparseTensor<T>> {
263        let shape = tensor.shape().to_vec();
264        let mut indices_per_dim = vec![Vec::new(); shape.len()];
265        let mut values = Vec::new();
266
267        let flat_tensor = tensor.as_slice().unwrap();
268
269        for &flat_idx in kept_indices {
270            if flat_idx >= flat_tensor.len() {
271                continue;
272            }
273
274            let value = flat_tensor[flat_idx];
275            if !value.is_zero() {
276                values.push(value);
277
278                // Convert flat index to multi-dimensional coordinates
279                let mut remaining_idx = flat_idx;
280                for (dim, &dim_size) in shape.iter().enumerate().rev() {
281                    let coord = remaining_idx % dim_size;
282                    indices_per_dim[shape.len() - 1 - dim].push(coord);
283                    remaining_idx /= dim_size;
284                }
285            }
286        }
287
288        let indices: Vec<Array1<usize>> = indices_per_dim
289            .into_iter()
290            .map(|v| Array1::from_vec(v))
291            .collect();
292        let values_array = Array1::from_vec(values);
293
294        SparseTensor::from_coo(indices, values_array, shape)
295    }
296
297    /// Get current sparsity target based on schedule
298    /// スケジュールに基づく現在のスパース率目標を取得
299    fn get_current_sparsity(&self) -> f32 {
300        match &self.config.schedule {
301            Some(schedule) => schedule.current_sparsity(),
302            None => self.config.target_sparsity,
303        }
304    }
305
306    /// Update importance scores for gradient-based pruning
307    /// 勾配ベースプルーニング用重要度スコアを更新
308    pub fn update_importance_scores(&mut self, param_name: &str, gradients: &ArrayD<T>) {
309        // Calculate importance as magnitude of gradients
310        let importance: Array1<T> = gradients.iter().map(|&grad| grad.abs()).collect();
311
312        self.importance_scores
313            .insert(param_name.to_string(), importance);
314    }
315
316    /// Prune an entire model (collection of parameters)
317    /// モデル全体（パラメータ集合）をプルーニング
318    pub fn prune_model(
319        &mut self,
320        parameters: &HashMap<String, Variable<T>>,
321    ) -> RusTorchResult<HashMap<String, SparseTensor<T>>> {
322        let mut pruned_params = HashMap::new();
323
324        for (name, param) in parameters.iter() {
325            let param_tensor = param.data();
326            let param_guard = param_tensor.read().unwrap();
327
328            let sparse_param = self.prune_tensor(&param_guard.data)?;
329            pruned_params.insert(name.clone(), sparse_param);
330        }
331
332        // Update schedule if gradual pruning
333        if let Some(ref mut schedule) = self.config.schedule {
334            schedule.step();
335        }
336
337        Ok(pruned_params)
338    }
339}

341/// Specific pruning algorithms implementation
342/// 特定プルーニングアルゴリズム実装
343pub struct MagnitudePruner<T: Float> {
344    /// Global or layer-wise pruning
345    /// グローバルまたはレイヤー単位プルーニング
346    pub global: bool,
347    /// Target sparsity
348    /// 目標スパース率
349    pub sparsity: f32,
350    _phantom: std::marker::PhantomData<T>,
351}
352
353impl<T: Float + PartialOrd + Copy> MagnitudePruner<T> {
354    /// Create magnitude-based pruner
355    /// マグニチュードベースプルーナーを作成
356    pub fn new(sparsity: f32, global: bool) -> Self {
357        Self {
358            global,
359            sparsity: sparsity.clamp(0.0, 1.0),
360            _phantom: std::marker::PhantomData,
361        }
362    }
363
364    /// Apply magnitude pruning to tensor
365    /// テンソルにマグニチュードプルーニングを適用
366    pub fn prune(&self, tensor: &ArrayD<T>) -> RusTorchResult<ArrayD<T>> {
367        let flat_tensor = tensor.as_slice().unwrap();
368        let total_elements = flat_tensor.len();
369        let elements_to_zero = (self.sparsity * total_elements as f32) as usize;
370
371        // Calculate absolute values and sort indices
372        let mut magnitude_indices: Vec<(usize, T)> = flat_tensor
373            .iter()
374            .enumerate()
375            .map(|(i, &val)| (i, val.abs()))
376            .collect();
377
378        magnitude_indices.sort_by(|a, b| a.1.partial_cmp(&b.1).unwrap_or(Ordering::Equal));
379
380        // Create pruned tensor
381        let mut pruned = tensor.clone();
382        let pruned_flat = pruned.as_slice_mut().unwrap();
383
384        for i in 0..elements_to_zero.min(magnitude_indices.len()) {
385            let idx = magnitude_indices[i].0;
386            pruned_flat[idx] = T::zero();
387        }
388
389        Ok(pruned)
390    }
391}

393/// Structured pruning for neural network layers
394/// ニューラルネットワーク層の構造化プルーニング
395pub struct StructuredPruner<T: Float> {
396    /// Granularity: neuron, channel, or filter
397    /// 粒度：ニューロン、チャンネル、フィルター
398    pub granularity: StructuredGranularity,
399    /// Target pruning ratio
400    /// 目標プルーニング率
401    pub ratio: f32,
402    _phantom: std::marker::PhantomData<T>,
403}
404
405/// Structured pruning granularity options
406/// 構造化プルーニング粒度オプション
407#[derive(Debug, Clone, Copy, PartialEq)]
408pub enum StructuredGranularity {
409    /// Remove entire neurons (rows in weight matrix)
410    /// ニューロン全体を削除（重み行列の行）
411    Neuron,
412    /// Remove entire channels (for convolutional layers)
413    /// チャンネル全体を削除（畳み込み層用）
414    Channel,
415    /// Remove entire filters (for convolutional layers)
416    /// フィルター全体を削除（畳み込み層用）
417    Filter,
418}
419
420impl<T: Float + PartialOrd + Copy> StructuredPruner<T> {
421    /// Create structured pruner
422    /// 構造化プルーナーを作成
423    pub fn new(granularity: StructuredGranularity, ratio: f32) -> Self {
424        Self {
425            granularity,
426            ratio: ratio.clamp(0.0, 1.0),
427            _phantom: std::marker::PhantomData,
428        }
429    }
430
431    /// Apply structured pruning to 2D weight matrix
432    /// 2D重み行列に構造化プルーニングを適用
433    pub fn prune_linear_weights(&self, weights: &Array2<T>) -> RusTorchResult<Array2<T>> {
434        let (rows, cols) = weights.dim();
435
436        match self.granularity {
437            StructuredGranularity::Neuron => {
438                let neurons_to_prune = (self.ratio * rows as f32) as usize;
439
440                // Calculate L2 norm for each neuron (row)
441                let mut neuron_norms: Vec<(usize, T)> = (0..rows)
442                    .map(|i| {
443                        let row = weights.row(i);
444                        let norm_sq = row.iter().map(|&x| x * x).fold(T::zero(), |a, b| a + b);
445                        (i, norm_sq.sqrt())
446                    })
447                    .collect();
448
449                // Sort by norm (ascending - prune smallest)
450                neuron_norms.sort_by(|a, b| a.1.partial_cmp(&b.1).unwrap_or(Ordering::Equal));
451
452                // Zero out smallest neurons
453                let mut pruned_weights = weights.clone();
454                for i in 0..neurons_to_prune.min(neuron_norms.len()) {
455                    let neuron_idx = neuron_norms[i].0;
456                    for j in 0..cols {
457                        pruned_weights[[neuron_idx, j]] = T::zero();
458                    }
459                }
460
461                Ok(pruned_weights)
462            }
463            _ => {
464                // For other granularities, fall back to magnitude pruning
465                let flattened = weights.clone().into_dyn();
466                let magnitude_pruner = MagnitudePruner::new(self.ratio, false);
467                let pruned_flat = magnitude_pruner.prune(&flattened)?;
468
469                Ok(Array2::from_shape_vec(
470                    (rows, cols),
471                    pruned_flat.into_raw_vec_and_offset().0,
472                )?)
473            }
474        }
475    }
476}

478/// Fisher information-based pruning
479/// フィッシャー情報ベースプルーニング
480pub struct FisherPruner<T: Float> {
481    /// Accumulated Fisher information matrix
482    /// 累積フィッシャー情報行列
483    pub fisher_info: HashMap<String, ArrayD<T>>,
484    /// Number of samples used for Fisher estimation
485    /// フィッシャー推定に使用されたサンプル数
486    pub n_samples: usize,
487}
488
489impl<
490        T: Float + std::ops::AddAssign + Copy + ndarray::ScalarOperand + Send + Sync + FromPrimitive,
491    > FisherPruner<T>
492{
493    /// Create Fisher information pruner
494    /// フィッシャー情報プルーナーを作成
495    pub fn new() -> Self {
496        Self {
497            fisher_info: HashMap::new(),
498            n_samples: 0,
499        }
500    }
501
502    /// Update Fisher information with gradients
503    /// 勾配でフィッシャー情報を更新
504    pub fn update_fisher(&mut self, param_name: &str, gradients: &ArrayD<T>) {
505        // Fisher information approximation: E[∇log p(x|θ)²]
506        let squared_grads = gradients.mapv(|g| g * g);
507
508        match self.fisher_info.get_mut(param_name) {
509            Some(existing) => {
510                // Running average update
511                let alpha = T::one() / T::from(self.n_samples + 1).unwrap();
512                *existing = &*existing * (T::one() - alpha) + &squared_grads * alpha;
513            }
514            None => {
515                self.fisher_info
516                    .insert(param_name.to_string(), squared_grads);
517            }
518        }
519
520        self.n_samples += 1;
521    }
522
523    /// Prune based on Fisher information scores
524    /// フィッシャー情報スコアに基づくプルーニング
525    pub fn prune_with_fisher(
526        &self,
527        param_name: &str,
528        tensor: &ArrayD<T>,
529        target_sparsity: f32,
530    ) -> RusTorchResult<SparseTensor<T>> {
531        let fisher_scores =
532            self.fisher_info
533                .get(param_name)
534                .ok_or_else(|| RusTorchError::InvalidParameters {
535                    operation: "fisher_pruning".to_string(),
536                    message: format!(
537                        "No Fisher information available for parameter: {}",
538                        param_name
539                    ),
540                })?;
541
542        if fisher_scores.shape() != tensor.shape() {
543            return Err(RusTorchError::ShapeMismatch {
544                expected: tensor.shape().to_vec(),
545                actual: fisher_scores.shape().to_vec(),
546            });
547        }
548
549        let total_elements = tensor.len();
550        let elements_to_keep = ((1.0 - target_sparsity) * total_elements as f32) as usize;
551
552        // Sort by Fisher scores (keep highest importance)
553        let mut fisher_indices: Vec<(usize, T)> = fisher_scores
554            .iter()
555            .enumerate()
556            .map(|(i, &score)| (i, score))
557            .collect();
558
559        fisher_indices.sort_by(|a, b| b.1.partial_cmp(&a.1).unwrap_or(Ordering::Equal));
560
561        let kept_indices: Vec<usize> = fisher_indices
562            .iter()
563            .take(elements_to_keep)
564            .map(|(idx, _)| *idx)
565            .collect();
566
567        let model_pruner = ModelPruner::new(PruningConfig {
568            target_sparsity,
569            strategy: PruningStrategy::GradientBased,
570            structured: false,
571            schedule: None,
572        });
573
574        model_pruner.create_sparse_from_indices(tensor, &kept_indices)
575    }
576}
