17impl<T: Float + Copy + PartialOrd + Sum + std::fmt::Display> SparseAnalyzer<T> {
18    /// Analyze sparsity patterns in tensor
19    /// テンソルのスパースパターンを解析
20    pub fn analyze_pattern(tensor: &SparseTensor<T>) -> SparsePatternAnalysis<T> {
21        let mut analysis = SparsePatternAnalysis::new();
22
23        // Basic statistics
24        analysis.total_elements = tensor.dense_size();
25        analysis.non_zero_elements = tensor.nnz;
26        analysis.sparsity_ratio = tensor.sparsity();
27        analysis.format = tensor.format;
28
29        // Value distribution statistics
30        if !tensor.values.is_empty() {
31            let values_slice = tensor.values.as_slice().unwrap();
32            analysis.min_value = values_slice
33                .iter()
34                .fold(T::infinity(), |a, &b| if a < b { a } else { b });
35            analysis.max_value =
36                values_slice
37                    .iter()
38                    .fold(T::neg_infinity(), |a, &b| if a > b { a } else { b });
39            analysis.mean_abs_value =
40                values_slice.iter().map(|&x| x.abs()).sum::<T>() / T::from(tensor.nnz).unwrap();
41        }
42
43        // Pattern regularity analysis
44        analysis.pattern_regularity = Self::compute_pattern_regularity(tensor);
45
46        // Memory efficiency
47        let dense_memory = tensor.dense_size() * std::mem::size_of::<T>();
48        let sparse_memory = tensor.memory_usage();
49        analysis.memory_efficiency = 1.0 - (sparse_memory as f64 / dense_memory as f64);
50
51        analysis
52    }
53
54    /// Compute pattern regularity score (0.0 = random, 1.0 = highly structured)
55    /// パターン規則性スコアを計算（0.0 = ランダム, 1.0 = 高度に構造化）
56    fn compute_pattern_regularity(tensor: &SparseTensor<T>) -> f64 {
57        if tensor.format != SparseFormat::COO || tensor.shape.len() != 2 {
58            return 0.0; // Can only analyze 2D COO tensors for now
59        }
60
61        let row_indices = &tensor.indices[0];
62        let col_indices = &tensor.indices[1];
63
64        // Analyze row distribution uniformity
65        let mut row_counts = HashMap::new();
66        for &row in row_indices.iter() {
67            *row_counts.entry(row).or_insert(0) += 1;
68        }
69
70        // Calculate coefficient of variation for row distribution
71        let row_count_values: Vec<_> = row_counts.values().collect();
72        if row_count_values.is_empty() {
73            return 0.0;
74        }
75
76        let mean = row_count_values.iter().map(|&&x| x as f64).sum::<f64>()
77            / row_count_values.len() as f64;
78        let variance = row_count_values
79            .iter()
80            .map(|&&x| (x as f64 - mean).powi(2))
81            .sum::<f64>()
82            / row_count_values.len() as f64;
83
84        let cv = variance.sqrt() / mean;
85
86        // Lower coefficient of variation indicates more regular pattern
87        (1.0 / (1.0 + cv)).clamp(0.0, 1.0)
88    }
89
90    /// Suggest optimal sparse format for given access pattern
91    /// アクセスパターンに最適なスパース形式を提案
92    pub fn suggest_optimal_format(
93        tensor: &SparseTensor<T>,
94        access_pattern: AccessPattern,
95    ) -> SparseFormat {
96        match access_pattern {
97            AccessPattern::RowMajor | AccessPattern::MatrixVector => SparseFormat::CSR,
98            AccessPattern::ColumnMajor => SparseFormat::CSC,
99            AccessPattern::Random | AccessPattern::Unknown => {
100                // Choose based on sparsity level
101                if tensor.sparsity() > 0.95 {
102                    SparseFormat::COO // Very sparse - COO is more memory efficient
103                } else {
104                    SparseFormat::CSR // Moderately sparse - CSR for better access patterns
105                }
106            }
107        }
108    }

232/// Sparse tensor validation utilities
233/// スパーステンソル検証ユーティリティ
234pub struct SparseValidator;
235
236impl SparseValidator {
237    /// Validate sparse tensor integrity
238    /// スパーステンソル整合性を検証
239    pub fn validate<T: Float + Copy + PartialOrd>(tensor: &SparseTensor<T>) -> RusTorchResult<()> {
240        // Check basic consistency
241        if tensor.values.len() != tensor.nnz {
242            return Err(RusTorchError::InvalidParameters {
243                operation: "sparse_validation".to_string(),
244                message: "Values length doesn't match nnz count".to_string(),
245            });
246        }
247
248        // Check indices validity (format-specific)
249        match tensor.format {
250            SparseFormat::COO => {
251                // COO: all index arrays must have length nnz
252                for (dim, indices) in tensor.indices.iter().enumerate() {
253                    if indices.len() != tensor.nnz {
254                        return Err(RusTorchError::InvalidParameters {
255                            operation: "sparse_validation".to_string(),
256                            message: format!("COO indices dimension {} length mismatch", dim),
257                        });
258                    }
259
260                    // Check bounds for COO indices
261                    if dim < tensor.shape.len() {
262                        let max_allowed = tensor.shape[dim];
263                        for &idx in indices.iter() {
264                            if idx >= max_allowed {
265                                return Err(RusTorchError::InvalidParameters {
266                                    operation: "sparse_validation".to_string(),
267                                    message: format!(
268                                        "Index {} exceeds dimension {} size {}",
269                                        idx, dim, max_allowed
270                                    ),
271                                });
272                            }
273                        }
274                    }
275                }
276            }
277            SparseFormat::CSR => {
278                // CSR: row_ptr has length rows+1, col_indices has length nnz
279                if tensor.indices.len() != 2 {
280                    return Err(RusTorchError::InvalidParameters {
281                        operation: "sparse_validation".to_string(),
282                        message: "CSR format requires exactly 2 index arrays".to_string(),
283                    });
284                }
285                // Row pointer validation will be done in format-specific validation
286                if tensor.indices[1].len() != tensor.nnz {
287                    return Err(RusTorchError::InvalidParameters {
288                        operation: "sparse_validation".to_string(),
289                        message: "CSR col_indices length must match nnz".to_string(),
290                    });
291                }
292
293                // Check column indices bounds
294                let max_cols = tensor.shape[1];
295                for &col_idx in tensor.indices[1].iter() {
296                    if col_idx >= max_cols {
297                        return Err(RusTorchError::InvalidParameters {
298                            operation: "sparse_validation".to_string(),
299                            message: format!(
300                                "Column index {} exceeds matrix width {}",
301                                col_idx, max_cols
302                            ),
303                        });
304                    }
305                }
306            }
307            SparseFormat::CSC => {
308                // Similar to CSR but for columns
309                if tensor.indices.len() != 2 {
310                    return Err(RusTorchError::InvalidParameters {
311                        operation: "sparse_validation".to_string(),
312                        message: "CSC format requires exactly 2 index arrays".to_string(),
313                    });
314                }
315            }
316        }
317
318        // Format-specific validation
319        match tensor.format {
320            SparseFormat::CSR => Self::validate_csr(tensor)?,
321            SparseFormat::COO => Self::validate_coo(tensor)?,
322            SparseFormat::CSC => {
323                return Err(RusTorchError::NotImplemented {
324                    feature: "CSC format validation".to_string(),
325                });
326            }
327        }
328
329        Ok(())
330    }
331
332    /// Validate CSR format specific constraints
333    /// CSR形式特有制約を検証
334    fn validate_csr<T: Float>(tensor: &SparseTensor<T>) -> RusTorchResult<()> {
335        if tensor.shape.len() != 2 {
336            return Err(RusTorchError::InvalidParameters {
337                operation: "csr_validation".to_string(),
338                message: "CSR format requires 2D tensors".to_string(),
339            });
340        }
341
342        if tensor.indices.len() != 2 {
343            return Err(RusTorchError::InvalidParameters {
344                operation: "csr_validation".to_string(),
345                message: "CSR format requires exactly 2 index arrays".to_string(),
346            });
347        }
348
349        let row_ptr = &tensor.indices[0];
350        let col_indices = &tensor.indices[1];
351
352        // Validate row pointer
353        if row_ptr.len() != tensor.shape[0] + 1 {
354            return Err(RusTorchError::InvalidParameters {
355                operation: "csr_validation".to_string(),
356                message: "Row pointer length must be rows + 1".to_string(),
357            });
358        }
359
360        // Check row pointer is non-decreasing
361        for i in 1..row_ptr.len() {
362            if row_ptr[i] < row_ptr[i - 1] {
363                return Err(RusTorchError::InvalidParameters {
364                    operation: "csr_validation".to_string(),
365                    message: "Row pointer must be non-decreasing".to_string(),
366                });
367            }
368        }
369
370        // Check last row pointer equals nnz
371        if row_ptr[row_ptr.len() - 1] != tensor.nnz {
372            return Err(RusTorchError::InvalidParameters {
373                operation: "csr_validation".to_string(),
374                message: "Last row pointer must equal nnz".to_string(),
375            });
376        }
377
378        Ok(())
379    }
380
381    /// Validate COO format specific constraints
382    /// COO形式特有制約を検証
383    fn validate_coo<T: Float>(tensor: &SparseTensor<T>) -> RusTorchResult<()> {
384        if tensor.indices.len() != tensor.shape.len() {
385            return Err(RusTorchError::InvalidParameters {
386                operation: "coo_validation".to_string(),
387                message: "COO format requires one index array per dimension".to_string(),
388            });
389        }
390
391        // Check for duplicate indices (optional - could be expensive)
392        if tensor.shape.len() == 2 {
393            let mut coordinate_set = HashSet::new();
394
395            for i in 0..tensor.nnz {
396                let coord = (tensor.indices[0][i], tensor.indices[1][i]);
397                if coordinate_set.contains(&coord) {
398                    return Err(RusTorchError::InvalidParameters {
399                        operation: "coo_validation".to_string(),
400                        message: "Duplicate coordinates found in COO tensor".to_string(),
401                    });
402                }
403                coordinate_set.insert(coord);
404            }
405        }
406
407        Ok(())
408    }

411/// Sparse tensor format conversion utilities
412/// スパーステンソル形式変換ユーティリティ
413pub struct SparseConverter;
414
415impl SparseConverter {
416    /// Convert between sparse formats with validation
417    /// 検証付きスパース形式間変換
418    pub fn convert<T: Float + Zero + One + Copy + std::ops::AddAssign + FromPrimitive>(
419        tensor: &SparseTensor<T>,
420        target_format: SparseFormat,
421    ) -> RusTorchResult<SparseTensor<T>> {
422        // Validate input tensor first
423        SparseValidator::validate(tensor)?;
424
425        let result = match (tensor.format, target_format) {
426            (SparseFormat::COO, SparseFormat::CSR) => tensor.to_csr()?,
427            (SparseFormat::CSR, SparseFormat::COO) => tensor.to_coo()?,
428            (format, target) if format == target => tensor.clone(),
429            _ => {
430                return Err(RusTorchError::NotImplemented {
431                    feature: format!("Conversion from {:?} to {:?}", tensor.format, target_format),
432                });
433            }
434        };
435
436        // Validate result
437        SparseValidator::validate(&result)?;
438        Ok(result)
439    }
440
441    /// Batch convert multiple tensors efficiently
442    /// 複数テンソルの効率的バッチ変換
443    pub fn batch_convert<T: Float + Zero + One + Copy + std::ops::AddAssign + FromPrimitive>(
444        tensors: &[SparseTensor<T>],
445        target_format: SparseFormat,
446    ) -> RusTorchResult<Vec<SparseTensor<T>>> {
447        let mut results = Vec::with_capacity(tensors.len());
448
449        for tensor in tensors {
450            let converted = Self::convert(tensor, target_format)?;
451            results.push(converted);
452        }
453
454        Ok(results)
455    }
456}

483/// Performance benchmarking for sparse operations
484/// スパース演算のパフォーマンスベンチマーク
485pub struct SparseBenchmark<T: Float> {
486    /// Benchmark results storage
487    /// ベンチマーク結果格納
488    pub results: HashMap<String, BenchmarkResult>,
489    _phantom: std::marker::PhantomData<T>,

492/// Individual benchmark result
493/// 個別ベンチマーク結果
494#[derive(Debug, Clone)]
495pub struct BenchmarkResult {
496    /// Operation name
497    /// 演算名
498    pub operation: String,
499    /// Execution time in nanoseconds
500    /// 実行時間（ナノ秒）
501    pub time_ns: u64,
502    /// Memory usage in bytes
503    /// メモリ使用量（バイト）
504    pub memory_bytes: usize,
505    /// Throughput (operations per second)
506    /// スループット（1秒あたりの演算数）
507    pub throughput_ops: f64,
508}

510impl<
511        T: Float
512            + Copy
513            + Zero
514            + One
515            + std::ops::AddAssign
516            + PartialOrd
517            + Sum
518            + num_traits::FromPrimitive
519            + 'static,
520    > SparseBenchmark<T>
521{
522    /// Create new benchmark suite
523    /// 新しいベンチマークスイートを作成
524    pub fn new() -> Self {
525        Self {
526            results: HashMap::new(),
527            _phantom: std::marker::PhantomData,
528        }
529    }
530
531    /// Benchmark sparse matrix-vector multiplication
532    /// スパース行列ベクトル乗算のベンチマーク
533    pub fn benchmark_spmv(
534        &mut self,
535        tensor: &SparseTensor<T>,
536        vector: &Array1<T>,
537        iterations: usize,
538    ) -> RusTorchResult<()>
539    where
540        T: Zero + One + std::ops::AddAssign + num_traits::FromPrimitive,
541    {
542        let start_time = std::time::Instant::now();
543
544        for _ in 0..iterations {
545            let _ = tensor.spmv(vector)?;
546        }
547
548        let elapsed = start_time.elapsed();
549        let time_per_op = elapsed.as_nanos() / iterations as u128;
550
551        let result = BenchmarkResult {
552            operation: "spmv".to_string(),
553            time_ns: time_per_op as u64,
554            memory_bytes: tensor.memory_usage(),
555            throughput_ops: 1_000_000_000.0 / time_per_op as f64,
556        };
557
558        self.results.insert("spmv".to_string(), result);
559        Ok(())
560    }
561
562    /// Compare sparse vs dense operation performance
563    /// スパースvs密演算パフォーマンスを比較
564    pub fn compare_with_dense(
565        &mut self,
566        sparse_tensor: &SparseTensor<T>,
567        dense_equivalent: &Array2<T>,
568        vector: &Array1<T>,
569    ) -> RusTorchResult<f64> {
570        // Benchmark sparse operation
571        self.benchmark_spmv(sparse_tensor, vector, 100)?;
572        let sparse_time = self.results["spmv"].time_ns;
573
574        // Benchmark dense operation
575        let start_time = std::time::Instant::now();
576        for _ in 0..100 {
577            let _ = dense_equivalent.dot(vector);
578        }
579        let dense_time = start_time.elapsed().as_nanos() / 100;
580
581        // Return speedup ratio (> 1.0 means sparse is faster)
582        Ok(dense_time as f64 / sparse_time as f64)
583    }

585    /// Generate comprehensive benchmark report
586    /// 包括的ベンチマークレポートを生成
587    pub fn report(&self) -> String {
588        let mut report = String::from("Sparse Operations Benchmark Report:\n");
589        report.push_str("=====================================\n");
590
591        for (op, result) in &self.results {
592            report.push_str(&format!(
593                "{}: {:.2}μs, {:.1}MB/s throughput\n",
594                op,
595                result.time_ns as f64 / 1000.0,
596                result.throughput_ops / 1_000_000.0
597            ));
598        }
599
600        report
601    }

657            }
658            Err(e) => {
659                println!("CSR conversion failed: {:?}", e);
660                panic!("CSR conversion should work");
661            }
662        }
663    }
