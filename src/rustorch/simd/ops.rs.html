26/// CPU feature detection and optimization info
27/// CPU機能検出と最適化情報
28pub fn get_optimization_info() -> String {
29    let mut info = String::new();
30    info.push_str("SIMD Optimization Status:\n");
31
32    if vectorized::is_avx2_available() {
33        info.push_str("✅ AVX2: Available (8x f32, 4x f64 parallel processing)\n");
34    } else {
35        info.push_str("❌ AVX2: Not available\n");
36    }
37
38    if vectorized::is_sse41_available() {
39        info.push_str("✅ SSE4.1: Available (4x f32 parallel processing)\n");
40    } else {
41        info.push_str("❌ SSE4.1: Not available\n");
42    }
43
44    info
45}
