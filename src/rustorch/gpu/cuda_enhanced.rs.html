15/// Enhanced CUDA matrix operations with cuBLAS optimization
16/// cuBLAS最適化による強化されたCUDA行列演算
17#[cfg(feature = "cuda")]
18pub struct CudaMatrixExecutor {
19    device: CudaDevice,
20    cublas: CudaBlas,
21    device_id: usize,
22    streams: Vec<cudarc::driver::CudaStream>,
23    memory_pool: Arc<Mutex<CudaMemoryPool>>,
24}

27impl CudaMatrixExecutor {
28    /// Create new CUDA matrix executor with cuBLAS
29    /// cuBLASを使用した新しいCUDA行列実行器を作成
30    pub fn new(device_id: usize) -> RusTorchResult<Self> {
31        let device = CudaDevice::new(device_id).map_err(|e| {
32            RusTorchError::tensor_op(format!(
33                "Failed to initialize CUDA device {}: {}",
34                device_id, e
35            ))
36        })?;
37
38        let cublas = CudaBlas::new(device.clone())
39            .map_err(|e| RusTorchError::tensor_op(format!("Failed to initialize cuBLAS: {}", e)))?;
40
41        // Create multiple streams for concurrent execution
42        let mut streams = Vec::new();
43        for i in 0..4 {
44            let stream = device.fork_default_stream().map_err(|e| {
45                RusTorchError::tensor_op(format!("Failed to create CUDA stream {}: {}", i, e))
46            })?;
47            streams.push(stream);
48        }
49
50        let memory_pool = Arc::new(Mutex::new(CudaMemoryPool::new()));
51
52        Ok(Self {
53            device,
54            cublas,
55            device_id,
56            streams,
57            memory_pool,
58        })
59    }

61    /// Optimized matrix multiplication using cuBLAS SGEMM
62    /// cuBLAS SGEMMを使用した最適化された行列乗算
63    pub fn matmul_f32(
64        &self,
65        a: &[f32],
66        b: &[f32],
67        c: &mut [f32],
68        m: usize,
69        n: usize,
70        k: usize,
71        use_tensor_cores: bool,
72    ) -> RusTorchResult<()> {
73        // Validate dimensions
74        if a.len() != m * k || b.len() != k * n || c.len() != m * n {
75            return Err(RusTorchError::shape_mismatch(
76                &[m, k, k, n, m, n],
77                &[a.len(), 0, b.len(), 0, c.len(), 0],
78            ));
79        }
80
81        // Allocate GPU memory
82        let a_gpu = self.device.htod_copy(a.to_vec()).map_err(|e| {
83            RusTorchError::tensor_op(format!("Failed to copy matrix A to GPU: {}", e))
84        })?;
85
86        let b_gpu = self.device.htod_copy(b.to_vec()).map_err(|e| {
87            RusTorchError::tensor_op(format!("Failed to copy matrix B to GPU: {}", e))
88        })?;
89
90        let mut c_gpu = self.device.alloc_zeros::<f32>(m * n).map_err(|e| {
91            RusTorchError::tensor_op(format!("Failed to allocate result matrix: {}", e))
92        })?;
93
94        // Configure cuBLAS for optimal performance
95        if use_tensor_cores {
96            // Use Tensor Core optimized GEMM for compatible hardware (Volta+)
97            self.perform_tensor_core_gemm(&a_gpu, &b_gpu, &mut c_gpu, m, n, k)?;
98        } else {
99            // Standard cuBLAS SGEMM
100            self.perform_standard_gemm(&a_gpu, &b_gpu, &mut c_gpu, m, n, k)?;
101        }
102
103        // Copy result back to host
104        let result = self.device.dtoh_sync_copy(&c_gpu).map_err(|e| {
105            RusTorchError::tensor_op(format!("Failed to copy result from GPU: {}", e))
106        })?;
107
108        c.copy_from_slice(&result);
109        Ok(())
110    }
111
112    /// Tensor Core optimized GEMM for Volta+ architectures
113    /// Volta+アーキテクチャ向けTensor Core最適化GEMM
114    fn perform_tensor_core_gemm(
115        &self,
116        a: &CudaSlice<f32>,
117        b: &CudaSlice<f32>,
118        c: &mut CudaSlice<f32>,
119        m: usize,
120        n: usize,
121        k: usize,
122    ) -> RusTorchResult<()> {
123        // Use mixed precision with Tensor Cores when available
124        // This requires converting f32 to f16, performing computation, then back to f32
125        // For simplicity, fall back to standard GEMM for now
126        self.perform_standard_gemm(a, b, c, m, n, k)
127    }
128
129    /// Standard cuBLAS SGEMM implementation
130    /// 標準cuBLAS SGEMM実装
131    fn perform_standard_gemm(
132        &self,
133        a: &CudaSlice<f32>,
134        b: &CudaSlice<f32>,
135        c: &mut CudaSlice<f32>,
136        m: usize,
137        n: usize,
138        k: usize,
139    ) -> RusTorchResult<()> {
140        // cuBLAS uses column-major order, so we need to transpose
141        // C = alpha * A * B + beta * C
142        // In column-major: C = alpha * B^T * A^T + beta * C
143        let alpha = 1.0f32;
144        let beta = 0.0f32;
145
146        // Use cuBLAS SGEMM
147        unsafe {
148            self.cublas
149                .gemm(
150                    cudarc::cublas::sys::cublasOperation_t::CUBLAS_OP_N, // B is not transposed
151                    cudarc::cublas::sys::cublasOperation_t::CUBLAS_OP_N, // A is not transposed
152                    n as i32, // number of rows of matrix op(B) and C
153                    m as i32, // number of columns of matrix op(A) and C
154                    k as i32, // number of columns of op(B) and rows of op(A)
155                    &alpha,   // scalar multiplier for A*B
156                    b,        // matrix B
157                    n as i32, // leading dimension of B
158                    a,        // matrix A
159                    k as i32, // leading dimension of A
160                    &beta,    // scalar multiplier for C
161                    c,        // matrix C (result)
162                    n as i32, // leading dimension of C
163                )
164                .map_err(|e| RusTorchError::tensor_op(format!("cuBLAS SGEMM failed: {}", e)))?;
165        }
166
167        Ok(())
168    }

170    /// Batch matrix multiplication for multiple matrices
171    /// 複数行列のバッチ行列乗算
172    pub fn batch_matmul_f32(
173        &self,
174        a_batch: &[Vec<f32>],
175        b_batch: &[Vec<f32>],
176        c_batch: &mut [Vec<f32>],
177        m: usize,
178        n: usize,
179        k: usize,
180    ) -> RusTorchResult<()> {
181        if a_batch.len() != b_batch.len() || a_batch.len() != c_batch.len() {
182            return Err(RusTorchError::shape_mismatch(
183                &[a_batch.len()],
184                &[b_batch.len(), c_batch.len()],
185            ));
186        }
187
188        let batch_size = a_batch.len();
189        let stream_count = self.streams.len().min(batch_size);
190
191        // Process batches in parallel using multiple streams
192        for (batch_idx, ((a, b), c)) in a_batch
193            .iter()
194            .zip(b_batch.iter())
195            .zip(c_batch.iter_mut())
196            .enumerate()
197        {
198            let stream_idx = batch_idx % stream_count;
199            let _stream = &self.streams[stream_idx];
200
201            // For now, process synchronously
202            // In production, we'd use asynchronous kernel launches with streams
203            self.matmul_f32(a, b, c, m, n, k, false)?;
204        }
205
206        // Synchronize all streams
207        self.device.synchronize().map_err(|e| {
208            RusTorchError::tensor_op(format!("Failed to synchronize CUDA device: {}", e))
209        })?;
210
211        Ok(())
212    }

214    /// Get device memory information
215    /// デバイスメモリ情報を取得
216    pub fn get_memory_info(&self) -> RusTorchResult<(usize, usize)> {
217        let total = self
218            .device
219            .total_memory()
220            .map_err(|e| RusTorchError::tensor_op(format!("Failed to get total memory: {}", e)))?;
221
222        // Free memory calculation (simplified)
223        let free = total; // In practice, we'd calculate actual free memory
224
225        Ok((free, total))
226    }

228    /// Get compute capability
229    /// 計算能力を取得
230    pub fn get_compute_capability(&self) -> RusTorchResult<(i32, i32)> {
231        // Get major and minor compute capability
232        let major = self.device.get_attribute(
233            cudarc::driver::sys::CUdevice_attribute::CU_DEVICE_ATTRIBUTE_COMPUTE_CAPABILITY_MAJOR,
234        ).map_err(|e| {
235            RusTorchError::tensor_op(format!("Failed to get compute capability major: {}", e))
236        })? as i32;
237
238        let minor = self.device.get_attribute(
239            cudarc::driver::sys::CUdevice_attribute::CU_DEVICE_ATTRIBUTE_COMPUTE_CAPABILITY_MINOR,
240        ).map_err(|e| {
241            RusTorchError::tensor_op(format!("Failed to get compute capability minor: {}", e))
242        })? as i32;
243
244        Ok((major, minor))
245    }

248/// CUDA memory pool for efficient memory management (future implementation)
249/// 効率的なメモリ管理のためのCUDAメモリプール（将来実装）
250#[allow(dead_code)]
251struct CudaMemoryPool {
252    buffers: HashMap<usize, Vec<*mut u8>>,
253}
254
255#[allow(dead_code)]
256impl CudaMemoryPool {
257    fn new() -> Self {
258        Self {
259            buffers: HashMap::new(),
260        }
261    }
262
263    fn get_buffer(&mut self, size: usize) -> Option<*mut u8> {
264        self.buffers.get_mut(&size)?.pop()
265    }
266
267    fn return_buffer(&mut self, ptr: *mut u8, size: usize) {
268        self.buffers.entry(size).or_default().push(ptr);
269    }
270}

338/// Public interface functions for CUDA operations
339/// CUDA演算のためのパブリックインターフェース関数

341/// Execute CUDA matrix multiplication with cuBLAS
342/// cuBLASを使用したCUDA行列乗算を実行

344pub fn cuda_matmul_f32(
345    a: &[f32],
346    b: &[f32],
347    c: &mut [f32],
348    m: usize,
349    n: usize,
350    k: usize,
351    device_id: usize,

353) -> RusTorchResult<()> {

363            "CUDA not available".to_string(),
364        ))
