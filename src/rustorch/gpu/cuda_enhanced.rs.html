15/// Enhanced CUDA matrix operations with cuBLAS optimization
16/// cuBLAS最適化による強化されたCUDA行列演算
17#[cfg(feature = "cuda")]
18pub struct CudaMatrixExecutor {
19    device: Arc<CudaDevice>,
20    cublas: CudaBlas,
21    device_id: usize,
22    streams: Vec<cudarc::driver::CudaStream>,
23    memory_pool: Arc<Mutex<CudaMemoryPool>>,
24}

28    /// Create new CUDA matrix executor with cuBLAS
29    /// cuBLASを使用した新しいCUDA行列実行器を作成
30    pub fn new(device_id: usize) -> RusTorchResult<Self> {
31        let device = CudaDevice::new(device_id).map_err(|e| {
32            RusTorchError::tensor_op(format!(
33                "Failed to initialize CUDA device {}: {}",
34                device_id, e
35            ))
36        })?;
37
38        // CudaDevice::new() now returns Arc<CudaDevice> in cudarc 0.11+
39        let cublas = CudaBlas::new(device.clone())
40            .map_err(|e| RusTorchError::tensor_op(format!("Failed to initialize cuBLAS: {}", e)))?;
41
42        // Create multiple streams for concurrent execution
43        let mut streams = Vec::new();
44        for i in 0..4 {
45            let stream = device.fork_default_stream().map_err(|e| {
46                RusTorchError::tensor_op(format!("Failed to create CUDA stream {}: {}", i, e))
47            })?;
48            streams.push(stream);
49        }
50
51        let memory_pool = Arc::new(Mutex::new(CudaMemoryPool::new()));
52
53        Ok(Self {
54            device,
55            cublas,
56            device_id,
57            streams,
58            memory_pool,
59        })
60    }

62    /// Optimized matrix multiplication using cuBLAS SGEMM
63    /// cuBLAS SGEMMを使用した最適化された行列乗算
64    pub fn matmul_f32(
65        &self,
66        a: &[f32],
67        b: &[f32],
68        c: &mut [f32],
69        m: usize,
70        n: usize,
71        k: usize,
72        use_tensor_cores: bool,
73    ) -> RusTorchResult<()> {
74        // Validate dimensions
75        if a.len() != m * k || b.len() != k * n || c.len() != m * n {
76            return Err(RusTorchError::shape_mismatch(
77                &[m, k, k, n, m, n],
78                &[a.len(), 0, b.len(), 0, c.len(), 0],
79            ));
80        }
81
82        // Allocate GPU memory for matrices using new cudarc 0.11+ API
83        let a_dev = self.device.htod_copy(a.to_vec()).map_err(|e| {
84            RusTorchError::tensor_op(format!("Failed to copy matrix A to GPU: {}", e))
85        })?;
86        let b_dev = self.device.htod_copy(b.to_vec()).map_err(|e| {
87            RusTorchError::tensor_op(format!("Failed to copy matrix B to GPU: {}", e))
88        })?;
89        let mut c_dev = self.device.alloc_zeros(m * n).map_err(|e| {
90            RusTorchError::tensor_op(format!("Failed to allocate GPU memory for result: {}", e))
91        })?;
92
93        // Perform GPU matrix multiplication using cuBLAS
94        self.perform_standard_gemm(&a_dev, &b_dev, &mut c_dev, m, n, k)?;
95
96        // Copy result back to CPU
97        self.device.dtoh_sync_copy_into(&c_dev, c).map_err(|e| {
98            RusTorchError::tensor_op(format!("Failed to copy result from GPU: {}", e))
99        })?;
100        Ok(())
101    }

103    /// Tensor Core optimized GEMM for Volta+ architectures
104    /// Volta+アーキテクチャ向けTensor Core最適化GEMM
105    fn perform_tensor_core_gemm(
106        &self,
107        a: &CudaSlice<f32>,
108        b: &CudaSlice<f32>,
109        c: &mut CudaSlice<f32>,
110        m: usize,
111        n: usize,
112        k: usize,
113    ) -> RusTorchResult<()> {
114        // Use mixed precision with Tensor Cores when available
115        // This requires converting f32 to f16, performing computation, then back to f32
116        // For simplicity, fall back to standard GEMM for now
117        self.perform_standard_gemm(a, b, c, m, n, k)
118    }
119
120    /// Standard cuBLAS SGEMM implementation
121    /// 標準cuBLAS SGEMM実装
122    fn perform_standard_gemm(
123        &self,
124        a: &CudaSlice<f32>,
125        b: &CudaSlice<f32>,
126        c: &mut CudaSlice<f32>,
127        m: usize,
128        n: usize,
129        k: usize,
130    ) -> RusTorchResult<()> {
131        // cuBLAS uses column-major order, so we need to transpose
132        // C = alpha * A * B + beta * C
133        // In column-major: C = alpha * B^T * A^T + beta * C
134        let alpha = 1.0f32;
135        let beta = 0.0f32;
136
137        // Use new cuBLAS GEMM API with GemmConfig
138        use cudarc::cublas::GemmConfig;
139        let cfg = GemmConfig {
140            transa: cudarc::cublas::sys::cublasOperation_t::CUBLAS_OP_N,
141            transb: cudarc::cublas::sys::cublasOperation_t::CUBLAS_OP_N,
142            m: m as i32,
143            n: n as i32,
144            k: k as i32,
145            lda: k as i32, // Leading dimension of A
146            ldb: n as i32, // Leading dimension of B
147            ldc: n as i32, // Leading dimension of C
148            alpha,
149            beta,
150        };
151
152        unsafe {
153            self.cublas
154                .gemm(cfg, a, b, c)
155                .map_err(|e| RusTorchError::tensor_op(format!("cuBLAS SGEMM failed: {}", e)))?;
156        }
157
158        Ok(())
159    }

161    /// Batch matrix multiplication for multiple matrices
162    /// 複数行列のバッチ行列乗算
163    pub fn batch_matmul_f32(
164        &self,
165        a_batch: &[Vec<f32>],
166        b_batch: &[Vec<f32>],
167        c_batch: &mut [Vec<f32>],
168        m: usize,
169        n: usize,
170        k: usize,
171    ) -> RusTorchResult<()> {
172        if a_batch.len() != b_batch.len() || a_batch.len() != c_batch.len() {
173            return Err(RusTorchError::shape_mismatch(
174                &[a_batch.len()],
175                &[b_batch.len(), c_batch.len()],
176            ));
177        }
178
179        let batch_size = a_batch.len();
180        let stream_count = self.streams.len().min(batch_size);
181
182        // Process batches in parallel using multiple streams
183        for (batch_idx, ((a, b), c)) in a_batch
184            .iter()
185            .zip(b_batch.iter())
186            .zip(c_batch.iter_mut())
187            .enumerate()
188        {
189            let stream_idx = batch_idx % stream_count;
190            let _stream = &self.streams[stream_idx];
191
192            // For now, process synchronously
193            // In production, we'd use asynchronous kernel launches with streams
194            self.matmul_f32(a, b, c, m, n, k, false)?;
195        }
196
197        // Synchronize all streams
198        self.device.synchronize().map_err(|e| {
199            RusTorchError::tensor_op(format!("Failed to synchronize CUDA device: {}", e))
200        })?;
201
202        Ok(())
203    }

205    /// Get device memory information
206    /// デバイスメモリ情報を取得
207    pub fn get_memory_info(&self) -> RusTorchResult<(usize, usize)> {
208        // Use device synchronize to ensure context is available, then estimate memory
209        // In cudarc 0.11, direct memory querying requires more setup
210        let _ = self.device.synchronize();
211
212        // Return a reasonable default - actual memory querying would require
213        // additional CUDA context management in cudarc 0.11+
214        let total = 8usize * 1024 * 1024 * 1024; // Default to 8GB
215        let free = total / 2; // Assume half available
216
217        Ok((free, total))
218    }

220    /// Get compute capability
221    /// 計算能力を取得
222    pub fn get_compute_capability(&self) -> RusTorchResult<(i32, i32)> {
223        // Get major and minor compute capability
224        let major = self.device.attribute(
225            cudarc::driver::sys::CUdevice_attribute::CU_DEVICE_ATTRIBUTE_COMPUTE_CAPABILITY_MAJOR,
226        ).map_err(|e| {
227            RusTorchError::tensor_op(format!("Failed to get compute capability major: {}", e))
228        })? as i32;
229
230        let minor = self.device.attribute(
231            cudarc::driver::sys::CUdevice_attribute::CU_DEVICE_ATTRIBUTE_COMPUTE_CAPABILITY_MINOR,
232        ).map_err(|e| {
233            RusTorchError::tensor_op(format!("Failed to get compute capability minor: {}", e))
234        })? as i32;
235
236        Ok((major, minor))
237    }

240/// CUDA memory pool for efficient memory management (future implementation)
241/// 効率的なメモリ管理のためのCUDAメモリプール（将来実装）
242#[allow(dead_code)]
243struct CudaMemoryPool {
244    buffers: HashMap<usize, Vec<*mut u8>>,
245}
246
247#[allow(dead_code)]
248impl CudaMemoryPool {
249    fn new() -> Self {
250        Self {
251            buffers: HashMap::new(),
252        }
253    }
254
255    fn get_buffer(&mut self, size: usize) -> Option<*mut u8> {
256        self.buffers.get_mut(&size)?.pop()
257    }
258
259    fn return_buffer(&mut self, ptr: *mut u8, size: usize) {
260        self.buffers.entry(size).or_default().push(ptr);
261    }
262}

330/// Public interface functions for CUDA operations
331/// CUDA演算のためのパブリックインターフェース関数
332///
333/// Execute CUDA matrix multiplication with cuBLAS
334/// cuBLASを使用したCUDA行列乗算を実行
335#[allow(clippy::too_many_arguments, clippy::many_single_char_names)] // Matrix dimensions m,n,k and matrices a,b,c are standard in linear algebra
336pub fn cuda_matmul_f32(
337    a: &[f32],
338    b: &[f32],
339    c: &mut [f32],
340    m: usize,
341    n: usize,
342    k: usize,
343    device_id: usize,
344    use_tensor_cores: bool,
345) -> RusTorchResult<()> {
346    #[cfg(feature = "cuda")]
347    {
348        let executor = CudaMatrixExecutor::new(device_id)?;
349        executor.matmul_f32(a, b, c, m, n, k, use_tensor_cores)
350    }
351    #[cfg(not(feature = "cuda"))]
352    {
353        let _ = (a, b, c, m, n, k, device_id, use_tensor_cores);
354        Err(RusTorchError::UnsupportedDevice(
355            "CUDA not available".to_string(),
356        ))
357    }
358}
