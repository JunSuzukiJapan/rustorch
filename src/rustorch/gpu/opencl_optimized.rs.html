66    /// Create new OpenCL matrix executor with automatic device selection
67    /// 自動デバイス選択による新しいOpenCL行列実行器を作成
68    pub fn new() -> RusTorchResult<Self> {
69        let platforms = get_platforms().map_err(|e| {
70            RusTorchError::tensor_op(format!("Failed to get OpenCL platforms: {:?}", e))
71        })?;
72
73        if platforms.is_empty() {
74            return Err(RusTorchError::UnsupportedDevice(
75                "No OpenCL platforms available".to_string(),
76            ));
77        }
78
79        // Find best GPU device across all platforms
80        let (device, platform) = Self::select_best_device(&platforms)?;
81
82        // Create context
83        let context = Context::from_device(&device).map_err(|e| {
84            RusTorchError::tensor_op(format!("Failed to create OpenCL context: {:?}", e))
85        })?;
86
87        // Create command queue with profiling
88        let command_queue =
89            CommandQueue::create_default_with_properties(&context, CL_QUEUE_PROFILING_ENABLE, 0)
90                .map_err(|e| {
91                    RusTorchError::tensor_op(format!("Failed to create command queue: {:?}", e))
92                })?;
93
94        // Get device information - need to extract device_id from device
95        let device_info = OpenClDeviceInfo {
96            name: "OpenCL Device".to_string(),
97            vendor: "Unknown".to_string(),
98            compute_units: 1,
99            max_work_group_size: 256,
100            global_mem_size: 1024 * 1024 * 1024, // 1GB default
101            local_mem_size: 64 * 1024,           // 64KB default
102            max_clock_frequency: 1000,           // 1GHz default
103            device_type: "GPU".to_string(),
104        };
105
106        println!(
107            "Selected OpenCL device: {} by {}",
108            device_info.name, device_info.vendor
109        );
110        println!(
111            "Compute units: {}, Max work group size: {}",
112            device_info.compute_units, device_info.max_work_group_size
113        );
114
115        Ok(Self {
116            context,
117            command_queue,
118            device,
119            device_info,
120        })
121    }

123    /// Select best GPU device from available platforms
124    /// 利用可能なプラットフォームから最適なGPUデバイスを選択
125    fn select_best_device(platforms: &[Platform]) -> RusTorchResult<(Device, Platform)> {
126        let mut best_device = None;
127        let mut best_platform = None;
128        let mut best_score = 0f64;
129
130        for platform in platforms {
131            let devices = platform
132                .get_devices(CL_DEVICE_TYPE_GPU)
133                .or_else(|_| platform.get_devices(CL_DEVICE_TYPE_ALL))
134                .map_err(|e| RusTorchError::tensor_op(format!("Failed to get devices: {:?}", e)))?;
135
136            for device_id in devices {
137                let device = Device::new(device_id);
138                let info = Self::get_opencl_device_info_from_id(device_id)?;
139
140                // Score devices based on compute capability
141                let score = Self::score_device(&info);
142
143                if score > best_score {
144                    best_score = score;
145                    best_device = Some(device);
146                    best_platform = Some(platform.clone());
147                }
148            }
149        }
150
151        match (best_device, best_platform) {
152            (Some(device), Some(platform)) => Ok((device, platform)),
153            _ => Err(RusTorchError::UnsupportedDevice(
154                "No suitable OpenCL device found".to_string(),
155            )),
156        }
157    }

202    /// Score device based on compute capability
203    /// 計算能力に基づいてデバイスをスコア化
204    fn score_device(info: &OpenClDeviceInfo) -> f64 {
205        let mut score = 0.0;
206
207        // Base score from compute units and frequency
208        score += info.compute_units as f64 * info.max_clock_frequency as f64 / 1000.0;
209
210        // Bonus for memory size (GB)
211        score += (info.global_mem_size as f64 / 1024.0 / 1024.0 / 1024.0) * 10.0;
212
213        // Vendor-specific bonuses
214        if info.vendor.contains("NVIDIA") {
215            score *= 1.2; // NVIDIA typically has better OpenCL support
216        } else if info.vendor.contains("AMD") {
217            score *= 1.1; // AMD also good for OpenCL
218        }
219
220        score
221    }

223    /// Optimized matrix multiplication with vendor-specific kernels
224    /// ベンダー固有カーネルによる最適化行列乗算
225    pub fn matmul_f32(
226        &mut self,
227        a: &[f32],
228        b: &[f32],
229        c: &mut [f32],
230        m: usize,
231        n: usize,
232        k: usize,
233    ) -> RusTorchResult<()> {
234        // Validate dimensions
235        if a.len() != m * k || b.len() != k * n || c.len() != m * n {
236            return Err(RusTorchError::shape_mismatch(
237                &[m, k, k, n, m, n],
238                &[a.len(), 0, b.len(), 0, c.len(), 0],
239            ));
240        }
241
242        // Select optimal kernel based on matrix size and device
243        let kernel_source = self.generate_matmul_kernel(m, n, k)?;
244        let kernel = self.compile_kernel("matmul_f32", &kernel_source)?;
245
246        // Create buffers
247        let mut buffer_a = unsafe {
248            Buffer::<f32>::create(&self.context, CL_MEM_READ_ONLY, m * k, std::ptr::null_mut())
249        }
250        .map_err(|e| RusTorchError::tensor_op(format!("Failed to create buffer A: {:?}", e)))?;
251
252        let mut buffer_b = unsafe {
253            Buffer::<f32>::create(&self.context, CL_MEM_READ_ONLY, k * n, std::ptr::null_mut())
254        }
255        .map_err(|e| RusTorchError::tensor_op(format!("Failed to create buffer B: {:?}", e)))?;
256
257        let buffer_c = unsafe {
258            Buffer::<f32>::create(
259                &self.context,
260                CL_MEM_WRITE_ONLY,
261                m * n,
262                std::ptr::null_mut(),
263            )
264        }
265        .map_err(|e| RusTorchError::tensor_op(format!("Failed to create buffer C: {:?}", e)))?;
266
267        // Copy data to device
268        unsafe {
269            self.command_queue
270                .enqueue_write_buffer(&mut buffer_a, CL_NON_BLOCKING, 0, a, &[])
271                .map_err(|e| {
272                    RusTorchError::tensor_op(format!("Failed to write buffer A: {:?}", e))
273                })?;
274
275            self.command_queue
276                .enqueue_write_buffer(&mut buffer_b, CL_NON_BLOCKING, 0, b, &[])
277                .map_err(|e| {
278                    RusTorchError::tensor_op(format!("Failed to write buffer B: {:?}", e))
279                })?;
280        }
281
282        // Calculate optimal work group sizes
283        let (global_work_size, local_work_size) = self.calculate_work_sizes(m, n)?;
284
285        // Set kernel arguments and execute
286        unsafe {
287            ExecuteKernel::new(&kernel)
288                .set_arg(&buffer_a)
289                .set_arg(&buffer_b)
290                .set_arg(&buffer_c)
291                .set_arg(&(m as u32))
292                .set_arg(&(n as u32))
293                .set_arg(&(k as u32))
294                .set_global_work_sizes(&global_work_size)
295                .set_local_work_sizes(&local_work_size)
296                .enqueue_nd_range(&self.command_queue)
297                .map_err(|e| {
298                    RusTorchError::tensor_op(format!("Failed to execute kernel: {:?}", e))
299                })?;
300        }
301
302        // Read result back
303        unsafe {
304            self.command_queue
305                .enqueue_read_buffer(&buffer_c, CL_NON_BLOCKING, 0, c, &[])
306                .map_err(|e| RusTorchError::tensor_op(format!("Failed to read result: {:?}", e)))?;
307        }
308
309        Ok(())
310    }

312    /// Generate optimized matrix multiplication kernel based on device and size
313    /// デバイスとサイズに基づく最適化行列乗算カーネルの生成
314    fn generate_matmul_kernel(&self, m: usize, n: usize, k: usize) -> RusTorchResult<String> {
315        let tile_size = self.calculate_optimal_tile_size(m, n, k);
316
317        // Vendor-specific optimizations
318        let kernel_source = if self.device_info.vendor.contains("AMD") {
319            self.generate_amd_optimized_kernel(tile_size)
320        } else if self.device_info.vendor.contains("NVIDIA") {
321            self.generate_nvidia_optimized_kernel(tile_size)
322        } else if self.device_info.vendor.contains("Intel") {
323            self.generate_intel_optimized_kernel(tile_size)
324        } else {
325            self.generate_generic_kernel(tile_size)
326        };
327
328        Ok(kernel_source)
329    }

331    /// Calculate optimal tile size based on device characteristics
332    /// デバイス特性に基づく最適タイルサイズの計算
333    fn calculate_optimal_tile_size(&self, m: usize, n: usize, k: usize) -> usize {
334        let max_work_group = self.device_info.max_work_group_size;
335        let local_mem_size = self.device_info.local_mem_size as usize;
336
337        // Calculate tile size based on local memory constraints
338        let max_tile_from_memory =
339            (((local_mem_size / 2) / (2 * std::mem::size_of::<f32>())) as f64).sqrt() as usize;
340        let max_tile_from_workgroup = (max_work_group as f64).sqrt() as usize;
341
342        let optimal_tile = max_tile_from_memory.min(max_tile_from_workgroup).min(32);
343
344        // Ensure tile size is a power of 2 and at least 8
345        let tile_size = (optimal_tile.next_power_of_two().max(8)).min(32);
346
347        tile_size
348    }

350    /// Generate AMD-optimized kernel
351    /// AMD最適化カーネルの生成
352    fn generate_amd_optimized_kernel(&self, tile_size: usize) -> String {
353        format!(
354            r#"
355__kernel void matmul_f32(
356    __global const float* A,
357    __global const float* B,
358    __global float* C,
359    const unsigned int M,
360    const unsigned int N,
361    const unsigned int K
362) {{
363    // AMD GCN architecture optimizations
364    const int TILE_SIZE = {tile_size};
365    
366    // Get work group and local IDs
367    const int group_x = get_group_id(0);
368    const int group_y = get_group_id(1);
369    const int local_x = get_local_id(0);
370    const int local_y = get_local_id(1);
371    
372    // Shared memory for tiles
373    __local float As[{tile_size}][{tile_size}];
374    __local float Bs[{tile_size}][{tile_size}];
375    
376    // Calculate global position
377    const int global_row = group_y * TILE_SIZE + local_y;
378    const int global_col = group_x * TILE_SIZE + local_x;
379    
380    float sum = 0.0f;
381    
382    // Loop over tiles
383    for (int tile = 0; tile < (K + TILE_SIZE - 1) / TILE_SIZE; tile++) {{
384        // Load tiles into shared memory with bounds checking
385        int a_row = global_row;
386        int a_col = tile * TILE_SIZE + local_x;
387        int b_row = tile * TILE_SIZE + local_y;
388        int b_col = global_col;
389        
390        As[local_y][local_x] = (a_row < M && a_col < K) ? A[a_row * K + a_col] : 0.0f;
391        Bs[local_y][local_x] = (b_row < K && b_col < N) ? B[b_row * N + b_col] : 0.0f;
392        
393        barrier(CLK_LOCAL_MEM_FENCE);
394        
395        // Compute partial sum with loop unrolling for AMD GCN
396        #pragma unroll 4
397        for (int k = 0; k < TILE_SIZE; k++) {{
398            sum = fma(As[local_y][k], Bs[k][local_x], sum);
399        }}
400        
401        barrier(CLK_LOCAL_MEM_FENCE);
402    }}
403    
404    // Write result
405    if (global_row < M && global_col < N) {{
406        C[global_row * N + global_col] = sum;
407    }}
408}}
409"#,
410            tile_size = tile_size
411        )
412    }

587    /// Compile kernel
588    /// カーネルのコンパイル
589    fn compile_kernel(&self, name: &str, source: &str) -> RusTorchResult<Kernel> {
590        // Compile program
591        let program = Program::create_and_build_from_source(&self.context, source, "")
592            .map_err(|e| RusTorchError::tensor_op(format!("Failed to compile kernel: {:?}", e)))?;
593
594        // Create kernel
595        let kernel = Kernel::create(&program, name)
596            .map_err(|e| RusTorchError::tensor_op(format!("Failed to create kernel: {:?}", e)))?;
597
598        Ok(kernel)
599    }
600
601    /// Calculate optimal work group sizes
602    /// 最適ワークグループサイズの計算
603    fn calculate_work_sizes(&self, m: usize, n: usize) -> RusTorchResult<([usize; 2], [usize; 2])> {
604        let tile_size = self.calculate_optimal_tile_size(m, n, m.max(n));
605
606        let global_work_size = [
607            n.div_ceil(tile_size) * tile_size,
608            m.div_ceil(tile_size) * tile_size,
609        ];
610
611        let local_work_size = [tile_size, tile_size];
612
613        Ok((global_work_size, local_work_size))
614    }

616    /// Get device information for display
617    /// 表示用デバイス情報の取得
618    pub fn get_device_info_ref(&self) -> &OpenClDeviceInfo {
619        &self.device_info
620    }
621}
