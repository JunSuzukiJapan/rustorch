50/// GPU backend for managing multiple devices
51pub struct GpuBackend {
52    devices: Vec<Arc<dyn GpuDevice>>,
53}
54
55impl GpuBackend {
56    /// Create new GPU backend
57    pub fn new() -> Self {
58        #[allow(unused_mut)]
59        let mut devices: Vec<Arc<dyn GpuDevice>> = vec![Arc::new(CpuDevice::new())];
60
61        #[cfg(feature = "cuda")]
62        {
63            // Add CUDA devices
64            for device_id in 0..Self::get_cuda_device_count() {
65                if let Ok(device) = CudaDevice::new(device_id) {
66                    devices.push(Arc::new(device));
67                }
68            }
69        }
70
71        #[cfg(feature = "metal")]
72        {
73            // Add Metal device
74            if let Ok(device) = MetalDevice::new() {
75                devices.push(Arc::new(device));
76            }
77        }
78
79        #[cfg(feature = "opencl")]
80        {
81            // Add OpenCL devices
82            for device_id in 0..Self::get_opencl_device_count() {
83                if let Ok(device) = OpenCLDevice::new(0, device_id) {
84                    devices.push(Arc::new(device));
85                }
86            }
87        }
88
89        Self { devices }
90    }
91
92    /// List all available devices
93    pub fn list_devices(&self) -> &[Arc<dyn GpuDevice>] {
94        &self.devices
95    }
96
97    /// Get device by ID
98    pub fn get_device(&self, id: usize) -> Option<&Arc<dyn GpuDevice>> {
99        self.devices.get(id)
100    }
101
102    #[cfg(feature = "cuda")]
103    fn get_cuda_device_count() -> usize {
104        // Try to detect CUDA devices
105        (0..8).filter(|&i| CudaDevice::new(i).is_ok()).count()
106    }
107
108    #[cfg(feature = "opencl")]
109    fn get_opencl_device_count() -> usize {
110        // Try to detect OpenCL devices
111        (0..8).filter(|&i| OpenCLDevice::new(0, i).is_ok()).count()
112    }
113}
114
115impl Default for GpuBackend {
116    fn default() -> Self {
117        Self::new()
118    }

645    /// Get optimal block size for given problem size
646    /// 指定された問題サイズに対する最適ブロックサイズを取得
647    pub fn optimal_block_size(&self, problem_size: usize) -> (u32, u32, u32) {
648        match self.device_type {
649            DeviceType::Cpu => (1, 1, 1),
650            DeviceType::Cuda(_) => {
651                // Simple heuristic for CUDA block size
652                let threads_per_block = if problem_size < 256 {
653                    128
654                } else if problem_size < 1024 {
655                    256
656                } else {
657                    512
658                };
659                (threads_per_block, 1, 1)
660            }
661            DeviceType::Metal(_) => {
662                // Metal threadgroup size
663                let threads_per_group = if problem_size < 256 {
664                    64
665                } else if problem_size < 1024 {
666                    128
667                } else {
668                    256
669                };
670                (threads_per_group, 1, 1)
671            }
672            DeviceType::OpenCL(_) => {
673                // OpenCL work group size
674                (64, 1, 1)
675            }
676        }
677    }

679    /// Get optimal grid size for given problem size and block size
680    /// 指定された問題サイズとブロックサイズに対する最適グリッドサイズを取得
681    pub fn optimal_grid_size(
682        &self,
683        problem_size: usize,
684        block_size: (u32, u32, u32),
685    ) -> (u32, u32, u32) {
686        let total_threads = block_size.0 * block_size.1 * block_size.2;
687        let num_blocks = (problem_size as u32).div_ceil(total_threads).max(1);
688
689        match self.device_type {
690            DeviceType::Cpu => (1, 1, 1),
691            _ => {
692                // Simple 1D grid for now
693                (num_blocks, 1, 1)
694            }
695        }
696    }

714    /// Get memory usage information

716    pub fn memory_info(&self) -> (usize, usize, f32) {
717        let total = self.capabilities.total_memory;
718        let available = self.capabilities.available_memory;
719        let usage_percent = if total > 0 {
720            ((total - available) as f32 / total as f32) * 100.0
721        } else {
722            0.0
723        };
724        (total, available, usage_percent)

780    /// Get best device for operation
781    /// 操作に最適なデバイスを取得
782    pub fn best_device_for_operation(&self, operation: &str, data_size: usize) -> DeviceType {
783        let mut best_device = DeviceType::Cpu;
784        let mut best_score = 0.0f32;
785
786        for (device_type, device_info) in &self.devices {
787            if !device_info.supports_operation(operation) {
788                continue;
789            }
790
791            let mut score = match device_type {
792                DeviceType::Cpu => 1.0,
793                DeviceType::Cuda(_) => 10.0,
794                DeviceType::Metal(_) => 8.0,
795                DeviceType::OpenCL(_) => 6.0,
796            };
797
798            // Adjust score based on data size
799            if data_size < 1000 {
800                // Small data might be better on CPU due to transfer overhead
801                if matches!(device_type, DeviceType::Cpu) {
802                    score *= 2.0;
803                } else {
804                    score *= 0.5;
805                }
806            }
807
808            // Adjust score based on memory availability
809            let (_, available_memory, _) = device_info.memory_info();
810            if available_memory > 0 {
811                let memory_score = (available_memory as f32 / (1024.0 * 1024.0 * 1024.0)).min(10.0);
812                score *= 1.0 + memory_score * 0.1;
813            }
814
815            if score > best_score {
816                best_score = score;
817                best_device = *device_type;
818            }
819        }
820
821        best_device
822    }

874    #[test]
875    fn test_best_device_selection() {
876        let registry = DeviceRegistry::new();
877
878        // Debug: Check registered devices
879        let devices = registry.list_devices();
880        println!("Registered devices: {:?}", devices);
881
882        // Small data - check what's actually selected
883        let device = registry.best_device_for_operation("matmul", 100);
884        println!("Selected device for small operation: {:?}", device);
885
886        // On macOS with Metal, Metal might still be preferred even for small data
887        #[cfg(all(target_os = "macos", feature = "metal"))]
888        {
889            if devices.contains(&DeviceType::Metal(0)) {
890                // Metal is available, accept either CPU or Metal for small data
891                assert!(device == DeviceType::Cpu || device == DeviceType::Metal(0));
892            } else {
893                assert_eq!(device, DeviceType::Cpu);
894            }
895        }
896
897        #[cfg(not(all(target_os = "macos", feature = "metal")))]
898        {
899            assert_eq!(device, DeviceType::Cpu);
900        }
901
902        // Large data should prefer GPU if available
903        let device = registry.best_device_for_operation("matmul", 1000000);
904        println!("Selected device for large operation: {:?}", device);
905
906        // Check if Metal is available on macOS
907        #[cfg(all(target_os = "macos", feature = "metal"))]
908        {
909            // On macOS with Metal feature, if Metal is registered, use it
910            if devices.contains(&DeviceType::Metal(0)) {
911                assert_eq!(device, DeviceType::Metal(0));
912            } else {
913                // Metal not available, should fall back to CPU
914                assert_eq!(device, DeviceType::Cpu);
915            }
916        }
917
918        #[cfg(not(all(target_os = "macos", feature = "metal")))]
919        {
920            // Without Metal, should fall back to CPU
921            assert_eq!(device, DeviceType::Cpu);
922        }
923    }
