1//! # GPU Acceleration Module
2//! RusTorchのGPU加速モジュール
3//!
4//! This module provides comprehensive GPU acceleration support for RusTorch,
5//! including device management, memory allocation, and kernel execution
6//! across multiple GPU backends (CUDA, Metal, OpenCL).
7//!
8//! ## Supported GPU Backends
9//!
10//! - **CUDA**: NVIDIA GPU acceleration with cuBLAS/cuDNN integration
11//! - **Metal**: Apple Silicon GPU acceleration with Metal Performance Shaders
12//! - **OpenCL**: Cross-platform GPU acceleration for AMD/Intel/NVIDIA
13//!
14//! ## Core Components
15//!
16//! - `DeviceType`: GPU device type enumeration (CUDA, Metal, OpenCL)
17//! - `GpuContext`: GPU context management and device state
18//! - `DeviceManager`: Global device management and selection
19//! - `GpuMemoryPool`: Efficient GPU memory allocation and pooling
20//! - `cuda_kernels`: CUDA-specific kernel implementations
21//! - `metal_kernels`: Metal-specific kernel implementations  
22//! - `opencl_kernels`: OpenCL-specific kernel implementations
23//!
24//! ## Key Features
25//!
26//! - **Automatic Device Detection**: Discovers available GPU devices at runtime
27//! - **Intelligent Device Selection**: Chooses optimal device based on workload
28//! - **Memory Management**: Efficient GPU memory allocation with pooling
29//! - **Error Handling**: Robust error handling with automatic fallback
30//! - **Cross-Platform**: Unified API across different GPU vendors
31//!
32//! ## Usage Examples
33//!
34//! ### Device Management
35//!
36//! ```no_run
37//! use rustorch::gpu::get_device_manager;
38//! // Example GPU usage (implementation dependent on backend)
39//!
40//! // Get available devices
41//! let manager = get_device_manager();
42//! let devices = manager.available_devices();
43//! println!("Available devices: {:?}", devices);
44//!
45//! // Check current device
46//! let current = manager.current_device();
47//! println!("Current device: {:?}", current);
48//! # Ok::<(), Box<dyn std::error::Error>>(())
49//! ```
50//!
51//! ### GPU Context Creation
52//!
53//! ```rust
54//! use rustorch::gpu::{GpuContext, DeviceType};
55//!
56//! // Create GPU context - will fallback to CPU if CUDA unavailable
57//! let context_result = GpuContext::new(DeviceType::Cuda(0));
58//! if let Ok(context) = context_result {
59//!     println!("Created context for device: {:?}", context.device());
60//!     // GPU context created successfully
61//! } else {
62//!     println!("CUDA not available, using CPU fallback");
63//! }
64//! # Ok::<(), Box<dyn std::error::Error>>(())
65//! ```
66//!
67//! ### Memory Pool Operations
68//!
69//! ```rust
70//! use rustorch::gpu::{DeviceType, memory::GpuMemoryPool};
71//!
72//! // Try to create GPU memory pool - fallback if CUDA unavailable
73//! let pool_result = GpuMemoryPool::new(DeviceType::Cuda(0), 1024 * 1024 * 100);
74//! if let Ok(mut pool) = pool_result {
75//!     if let Ok(buffer) = pool.allocate(1024 * 1024) {
76//!         println!("Allocated buffer size: {}", buffer.size);
77//!     }
78//! } else {
79//!     println!("GPU memory pool unavailable, using system memory");
80//! }
81//! # Ok::<(), Box<dyn std::error::Error>>(())
82//! ```
83//!
84//! ## Feature Flags
85//!
86//! GPU backends are controlled by Cargo feature flags:
87//!
88//! ```toml
89//! [dependencies]
90//! rustorch = { version = "0.1", features = ["cuda", "metal", "opencl"] }
91//! ```
92//!
93//! - `cuda`: Enable NVIDIA CUDA support
94//! - `metal`: Enable Apple Metal support
95//! - `opencl`: Enable OpenCL support
96//!
97//! ## Error Handling
98//!
99//! The GPU module provides comprehensive error handling through `GpuError`:
100//!
101//! - **DeviceNotFound**: Requested GPU device is not available
102//! - **OutOfMemory**: Insufficient GPU memory for operation
103//! - **KernelLaunchFailed**: GPU kernel execution failed
104//! - **DriverError**: GPU driver or runtime error
105//! - **UnsupportedOperation**: Operation not supported on current device
106//!
107//! ## Performance Considerations
108//!
109//! - **Memory Coalescing**: Optimize memory access patterns for GPU efficiency
110//! - **Kernel Occupancy**: Balance thread blocks for optimal GPU utilization
111//! - **Memory Bandwidth**: Minimize GPU-CPU data transfers
112//! - **Asynchronous Execution**: Use streams for overlapping computation and transfer

306impl DeviceCapability {
307    /// Check if device supports specific operation
308    /// デバイスが特定の演算をサポートするかチェック
309    pub fn supports_operation(&self, op_type: &OpType) -> bool {
310        self.supported_operations.contains(op_type)
311    }
312
313    /// Get CoreML capability
314    /// CoreML能力を取得
315    #[cfg(feature = "coreml")]
316    pub fn coreml_capability() -> Self {
317        let mut supported_ops = std::collections::HashSet::new();
318        supported_ops.insert(OpType::LinearAlgebra);
319        supported_ops.insert(OpType::Convolution);
320        supported_ops.insert(OpType::Activation);
321        supported_ops.insert(OpType::Reduction);
322        supported_ops.insert(OpType::Normalization);
323
324        Self {
325            device_type: DeviceType::CoreML(0),
326            supports_f16: true,
327            supports_f32: true,
328            supports_f64: false,         // CoreML limitation
329            supports_complex: false,     // CoreML limitation
330            supports_distributed: false, // CoreML limitation
331            max_memory_gb: 8.0,          // Typical Apple Silicon unified memory
332            supported_operations: supported_ops,
333        }
334    }
335}

337#[cfg(feature = "mac-hybrid")]
338impl DeviceType {
339    /// Intelligent device selection for Mac hybrid feature
340    /// Mac ハイブリッドフィーチャー用インテリジェント・デバイス選択
341    pub fn select_best_for_operation(op_type: &OpType, tensor_size: Option<usize>) -> DeviceType {
342        use crate::backends::DeviceManager;
343        use crate::gpu::metal_kernels::MetalKernelExecutor;
344
345        // Check if both backends are available
346        let coreml_available = DeviceManager::is_coreml_available();
347        let metal_available = MetalKernelExecutor::new().is_ok();
348
349        // If only one backend is available, use it
350        if coreml_available && !metal_available {
351            return DeviceType::CoreML(0);
352        }
353        if !coreml_available && metal_available {
354            return DeviceType::Metal(0);
355        }
356        if !coreml_available && !metal_available {
357            // mac-hybrid feature should not fall back to CPU!
358            // This is a configuration error - user enabled mac-hybrid but no hardware available
359            panic!("mac-hybrid feature enabled but neither CoreML nor Metal available. Check system configuration.");
360        }
361
362        // Both available: intelligent selection based on operation type
363        match op_type {
364            // CoreML preferred for optimized ML operations
365            OpType::Convolution | OpType::Activation if tensor_size.unwrap_or(0) > 1000 => {
366                DeviceType::CoreML(0)
367            }
368            // Metal preferred for custom computations and large matrices
369            OpType::LinearAlgebra if tensor_size.unwrap_or(0) > 10000 => DeviceType::Metal(0),
370            // CoreML unsupported operations → Metal
371            OpType::ComplexMath
372            | OpType::Distribution
373            | OpType::CustomKernel
374            | OpType::DistributedOps => DeviceType::Metal(0),
375            // Default: prefer CoreML for power efficiency
376            _ => DeviceType::CoreML(0),
377        }
378    }
379}

381impl Default for DeviceType {
382    fn default() -> Self {
383        // Auto-select best available device
384        #[cfg(feature = "coreml")]
385        if crate::backends::DeviceManager::is_coreml_available() {
386            return DeviceType::CoreML(0);
387        }
388
389        #[cfg(feature = "cuda")]
390        if crate::backends::DeviceManager::is_cuda_available() {
391            return DeviceType::Cuda(0);
392        }
393
394        #[cfg(feature = "metal")]
395        {
396            use crate::gpu::metal_kernels::MetalKernelExecutor;
397            if MetalKernelExecutor::new().is_ok() {
398                return DeviceType::Metal(0);
399            }
400        }
401
402        DeviceType::Cpu
403    }
404}

429impl DeviceType {
430    /// Check if the device is available
431    /// デバイスが利用可能かチェック
432    pub fn is_available(&self) -> bool {
433        match self {
434            DeviceType::Cpu => true,
435            DeviceType::Cuda(_) => {
436                #[cfg(feature = "cuda")]
437                {
438                    use crate::gpu::cuda_kernels::CudaKernelExecutor;
439                    CudaKernelExecutor::new(0).is_ok()
440                }
441                #[cfg(not(feature = "cuda"))]
442                false
443            }
444            DeviceType::Metal(_) => {
445                #[cfg(feature = "metal")]
446                {
447                    use crate::gpu::metal_kernels::MetalKernelExecutor;
448                    MetalKernelExecutor::new().is_ok()
449                }
450                #[cfg(not(feature = "metal"))]
451                false
452            }
453            DeviceType::OpenCL(_) => {
454                #[cfg(feature = "opencl")]
455                {
456                    use crate::gpu::opencl_kernels::OpenClKernelExecutor;
457                    OpenClKernelExecutor::new(0).is_ok()
458                }
459                #[cfg(not(feature = "opencl"))]
460                false
461            }
462            #[cfg(feature = "coreml")]
463            DeviceType::CoreML(_) => {
464                // Check if CoreML is available on this platform
465                cfg!(target_os = "macos")
466            }
467            #[cfg(feature = "coreml-hybrid")]
468            DeviceType::CoreMLHybrid { fallback_gpu, .. } => {
469                // Check if CoreML is available, or if fallback GPU is available
470                cfg!(target_os = "macos")
471                    || fallback_gpu.map_or(false, |gpu| match gpu {
472                        GpuDevice::Cuda(id) => DeviceType::Cuda(id).is_available(),
473                        GpuDevice::Metal(id) => DeviceType::Metal(id).is_available(),
474                        GpuDevice::OpenCL(id) => DeviceType::OpenCL(id).is_available(),
475                    })
476            }
477            #[cfg(feature = "mac-hybrid")]
478            DeviceType::MacHybrid => {
479                // MacHybrid is available if either Metal or CoreML is available
480                cfg!(target_os = "macos")
481                    && (DeviceType::Metal(0).is_available() || DeviceType::CoreML(0).is_available())
482            }
483            DeviceType::Auto => true, // Auto always "available" - selects best device
484        }
485    }
486}

513            DeviceType::Cpu => Ok(GpuContext {
514                device,
515                memory_pool_size: 0,

518            DeviceType::Cuda(_) => {
519                #[cfg(feature = "cuda")]

521                    // CUDA initialization would go here
522                    Ok(GpuContext {
523                        device,
524                        memory_pool_size: 1024 * 1024 * 1024, // 1GB default
525                        stream_count: 4,
526                    })
527                }
528                #[cfg(not(feature = "cuda"))]
529                {
530                    Err(crate::error::RusTorchError::gpu(
531                        "CUDA support not compiled",
532                    ))
533                }
534            }
535            DeviceType::Metal(_) => {
536                #[cfg(feature = "metal")]
537                {
538                    // Metal initialization would go here
539                    Ok(GpuContext {
540                        device,
541                        memory_pool_size: 512 * 1024 * 1024, // 512MB default
542                        stream_count: 2,
543                    })
544                }
545                #[cfg(not(feature = "metal"))]
546                {
547                    Err(crate::error::RusTorchError::gpu(
548                        "Metal support not compiled",
549                    ))
550                }
551            }
552            DeviceType::OpenCL(_) => {
553                #[cfg(feature = "opencl")]
554                {
555                    // OpenCL initialization would go here
556                    Ok(GpuContext {
557                        device,
558                        memory_pool_size: 256 * 1024 * 1024, // 256MB default
559                        stream_count: 2,
560                    })
561                }
562                #[cfg(not(feature = "opencl"))]
563                {
564                    Err(crate::error::RusTorchError::gpu(
565                        "OpenCL support not compiled",
566                    ))
567                }
568            }
569            #[cfg(feature = "coreml")]
570            DeviceType::CoreML(_) => {
571                Ok(GpuContext {
572                    device,
573                    memory_pool_size: 1024 * 1024 * 1024, // 1GB default
574                    stream_count: 1,
575                })
576            }
577            #[cfg(feature = "coreml-hybrid")]
578            DeviceType::CoreMLHybrid { .. } => {
579                Ok(GpuContext {
580                    device,
581                    memory_pool_size: 1024 * 1024 * 1024, // 1GB default
582                    stream_count: 2,
583                })
584            }
585            #[cfg(feature = "mac-hybrid")]
586            DeviceType::MacHybrid => {
587                Ok(GpuContext {
588                    device,
589                    memory_pool_size: 1024 * 1024 * 1024, // 1GB default for hybrid
590                    stream_count: 4, // Support multiple streams for hybrid operations
591                })
592            }
593            DeviceType::Auto => {
594                // Auto-select best available device
595                let best_device = DeviceType::default();
596                Self::new(best_device)
597            }
598        }
599    }

640/// GPU device manager
641/// GPUデバイスマネージャー
642pub struct DeviceManager {
643    contexts: Vec<GpuContext>,
644    current_device: usize,
645}
646
647impl DeviceManager {
648    /// Create a new device manager
649    /// 新しいデバイスマネージャーを作成
650    pub fn new() -> Self {
651        let mut contexts = Vec::new();
652
653        // Always add CPU context
654        if let Ok(cpu_context) = GpuContext::new(DeviceType::Cpu) {
655            contexts.push(cpu_context);
656        }
657
658        // Try to add GPU contexts
659        #[cfg(feature = "cuda")]
660        {
661            for device_id in 0..Self::get_cuda_device_count() {
662                if let Ok(cuda_context) = GpuContext::new(DeviceType::Cuda(device_id)) {
663                    contexts.push(cuda_context);
664                }
665            }
666        }
667
668        #[cfg(feature = "metal")]
669        {
670            if let Ok(metal_context) = GpuContext::new(DeviceType::Metal(0)) {
671                contexts.push(metal_context);
672            }
673        }
674
675        DeviceManager {
676            contexts,
677            current_device: 0,
678        }
679    }
680
681    /// Get available devices
682    /// 利用可能なデバイスを取得
683    pub fn available_devices(&self) -> Vec<DeviceType> {
684        self.contexts.iter().map(|ctx| ctx.device()).collect()
685    }
686
687    /// Set current device
688    /// 現在のデバイスを設定
689    pub fn set_device(&mut self, device: DeviceType) -> crate::error::RusTorchResult<()> {
690        if let Some(index) = self.contexts.iter().position(|ctx| ctx.device() == device) {
691            self.current_device = index;
692            Ok(())
693        } else {
694            Err(crate::error::RusTorchError::device_not_available(
695                device.to_string(),
696            ))
697        }
698    }
699
700    /// Get current device
701    /// 現在のデバイスを取得
702    pub fn current_device(&self) -> DeviceType {
703        self.contexts[self.current_device].device()
704    }
705
706    /// Get current context
707    /// 現在のコンテキストを取得
708    pub fn current_context(&self) -> &GpuContext {
709        &self.contexts[self.current_device]
710    }
711
712    /// Check if CUDA is available
713    /// CUDAが利用可能かチェック
714    pub fn is_cuda_available() -> bool {
715        #[cfg(feature = "cuda")]
716        {
717            Self::get_cuda_device_count() > 0
718        }
719        #[cfg(not(feature = "cuda"))]
720        {
721            false
722        }
723    }
724
725    /// Check if Metal is available
726    /// Metalが利用可能かチェック
727    pub fn is_metal_available() -> bool {
728        #[cfg(feature = "metal")]
729        {
730            use crate::gpu::metal_kernels::MetalKernelExecutor;
731            cfg!(target_os = "macos") && MetalKernelExecutor::new().is_ok()
732        }
733        #[cfg(not(feature = "metal"))]
734        {
735            false
736        }
737    }
738
739    #[cfg(feature = "cuda")]
740    fn get_cuda_device_count() -> usize {
741        use crate::gpu::cuda_kernels::CudaKernelExecutor;
742        // Try to create CUDA executors for devices 0-7 and count successful ones
743        (0..8)
744            .filter(|&i| CudaKernelExecutor::new(i).is_ok())
745            .count()
746    }
747}

750    fn default() -> Self {
751        Self::new()
752    }

755/// Global device manager instance
756/// グローバルデバイスマネージャーインスタンス
757static mut DEVICE_MANAGER: Option<DeviceManager> = None;
758static DEVICE_MANAGER_INIT: std::sync::Once = std::sync::Once::new();
759
760/// Get the global device manager
761/// グローバルデバイスマネージャーを取得
762pub fn get_device_manager() -> &'static mut DeviceManager {
763    unsafe {
764        DEVICE_MANAGER_INIT.call_once(|| {
765            DEVICE_MANAGER = Some(DeviceManager::new());
766        });
767        #[allow(static_mut_refs)]
768        {
769            DEVICE_MANAGER.as_mut().unwrap()
770        }
771    }
772}
773
774/// Set the current device globally
775/// グローバルに現在のデバイスを設定
776pub fn set_device(device: DeviceType) -> crate::error::RusTorchResult<()> {
777    get_device_manager().set_device(device)
778}

780/// Get the current device
781/// 現在のデバイスを取得
782pub fn current_device() -> DeviceType {
783    get_device_manager().current_device()
784}

786/// Check if any GPU is available
787/// 何らかのGPUが利用可能かチェック
788pub fn is_gpu_available() -> bool {
789    DeviceManager::is_cuda_available() || DeviceManager::is_metal_available()
790}
