1//! # GPU Acceleration Module
2//! RusTorchのGPU加速モジュール
3//!
4//! This module provides comprehensive GPU acceleration support for RusTorch,
5//! including device management, memory allocation, and kernel execution
6//! across multiple GPU backends (CUDA, Metal, OpenCL).
7//!
8//! ## Supported GPU Backends
9//!
10//! - **CUDA**: NVIDIA GPU acceleration with cuBLAS/cuDNN integration
11//! - **Metal**: Apple Silicon GPU acceleration with Metal Performance Shaders
12//! - **OpenCL**: Cross-platform GPU acceleration for AMD/Intel/NVIDIA
13//!
14//! ## Core Components
15//!
16//! - `DeviceType`: GPU device type enumeration (CUDA, Metal, OpenCL)
17//! - `GpuContext`: GPU context management and device state
18//! - `DeviceManager`: Global device management and selection
19//! - `GpuMemoryPool`: Efficient GPU memory allocation and pooling
20//! - `cuda_kernels`: CUDA-specific kernel implementations
21//! - `metal_kernels`: Metal-specific kernel implementations  
22//! - `opencl_kernels`: OpenCL-specific kernel implementations
23//!
24//! ## Key Features
25//!
26//! - **Automatic Device Detection**: Discovers available GPU devices at runtime
27//! - **Intelligent Device Selection**: Chooses optimal device based on workload
28//! - **Memory Management**: Efficient GPU memory allocation with pooling
29//! - **Error Handling**: Robust error handling with automatic fallback
30//! - **Cross-Platform**: Unified API across different GPU vendors
31//!
32//! ## Usage Examples
33//!
34//! ### Device Management
35//!
36//! ```no_run
37//! use rustorch::gpu::get_device_manager;
38//! // Example GPU usage (implementation dependent on backend)
39//!
40//! // Get available devices
41//! let manager = get_device_manager();
42//! let devices = manager.available_devices();
43//! println!("Available devices: {:?}", devices);
44//!
45//! // Check current device
46//! let current = manager.current_device();
47//! println!("Current device: {:?}", current);
48//! # Ok::<(), Box<dyn std::error::Error>>(())
49//! ```
50//!
51//! ### GPU Context Creation
52//!
53//! ```rust
54//! use rustorch::gpu::{GpuContext, DeviceType};
55//!
56//! // Create GPU context - will fallback to CPU if CUDA unavailable
57//! let context_result = GpuContext::new(DeviceType::Cuda(0));
58//! if let Ok(context) = context_result {
59//!     println!("Created context for device: {:?}", context.device());
60//!     // GPU context created successfully
61//! } else {
62//!     println!("CUDA not available, using CPU fallback");
63//! }
64//! # Ok::<(), Box<dyn std::error::Error>>(())
65//! ```
66//!
67//! ### Memory Pool Operations
68//!
69//! ```rust
70//! use rustorch::gpu::{DeviceType, memory::GpuMemoryPool};
71//!
72//! // Try to create GPU memory pool - fallback if CUDA unavailable
73//! let pool_result = GpuMemoryPool::new(DeviceType::Cuda(0), 1024 * 1024 * 100);
74//! if let Ok(mut pool) = pool_result {
75//!     if let Ok(buffer) = pool.allocate(1024 * 1024) {
76//!         println!("Allocated buffer size: {}", buffer.size);
77//!     }
78//! } else {
79//!     println!("GPU memory pool unavailable, using system memory");
80//! }
81//! # Ok::<(), Box<dyn std::error::Error>>(())
82//! ```
83//!
84//! ## Feature Flags
85//!
86//! GPU backends are controlled by Cargo feature flags:
87//!
88//! ```toml
89//! [dependencies]
90//! rustorch = { version = "0.1", features = ["cuda", "metal", "opencl"] }
91//! ```
92//!
93//! - `cuda`: Enable NVIDIA CUDA support
94//! - `metal`: Enable Apple Metal support
95//! - `opencl`: Enable OpenCL support
96//!
97//! ## Error Handling
98//!
99//! The GPU module provides comprehensive error handling through `GpuError`:
100//!
101//! - **DeviceNotFound**: Requested GPU device is not available
102//! - **OutOfMemory**: Insufficient GPU memory for operation
103//! - **KernelLaunchFailed**: GPU kernel execution failed
104//! - **DriverError**: GPU driver or runtime error
105//! - **UnsupportedOperation**: Operation not supported on current device
106//!
107//! ## Performance Considerations
108//!
109//! - **Memory Coalescing**: Optimize memory access patterns for GPU efficiency
110//! - **Kernel Occupancy**: Balance thread blocks for optimal GPU utilization
111//! - **Memory Bandwidth**: Minimize GPU-CPU data transfers
112//! - **Asynchronous Execution**: Use streams for overlapping computation and transfer

186/// GPU device types
187/// GPU デバイスタイプ
188#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Default)]
189pub enum DeviceType {
190    /// CPU device
191    /// CPUデバイス
192    #[default]
193    Cpu,
194    /// CUDA GPU device
195    /// CUDA GPUデバイス
196    Cuda(usize), // device_id
197    /// Metal GPU device (for Apple Silicon)
198    /// Metal GPUデバイス（Apple Silicon用）
199    Metal(usize),
200    /// OpenCL GPU device
201    /// OpenCL GPUデバイス
202    OpenCL(usize),
203}
204
205impl fmt::Display for DeviceType {
206    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
207        match self {
208            DeviceType::Cpu => write!(f, "cpu"),
209            DeviceType::Cuda(id) => write!(f, "cuda:{}", id),
210            DeviceType::Metal(id) => write!(f, "metal:{}", id),
211            DeviceType::OpenCL(id) => write!(f, "opencl:{}", id),
212        }
213    }
214}
215
216impl DeviceType {
217    /// Check if the device is available
218    /// デバイスが利用可能かチェック
219    pub fn is_available(&self) -> bool {
220        match self {
221            DeviceType::Cpu => true,
222            DeviceType::Cuda(_) => {
223                #[cfg(feature = "cuda")]
224                {
225                    use crate::gpu::cuda_kernels::CudaKernelExecutor;
226                    CudaKernelExecutor::new(0).is_ok()
227                }
228                #[cfg(not(feature = "cuda"))]
229                false
230            }
231            DeviceType::Metal(_) => {
232                #[cfg(feature = "metal")]
233                {
234                    use crate::gpu::metal_kernels::MetalKernelExecutor;
235                    MetalKernelExecutor::new().is_ok()
236                }
237                #[cfg(not(feature = "metal"))]
238                false
239            }
240            DeviceType::OpenCL(_) => {
241                #[cfg(feature = "opencl")]
242                {
243                    use crate::gpu::opencl_kernels::OpenClKernelExecutor;
244                    OpenClKernelExecutor::new(0).is_ok()
245                }
246                #[cfg(not(feature = "opencl"))]
247                false
248            }
249        }
250    }
251}

265/// GPU operation context
266/// GPU演算コンテキスト
267pub struct GpuContext {
268    device: DeviceType,
269    memory_pool_size: usize,
270    stream_count: usize,
271}
272
273impl GpuContext {
274    /// Create a new GPU context
275    /// 新しいGPUコンテキストを作成
276    pub fn new(device: DeviceType) -> crate::error::RusTorchResult<Self> {
277        match device {
278            DeviceType::Cpu => Ok(GpuContext {
279                device,
280                memory_pool_size: 0,
281                stream_count: 1,
282            }),
283            DeviceType::Cuda(_) => {
284                #[cfg(feature = "cuda")]
285                {
286                    // CUDA initialization would go here
287                    Ok(GpuContext {
288                        device,
289                        memory_pool_size: 1024 * 1024 * 1024, // 1GB default
290                        stream_count: 4,
291                    })
292                }
293                #[cfg(not(feature = "cuda"))]
294                {
295                    Err(crate::error::RusTorchError::gpu(
296                        "CUDA support not compiled",
297                    ))
298                }
299            }
300            DeviceType::Metal(_) => {
301                #[cfg(feature = "metal")]
302                {
303                    // Metal initialization would go here
304                    Ok(GpuContext {
305                        device,
306                        memory_pool_size: 512 * 1024 * 1024, // 512MB default
307                        stream_count: 2,
308                    })
309                }
310                #[cfg(not(feature = "metal"))]
311                {
312                    Err(crate::error::RusTorchError::gpu(
313                        "Metal support not compiled",
314                    ))
315                }
316            }
317            DeviceType::OpenCL(_) => {
318                #[cfg(feature = "opencl")]
319                {
320                    // OpenCL initialization would go here
321                    Ok(GpuContext {
322                        device,
323                        memory_pool_size: 256 * 1024 * 1024, // 256MB default
324                        stream_count: 2,
325                    })
326                }
327                #[cfg(not(feature = "opencl"))]
328                {
329                    Err(crate::error::RusTorchError::gpu(
330                        "OpenCL support not compiled",
331                    ))
332                }
333            }
334        }
335    }
336
337    /// Get the device type
338    /// デバイスタイプを取得
339    pub fn device(&self) -> DeviceType {
340        self.device
341    }
342
343    /// Check if GPU is available
344    /// GPUが利用可能かチェック
345    pub fn is_gpu_available(&self) -> bool {
346        !matches!(self.device, DeviceType::Cpu)
347    }
348
349    /// Get memory pool size
350    /// メモリプールサイズを取得
351    pub fn memory_pool_size(&self) -> usize {
352        self.memory_pool_size
353    }
354
355    /// Get number of streams
356    /// ストリーム数を取得
357    pub fn stream_count(&self) -> usize {
358        self.stream_count
359    }
360}

376/// GPU device manager
377/// GPUデバイスマネージャー
378pub struct DeviceManager {
379    contexts: Vec<GpuContext>,
380    current_device: usize,
381}
382
383impl DeviceManager {
384    /// Create a new device manager
385    /// 新しいデバイスマネージャーを作成
386    pub fn new() -> Self {
387        let mut contexts = Vec::new();
388
389        // Always add CPU context
390        if let Ok(cpu_context) = GpuContext::new(DeviceType::Cpu) {
391            contexts.push(cpu_context);
392        }
393
394        // Try to add GPU contexts
395        #[cfg(feature = "cuda")]
396        {
397            for device_id in 0..Self::get_cuda_device_count() {
398                if let Ok(cuda_context) = GpuContext::new(DeviceType::Cuda(device_id)) {
399                    contexts.push(cuda_context);
400                }
401            }
402        }
403
404        #[cfg(feature = "metal")]
405        {
406            if let Ok(metal_context) = GpuContext::new(DeviceType::Metal(0)) {
407                contexts.push(metal_context);
408            }
409        }
410
411        DeviceManager {
412            contexts,
413            current_device: 0,
414        }
415    }
416
417    /// Get available devices
418    /// 利用可能なデバイスを取得
419    pub fn available_devices(&self) -> Vec<DeviceType> {
420        self.contexts.iter().map(|ctx| ctx.device()).collect()
421    }
422
423    /// Set current device
424    /// 現在のデバイスを設定
425    pub fn set_device(&mut self, device: DeviceType) -> crate::error::RusTorchResult<()> {
426        if let Some(index) = self.contexts.iter().position(|ctx| ctx.device() == device) {
427            self.current_device = index;
428            Ok(())
429        } else {
430            Err(crate::error::RusTorchError::device_not_available(
431                device.to_string(),
432            ))
433        }
434    }
435
436    /// Get current device
437    /// 現在のデバイスを取得
438    pub fn current_device(&self) -> DeviceType {
439        self.contexts[self.current_device].device()
440    }
441
442    /// Get current context
443    /// 現在のコンテキストを取得
444    pub fn current_context(&self) -> &GpuContext {
445        &self.contexts[self.current_device]
446    }
447
448    /// Check if CUDA is available
449    /// CUDAが利用可能かチェック
450    pub fn is_cuda_available() -> bool {
451        #[cfg(feature = "cuda")]
452        {
453            Self::get_cuda_device_count() > 0
454        }
455        #[cfg(not(feature = "cuda"))]
456        {
457            false
458        }
459    }
460
461    /// Check if Metal is available
462    /// Metalが利用可能かチェック
463    pub fn is_metal_available() -> bool {
464        #[cfg(feature = "metal")]
465        {
466            use crate::gpu::metal_kernels::MetalKernelExecutor;
467            cfg!(target_os = "macos") && MetalKernelExecutor::new().is_ok()
468        }
469        #[cfg(not(feature = "metal"))]
470        {
471            false
472        }
473    }
474
475    #[cfg(feature = "cuda")]
476    fn get_cuda_device_count() -> usize {
477        use crate::gpu::cuda_kernels::CudaKernelExecutor;
478        // Try to create CUDA executors for devices 0-7 and count successful ones
479        (0..8)
480            .filter(|&i| CudaKernelExecutor::new(i).is_ok())
481            .count()
482    }
483}

491/// Global device manager instance
492/// グローバルデバイスマネージャーインスタンス
493static mut DEVICE_MANAGER: Option<DeviceManager> = None;
494static DEVICE_MANAGER_INIT: std::sync::Once = std::sync::Once::new();
495
496/// Get the global device manager
497/// グローバルデバイスマネージャーを取得
498pub fn get_device_manager() -> &'static mut DeviceManager {
499    unsafe {
500        DEVICE_MANAGER_INIT.call_once(|| {
501            DEVICE_MANAGER = Some(DeviceManager::new());
502        });
503        #[allow(static_mut_refs)]
504        {
505            DEVICE_MANAGER.as_mut().unwrap()
506        }
507    }
508}
