209    /// Execute training step across all GPUs
210    pub fn training_step(
211        &mut self,
212        model_parameters: &HashMap<String, Tensor<f32>>,
213        gradients: HashMap<String, Vec<Tensor<f32>>>,
214    ) -> RusTorchResult<HashMap<String, Tensor<f32>>> {
215        let step_start = Instant::now();
216
217        // Synchronize gradients across GPUs
218        let sync_start = Instant::now();
219        let synchronized_gradients = self.synchronize_gradients(gradients)?;
220        let sync_time = sync_start.elapsed();
221
222        // Apply gradient compression if configured
223        let compressed_gradients = self.compress_gradients(synchronized_gradients)?;
224
225        // Update parameters using parameter server or local updates
226        let updated_parameters = if let Some(ref mut param_server) = self.parameter_server {
227            param_server.apply_updates(compressed_gradients)?
228        } else {
229            self.apply_local_updates(model_parameters, compressed_gradients)?
230        };
231
232        // Update metrics
233        self.update_metrics(step_start.elapsed(), sync_time);
234
235        // Profile this training step if profiler is enabled
236        if let Some(ref profiler) = self.profiler {
237            let _ = profiler.record_training_step(
238                Duration::from_millis(50), // Mock forward time
239                Duration::from_millis(30), // Mock backward time
240                Duration::from_millis(10), // Mock update time
241                sync_time,
242                100.0, // Mock throughput
243            );
244        }
245
246        Ok(updated_parameters)
247    }

249    /// Synchronize gradients across all GPUs
250    fn synchronize_gradients(
251        &self,
252        gradients: HashMap<String, Vec<Tensor<f32>>>,
253    ) -> RusTorchResult<HashMap<String, Vec<Tensor<f32>>>> {
254        let mut synchronized = HashMap::new();
255
256        for (param_name, grad_tensors) in gradients {
257            // Perform all-reduce on gradient tensors
258            let reduced_grads = self.context.all_reduce(grad_tensors)?;
259            synchronized.insert(param_name, reduced_grads);
260        }
261
262        Ok(synchronized)
263    }

265    /// Apply gradient compression
266    fn compress_gradients(
267        &self,
268        gradients: HashMap<String, Vec<Tensor<f32>>>,
269    ) -> RusTorchResult<HashMap<String, Vec<Tensor<f32>>>> {
270        match self.config.compression {
271            GradientCompression::TopK(k) => {
272                // Top-K sparsification implementation
273                self.apply_topk_compression(gradients, k)
274            }
275            GradientCompression::Random(ratio) => {
276                // Random sparsification implementation
277                self.apply_random_compression(gradients, ratio)
278            }
279            GradientCompression::Quantization(bits) => {
280                // Quantization implementation
281                self.apply_quantization(gradients, bits)
282            }
283            GradientCompression::ErrorFeedback => {
284                // Error feedback compression
285                self.apply_error_feedback_compression(gradients)
286            }
287        }
288    }

290    /// Apply Top-K gradient compression
291    fn apply_topk_compression(
292        &self,
293        gradients: HashMap<String, Vec<Tensor<f32>>>,
294        k: usize,
295    ) -> RusTorchResult<HashMap<String, Vec<Tensor<f32>>>> {
296        let mut compressed = HashMap::new();
297
298        for (param_name, grad_tensors) in gradients {
299            let mut compressed_tensors = Vec::new();
300
301            for tensor in grad_tensors {
302                let compressed_tensor = tensor.clone();
303                let element_count = tensor.numel();
304                let k_actual = std::cmp::min(k, element_count);
305
306                // Get top-k elements by absolute value
307                let mut indices_values: Vec<(usize, f32)> = (0..element_count)
308                    .map(|i| unsafe {
309                        let val = *((tensor.as_ptr() as *const f32).add(i));
310                        (i, val.abs())
311                    })
312                    .collect();
313
314                indices_values.sort_by(|a, b| b.1.partial_cmp(&a.1).unwrap());
315
316                // Zero out all but top-k elements
317                unsafe {
318                    let as_ptr = compressed_tensor.as_ptr() as *mut f32;
319                    for i in 0..element_count {
320                        *as_ptr.add(i) = 0.0;
321                    }
322
323                    for &(idx, _) in indices_values.iter().take(k_actual) {
324                        let original_val = *((tensor.as_ptr() as *const f32).add(idx));
325                        *as_ptr.add(idx) = original_val;
326                    }
327                }
328
329                compressed_tensors.push(compressed_tensor);
330            }
331
332            compressed.insert(param_name, compressed_tensors);
333        }
334
335        Ok(compressed)
336    }

367    /// Apply parameter updates locally
368    fn apply_local_updates(
369        &self,
370        parameters: &HashMap<String, Tensor<f32>>,
371        gradients: HashMap<String, Vec<Tensor<f32>>>,
372    ) -> RusTorchResult<HashMap<String, Tensor<f32>>> {
373        let mut updated_params = HashMap::new();
374        let learning_rate = self.get_current_learning_rate();
375
376        for (param_name, param_tensor) in parameters {
377            if let Some(grad_tensors) = gradients.get(param_name) {
378                // Use first GPU's gradient (they should be synchronized)
379                let grad = &grad_tensors[0];
380                let updated_param = param_tensor.clone();
381
382                // Apply gradient descent: param = param - lr * grad
383                unsafe {
384                    let param_ptr = updated_param.as_ptr() as *mut f32;
385                    let grad_ptr = grad.as_ptr() as *const f32;
386
387                    for i in 0..param_tensor.numel() {
388                        let current_param = *param_ptr.add(i);
389                        let gradient = *grad_ptr.add(i);
390                        *param_ptr.add(i) = current_param - learning_rate * gradient;
391                    }
392                }
393
394                updated_params.insert(param_name.clone(), updated_param);
395            }
396        }
397
398        Ok(updated_params)
399    }

401    /// Get current learning rate based on schedule
402    fn get_current_learning_rate(&self) -> f32 {
403        let metrics = self.metrics.read().unwrap();
404        let current_step = metrics.steps_completed;
405
406        match &self.config.lr_schedule {
407            LearningRateSchedule::Constant(lr) => *lr,
408            LearningRateSchedule::LinearDecay {
409                initial,
410                final_rate,
411                steps,
412            } => {
413                if current_step >= *steps {
414                    *final_rate
415                } else {
416                    let progress = current_step as f32 / *steps as f32;
417                    initial * (1.0 - progress) + final_rate * progress
418                }
419            }
420            LearningRateSchedule::ExponentialDecay {
421                initial,
422                decay_rate,
423                decay_steps,
424            } => {
425                let decay_factor = (current_step / decay_steps) as f32;
426                initial * decay_rate.powf(decay_factor)
427            }
428            LearningRateSchedule::CosineAnnealing {
429                initial,
430                min_rate,
431                period,
432            } => {
433                let progress = (current_step % period) as f32 / *period as f32;
434                let cosine_factor = (1.0 + (std::f32::consts::PI * progress).cos()) / 2.0;
435                min_rate + (initial - min_rate) * cosine_factor
436            }
437            LearningRateSchedule::WarmupDecay {
438                warmup_steps,
439                initial,
440                decay_rate,
441            } => {
442                if current_step < *warmup_steps {
443                    // Linear warmup
444                    initial * (current_step as f32 / *warmup_steps as f32)
445                } else {
446                    // Exponential decay after warmup
447                    let decay_steps = current_step - warmup_steps;
448                    initial * decay_rate.powf(decay_steps as f32 / 1000.0)
449                }
450            }
451        }
452    }

454    /// Update training metrics
455    fn update_metrics(&self, step_time: Duration, sync_time: Duration) {
456        let mut metrics = self.metrics.write().unwrap();
457        metrics.steps_completed += 1;
458        metrics.total_time += step_time;
459        metrics.gradient_sync_time += sync_time;
460
461        // Update average step time with exponential moving average
462        let alpha = 0.1; // Smoothing factor
463        if metrics.steps_completed == 1 {
464            metrics.avg_step_time = step_time;
465        } else {
466            let current_avg_ms = metrics.avg_step_time.as_millis() as f32;
467            let new_step_ms = step_time.as_millis() as f32;
468            let new_avg_ms = alpha * new_step_ms + (1.0 - alpha) * current_avg_ms;
469            metrics.avg_step_time = Duration::from_millis(new_avg_ms as u64);
470        }
471
472        // Update communication overhead
473        metrics.communication_overhead += sync_time;
474
475        // Calculate throughput (simplified)
476        if metrics.total_time.as_secs_f32() > 0.0 {
477            metrics.throughput = metrics.steps_completed as f32 / metrics.total_time.as_secs_f32();
478        }
479    }

537    /// Apply parameter updates from all GPUs
538    pub fn apply_updates(
539        &mut self,
540        gradients: HashMap<String, Vec<Tensor<f32>>>,
541    ) -> RusTorchResult<HashMap<String, Tensor<f32>>> {
542        let mut updated_parameters = HashMap::new();
543
544        for (param_name, grad_tensors) in gradients {
545            // Average gradients from all GPUs
546            if let Some(first_grad) = grad_tensors.first() {
547                let avg_gradient = first_grad.clone();
548                let element_count = first_grad.numel();
549
550                if grad_tensors.len() > 1 {
551                    unsafe {
552                        let avg_ptr = avg_gradient.as_ptr() as *mut f32;
553
554                        // Sum all gradients
555                        for grad in grad_tensors.iter().skip(1) {
556                            let grad_ptr = grad.as_ptr() as *const f32;
557                            for i in 0..element_count {
558                                *avg_ptr.add(i) += *grad_ptr.add(i);
559                            }
560                        }
561
562                        // Average
563                        let num_grads = grad_tensors.len() as f32;
564                        for i in 0..element_count {
565                            *avg_ptr.add(i) /= num_grads;
566                        }
567                    }
568                }
569
570                // Store parameter (simplified - would integrate with actual parameter storage)
571                updated_parameters.insert(param_name, avg_gradient);
572            }
573        }
574
575        Ok(updated_parameters)
576    }

599    /// Get efficiency ratio (useful computation vs total time)
600    pub fn efficiency_ratio(&self) -> f32 {
601        if self.total_time.as_secs_f32() > 0.0 {
602            let compute_time =
603                self.total_time.as_secs_f32() - self.communication_overhead.as_secs_f32();
604            compute_time / self.total_time.as_secs_f32()
605        } else {
606            0.0
607        }
608    }

610    /// Get average GPU utilization
611    pub fn avg_gpu_utilization(&self) -> f32 {
612        if self.gpu_utilization.is_empty() {
613            0.0
614        } else {
615            self.gpu_utilization.values().sum::<f32>() / self.gpu_utilization.len() as f32
616        }
