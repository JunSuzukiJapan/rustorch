16/// Performance metrics for GPU operations

21    pub execution_time: Duration,
22    pub memory_usage: usize,
23    pub device_type: DeviceType,
24    pub tensor_size: usize,
25    pub timestamp: Instant,

30/// Device thermal state monitoring
31/// デバイス熱状態監視
32#[derive(Debug, Clone, PartialEq)]
33pub enum ThermalState {
34    Normal,   // < 60°C
35    Elevated, // 60-80°C
36    Critical, // > 80°C
37    Unknown,
38}

40/// Optimization strategies for different scenarios
41/// 異なるシナリオ用の最適化戦略
42#[derive(Debug, Clone, PartialEq)]
43pub enum OptimizationStrategy {
44    LatencyOptimized,    // Minimize execution time
45    ThroughputOptimized, // Maximize operations per second
46    EnergyOptimized,     // Minimize power consumption
47    MemoryOptimized,     // Minimize memory usage
48    Balanced,            // Balance all factors
49    Auto,                // Automatically choose best strategy
50}

144    /// Record performance metrics for an operation
145    /// 演算のパフォーマンス指標を記録
146    pub fn record_metrics(&self, metrics: PerformanceMetrics) -> RusTorchResult<()> {
147        let mut history = self
148            .metrics_history
149            .write()
150            .map_err(|_| RusTorchError::TensorOp {
151                message: "Failed to acquire metrics lock".to_string(),
152                source: None,
153            })?;
154
155        history.push(metrics.clone());
156
157        // Keep only last 1000 metrics to prevent memory bloat
158        // メモリ肥大化防止のため最新1000件のみ保持
159        let history_len = history.len();
160        if history_len > 1000 {
161            history.drain(0..history_len - 1000);
162        }
163
164        // Update thermal monitoring
165        // 熱監視を更新
166        let device_type = metrics.device_type;
167        let thermal_state = metrics.thermal_state.clone();
168        if let Ok(mut thermal) = self.thermal_monitoring.lock() {
169            thermal.insert(device_type, thermal_state);
170        }
171
172        // Auto-tune if enabled
173        // 自動調整が有効な場合は実行
174        if self.auto_tuning_enabled {
175            self.auto_tune_strategy(&metrics)?;
176        }
177
178        Ok(())
179    }

181    /// Select optimal device for a given operation
182    /// 指定された演算に最適なデバイスを選択
183    pub fn select_optimal_device(
184        &self,
185        operation_type: &str,
186        tensor_size: usize,
187    ) -> RusTorchResult<DeviceType> {
188        let strategy = self
189            .optimization_strategy
190            .read()
191            .map_err(|_| RusTorchError::TensorOp {
192                message: "Failed to acquire strategy lock".to_string(),
193                source: None,
194            })?;
195
196        let profiles = self
197            .device_profiles
198            .read()
199            .map_err(|_| RusTorchError::TensorOp {
200                message: "Failed to acquire profiles lock".to_string(),
201                source: None,
202            })?;
203
204        let thermal = self
205            .thermal_monitoring
206            .lock()
207            .map_err(|_| RusTorchError::TensorOp {
208                message: "Failed to acquire thermal lock".to_string(),
209                source: None,
210            })?;
211
212        let mut best_device = DeviceType::Cpu;
213        let mut best_score = 0.0;
214
215        for (device_type, profile) in profiles.iter() {
216            // Skip devices with critical thermal state
217            // 臨界熱状態のデバイスはスキップ
218            if thermal.get(device_type) == Some(&ThermalState::Critical) {
219                continue;
220            }
221
222            let score = self.calculate_device_score(
223                &strategy,
224                profile,
225                operation_type,
226                tensor_size,
227                thermal.get(device_type).unwrap_or(&ThermalState::Unknown),
228            );
229
230            if score > best_score {
231                best_score = score;
232                best_device = *device_type;
233            }
234        }
235
236        Ok(best_device)
237    }

239    /// Calculate device suitability score based on strategy
240    /// 戦略に基づくデバイス適合性スコアを計算
241    fn calculate_device_score(
242        &self,
243        strategy: &OptimizationStrategy,
244        profile: &DeviceProfile,
245        operation_type: &str,
246        tensor_size: usize,
247        thermal_state: &ThermalState,
248    ) -> f64 {
249        let mut score = 0.0;
250
251        // Base compute score
252        // 基本計算スコア
253        score += profile.compute_score * 0.4;
254
255        // Memory adequacy
256        // メモリ適正性
257        let memory_ratio = tensor_size as f64 / profile.memory_capacity as f64;
258        let memory_score = if memory_ratio > 1.0 {
259            0.0 // Insufficient memory
260        } else {
261            (1.0 - memory_ratio).min(1.0)
262        };
263        score += memory_score * 0.2;
264
265        // Strategy-specific adjustments
266        // 戦略固有の調整
267        match strategy {
268            OptimizationStrategy::LatencyOptimized => {
269                score += profile.compute_score * 0.3;
270                score += (profile.memory_bandwidth / 1000.0).min(1.0) * 0.1;
271            }
272            OptimizationStrategy::ThroughputOptimized => {
273                score += profile.compute_score * 0.2;
274                score += (profile.memory_bandwidth / 1000.0).min(1.0) * 0.2;
275            }
276            OptimizationStrategy::EnergyOptimized => {
277                score += (profile.power_efficiency / 100.0).min(1.0) * 0.4;
278            }
279            OptimizationStrategy::MemoryOptimized => {
280                score += memory_score * 0.3;
281                score += (profile.memory_bandwidth / 1000.0).min(1.0) * 0.1;
282            }
283            OptimizationStrategy::Balanced => {
284                score += profile.compute_score * 0.15;
285                score += memory_score * 0.1;
286                score += (profile.power_efficiency / 100.0).min(1.0) * 0.05;
287            }
288            OptimizationStrategy::Auto => {
289                // Auto strategy adapts based on operation type
290                // 自動戦略は演算タイプに基づいて適応
291                match operation_type {
292                    "matmul" | "conv2d" => {
293                        score += profile.compute_score * 0.3;
294                    }
295                    "activation" | "elementwise" => {
296                        score += (profile.memory_bandwidth / 1000.0).min(1.0) * 0.2;
297                    }
298                    _ => {
299                        score += profile.compute_score * 0.2;
300                    }
301                }
302            }
303        }
304
305        // Thermal penalties
306        // 熱ペナルティ
307        match thermal_state {
308            ThermalState::Normal => {}
309            ThermalState::Elevated => score *= 0.8,
310            ThermalState::Critical => score = 0.0,
311            ThermalState::Unknown => score *= 0.9,
312        }
313
314        score.max(0.0).min(1.0)
315    }

317    /// Auto-tune optimization strategy based on performance history
318    /// パフォーマンス履歴に基づく最適化戦略の自動調整
319    fn auto_tune_strategy(&self, latest_metrics: &PerformanceMetrics) -> RusTorchResult<()> {
320        let history = self
321            .metrics_history
322            .read()
323            .map_err(|_| RusTorchError::TensorOp {
324                message: "Failed to acquire metrics lock".to_string(),
325                source: None,
326            })?;
327
328        if history.len() < 10 {
329            return Ok(()); // Not enough data for tuning
330        }
331
332        // Analyze recent performance trends
333        // 最近のパフォーマンス傾向を分析
334        let recent_metrics: Vec<&PerformanceMetrics> = history.iter().rev().take(10).collect();
335
336        let avg_execution_time: Duration = recent_metrics
337            .iter()
338            .map(|m| m.execution_time)
339            .sum::<Duration>()
340            / recent_metrics.len() as u32;
341
342        let avg_energy: f64 = recent_metrics
343            .iter()
344            .map(|m| m.energy_consumption)
345            .sum::<f64>()
346            / recent_metrics.len() as f64;
347
348        // Adjust strategy based on trends
349        // 傾向に基づく戦略調整
350        let mut strategy =
351            self.optimization_strategy
352                .write()
353                .map_err(|_| RusTorchError::TensorOp {
354                    message: "Failed to acquire strategy lock".to_string(),
355                    source: None,
356                })?;
357
358        if *strategy == OptimizationStrategy::Auto {
359            let new_strategy = if avg_execution_time > Duration::from_millis(100) {
360                OptimizationStrategy::LatencyOptimized
361            } else if avg_energy > 50.0 {
362                OptimizationStrategy::EnergyOptimized
363            } else {
364                OptimizationStrategy::Balanced
365            };
366
367            *strategy = new_strategy;
368        }
369
370        Ok(())
371    }

373    /// Get performance statistics
374    /// パフォーマンス統計を取得
375    pub fn get_performance_stats(&self) -> RusTorchResult<PerformanceStats> {
376        let history = self
377            .metrics_history
378            .read()
379            .map_err(|_| RusTorchError::TensorOp {
380                message: "Failed to acquire metrics lock".to_string(),
381                source: None,
382            })?;
383
384        if history.is_empty() {
385            return Ok(PerformanceStats::default());
386        }
387
388        let total_operations = history.len();
389        let avg_execution_time =
390            history.iter().map(|m| m.execution_time).sum::<Duration>() / total_operations as u32;
391
392        let avg_memory_usage =
393            history.iter().map(|m| m.memory_usage).sum::<usize>() / total_operations;
394
395        let avg_energy_consumption =
396            history.iter().map(|m| m.energy_consumption).sum::<f64>() / total_operations as f64;
397
398        // Device usage distribution
399        // デバイス使用分布
400        let mut device_usage = HashMap::new();
401        for metrics in history.iter() {
402            *device_usage.entry(metrics.device_type).or_insert(0) += 1;
403        }
404
405        let thermal = self
406            .thermal_monitoring
407            .lock()
408            .map_err(|_| RusTorchError::TensorOp {
409                message: "Failed to acquire thermal lock".to_string(),
410                source: None,
411            })?;
412
413        Ok(PerformanceStats {
414            total_operations,
415            avg_execution_time,
416            avg_memory_usage,
417            avg_energy_consumption,
418            device_usage,
419            current_thermal_states: thermal.clone(),
420        })
421    }
