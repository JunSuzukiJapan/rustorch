18/// Metal kernel types
19/// Metalカーネルタイプ
20#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
21pub enum MetalKernelType {
22    /// Element-wise operations (add, mul, etc.)
23    /// 要素ごとの演算（加算、乗算など）
24    ElementWise,
25    /// Matrix multiplication operations
26    /// 行列乗算演算
27    MatMul,
28    /// Reduction operations (sum, mean, etc.)
29    /// リダクション演算（合計、平均など）
30    Reduction,
31    /// Convolution operations
32    /// 畳み込み演算
33    Convolution,
34    /// Batch normalization operations
35    /// バッチ正規化演算
36    BatchNorm,
37}

60/// Metal buffer wrapper
61/// Metalバッファラッパー
62pub struct MetalBuffer<T> {
63    _buffer: *mut c_void,
64    size: usize,
65    _phantom: PhantomData<T>,
66}
67
68impl<T> MetalBuffer<T> {
69    /// Create a new Metal buffer
70    /// 新しいMetalバッファを作成
71    #[cfg(feature = "metal")]
72    pub fn new(size: usize, device: &Device) -> RusTorchResult<Self> {
73        let buffer_size = size * std::mem::size_of::<T>();
74        let buffer = device.new_buffer(buffer_size as u64, MTLResourceOptions::StorageModeShared);
75
76        Ok(Self {
77            _buffer: buffer.as_ptr() as *mut c_void,
78            size,
79            _phantom: PhantomData,
80        })
81    }
82
83    #[cfg(not(feature = "metal"))]
84    /// Create a new Metal buffer with specified size
85    /// 指定されたサイズで新しいMetalバッファを作成
86    pub fn new(_size: usize, _device: &()) -> RusTorchResult<Self> {
87        Err(RusTorchError::UnsupportedDevice(
88            "Metal not available".to_string(),
89        ))
90    }
91
92    /// Copy data from host to Metal buffer
93    /// ホストからMetalバッファへデータをコピー
94    pub fn copy_from_host(&mut self, host_data: &[T]) -> RusTorchResult<()> {
95        if host_data.len() != self.size {
96            return Err(RusTorchError::InvalidOperation(
97                "Size mismatch in host-to-device copy".to_string(),
98            ));
99        }
100
101        #[cfg(feature = "metal")]
102        {
103            unsafe {
104                let buffer_ptr = self._buffer as *mut T;
105                std::ptr::copy_nonoverlapping(host_data.as_ptr(), buffer_ptr, self.size);
106            }
107            Ok(())
108        }
109        #[cfg(not(feature = "metal"))]
110        {
111            Err(RusTorchError::UnsupportedDevice(
112                "Metal not available".to_string(),
113            ))
114        }
115    }
116
117    /// Copy data from Metal buffer to host
118    /// Metalバッファからホストへデータをコピー
119    pub fn copy_to_host(&self, host_data: &mut [T]) -> RusTorchResult<()> {
120        if host_data.len() != self.size {
121            return Err(RusTorchError::InvalidOperation(
122                "Size mismatch in device-to-host copy".to_string(),
123            ));
124        }
125
126        #[cfg(feature = "metal")]
127        {
128            unsafe {
129                let buffer_ptr = self._buffer as *const T;
130                std::ptr::copy_nonoverlapping(buffer_ptr, host_data.as_mut_ptr(), self.size);
131            }
132            Ok(())
133        }
134        #[cfg(not(feature = "metal"))]
135        {
136            Err(RusTorchError::UnsupportedDevice(
137                "Metal not available".to_string(),
138            ))
139        }
140    }

143/// Metal kernel executor for high-performance GPU operations
144/// 高性能GPU演算のためのMetalカーネル実行器
145#[cfg(feature = "metal")]
146pub struct MetalKernelExecutor {
147    device: Device,
148    command_queue: CommandQueue,
149    library: Library,
150    pipeline_states: HashMap<MetalKernelType, ComputePipelineState>,
151}

153#[cfg(feature = "metal")]
154impl MetalKernelExecutor {
155    /// Create a new Metal kernel executor
156    /// 新しいMetalカーネル実行器を作成
157    pub fn new() -> RusTorchResult<Self> {
158        let device = Device::system_default()
159            .ok_or_else(|| RusTorchError::tensor_op("No Metal device available"))?;
160
161        let command_queue = device.new_command_queue();
162
163        // Metal shader library source optimized for AMD Radeon Pro Vega 56
164        let shader_source = r#"
165        #include <metal_stdlib>
166        using namespace metal;
167        
168        // Optimized element-wise addition with vectorization
169        kernel void elementwise_add_f32(
170            device const float* a [[buffer(0)]],
171            device const float* b [[buffer(1)]],
172            device float* result [[buffer(2)]],
173            uint index [[thread_position_in_grid]]
174        ) {
175            result[index] = a[index] + b[index];
176        }
177        
178        // Optimized element-wise multiplication with vectorization
179        kernel void elementwise_mul_f32(
180            device const float* a [[buffer(0)]],
181            device const float* b [[buffer(1)]],
182            device float* result [[buffer(2)]],
183            uint index [[thread_position_in_grid]]
184        ) {
185            result[index] = a[index] * b[index];
186        }
187        
188        // High-performance matrix multiplication optimized for Vega 56 architecture
189        kernel void matrix_multiply_f32(
190            device const float* a [[buffer(0)]],
191            device const float* b [[buffer(1)]],
192            device float* c [[buffer(2)]],
193            constant uint& M [[buffer(3)]],
194            constant uint& N [[buffer(4)]],
195            constant uint& K [[buffer(5)]],
196            uint2 gid [[thread_position_in_grid]]
197        ) {
198            uint row = gid.y;
199            uint col = gid.x;
200            
201            if (row >= M || col >= N) return;
202            
203            float sum = 0.0;
204            // Unroll loop for better performance on Vega architecture
205            for (uint k = 0; k < K; k++) {
206                sum += a[row * K + k] * b[k * N + col];
207            }
208            c[row * N + col] = sum;
209        }
210        
211        // Optimized tiled matrix multiplication for large matrices
212        kernel void tiled_matrix_multiply_f32(
213            device const float* a [[buffer(0)]],
214            device const float* b [[buffer(1)]],
215            device float* c [[buffer(2)]],
216            constant uint& M [[buffer(3)]],
217            constant uint& N [[buffer(4)]],
218            constant uint& K [[buffer(5)]],
219            threadgroup float* tile_a [[threadgroup(0)]],
220            threadgroup float* tile_b [[threadgroup(1)]],
221            uint2 gid [[thread_position_in_grid]],
222            uint2 lid [[thread_position_in_threadgroup]]
223        ) {
224            const uint TILE_SIZE = 16; // Optimized for Vega 56 memory hierarchy
225            
226            uint row = gid.y;
227            uint col = gid.x;
228            uint local_row = lid.y;
229            uint local_col = lid.x;
230            
231            float sum = 0.0;
232            
233            // Tiled computation for better memory bandwidth utilization
234            for (uint tile = 0; tile < (K + TILE_SIZE - 1) / TILE_SIZE; tile++) {
235                // Load tiles into shared memory
236                uint a_idx = row * K + tile * TILE_SIZE + local_col;
237                uint b_idx = (tile * TILE_SIZE + local_row) * N + col;
238                
239                tile_a[local_row * TILE_SIZE + local_col] = 
240                    (row < M && tile * TILE_SIZE + local_col < K) ? a[a_idx] : 0.0;
241                tile_b[local_row * TILE_SIZE + local_col] = 
242                    (tile * TILE_SIZE + local_row < K && col < N) ? b[b_idx] : 0.0;
243                
244                threadgroup_barrier(mem_flags::mem_threadgroup);
245                
246                // Compute partial result
247                for (uint k = 0; k < TILE_SIZE; k++) {
248                    sum += tile_a[local_row * TILE_SIZE + k] * 
249                           tile_b[k * TILE_SIZE + local_col];
250                }
251                
252                threadgroup_barrier(mem_flags::mem_threadgroup);
253            }
254            
255            if (row < M && col < N) {
256                c[row * N + col] = sum;
257            }
258        }
259        
260        kernel void reduce_sum_f32(
261            device const float* input [[buffer(0)]],
262            device float* output [[buffer(1)]],
263            threadgroup float* shared_data [[threadgroup(0)]],
264            uint tid [[thread_index_in_threadgroup]],
265            uint bid [[threadgroup_position_in_grid]],
266            uint block_size [[threads_per_threadgroup]],
267            uint grid_size [[threadgroups_per_grid]]
268        ) {
269            uint index = bid * block_size + tid;
270            
271            // Load data into shared memory
272            shared_data[tid] = (index < grid_size * block_size) ? input[index] : 0.0;
273            threadgroup_barrier(mem_flags::mem_threadgroup);
274            
275            // Reduction in shared memory
276            for (uint s = block_size / 2; s > 0; s >>= 1) {
277                if (tid < s) {
278                    shared_data[tid] += shared_data[tid + s];
279                }
280                threadgroup_barrier(mem_flags::mem_threadgroup);
281            }
282            
283            // Write result for this block
284            if (tid == 0) {
285                output[bid] = shared_data[0];
286            }
287        }
288        "#;
289
290        let library = device
291            .new_library_with_source(shader_source, &CompileOptions::new())
292            .map_err(|e| {
293                RusTorchError::KernelError(format!("Failed to compile Metal shaders: {:?}", e))
294            })?;
295
296        let mut pipeline_states = HashMap::new();
297
298        // Create compute pipeline states
299        let add_function = library
300            .get_function("elementwise_add_f32", None)
301            .map_err(|e| {
302                RusTorchError::KernelError(format!("Failed to get add function: {:?}", e))
303            })?;
304        let add_pipeline = device
305            .new_compute_pipeline_state_with_function(&add_function)
306            .map_err(|e| {
307                RusTorchError::KernelError(format!("Failed to create add pipeline: {:?}", e))
308            })?;
309        pipeline_states.insert(MetalKernelType::ElementWise, add_pipeline);
310
311        let matmul_function = library
312            .get_function("matrix_multiply_f32", None)
313            .map_err(|e| {
314                RusTorchError::KernelError(format!("Failed to get matmul function: {:?}", e))
315            })?;
316        let matmul_pipeline = device
317            .new_compute_pipeline_state_with_function(&matmul_function)
318            .map_err(|e| {
319                RusTorchError::KernelError(format!("Failed to create matmul pipeline: {:?}", e))
320            })?;
321        pipeline_states.insert(MetalKernelType::MatMul, matmul_pipeline);
322
323        let tiled_matmul_function = library
324            .get_function("tiled_matrix_multiply_f32", None)
325            .map_err(|e| {
326                RusTorchError::KernelError(format!("Failed to get tiled matmul function: {:?}", e))
327            })?;
328        let tiled_matmul_pipeline = device
329            .new_compute_pipeline_state_with_function(&tiled_matmul_function)
330            .map_err(|e| {
331                RusTorchError::KernelError(format!(
332                    "Failed to create tiled matmul pipeline: {:?}",
333                    e
334                ))
335            })?;
336        pipeline_states.insert(MetalKernelType::Convolution, tiled_matmul_pipeline);
337
338        let reduce_function = library.get_function("reduce_sum_f32", None).map_err(|e| {
339            RusTorchError::KernelError(format!("Failed to get reduce function: {:?}", e))
340        })?;
341        let reduce_pipeline = device
342            .new_compute_pipeline_state_with_function(&reduce_function)
343            .map_err(|e| {
344                RusTorchError::KernelError(format!("Failed to create reduce pipeline: {:?}", e))
345            })?;
346        pipeline_states.insert(MetalKernelType::Reduction, reduce_pipeline);
347
348        Ok(Self {
349            device,
350            command_queue,
351            library,
352            pipeline_states,
353        })
354    }

356    /// Execute element-wise addition using Metal Performance Shaders
357    /// Metal Performance Shadersを使用して要素ごと加算を実行
358    pub fn elementwise_add_f32(&self, a: &[f32], b: &[f32], c: &mut [f32]) -> RusTorchResult<()> {
359        let size = a.len();
360        if b.len() != size || c.len() != size {
361            return Err(RusTorchError::InvalidOperation(
362                "Array size mismatch in element-wise addition".to_string(),
363            ));
364        }
365
366        // Create Metal buffers
367        let a_buffer = self.device.new_buffer_with_data(
368            a.as_ptr() as *const c_void,
369            (size * std::mem::size_of::<f32>()) as u64,
370            MTLResourceOptions::StorageModeShared,
371        );
372
373        let b_buffer = self.device.new_buffer_with_data(
374            b.as_ptr() as *const c_void,
375            (size * std::mem::size_of::<f32>()) as u64,
376            MTLResourceOptions::StorageModeShared,
377        );
378
379        let c_buffer = self.device.new_buffer(
380            (size * std::mem::size_of::<f32>()) as u64,
381            MTLResourceOptions::StorageModeShared,
382        );
383
384        // Get pipeline state
385        let pipeline_state = self
386            .pipeline_states
387            .get(&MetalKernelType::ElementWise)
388            .ok_or_else(|| {
389                RusTorchError::KernelError("ElementWise pipeline not found".to_string())
390            })?;
391
392        // Create command buffer and encoder
393        let command_buffer = self.command_queue.new_command_buffer();
394        let compute_encoder = command_buffer.new_compute_command_encoder();
395
396        compute_encoder.set_compute_pipeline_state(pipeline_state);
397        compute_encoder.set_buffer(0, Some(&a_buffer), 0);
398        compute_encoder.set_buffer(1, Some(&b_buffer), 0);
399        compute_encoder.set_buffer(2, Some(&c_buffer), 0);
400
401        // Calculate thread configuration
402        let threads_per_threadgroup = MTLSize::new(256, 1, 1);
403        let threadgroups_per_grid = MTLSize::new(((size + 255) / 256) as u64, 1, 1);
404
405        compute_encoder.dispatch_thread_groups(threadgroups_per_grid, threads_per_threadgroup);
406        compute_encoder.end_encoding();
407
408        command_buffer.commit();
409        command_buffer.wait_until_completed();
410
411        // Copy result back to host
412        unsafe {
413            let result_ptr = c_buffer.contents() as *const f32;
414            std::ptr::copy_nonoverlapping(result_ptr, c.as_mut_ptr(), size);
415        }
416
417        Ok(())
418    }

420    /// Execute tiled matrix multiplication using Metal for large matrices
421    /// 大行列に対してMetalを使用してタイル化行列乗算を実行
422    pub fn tiled_matmul_f32(
423        &self,
424        a: &[f32],
425        b: &[f32],
426        c: &mut [f32],
427        m: usize,
428        n: usize,
429        k: usize,
430    ) -> RusTorchResult<()> {
431        const TILE_SIZE: usize = 16; // Optimized for Vega 56
432
433        // Create Metal buffers
434        let a_buffer = self.device.new_buffer_with_data(
435            a.as_ptr() as *const c_void,
436            (m * k * std::mem::size_of::<f32>()) as u64,
437            MTLResourceOptions::StorageModeShared,
438        );
439
440        let b_buffer = self.device.new_buffer_with_data(
441            b.as_ptr() as *const c_void,
442            (k * n * std::mem::size_of::<f32>()) as u64,
443            MTLResourceOptions::StorageModeShared,
444        );
445
446        let c_buffer = self.device.new_buffer(
447            (m * n * std::mem::size_of::<f32>()) as u64,
448            MTLResourceOptions::StorageModeShared,
449        );
450
451        // Create parameter buffers
452        let m_u32 = m as u32;
453        let n_u32 = n as u32;
454        let k_u32 = k as u32;
455        let m_buffer = self.device.new_buffer_with_data(
456            &m_u32 as *const u32 as *const c_void,
457            std::mem::size_of::<u32>() as u64,
458            MTLResourceOptions::StorageModeShared,
459        );
460        let n_buffer = self.device.new_buffer_with_data(
461            &n_u32 as *const u32 as *const c_void,
462            std::mem::size_of::<u32>() as u64,
463            MTLResourceOptions::StorageModeShared,
464        );
465        let k_buffer = self.device.new_buffer_with_data(
466            &k_u32 as *const u32 as *const c_void,
467            std::mem::size_of::<u32>() as u64,
468            MTLResourceOptions::StorageModeShared,
469        );
470
471        // Get tiled pipeline state
472        let pipeline_state = self
473            .pipeline_states
474            .get(&MetalKernelType::Convolution)
475            .ok_or_else(|| {
476                RusTorchError::KernelError("Tiled MatMul pipeline not found".to_string())
477            })?;
478
479        // Create command buffer and encoder
480        let command_buffer = self.command_queue.new_command_buffer();
481        let compute_encoder = command_buffer.new_compute_command_encoder();
482
483        compute_encoder.set_compute_pipeline_state(pipeline_state);
484        compute_encoder.set_buffer(0, Some(&a_buffer), 0);
485        compute_encoder.set_buffer(1, Some(&b_buffer), 0);
486        compute_encoder.set_buffer(2, Some(&c_buffer), 0);
487        compute_encoder.set_buffer(3, Some(&m_buffer), 0);
488        compute_encoder.set_buffer(4, Some(&n_buffer), 0);
489        compute_encoder.set_buffer(5, Some(&k_buffer), 0);
490
491        // Set threadgroup memory for tiles (2 tiles * TILE_SIZE * TILE_SIZE * sizeof(f32))
492        let threadgroup_memory_size = 2 * TILE_SIZE * TILE_SIZE * std::mem::size_of::<f32>();
493        compute_encoder.set_threadgroup_memory_length(0, threadgroup_memory_size as u64);
494        compute_encoder.set_threadgroup_memory_length(1, threadgroup_memory_size as u64);
495
496        // Calculate thread configuration for tiled execution
497        let threads_per_threadgroup = MTLSize::new(TILE_SIZE as u64, TILE_SIZE as u64, 1);
498        let threadgroups_per_grid = MTLSize::new(
499            n.div_ceil(TILE_SIZE) as u64,
500            m.div_ceil(TILE_SIZE) as u64,
501            1,
502        );
503
504        compute_encoder.dispatch_thread_groups(threadgroups_per_grid, threads_per_threadgroup);
505        compute_encoder.end_encoding();
506
507        command_buffer.commit();
508        command_buffer.wait_until_completed();
509
510        // Copy result back to host
511        unsafe {
512            let result_ptr = c_buffer.contents() as *const f32;
513            std::ptr::copy_nonoverlapping(result_ptr, c.as_mut_ptr(), m * n);
514        }
515
516        Ok(())
517    }

519    /// Execute standard matrix multiplication using Metal
520    /// Metalを使用して標準行列乗算を実行
521    pub fn matmul_f32(
522        &self,
523        a: &[f32],
524        b: &[f32],
525        c: &mut [f32],
526        m: usize,
527        n: usize,
528        k: usize,
529    ) -> RusTorchResult<()> {
530        // Use tiled version for large matrices for better performance
531        if m >= 256 || n >= 256 || k >= 256 {
532            return self.tiled_matmul_f32(a, b, c, m, n, k);
533        }
534        // Create Metal buffers
535        let a_buffer = self.device.new_buffer_with_data(
536            a.as_ptr() as *const c_void,
537            (m * k * std::mem::size_of::<f32>()) as u64,
538            MTLResourceOptions::StorageModeShared,
539        );
540
541        let b_buffer = self.device.new_buffer_with_data(
542            b.as_ptr() as *const c_void,
543            (k * n * std::mem::size_of::<f32>()) as u64,
544            MTLResourceOptions::StorageModeShared,
545        );
546
547        let c_buffer = self.device.new_buffer(
548            (m * n * std::mem::size_of::<f32>()) as u64,
549            MTLResourceOptions::StorageModeShared,
550        );
551
552        // Create parameter buffers
553        let m_u32 = m as u32;
554        let n_u32 = n as u32;
555        let k_u32 = k as u32;
556        let m_buffer = self.device.new_buffer_with_data(
557            &m_u32 as *const u32 as *const c_void,
558            std::mem::size_of::<u32>() as u64,
559            MTLResourceOptions::StorageModeShared,
560        );
561        let n_buffer = self.device.new_buffer_with_data(
562            &n_u32 as *const u32 as *const c_void,
563            std::mem::size_of::<u32>() as u64,
564            MTLResourceOptions::StorageModeShared,
565        );
566        let k_buffer = self.device.new_buffer_with_data(
567            &k_u32 as *const u32 as *const c_void,
568            std::mem::size_of::<u32>() as u64,
569            MTLResourceOptions::StorageModeShared,
570        );
571
572        // Get pipeline state
573        let pipeline_state = self
574            .pipeline_states
575            .get(&MetalKernelType::MatMul)
576            .ok_or_else(|| RusTorchError::KernelError("MatMul pipeline not found".to_string()))?;
577
578        // Create command buffer and encoder
579        let command_buffer = self.command_queue.new_command_buffer();
580        let compute_encoder = command_buffer.new_compute_command_encoder();
581
582        compute_encoder.set_compute_pipeline_state(pipeline_state);
583        compute_encoder.set_buffer(0, Some(&a_buffer), 0);
584        compute_encoder.set_buffer(1, Some(&b_buffer), 0);
585        compute_encoder.set_buffer(2, Some(&c_buffer), 0);
586        compute_encoder.set_buffer(3, Some(&m_buffer), 0);
587        compute_encoder.set_buffer(4, Some(&n_buffer), 0);
588        compute_encoder.set_buffer(5, Some(&k_buffer), 0);
589
590        // Calculate thread configuration
591        let threads_per_threadgroup = MTLSize::new(16, 16, 1);
592        let threadgroups_per_grid = MTLSize::new(((n + 15) / 16) as u64, ((m + 15) / 16) as u64, 1);
593
594        compute_encoder.dispatch_thread_groups(threadgroups_per_grid, threads_per_threadgroup);
595        compute_encoder.end_encoding();
596
597        command_buffer.commit();
598        command_buffer.wait_until_completed();
599
600        // Copy result back to host
601        unsafe {
602            let result_ptr = c_buffer.contents() as *const f32;
603            std::ptr::copy_nonoverlapping(result_ptr, c.as_mut_ptr(), m * n);
604        }
605
606        Ok(())
607    }

609    /// Perform matrix multiplication using Metal with result return
610    /// Metalを使用して行列乗算を実行し結果を返す  
611    pub fn matrix_multiply_f32(
612        &self,
613        a: &[f32],
614        b: &[f32],
615        m: usize,
616        n: usize,
617        k: usize,
618    ) -> RusTorchResult<Vec<f32>> {
619        if a.len() != m * k || b.len() != k * n {
620            return Err(RusTorchError::InvalidOperation(
621                "Matrix dimension mismatch".to_string(),
622            ));
623        }
624
625        let mut result = vec![0.0f32; m * n];
626        self.matmul_f32(a, b, &mut result, m, n, k)?;
627        Ok(result)
628    }
629
630    /// Execute reduction operation (sum) using Metal
631    /// Metalを使用してリダクション演算（合計）を実行
632    pub fn reduce_sum_f32(&self, input: &[f32]) -> RusTorchResult<f32> {
633        let size = input.len();
634        let block_size = 256;
635        let grid_size = size.div_ceil(block_size);
636
637        // Create Metal buffers
638        let input_buffer = self.device.new_buffer_with_data(
639            input.as_ptr() as *const c_void,
640            (size * std::mem::size_of::<f32>()) as u64,
641            MTLResourceOptions::StorageModeShared,
642        );
643
644        let output_buffer = self.device.new_buffer(
645            (grid_size * std::mem::size_of::<f32>()) as u64,
646            MTLResourceOptions::StorageModeShared,
647        );
648
649        // Get pipeline state
650        let pipeline_state = self
651            .pipeline_states
652            .get(&MetalKernelType::Reduction)
653            .ok_or_else(|| {
654                RusTorchError::KernelError("Reduction pipeline not found".to_string())
655            })?;
656
657        // Create command buffer and encoder
658        let command_buffer = self.command_queue.new_command_buffer();
659        let compute_encoder = command_buffer.new_compute_command_encoder();
660
661        compute_encoder.set_compute_pipeline_state(pipeline_state);
662        compute_encoder.set_buffer(0, Some(&input_buffer), 0);
663        compute_encoder.set_buffer(1, Some(&output_buffer), 0);
664
665        // Set threadgroup memory
666        compute_encoder
667            .set_threadgroup_memory_length(0, (block_size * std::mem::size_of::<f32>()) as u64);
668
669        // Calculate thread configuration
670        let threads_per_threadgroup = MTLSize::new(block_size as u64, 1, 1);
671        let threadgroups_per_grid = MTLSize::new(grid_size as u64, 1, 1);
672
673        compute_encoder.dispatch_thread_groups(threadgroups_per_grid, threads_per_threadgroup);
674        compute_encoder.end_encoding();
675
676        command_buffer.commit();
677        command_buffer.wait_until_completed();
678
679        // Copy partial results back and sum on CPU
680        let mut partial_results = vec![0.0f32; grid_size];
681        unsafe {
682            let result_ptr = output_buffer.contents() as *const f32;
683            std::ptr::copy_nonoverlapping(result_ptr, partial_results.as_mut_ptr(), grid_size);
684        }
685
686        Ok(partial_results.iter().sum())
687    }
688}
