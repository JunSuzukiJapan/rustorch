289/// Pooling operations
290pub struct GpuPoolingExecutor<T: Float + FromPrimitive + ScalarOperand + 'static> {
291    device_type: super::DeviceType,
292    _phantom: std::marker::PhantomData<T>,
293}
294
295impl<T: Float + FromPrimitive + ScalarOperand + 'static> GpuPoolingExecutor<T> {
296    /// Create new GPU pooling executor
297    pub fn new(device_type: super::DeviceType) -> RusTorchResult<Self> {
298        Ok(Self {
299            device_type,
300            _phantom: std::marker::PhantomData,
301        })
302    }
303
304    /// Perform max pooling
305    pub fn max_pool2d(
306        &self,
307        input: &Tensor<T>,
308        kernel_size: [usize; 2],
309        stride: [usize; 2],
310        padding: [usize; 2],
311    ) -> RusTorchResult<Tensor<T>> {
312        match self.device_type {
313            super::DeviceType::Cpu => self.cpu_max_pool2d(input, kernel_size, stride, padding),
314            _ => {
315                // For now, fall back to CPU
316                self.cpu_max_pool2d(input, kernel_size, stride, padding)
317            }
318        }
319    }
320
321    /// CPU max pooling fallback
322    fn cpu_max_pool2d(
323        &self,
324        input: &Tensor<T>,
325        kernel_size: [usize; 2],
326        stride: [usize; 2],
327        padding: [usize; 2],
328    ) -> RusTorchResult<Tensor<T>> {
329        // Basic max pooling implementation
330        let input_shape = input.shape();
331        if input_shape.len() != 4 {
332            return Err(RusTorchError::gpu("MaxPool2D requires 4D input"));
333        }
334
335        let [batch, channels, height, width] = [
336            input_shape[0],
337            input_shape[1],
338            input_shape[2],
339            input_shape[3],
340        ];
341
342        // Calculate output dimensions (corrected formula)
343        let out_height = (height + 2 * padding[0] - kernel_size[0]) / stride[0] + 1;
344        let out_width = (width + 2 * padding[1] - kernel_size[1]) / stride[1] + 1;
345
346        // Create output tensor
347        let output_shape = vec![batch, channels, out_height, out_width];
348        let mut output_data = vec![T::zero(); output_shape.iter().product()];
349
350        // Perform max pooling (simplified implementation)
351        for b in 0..batch {
352            for c in 0..channels {
353                for oh in 0..out_height {
354                    for ow in 0..out_width {
355                        let mut max_val = T::neg_infinity();
356
357                        for kh in 0..kernel_size[0] {
358                            for kw in 0..kernel_size[1] {
359                                let ih = oh * stride[0] + kh;
360                                let iw = ow * stride[1] + kw;
361
362                                if ih < height && iw < width {
363                                    // Use as_array for safe indexing
364                                    if let Some(val) = input.as_array().get([b, c, ih, iw]) {
365                                        if *val > max_val {
366                                            max_val = *val;
367                                        }
368                                    }
369                                }
370                            }
371                        }
372
373                        let output_idx = ((b * channels + c) * out_height + oh) * out_width + ow;
374                        output_data[output_idx] = max_val;
375                    }
376                }
377            }
378        }
379
380        Ok(Tensor::from_vec(output_data, output_shape))
381    }
382}
