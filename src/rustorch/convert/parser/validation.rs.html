13impl ModelValidator {
14    /// Validate model graph for consistency
15    /// モデルグラフの一貫性を検証
16    pub fn validate_graph(graph: &ModelGraph) -> Result<(), ParsingError> {
17        // Check for cycles
18        Self::check_cycles(graph)?;
19
20        // Check layer compatibility
21        Self::check_layer_compatibility(graph)?;
22
23        // Ensure all referenced layers exist
24        for (from_layer, to_layers) in &graph.connections {
25            if !graph.layers.contains_key(from_layer) {
26                return Err(ParsingError::MissingConnection(format!(
27                    "Source layer '{}' not found",
28                    from_layer
29                )));
30            }
31
32            for to_layer in to_layers {
33                if !graph.layers.contains_key(to_layer) {
34                    return Err(ParsingError::MissingConnection(format!(
35                        "Target layer '{}' not found",
36                        to_layer
37                    )));
38                }
39            }
40        }
41
42        Ok(())
43    }
44
45    /// Check for cycles in the model graph
46    /// モデルグラフの循環をチェック
47    fn check_cycles(graph: &ModelGraph) -> Result<(), ParsingError> {
48        let mut visited = HashSet::new();
49        let mut rec_stack = HashSet::new();
50
51        fn dfs_cycle_check(
52            node: &str,
53            connections: &HashMap<String, Vec<String>>,
54            visited: &mut HashSet<String>,
55            rec_stack: &mut HashSet<String>,
56        ) -> Result<(), String> {
57            visited.insert(node.to_string());
58            rec_stack.insert(node.to_string());
59
60            if let Some(neighbors) = connections.get(node) {
61                for neighbor in neighbors {
62                    if !visited.contains(neighbor) {
63                        dfs_cycle_check(neighbor, connections, visited, rec_stack)?;
64                    } else if rec_stack.contains(neighbor) {
65                        return Err(format!("Cycle detected: {} -> {}", node, neighbor));
66                    }
67                }
68            }
69
70            rec_stack.remove(node);
71            Ok(())
72        }
73
74        for layer in graph.layers.keys() {
75            if !visited.contains(layer) {
76                if let Err(cycle_info) =
77                    dfs_cycle_check(layer, &graph.connections, &mut visited, &mut rec_stack)
78                {
79                    return Err(ParsingError::CircularDependency(cycle_info));
80                }
81            }
82        }
83
84        Ok(())
85    }

87    /// Check layer dimension compatibility
88    /// レイヤー次元の互換性をチェック
89    fn check_layer_compatibility(graph: &ModelGraph) -> Result<(), ParsingError> {
90        for (from_layer, to_layers) in &graph.connections {
91            let from_info = match graph.layers.get(from_layer) {
92                Some(info) => info,
93                None => continue, // Skip missing layer
94            };
95
96            for to_layer in to_layers {
97                let to_info = match graph.layers.get(to_layer) {
98                    Some(info) => info,
99                    None => continue, // Skip missing layer
100                };
101
102                // Check if output shape of from_layer matches input shape of to_layer
103                if let (Some(output_shape), Some(input_shape)) =
104                    (&from_info.output_shape, &to_info.input_shape)
105                {
106                    if !Self::shapes_compatible(output_shape, input_shape) {
107                        return Err(ParsingError::IncompatibleDimensions {
108                            layer1: from_layer.clone(),
109                            layer2: to_layer.clone(),
110                        });
111                    }
112                }
113            }
114        }
115
116        Ok(())
117    }
118
119    /// Check if two shapes are compatible for layer connection
120    /// 2つの形状がレイヤー接続に対して互換性があるかチェック
121    fn shapes_compatible(output_shape: &[usize], input_shape: &[usize]) -> bool {
122        // Simple compatibility check - in practice, this would be more sophisticated
123        // considering transformations like flatten, reshape, etc.
124
125        if output_shape.len() == 1 && input_shape.len() == 1 {
126            // Both are 1D - must match exactly
127            output_shape[0] == input_shape[0]
128        } else if output_shape.len() > 1 && input_shape.len() == 1 {
129            // Output is multi-dimensional, input expects 1D (e.g., after flatten)
130            let output_size: usize = output_shape.iter().product();
131            output_size == input_shape[0]
132        } else {
133            // For now, assume compatible if we can't determine
134            true
135        }
136    }

138    /// Validate that all referenced layers exist
139    /// 参照されるすべてのレイヤーが存在することを検証
140    pub fn validate_layer_references(
141        desc: &super::formats::ArchitectureDescription,
142        layers: &HashMap<String, super::types::LayerInfo>,
143    ) -> Result<(), ParsingError> {
144        let layer_names: HashSet<String> = desc.layers.iter().map(|l| l.name.clone()).collect();
145
146        for connection in &desc.connections {
147            if !layer_names.contains(&connection.from) {
148                return Err(ParsingError::MissingConnection(format!(
149                    "Connection references unknown source layer: {}",
150                    connection.from
151                )));
152            }
153
154            if !layer_names.contains(&connection.to) {
155                return Err(ParsingError::MissingConnection(format!(
156                    "Connection references unknown target layer: {}",
157                    connection.to
158                )));
159            }
160        }
161
162        Ok(())
163    }
