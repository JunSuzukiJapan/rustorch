1//! Error types for model parsing
2//! モデル解析エラータイプ
3
4use std::error::Error;
5use std::fmt;
6
7/// Type alias for execution result containing layer order and dependencies
8pub type ExecutionResult = (Vec<String>, std::collections::HashMap<String, Vec<String>>);
9/// Type alias for parsing result containing layer order and dependencies  
10pub type ParsingResult = Result<ExecutionResult, ParsingError>;
11
12/// Model parsing errors
13/// モデル解析エラー
14#[derive(Debug)]
15pub enum ParsingError {
16    /// Invalid architecture format
17    /// 無効なアーキテクチャ形式
18    InvalidArchitecture(String),
19    /// Circular dependency detected
20    /// 循環依存を検出
21    CircularDependency(String),
22    /// Missing layer connection
23    /// レイヤー接続が見つからない
24    MissingConnection(String),
25    /// Incompatible layer dimensions
26    /// 互換性のないレイヤー次元
27    IncompatibleDimensions {
28        /// First layer name
29        layer1: String,
30        /// Second layer name
31        layer2: String,
32    },
33}
34
35impl fmt::Display for ParsingError {
36    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
37        match self {
38            ParsingError::InvalidArchitecture(msg) => write!(f, "Invalid architecture: {}", msg),
39            ParsingError::CircularDependency(msg) => write!(f, "Circular dependency: {}", msg),
40            ParsingError::MissingConnection(msg) => write!(f, "Missing connection: {}", msg),
41            ParsingError::IncompatibleDimensions { layer1, layer2 } => write!(
42                f,
43                "Incompatible dimensions between {} and {}",
44                layer1, layer2
45            ),
46        }
47    }
48}
49
50impl Error for ParsingError {}
