7/// Layer information extracted from PyTorch model
8/// PyTorchモデルから抽出されたレイヤー情報
9#[derive(Debug, Clone)]
10pub struct LayerInfo {
11    /// Layer name
12    /// レイヤー名
13    pub name: String,
14    /// Layer type (Linear, Conv2d, etc.)
15    /// レイヤータイプ（Linear、Conv2dなど）
16    pub layer_type: LayerType,
17    /// Input shape
18    /// 入力形状
19    pub input_shape: Option<Vec<usize>>,
20    /// Output shape
21    /// 出力形状
22    pub output_shape: Option<Vec<usize>>,
23    /// Layer parameters
24    /// レイヤーパラメータ
25    pub parameters: HashMap<String, Vec<usize>>, // parameter name -> shape
26    /// Number of parameters
27    /// パラメータ数
28    pub num_parameters: usize,
29}
30
31/// Supported layer types
32/// サポートされているレイヤータイプ
33#[derive(Debug, Clone, PartialEq)]
34pub enum LayerType {
35    /// Linear/Dense layer
36    /// Linear/Denseレイヤー
37    Linear {
38        /// Input features
39        in_features: usize,
40        /// Output features
41        out_features: usize,
42    },
43    /// 1D Convolution layer
44    /// 1D畳み込みレイヤー
45    Conv1d {
46        /// Input channels
47        in_channels: usize,
48        /// Output channels
49        out_channels: usize,
50        /// Kernel size
51        kernel_size: usize,
52        /// Stride
53        stride: usize,
54        /// Padding
55        padding: usize,
56    },
57    /// 2D Convolution layer
58    /// 2D畳み込みレイヤー
59    Conv2d {
60        /// Input channels
61        in_channels: usize,
62        /// Output channels
63        out_channels: usize,
64        /// Kernel size
65        kernel_size: (usize, usize),
66        /// Stride
67        stride: (usize, usize),
68        /// Padding
69        padding: (usize, usize),
70    },
71    /// 3D Convolution layer
72    /// 3D畳み込みレイヤー
73    Conv3d {
74        /// Input channels
75        in_channels: usize,
76        /// Output channels
77        out_channels: usize,
78        /// Kernel size
79        kernel_size: (usize, usize, usize),
80        /// Stride
81        stride: (usize, usize, usize),
82        /// Padding
83        padding: (usize, usize, usize),
84    },
85    /// 2D Batch Normalization
86    /// 2Dバッチ正規化
87    BatchNorm2d {
88        /// Number of features
89        num_features: usize,
90    },
91    /// ReLU activation
92    /// ReLU活性化
93    ReLU,
94    /// Dropout layer
95    /// Dropoutレイヤー
96    Dropout {
97        /// Dropout probability
98        p: f64,
99    },
100    /// 2D Max Pooling
101    /// 2D最大プーリング
102    MaxPool2d {
103        /// Kernel size
104        kernel_size: (usize, usize),
105        /// Stride
106        stride: (usize, usize),
107    },
108    /// 2D Average Pooling
109    /// 2D平均プーリング
110    AvgPool2d {
111        /// Kernel size
112        kernel_size: (usize, usize),
113        /// Stride
114        stride: (usize, usize),
115    },
116    /// Flatten layer
117    /// Flattenレイヤー
118    Flatten,
119    /// Unknown layer type
120    /// 不明なレイヤータイプ
121    Unknown(String),
122}

124/// Model architecture graph
125/// モデルアーキテクチャグラフ
126#[derive(Debug)]
127pub struct ModelGraph {
128    /// Layers in the model
129    /// モデル内のレイヤー
130    pub layers: HashMap<String, LayerInfo>,
131    /// Layer execution order
132    /// レイヤー実行順序
133    pub execution_order: Vec<String>,
134    /// Layer connections (from -> to)
135    /// レイヤー接続（from -> to）
136    pub connections: HashMap<String, Vec<String>>,
137    /// Model input layers
138    /// モデル入力レイヤー
139    pub input_layers: Vec<String>,
140    /// Model output layers
141    /// モデル出力レイヤー
142    pub output_layers: Vec<String>,
143}
