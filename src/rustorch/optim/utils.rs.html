13    /// Ensure numerical stability by clamping values to a safe range
14    /// 値を安全な範囲にクランプして数値安定性を確保
15    pub fn clamp_for_stability(tensor: &Tensor<f32>, min_val: f32, max_val: f32) -> Tensor<f32> {
16        let data = tensor.as_slice().unwrap();
17        let clamped_data: Vec<f32> = data.iter().map(|&x| x.max(min_val).min(max_val)).collect();
18        Tensor::from_vec(clamped_data, tensor.shape().to_vec())
19    }

21    /// Apply gradient clipping by norm to prevent gradient explosion
22    /// 勾配爆発を防ぐためにノルムによる勾配クリッピングを適用
23    pub fn clip_gradient_norm(grad: &Tensor<f32>, max_norm: f32) -> Tensor<f32> {
24        let grad_squared = grad * grad;
25        let grad_norm = grad_squared.sum().sqrt();
26
27        if grad_norm > max_norm {
28            let scale_factor = max_norm / grad_norm;
29            grad * scale_factor
30        } else {
31            grad.clone()
32        }
33    }

35    /// Apply gradient clipping by value to prevent numerical instability
36    /// 数値不安定性を防ぐために値による勾配クリッピングを適用
37    pub fn clip_gradient_value(grad: &Tensor<f32>, min_val: f32, max_val: f32) -> Tensor<f32> {
38        Self::clamp_for_stability(grad, min_val, max_val)
39    }

42    /// イプシロンによる数値安定な平方根を計算
43    pub fn stable_sqrt(tensor: &Tensor<f32>, eps: f32) -> Tensor<f32> {
44        let data = tensor.as_slice().unwrap();
45        let sqrt_data: Vec<f32> = data.iter().map(|&x| (x + eps).sqrt()).collect();

49    /// Apply exponential moving average efficiently
50    /// 指数移動平均を効率的に適用
51    pub fn ema_update(current: &mut Tensor<f32>, new_value: &Tensor<f32>, decay: f32) {
52        // current = decay * current + (1 - decay) * new_value
53        let decay_term = &*current * decay;
54        let new_term = new_value * (1.0 - decay);
55        let updated = &decay_term + &new_term;
56        current.copy_from(&updated);
57    }

59    /// Check if tensor contains NaN or infinity values
60    /// テンソルにNaNや無限大値が含まれているかチェック
61    pub fn has_invalid_values(tensor: &Tensor<f32>) -> bool {
62        let data = tensor.as_slice().unwrap();
63        data.iter().any(|&x| x.is_nan() || x.is_infinite())
64    }

66    /// Replace NaN and infinity values with specified defaults
67    /// NaNと無限大値を指定されたデフォルト値で置換
68    pub fn sanitize_tensor(
69        tensor: &Tensor<f32>,
70        nan_replacement: f32,
71        inf_replacement: f32,
72    ) -> Tensor<f32> {
73        let data = tensor.as_slice().unwrap();
74        let sanitized_data: Vec<f32> = data
75            .iter()
76            .map(|&x| {
77                if x.is_nan() {
78                    nan_replacement
79                } else if x.is_infinite() {
80                    if x > 0.0 {
81                        inf_replacement
82                    } else {
83                        -inf_replacement
84                    }
85                } else {
86                    x
87                }
88            })
89            .collect();
90        Tensor::from_vec(sanitized_data, tensor.shape().to_vec())
91    }

117        param: &Tensor<f32>,
118        weight_decay: f32,
119    ) -> Tensor<f32> {
120        if weight_decay > 0.0 {

125        }

128    /// Update momentum (first moment) with numerical stability
129    /// 数値安定性を考慮したモーメンタム（第一モーメント）更新
130    pub fn update_momentum(
131        momentum: &mut Tensor<f32>,
132        grad: &Tensor<f32>,
133        beta1: f32,
134    ) -> RusTorchResult<()> {
135        if !(0.0..1.0).contains(&beta1) {
136            return Err(RusTorchError::InvalidParameters {
137                operation: "momentum update".to_string(),
138                message: format!("Beta1 must be in [0, 1), got {}", beta1),
139            });
140        }
141
142        let beta1_term = &*momentum * beta1;
143        let grad_term = grad * (1.0 - beta1);
144        let updated = &beta1_term + &grad_term;
145
146        // Check for numerical stability
147        if Self::has_invalid_values(&updated) {
148            return Err(RusTorchError::InvalidParameters {
149                operation: "momentum update".to_string(),
150                message: "Numerical instability detected in momentum update".to_string(),
151            });
152        }
153
154        momentum.copy_from(&updated);
155        Ok(())
156    }

158    /// Update velocity (second moment) with numerical stability
159    /// 数値安定性を考慮したベロシティ（第二モーメント）更新  
160    pub fn update_velocity(
161        velocity: &mut Tensor<f32>,
162        grad: &Tensor<f32>,
163        beta2: f32,
164    ) -> RusTorchResult<()> {
165        if !(0.0..1.0).contains(&beta2) {
166            return Err(RusTorchError::InvalidParameters {
167                operation: "velocity update".to_string(),
168                message: format!("Beta2 must be in [0, 1), got {}", beta2),
169            });
170        }
171
172        let beta2_term = &*velocity * beta2;
173        let grad_squared = grad * grad;
174        let grad_term = &grad_squared * (1.0 - beta2);
175        let updated = &beta2_term + &grad_term;
176
177        // Check for numerical stability
178        if Self::has_invalid_values(&updated) {
179            return Err(RusTorchError::InvalidParameters {
180                operation: "velocity update".to_string(),
181                message: "Numerical instability detected in velocity update".to_string(),
182            });
183        }
184
185        velocity.copy_from(&updated);
186        Ok(())
187    }

189    /// Compute bias correction factor
190    /// バイアス補正係数を計算
191    pub fn bias_correction(beta: f32, step: usize) -> f32 {
192        if step == 0 || beta == 0.0 {
193            1.0
194        } else {
195            1.0 - beta.powi(step as i32)
196        }
197    }

199    /// Apply bias correction to tensor
200    /// テンソルにバイアス補正を適用
201    pub fn apply_bias_correction(tensor: &Tensor<f32>, correction: f32) -> Tensor<f32> {
202        if correction.abs() < 1e-12 {
203            // Avoid division by very small numbers
204            tensor.clone()
205        } else {
206            tensor / correction
207        }
208    }

221    /// Compute element-wise maximum between two tensors (for Adamax)
222    /// 二つのテンソル間の要素毎最大値を計算（Adamax用）
223    pub fn tensor_max(a: &Tensor<f32>, b: &Tensor<f32>) -> RusTorchResult<Tensor<f32>> {
224        let a_data = a
225            .as_slice()
226            .ok_or_else(|| RusTorchError::InvalidParameters {
227                operation: "tensor_max".to_string(),
228                message: "Failed to access tensor A data".to_string(),
229            })?;
230        let b_data = b
231            .as_slice()
232            .ok_or_else(|| RusTorchError::InvalidParameters {
233                operation: "tensor_max".to_string(),
234                message: "Failed to access tensor B data".to_string(),
235            })?;
236
237        if a_data.len() != b_data.len() || a.shape() != b.shape() {
238            return Err(RusTorchError::InvalidParameters {
239                operation: "tensor_max".to_string(),
240                message: "Tensor shapes must match".to_string(),
241            });
242        }
243
244        let max_data: Vec<f32> = a_data
245            .iter()
246            .zip(b_data.iter())
247            .map(|(&a_val, &b_val)| a_val.max(b_val))
248            .collect();
249
250        Ok(Tensor::from_vec(max_data, a.shape().to_vec()))
251    }

267    /// Advanced exponential moving average with momentum scheduling
268    /// モーメンタムスケジューリング付き高度な指数移動平均
269    pub fn advanced_ema_update(
270        current: &mut Tensor<f32>,
271        new_value: &Tensor<f32>,
272        base_decay: f32,
273        step: usize,
274        warmup_steps: usize,
275    ) -> RusTorchResult<()> {
276        // Apply warmup scheduling
277        let effective_decay = if step < warmup_steps {
278            let warmup_factor = (step as f32) / (warmup_steps as f32);
279            base_decay * warmup_factor
280        } else {
281            base_decay
282        };
283
284        let decay_term = &*current * effective_decay;
285        let new_term = new_value * (1.0 - effective_decay);
286        let updated = &decay_term + &new_term;
287
288        if Self::has_invalid_values(&updated) {
289            return Err(RusTorchError::InvalidParameters {
290                operation: "advanced_ema_update".to_string(),
291                message: "Numerical instability detected".to_string(),
292            });
293        }
294
295        current.copy_from(&updated);
296        Ok(())
297    }

299    /// Compute learning rate with cosine annealing
300    /// コサインアニーリングによる学習率計算
301    pub fn cosine_annealing_lr(
302        base_lr: f32,
303        current_step: usize,
304        total_steps: usize,
305        min_lr: f32,
306    ) -> f32 {
307        if total_steps == 0 {
308            return base_lr;
309        }
310
311        let progress = (current_step.min(total_steps) as f32) / (total_steps as f32);
312        let cosine_factor = 0.5 * (1.0 + (std::f32::consts::PI * progress).cos());
313        min_lr + (base_lr - min_lr) * cosine_factor
314    }
315
316    /// Compute adaptive learning rate scaling based on gradient statistics
317    /// 勾配統計に基づく適応学習率スケーリング計算
318    pub fn adaptive_lr_scaling(grad_norm: f32, velocity_norm: f32, trust_ratio: f32) -> f32 {
319        if velocity_norm < 1e-12 {
320            trust_ratio
321        } else {
322            trust_ratio * (grad_norm / velocity_norm).min(1.0)
323        }
324    }

327/// Memory-efficient state management for optimizers
328/// オプティマイザのメモリ効率的な状態管理
329#[derive(Debug, Clone)]
330pub struct OptimizerState {
331    /// Parameter states indexed by parameter ID
332    states: HashMap<usize, ParameterState>,
333    /// Global step counter
334    global_step: usize,
335    /// Memory usage threshold for cleanup (in MB)
336    memory_threshold_mb: usize,
337}

339/// Individual parameter state for memory-efficient storage
340/// メモリ効率的な保存のための個別パラメータ状態
341#[derive(Debug, Clone)]
342pub struct ParameterState {
343    /// First moment (momentum)
344    pub momentum: Option<Tensor<f32>>,
345    /// Second moment (velocity)
346    pub velocity: Option<Tensor<f32>>,
347    /// Additional state for specialized optimizers
348    pub extra_state: HashMap<String, Tensor<f32>>,
349    /// Last update step
350    pub last_step: usize,
351}
352
353impl OptimizerState {
354    /// Create new optimizer state with memory management
355    pub fn new(memory_threshold_mb: usize) -> Self {
356        Self {
357            states: HashMap::new(),
358            global_step: 0,
359            memory_threshold_mb,
360        }
361    }
362
363    /// Get or create parameter state
364    pub fn get_or_create_state(
365        &mut self,
366        param_id: usize,
367        param_shape: &[usize],
368    ) -> &mut ParameterState {
369        self.states
370            .entry(param_id)
371            .or_insert_with(|| ParameterState {
372                momentum: None,
373                velocity: None,
374                extra_state: HashMap::new(),
375                last_step: 0,
376            })
377    }
378
379    /// Initialize momentum for parameter
380    pub fn init_momentum(&mut self, param_id: usize, param_shape: &[usize]) {
381        if let Some(state) = self.states.get_mut(&param_id) {
382            if state.momentum.is_none() {
383                state.momentum = Some(Tensor::zeros(param_shape));
384            }
385        }
386    }
387
388    /// Initialize velocity for parameter
389    pub fn init_velocity(&mut self, param_id: usize, param_shape: &[usize]) {
390        if let Some(state) = self.states.get_mut(&param_id) {
391            if state.velocity.is_none() {
392                state.velocity = Some(Tensor::zeros(param_shape));
393            }
394        }
395    }
396
397    /// Clean up states for parameters not used recently
398    pub fn cleanup_stale_states(&mut self, steps_threshold: usize) {
399        let current_step = self.global_step;
400        self.states
401            .retain(|_, state| current_step - state.last_step < steps_threshold);
402    }
403
404    /// Increment global step counter
405    pub fn step(&mut self) {
406        self.global_step += 1;
407
408        // Periodic cleanup every 1000 steps
409        if self.global_step % 1000 == 0 {
410            self.cleanup_stale_states(5000);
411        }
412    }
413
414    /// Get current global step
415    pub fn get_step(&self) -> usize {
416        self.global_step
417    }
418
419    /// Estimate memory usage (rough approximation)
420    pub fn estimate_memory_mb(&self) -> usize {
421        let mut total_elements = 0;
422        for state in self.states.values() {
423            if let Some(ref momentum) = state.momentum {
424                total_elements += momentum.as_slice().unwrap().len();
425            }
426            if let Some(ref velocity) = state.velocity {
427                total_elements += velocity.as_slice().unwrap().len();
428            }
429            for tensor in state.extra_state.values() {
430                total_elements += tensor.as_slice().unwrap().len();
431            }
432        }
433        // Rough estimate: 4 bytes per f32 element
434        (total_elements * 4) / (1024 * 1024)
435    }
436}

438/// Numerical stability configuration for optimizers
439/// オプティマイザの数値安定性設定
440#[derive(Debug, Clone)]
441pub struct StabilityConfig {
442    /// Minimum epsilon for numerical stability
443    pub min_eps: f32,
444    /// Maximum gradient norm for clipping
445    pub max_grad_norm: f32,
446    /// Maximum parameter value to prevent overflow
447    pub max_param_value: f32,
448    /// Enable automatic NaN detection and correction
449    pub auto_nan_correction: bool,
450    /// Enable gradient clipping
451    pub gradient_clipping: bool,
452}
453
454impl Default for StabilityConfig {
455    fn default() -> Self {
456        Self {
457            min_eps: 1e-8,
458            max_grad_norm: 10.0,
459            max_param_value: 1e6,
460            auto_nan_correction: true,
461            gradient_clipping: true,
462        }
463    }
464}
465
466impl StabilityConfig {
467    /// Apply stability measures to gradient
468    pub fn stabilize_gradient(&self, grad: &Tensor<f32>) -> Tensor<f32> {
469        let mut stabilized_grad = grad.clone();
470
471        // Check for invalid values
472        if self.auto_nan_correction && OptimizerUtils::has_invalid_values(&stabilized_grad) {
473            stabilized_grad =
474                OptimizerUtils::sanitize_tensor(&stabilized_grad, 0.0, self.max_grad_norm);
475        }
476
477        // Apply gradient clipping
478        if self.gradient_clipping {
479            stabilized_grad =
480                OptimizerUtils::clip_gradient_norm(&stabilized_grad, self.max_grad_norm);
481        }
482
483        stabilized_grad
484    }
485
486    /// Apply stability measures to parameter update
487    pub fn stabilize_parameter(&self, param: &Tensor<f32>) -> Tensor<f32> {
488        let mut stabilized_param = param.clone();
489
490        // Check for invalid values
491        if self.auto_nan_correction && OptimizerUtils::has_invalid_values(&stabilized_param) {
492            stabilized_param =
493                OptimizerUtils::sanitize_tensor(&stabilized_param, 0.0, self.max_param_value);
494        }
495
496        // Clamp parameter values
497        stabilized_param = OptimizerUtils::clamp_for_stability(
498            &stabilized_param,
499            -self.max_param_value,
500            self.max_param_value,
501        );
502
503        stabilized_param
504    }
505}

507/// Advanced optimizer metadata and statistics tracking
508/// 高度なオプティマイザメタデータと統計追跡
509#[derive(Debug, Clone)]
510pub struct OptimizerMetrics {
511    /// Gradient norm history for convergence analysis
512    gradient_norms: Vec<f32>,
513    /// Parameter change norms for step size analysis
514    param_change_norms: Vec<f32>,
515    /// Learning rate history
516    learning_rates: Vec<f32>,
517    /// Step timings for performance analysis
518    step_times: Vec<f32>,
519    /// Maximum history length
520    max_history: usize,
521    /// Current step count
522    step_count: usize,
523}
524
525impl OptimizerMetrics {
526    /// Create new optimizer metrics tracker
527    pub fn new(max_history: usize) -> Self {
528        Self {
529            gradient_norms: Vec::with_capacity(max_history),
530            param_change_norms: Vec::with_capacity(max_history),
531            learning_rates: Vec::with_capacity(max_history),
532            step_times: Vec::with_capacity(max_history),
533            max_history,
534            step_count: 0,
535        }
536    }
537
538    /// Record step metrics
539    pub fn record_step(
540        &mut self,
541        grad_norm: f32,
542        param_change_norm: f32,
543        learning_rate: f32,
544        step_time: f32,
545    ) {
546        self.gradient_norms.push(grad_norm);
547        self.param_change_norms.push(param_change_norm);
548        self.learning_rates.push(learning_rate);
549        self.step_times.push(step_time);
550
551        // Maintain maximum history size
552        if self.gradient_norms.len() > self.max_history {
553            self.gradient_norms.remove(0);
554            self.param_change_norms.remove(0);
555            self.learning_rates.remove(0);
556            self.step_times.remove(0);
557        }
558
559        self.step_count += 1;
560    }
561
562    /// Get gradient norm statistics
563    pub fn gradient_stats(&self) -> (f32, f32, f32) {
564        if self.gradient_norms.is_empty() {
565            return (0.0, 0.0, 0.0);
566        }
567
568        let mean = self.gradient_norms.iter().sum::<f32>() / self.gradient_norms.len() as f32;
569        let min = *self
570            .gradient_norms
571            .iter()
572            .min_by(|a, b| a.partial_cmp(b).unwrap())
573            .unwrap();
574        let max = *self
575            .gradient_norms
576            .iter()
577            .max_by(|a, b| a.partial_cmp(b).unwrap())
578            .unwrap();
579
580        (mean, min, max)
581    }
582
583    /// Check for convergence based on recent gradient norms
584    pub fn check_convergence(&self, threshold: f32, window_size: usize) -> bool {
585        if self.gradient_norms.len() < window_size {
586            return false;
587        }
588
589        let recent_norms =
590            &self.gradient_norms[self.gradient_norms.len().saturating_sub(window_size)..];
591        let avg_norm = recent_norms.iter().sum::<f32>() / recent_norms.len() as f32;
592
593        avg_norm < threshold
594    }
595
596    /// Detect optimization issues (gradient explosion, vanishing, etc.)
597    pub fn detect_issues(&self) -> Vec<String> {
598        let mut issues = Vec::new();
599
600        if let Some(&latest_grad_norm) = self.gradient_norms.last() {
601            if latest_grad_norm > 100.0 {
602                issues.push("Gradient explosion detected".to_string());
603            }
604            if latest_grad_norm < 1e-8 {
605                issues.push("Vanishing gradients detected".to_string());
606            }
607        }
608
609        if self.gradient_norms.len() > 10 {
610            let recent = &self.gradient_norms[self.gradient_norms.len() - 10..];
611            let variance = {
612                let mean = recent.iter().sum::<f32>() / recent.len() as f32;
613                recent.iter().map(|x| (x - mean).powi(2)).sum::<f32>() / recent.len() as f32
614            };
615
616            if variance < 1e-12 {
617                issues.push("Optimization stagnation detected".to_string());
618            }
619        }
620
621        issues
622    }
623
624    /// Get current step count
625    pub fn step_count(&self) -> usize {
626        self.step_count
627    }
628
629    /// Reset metrics
630    pub fn reset(&mut self) {
631        self.gradient_norms.clear();
632        self.param_change_norms.clear();
633        self.learning_rates.clear();
634        self.step_times.clear();
635        self.step_count = 0;
636    }
637}

639/// Unified optimizer factory for creating optimizers with consistent configuration
640/// 一貫した設定でオプティマイザを作成するための統合オプティマイザファクトリ
641pub struct OptimizerFactory;
642
643impl OptimizerFactory {
644    /// Create optimizer configuration with validation
645    pub fn validate_config(learning_rate: f32, weight_decay: f32, eps: f32) -> RusTorchResult<()> {
646        if learning_rate <= 0.0 {
647            return Err(RusTorchError::InvalidParameters {
648                operation: "optimizer creation".to_string(),
649                message: "Learning rate must be positive".to_string(),
650            });
651        }
652
653        if weight_decay < 0.0 {
654            return Err(RusTorchError::InvalidParameters {
655                operation: "optimizer creation".to_string(),
656                message: "Weight decay must be non-negative".to_string(),
657            });
658        }
659
660        if eps <= 0.0 {
661            return Err(RusTorchError::InvalidParameters {
662                operation: "optimizer creation".to_string(),
663                message: "Epsilon must be positive".to_string(),
664            });
665        }
666
667        Ok(())
668    }
669
670    /// Suggest optimal parameters based on problem characteristics
671    pub fn suggest_parameters(problem_type: &str, model_size: usize) -> (f32, f32, f32) {
672        // (learning_rate, weight_decay, eps)
673        match problem_type {
674            "vision" => {
675                if model_size > 50_000_000 {
676                    // Large model
677                    (1e-4, 1e-4, 1e-8)
678                } else {
679                    (1e-3, 1e-4, 1e-8)
680                }
681            }
682            "nlp" => {
683                if model_size > 100_000_000 {
684                    // Large language model
685                    (5e-5, 1e-2, 1e-6)
686                } else {
687                    (2e-4, 1e-3, 1e-8)
688                }
689            }
690            "reinforcement_learning" => (3e-4, 0.0, 1e-5),
691            "fine_tuning" => (5e-5, 1e-5, 1e-8),
692            _ => (1e-3, 1e-4, 1e-8), // Default
693        }
694    }
