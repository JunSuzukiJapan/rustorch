149    /// Benchmark a single optimizer with given configuration
150    /// æŒ‡å®šã•ã‚ŒãŸè¨­å®šã§å˜ä¸€ã®æœ€é©åŒ–å™¨ã‚’ãƒ™ãƒ³ãƒãƒãƒ¼ã‚¯
151    fn benchmark_optimizer(
152        &self,
153        config: &BenchmarkConfig,
154        mut optimizer: Box<dyn Optimizer>,
155    ) -> BenchmarkResult {
156        let mut step_times = Vec::new();
157        let mut metrics = OptimizerMetrics::new(config.num_steps + config.warmup_steps);
158
159        let start_total = Instant::now();
160
161        for iteration in 0..config.num_iterations {
162            // Create test data for this iteration
163            let param = Tensor::ones(&config.param_shape);
164            let mut grad_norms = Vec::new();
165
166            // Warmup phase
167            for step in 0..config.warmup_steps {
168                let grad = self.generate_synthetic_gradient(&config.param_shape, step);
169                let grad_norm = OptimizerUtils::l2_norm(&grad);
170                optimizer.step(&param, &grad);
171                grad_norms.push(grad_norm);
172            }
173
174            // Timed benchmark phase
175            for step in 0..config.num_steps {
176                let grad = self.generate_synthetic_gradient(&config.param_shape, step);
177                let grad_norm = OptimizerUtils::l2_norm(&grad);
178
179                let step_start = Instant::now();
180                optimizer.step(&param, &grad);
181                let step_time = step_start.elapsed();
182
183                if iteration == 0 {
184                    // Only collect timing for first iteration
185                    step_times.push(step_time.as_micros() as f64);
186                }
187
188                // Record metrics
189                metrics.record_step(
190                    grad_norm,
191                    0.1, // Dummy parameter change norm
192                    optimizer.learning_rate(),
193                    step_time.as_secs_f32(),
194                );
195
196                grad_norms.push(grad_norm);
197            }
198        }
199
200        let total_duration = start_total.elapsed();
201
202        // Calculate statistics
203        let avg_step_time = step_times.iter().sum::<f64>() / step_times.len() as f64;
204        let variance = step_times
205            .iter()
206            .map(|&t| (t - avg_step_time).powi(2))
207            .sum::<f64>()
208            / step_times.len() as f64;
209        let std_dev = variance.sqrt();
210
211        // Estimate memory usage (rough approximation)
212        let total_elements = config.param_shape.iter().product::<usize>();
213        let estimated_memory_mb = (total_elements * 4 * 3) / (1024 * 1024); // 3 tensors (param, momentum, velocity) * 4 bytes per f32
214
215        BenchmarkResult {
216            avg_step_time_us: avg_step_time,
217            step_time_std_us: std_dev,
218            peak_memory_mb: estimated_memory_mb,
219            convergence_rate: 0.95, // Placeholder - would need proper convergence analysis
220            total_duration_ms: total_duration.as_millis() as u64,
221            steps_per_second: (config.num_steps as f64) / total_duration.as_secs_f64()
222                * config.num_iterations as f64,
223        }
224    }

226    /// Generate synthetic gradient for testing
227    /// ãƒ†ã‚¹ãƒˆç”¨ã®åˆæˆå‹¾é…ã‚’ç”Ÿæˆ
228    fn generate_synthetic_gradient(&self, shape: &[usize], step: usize) -> Tensor<f32> {
229        let total_elements = shape.iter().product();
230        let mut data = Vec::with_capacity(total_elements);
231
232        // Generate gradient that simulates realistic training behavior
233        let step_factor = 1.0 / (1.0 + step as f32 * 0.001); // Gradually decreasing gradients
234
235        for i in 0..total_elements {
236            let noise = (i as f32 * 0.123 + step as f32 * 0.456).sin() * 0.1;
237            let gradient = (0.1 + noise) * step_factor;
238            data.push(gradient);
239        }
240
241        Tensor::from_vec(data, shape.to_vec())
242    }

323            }
324            if let Some((name, memory)) = most_memory_efficient {
325                most_efficient_memory.insert(config_name, (name, memory));
326            }
327        }
328
329        report.push_str("### âš¡ é€Ÿåº¦ãƒãƒ£ãƒ³ãƒ”ã‚ªãƒ³:\n");
330        for (config, (optimizer, time)) in fastest_by_config {
331            report.push_str(&format!(
332                "- **{}**: {} ({:.2}Î¼s/ã‚¹ãƒ†ãƒƒãƒ—)\n",
333                config, optimizer, time

335        }

338        for (config, (optimizer, memory)) in most_efficient_memory {

340        }

343    }
