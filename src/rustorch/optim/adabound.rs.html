1//! AdaBound optimizer
2//! AdaBoundオプティマイザー
3//!
4//! AdaBound bridges the gap between Adam and SGD by dynamically adjusting
5//! the learning rate bounds, providing better convergence properties.
6//!
7//! AdaBoundは学習率の境界を動的に調整することでAdamとSGDの
8//! ギャップを埋め、より良い収束特性を提供します。

14/// AdaBound optimizer
15///
16/// AdaBound dynamically adjusts the bounds of learning rates to combine
17/// the fast convergence of Adam with the good generalization of SGD.
18#[derive(Debug, Clone)]
19pub struct AdaBound {
20    learning_rate: f32,
21    beta1: f32,
22    beta2: f32,
23    epsilon: f32,
24    weight_decay: f32,
25    final_lr: f32,
26    gamma: f32,
27    step_count: usize,
28    exp_avg: HashMap<usize, Tensor<f32>>, // First moment estimate
29    exp_avg_sq: HashMap<usize, Tensor<f32>>, // Second moment estimate
30}
31
32impl AdaBound {
33    /// Create new AdaBound optimizer with default parameters
34    /// デフォルトパラメータで新しいAdaBoundオプティマイザーを作成
35    pub fn new(learning_rate: f32) -> Self {
36        Self::with_params(learning_rate, 0.1, 0.9, 0.999, 1e-8, 0.0, 1e-3)
37    }
38
39    /// Create AdaBound optimizer with custom parameters
40    /// カスタムパラメータでAdaBoundオプティマイザーを作成
41    pub fn with_params(
42        learning_rate: f32,
43        final_lr: f32,
44        beta1: f32,
45        beta2: f32,
46        epsilon: f32,
47        weight_decay: f32,
48        gamma: f32,
49    ) -> Self {
50        Self {
51            learning_rate,
52            beta1,
53            beta2,
54            epsilon,
55            weight_decay,
56            final_lr,
57            gamma,
58            step_count: 0,
59            exp_avg: HashMap::new(),
60            exp_avg_sq: HashMap::new(),
61        }
62    }
63
64    /// Set final learning rate for convergence
65    /// 収束用の最終学習率を設定
66    pub fn set_final_lr(&mut self, final_lr: f32) {
67        self.final_lr = final_lr;
68    }
69
70    /// Set gamma parameter for bounds adjustment
71    /// 境界調整用のガンマパラメータを設定  
72    pub fn set_gamma(&mut self, gamma: f32) {
73        self.gamma = gamma;
74    }
75
76    /// Get current step count
77    /// 現在のステップ数を取得
78    pub fn step_count(&self) -> usize {
79        self.step_count
80    }
81
82    /// Calculate dynamic learning rate bounds
83    /// 動的学習率境界を計算
84    fn compute_bounds(&self) -> (f32, f32) {
85        let t = self.step_count as f32;
86        let base_lr = self.learning_rate;
87        let final_lr = self.final_lr;
88
89        // Dynamic bound calculation based on step count
90        // ステップ数に基づく動的境界計算
91        let bound_scale = (1.0 + t * self.gamma).ln();
92        let lower_bound = final_lr * (1.0 - 1.0 / bound_scale);
93        let upper_bound = final_lr * (1.0 + 1.0 / bound_scale);
94
95        (lower_bound.max(0.0), upper_bound.min(base_lr))
96    }
97}

99impl Optimizer for AdaBound {
100    fn step(&mut self, param: &Tensor<f32>, grad: &Tensor<f32>) {
101        let param_id = param.as_ptr() as usize;
102        self.step_count += 1;
103
104        let mut d_p = grad.clone();
105
106        // Apply weight decay
107        // 重み減衰を適用
108        if self.weight_decay != 0.0 {
109            let weight_decay_term = param * self.weight_decay;
110            d_p = &d_p + &weight_decay_term;
111        }
112
113        // Get or initialize momentum buffers
114        // モーメンタムバッファを取得または初期化
115        let exp_avg = if let Some(avg) = self.exp_avg.get(&param_id) {
116            let beta1_term = avg * self.beta1;
117            let one_minus_beta1_term = &d_p * (1.0 - self.beta1);
118            &beta1_term + &one_minus_beta1_term
119        } else {
120            d_p.clone() * (1.0 - self.beta1)
121        };
122
123        let exp_avg_sq = if let Some(avg_sq) = self.exp_avg_sq.get(&param_id) {
124            let beta2_term = avg_sq * self.beta2;
125            let d_p_squared = &d_p * &d_p;
126            let one_minus_beta2_term = &d_p_squared * (1.0 - self.beta2);
127            &beta2_term + &one_minus_beta2_term
128        } else {
129            let d_p_squared = &d_p * &d_p;
130            d_p_squared * (1.0 - self.beta2)
131        };
132
133        // Store updated momentum buffers
134        // 更新されたモーメンタムバッファを保存
135        self.exp_avg.insert(param_id, exp_avg.clone());
136        self.exp_avg_sq.insert(param_id, exp_avg_sq.clone());
137
138        // Bias correction
139        // バイアス補正
140        let bias_correction1 = 1.0 - self.beta1.powi(self.step_count as i32);
141        let bias_correction2 = 1.0 - self.beta2.powi(self.step_count as i32);
142
143        let corrected_exp_avg = &exp_avg / bias_correction1;
144        let corrected_exp_avg_sq = &exp_avg_sq / bias_correction2;
145
146        // Compute adaptive learning rates
147        // 適応学習率を計算
148        let sqrt_corrected_exp_avg_sq = corrected_exp_avg_sq.sqrt();
149        let raw_step_size = corrected_exp_avg / (&sqrt_corrected_exp_avg_sq + self.epsilon);
150
151        // Apply dynamic bounds (AdaBound's key innovation)
152        // 動的境界を適用（AdaBoundの主要革新）
153        let (lower_bound, upper_bound) = self.compute_bounds();
154
155        // Element-wise bound clipping would be ideal, but for simplicity we use scalar bounds
156        // 要素ごとの境界クリッピングが理想的ですが、簡単のためスカラー境界を使用
157        let step_size = self.learning_rate.max(lower_bound).min(upper_bound);
158
159        // Final parameter update
160        // 最終パラメータ更新
161        let scaled_update = &raw_step_size * step_size;
162        let updated_param = param - &scaled_update;
163
164        param.copy_from(&updated_param);
165    }
166
167    fn learning_rate(&self) -> f32 {
168        self.learning_rate
169    }
170
171    fn set_learning_rate(&mut self, lr: f32) {
172        self.learning_rate = lr;
173    }
174
175    fn state_dict(&self) -> HashMap<String, f32> {
176        let mut state = HashMap::new();
177        state.insert("learning_rate".to_string(), self.learning_rate);
178        state.insert("beta1".to_string(), self.beta1);
179        state.insert("beta2".to_string(), self.beta2);
180        state.insert("epsilon".to_string(), self.epsilon);
181        state.insert("weight_decay".to_string(), self.weight_decay);
182        state.insert("final_lr".to_string(), self.final_lr);
183        state.insert("gamma".to_string(), self.gamma);
184        state.insert("step_count".to_string(), self.step_count as f32);
185        state
186    }
187
188    fn load_state_dict(&mut self, state: HashMap<String, f32>) {
189        if let Some(&lr) = state.get("learning_rate") {
190            self.learning_rate = lr;
191        }
192        if let Some(&beta1) = state.get("beta1") {
193            self.beta1 = beta1;
194        }
195        if let Some(&beta2) = state.get("beta2") {
196            self.beta2 = beta2;
197        }
198        if let Some(&epsilon) = state.get("epsilon") {
199            self.epsilon = epsilon;
200        }
201        if let Some(&weight_decay) = state.get("weight_decay") {
202            self.weight_decay = weight_decay;
203        }
204        if let Some(&final_lr) = state.get("final_lr") {
205            self.final_lr = final_lr;
206        }
207        if let Some(&gamma) = state.get("gamma") {
208            self.gamma = gamma;
209        }
210        if let Some(&step_count) = state.get("step_count") {
211            self.step_count = step_count as usize;
212        }
213    }

242    fn test_adabound_bounds_computation() {
243        let mut optimizer = AdaBound::new(0.1);

254    #[test]
255    fn test_adabound_step() {
256        let mut optimizer = AdaBound::new(0.01);
257        let param = Tensor::<f32>::ones(&[2, 2]);
258        let grad = Tensor::<f32>::ones(&[2, 2]) * 0.1;
259
260        // Initial parameter values
261        let initial_param = param.clone();
262
263        // Perform optimization step
264        optimizer.step(&param, &grad);
265
266        // Verify step count increased
267        assert_eq!(optimizer.step_count(), 1);
268
269        // Verify parameters were updated
270        let updated_data = param.data.as_slice().unwrap();
271        let initial_data = initial_param.data.as_slice().unwrap();
272
273        assert_ne!(updated_data[0], initial_data[0]);
274    }

277    fn test_adabound_convergence_behavior() {
278        let mut optimizer = AdaBound::new(0.1);
279        let param = Tensor::<f32>::ones(&[2, 2]);
280        let grad = Tensor::<f32>::ones(&[2, 2]) * 0.05;

289

291        // Bounds should be reasonable, upper_bound should be less than or equal to initial LR
