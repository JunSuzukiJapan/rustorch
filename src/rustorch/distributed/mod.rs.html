171/// Communication operations for distributed training
172/// 分散学習用通信操作
173pub trait DistributedOps<T: Float> {
174    /// All-reduce operation across all processes
175    /// 全プロセス間でのall-reduce操作
176    fn all_reduce(&self, tensor: &mut Tensor<T>, op: ReduceOp) -> RusTorchResult<()>;
177
178    /// All-gather operation across all processes
179    /// 全プロセス間でのall-gather操作
180    fn all_gather(&self, tensor: &Tensor<T>) -> RusTorchResult<Vec<Tensor<T>>>;
181
182    /// Broadcast operation from root process
183    /// ルートプロセスからのブロードキャスト操作
184    fn broadcast(&self, tensor: &mut Tensor<T>, root: usize) -> RusTorchResult<()>;
185
186    /// Reduce operation to root process
187    /// ルートプロセスへのreduce操作
188    fn reduce(&self, tensor: &mut Tensor<T>, root: usize, op: ReduceOp) -> RusTorchResult<()>;
189
190    /// Scatter operation from root process
191    /// ルートプロセスからのscatter操作
192    fn scatter(&self, tensors: &[Tensor<T>], root: usize) -> RusTorchResult<Tensor<T>>;
193
194    /// Gather operation to root process
195    /// ルートプロセスへのgather操作
196    fn gather(&self, tensor: &Tensor<T>, root: usize) -> RusTorchResult<Vec<Tensor<T>>>;
197}

220/// Global distributed state
221/// グローバル分散状態
222static mut DISTRIBUTED_STATE: Option<Arc<Mutex<DistributedState>>> = None;
223static DISTRIBUTED_INIT: std::sync::Once = std::sync::Once::new();
224
225/// Distributed state management
226/// 分散状態管理
227#[derive(Debug)]
228pub struct DistributedState {
229    /// Current process group
230    /// 現在のプロセスグループ
231    pub process_group: Option<ProcessGroup>,
232    /// Available devices for distributed training
233    /// 分散学習で利用可能なデバイス
234    pub devices: Vec<DeviceType>,
235    /// Device mapping for each rank
236    /// 各ランクのデバイスマッピング
237    pub device_map: HashMap<usize, Vec<DeviceType>>,
238}
239
240impl Default for DistributedState {
241    fn default() -> Self {
242        Self::new()
243    }
244}
245
246impl DistributedState {
247    /// Create new distributed state
248    /// 新しい分散状態を作成
249    pub fn new() -> Self {
250        Self {
251            process_group: None,
252            devices: Vec::new(),
253            device_map: HashMap::new(),
254        }
255    }
256
257    /// Set process group
258    /// プロセスグループを設定
259    pub fn set_process_group(&mut self, pg: ProcessGroup) {
260        self.process_group = Some(pg);
261    }
262
263    /// Get current rank
264    /// 現在のランクを取得
265    pub fn rank(&self) -> Option<usize> {
266        self.process_group.as_ref().map(|pg| pg.rank)
267    }
268
269    /// Get world size
270    /// ワールドサイズを取得
271    pub fn world_size(&self) -> Option<usize> {
272        self.process_group.as_ref().map(|pg| pg.world_size)
273    }
274
275    /// Check if distributed training is initialized
276    /// 分散学習が初期化されているかチェック
277    pub fn is_initialized(&self) -> bool {
278        self.process_group.is_some()
279    }
280}

282/// Get global distributed state
283/// グローバル分散状態を取得
284pub fn get_distributed_state() -> &'static Arc<Mutex<DistributedState>> {
285    unsafe {
286        DISTRIBUTED_INIT.call_once(|| {
287            DISTRIBUTED_STATE = Some(Arc::new(Mutex::new(DistributedState::new())));
288        });
289        #[allow(static_mut_refs)]
290        DISTRIBUTED_STATE.as_ref().unwrap()
291    }
292}

294/// Initialize distributed training
295/// 分散学習を初期化
296pub fn init_distributed(
297    backend: DistributedBackend,
298    rank: usize,
299    world_size: usize,
300    master_addr: String,
301    master_port: u16,
302) -> RusTorchResult<()> {
303    let process_group = ProcessGroup::new(rank, world_size, backend, master_addr, master_port);
304    process_group.init()?;
305
306    let state = get_distributed_state();
307    let mut state_guard = state.lock().unwrap();
308    state_guard.set_process_group(process_group);
309
310    Ok(())
311}
312
313/// Check if distributed training is available
314/// 分散学習が利用可能かチェック
315pub fn is_available() -> bool {
316    let state = get_distributed_state();
317    let state_guard = state.lock().unwrap();
318    state_guard.is_initialized()
319}

321/// Get current rank in distributed training
322/// 分散学習での現在のランクを取得
323pub fn get_rank() -> Option<usize> {
324    let state = get_distributed_state();
325    let state_guard = state.lock().unwrap();
326    state_guard.rank()
327}
328
329/// Get world size in distributed training
330/// 分散学習でのワールドサイズを取得
331pub fn get_world_size() -> Option<usize> {
332    let state = get_distributed_state();
333    let state_guard = state.lock().unwrap();
334    state_guard.world_size()
335}
336
337/// Finalize distributed training
338/// 分散学習を終了
339pub fn finalize() -> RusTorchResult<()> {
340    let state = get_distributed_state();
341    let mut state_guard = state.lock().unwrap();
342    state_guard.process_group = None;
343    state_guard.devices.clear();
344    state_guard.device_map.clear();
345
346    Ok(())
347}
