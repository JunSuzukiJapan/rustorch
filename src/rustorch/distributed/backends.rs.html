10/// NCCL backend for NVIDIA GPU communication
11/// NVIDIA GPU通信用NCCLバックエンド
12#[cfg(feature = "nccl")]
13pub struct NCCLBackend {
14    process_group: ProcessGroup,
15    comm: *mut std::ffi::c_void,
16}
17
18#[cfg(feature = "nccl")]
19impl NCCLBackend {
20    pub fn new(process_group: ProcessGroup) -> RusTorchResult<Self> {
21        Ok(Self {
22            process_group,
23            comm: std::ptr::null_mut(),
24        })
25    }
26}
27
28#[cfg(feature = "nccl")]
29impl<T: Float + Send + Sync + 'static> DistributedOps<T> for NCCLBackend {
30    fn all_reduce(&self, tensor: &mut Tensor<T>, op: ReduceOp) -> RusTorchResult<()> {
31        CommonOps::validate_tensor(tensor)?;
32        CommonOps::default_all_reduce(tensor, op)
33    }
34
35    fn all_gather(&self, tensor: &Tensor<T>) -> RusTorchResult<Vec<Tensor<T>>> {
36        CommonOps::validate_tensor(tensor)?;
37        CommonOps::default_all_gather(tensor, self.process_group.world_size)
38    }
39
40    fn broadcast(&self, tensor: &mut Tensor<T>, root: usize) -> RusTorchResult<()> {
41        CommonOps::validate_tensor(tensor)?;
42        CommonOps::default_broadcast(tensor, root)
43    }
44
45    fn gather(&self, tensor: &Tensor<T>, root: usize) -> RusTorchResult<Vec<Tensor<T>>> {
46        CommonOps::validate_tensor(tensor)?;
47        CommonOps::default_gather(tensor, self.process_group.world_size, root)
48    }
49
50    fn scatter(&self, tensors: &[Tensor<T>], _root: usize) -> RusTorchResult<Tensor<T>> {
51        if tensors.is_empty() {
52            return Err(RusTorchError::tensor_op(
53                "Empty tensor array for scatter operation",
54            ));
55        }
56        Ok(tensors[0].clone())
57    }
58
59    fn reduce(&self, _tensor: &mut Tensor<T>, _root: usize, _op: ReduceOp) -> RusTorchResult<()> {
60        Ok(())
61    }
62}

64#[cfg(feature = "nccl")]
65impl<T: Float + Send + Sync + 'static> BackendOptimizations<T> for NCCLBackend {}

200/// Backend factory
201/// バックエンドファクトリ
202pub struct BackendFactory;
203
204impl BackendFactory {
205    /// Create backend instance based on process group configuration
206    /// プロセスグループ設定に基づいてバックエンドインスタンスを作成
207    pub fn create_backend<T: Float + Send + Sync + 'static>(
208        process_group: ProcessGroup,
209    ) -> RusTorchResult<Box<dyn DistributedOps<T> + Send + Sync>> {
210        match process_group.backend {
211            #[cfg(feature = "nccl")]
212            super::DistributedBackend::NCCL => {
213                let backend = NCCLBackend::new(process_group)?;
214                Ok(Box::new(backend))
215            }
216            super::DistributedBackend::Gloo => {
217                let backend = GlooBackend::new(process_group)?;
218                Ok(Box::new(backend))
219            }
220            super::DistributedBackend::TCP => {
221                let backend = TCPBackend::new(process_group)?;
222                Ok(Box::new(backend))
223            }
224            #[cfg(not(feature = "nccl"))]
225            super::DistributedBackend::NCCL => {
226                Err(RusTorchError::backend_unavailable("NCCL not compiled"))
227            }
228            super::DistributedBackend::MPI => {
229                Err(RusTorchError::backend_unavailable("MPI not implemented"))
230            }
231        }
232    }
233}
