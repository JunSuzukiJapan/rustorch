20/// DistributedDataParallel wrapper for PyTorch compatibility
21/// PyTorch互換性のためのDistributedDataParallelラッパー
22#[derive(Debug)]
23pub struct DistributedDataParallel<T: DistributedScalar, M: Module<T>> {
24    /// The wrapped module
25    /// ラップされたモジュール
26    module: Arc<Mutex<M>>,
27    /// Process group for communication
28    /// 通信用プロセスグループ
29    process_group: Option<ProcessGroup>,
30    /// Device IDs for this process
31    /// このプロセスのデバイスID
32    device_ids: Vec<usize>,
33    /// Output device ID
34    /// 出力デバイスID
35    output_device: Option<usize>,
36    /// Bucket size for gradient communication
37    /// 勾配通信用バケットサイズ
38    bucket_cap_mb: usize,
39    /// Find unused parameters
40    /// 未使用パラメータを検出
41    find_unused_parameters: bool,
42    /// Gradient as bucket view
43    /// バケットビューとしての勾配
44    gradient_as_bucket_view: bool,
45    /// Static graph optimization
46    /// 静的グラフ最適化
47    static_graph: bool,
48    /// Gradient accumulation state
49    /// 勾配累積状態
50    gradient_state: Arc<Mutex<GradientState<T>>>,
51}

53/// Gradient accumulation and synchronization state
54/// 勾配累積と同期状態
55#[derive(Debug)]
56struct GradientState<T: DistributedScalar> {
57    /// Accumulated gradients per parameter
58    /// パラメータ毎の累積勾配
59    accumulated_grads: HashMap<String, Tensor<T>>,
60    /// Whether gradients are ready for synchronization
61    /// 勾配が同期準備完了かどうか
62    ready_for_sync: bool,
63    /// Bucket management for efficient communication
64    /// 効率的な通信のためのバケット管理
65    buckets: Vec<GradientBucket<T>>,
66}

68/// Gradient bucket for efficient communication
69/// 効率的な通信のための勾配バケット
70#[derive(Debug, Clone)]
71struct GradientBucket<T: DistributedScalar> {
72    /// Parameters in this bucket
73    /// このバケット内のパラメータ
74    parameters: Vec<String>,
75    /// Combined gradient tensor
76    /// 結合勾配テンソル
77    gradient: Option<Tensor<T>>,
78    /// Bucket size in bytes
79    /// バケットサイズ（バイト）
80    size_bytes: usize,
81}

83impl<T: DistributedScalar, M: Module<T> + Send + Sync + 'static> DistributedDataParallel<T, M> {
84    /// Create a new DistributedDataParallel wrapper
85    /// 新しいDistributedDataParallelラッパーを作成
86    pub fn new(
87        module: M,
88        device_ids: Option<Vec<usize>>,
89        output_device: Option<usize>,
90        dim: Option<usize>,
91        broadcast_buffers: bool,
92        process_group: Option<ProcessGroup>,
93        bucket_cap_mb: Option<usize>,
94        find_unused_parameters: Option<bool>,
95        check_reduction: Option<bool>,
96        gradient_as_bucket_view: Option<bool>,
97        static_graph: Option<bool>,
98    ) -> RusTorchResult<Self> {
99        if !api::is_initialized() {
100            return Err(RusTorchError::distributed(
101                "Distributed process group not initialized. Call distributed::init_process_group() first."
102            ));
103        }
104
105        let device_ids = device_ids.unwrap_or_else(|| vec![0]);
106        let bucket_cap_mb = bucket_cap_mb.unwrap_or(25); // Default 25MB
107        let find_unused_parameters = find_unused_parameters.unwrap_or(false);
108        let gradient_as_bucket_view = gradient_as_bucket_view.unwrap_or(false);
109        let static_graph = static_graph.unwrap_or(false);
110
111        // Initialize gradient state
112        let gradient_state = Arc::new(Mutex::new(GradientState {
113            accumulated_grads: HashMap::new(),
114            ready_for_sync: false,
115            buckets: Vec::new(),
116        }));
117
118        let _ = (dim, broadcast_buffers, check_reduction); // Note: advanced DDP features not implemented in current version
119
120        Ok(Self {
121            module: Arc::new(Mutex::new(module)),
122            process_group,
123            device_ids,
124            output_device,
125            bucket_cap_mb,
126            find_unused_parameters,
127            gradient_as_bucket_view,
128            static_graph,
129            gradient_state,
130        })
131    }

133    /// Perform forward pass with distributed synchronization
134    /// 分散同期付きフォワードパス
135    pub fn forward(&self, input: &Variable<T>) -> RusTorchResult<Variable<T>> {
136        let module = self.module.lock().unwrap();
137        let output = module.forward(input);
138
139        // Register backward hook for gradient synchronization
140        // 勾配同期のためのバックワードフック登録
141        self.register_grad_hooks()?;
142
143        Ok(output)
144    }
145
146    /// Register gradient synchronization hooks
147    /// 勾配同期フック登録
148    fn register_grad_hooks(&self) -> RusTorchResult<()> {
149        // In a full implementation, this would register hooks on all parameters
150        // 完全な実装では、全パラメータにフックを登録
151        Ok(())
152    }
153
154    /// Synchronize gradients across all processes
155    /// 全プロセス間での勾配同期
156    pub fn sync_gradients(&self) -> RusTorchResult<()> {
157        let module = self.module.lock().unwrap();
158
159        // Get all parameters with gradients
160        let parameters = module.parameters();
161
162        for param in parameters {
163            let grad_lock = param.grad();
164            let mut grad_guard = grad_lock.write().unwrap();
165            if let Some(ref mut grad) = *grad_guard {
166                // Perform all-reduce on gradient
167                api::all_reduce(grad, ReduceOp::Average, self.process_group.as_ref(), false)?;
168            }
169        }
170
171        Ok(())
172    }
173
174    /// Get the wrapped module
175    /// ラップされたモジュールを取得
176    pub fn module(&self) -> Arc<Mutex<M>> {
177        Arc::clone(&self.module)
178    }
179
180    /// Get device IDs
181    /// デバイスIDを取得
182    pub fn device_ids(&self) -> &[usize] {
183        &self.device_ids
184    }
185
186    /// Check if this is a DDP module
187    /// DDPモジュールかどうかをチェック
188    pub fn is_ddp_module() -> bool {
189        true
190    }
191}

232impl<T: DistributedScalar> GradientState<T> {
233    /// Create gradient buckets for efficient communication
234    /// 効率的な通信のための勾配バケット作成
235    fn create_buckets(&mut self, bucket_size_mb: usize) -> RusTorchResult<()> {
236        let bucket_size_bytes = bucket_size_mb * 1024 * 1024;
237
238        // Group parameters into buckets based on size
239        // サイズに基づいてパラメータをバケットにグループ化
240        let mut current_bucket = GradientBucket {
241            parameters: Vec::new(),
242            gradient: None,
243            size_bytes: 0,
244        };
245
246        for param_name in self.accumulated_grads.keys() {
247            if let Some(grad) = self.accumulated_grads.get(param_name) {
248                let grad_size = grad.numel() * std::mem::size_of::<T>();
249
250                if current_bucket.size_bytes + grad_size > bucket_size_bytes
251                    && !current_bucket.parameters.is_empty()
252                {
253                    self.buckets.push(current_bucket.clone());
254                    current_bucket = GradientBucket {
255                        parameters: Vec::new(),
256                        gradient: None,
257                        size_bytes: 0,
258                    };
259                }
260
261                current_bucket.parameters.push(param_name.clone());
262                current_bucket.size_bytes += grad_size;
263            }
264        }
265
266        if !current_bucket.parameters.is_empty() {
267            self.buckets.push(current_bucket);
268        }
269
270        Ok(())
271    }
272}

274/// Convenience function to wrap a module in DDP
275/// モジュールをDDPでラップするための便利関数
276pub fn wrap_module<T: DistributedScalar, M: Module<T> + Send + Sync + 'static>(
277    module: M,
278    device_ids: Option<Vec<usize>>,
279) -> RusTorchResult<DistributedDataParallel<T, M>> {
280    DistributedDataParallel::new(
281        module, device_ids, None, // output_device
282        None, // dim
283        true, // broadcast_buffers
284        None, // process_group
285        None, // bucket_cap_mb
286        None, // find_unused_parameters
287        None, // check_reduction
288        None, // gradient_as_bucket_view
289        None, // static_graph
290    )
291}
