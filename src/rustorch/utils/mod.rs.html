6/// Converts a vector to a column vector (2D tensor with shape [n, 1])
7pub fn to_column_vector<T: Float>(data: Vec<T>) -> ArrayD<T> {
8    let n = data.len();
9    ArrayD::from_shape_vec(vec![n, 1], data).expect("Failed to create column vector")
10}
11
12/// Converts a vector to a row vector (2D tensor with shape [1, n])
13pub fn to_row_vector<T: Float>(data: Vec<T>) -> ArrayD<T> {
14    let n = data.len();
15    ArrayD::from_shape_vec(vec![1, n], data).expect("Failed to create row vector")
16}

18/// Initializes weights using Kaiming (He) initialization
19pub fn kaiming_init<T: Float>(shape: &[usize], fan_in: usize) -> Vec<T> {
20    let bound = T::from(1.0).unwrap() / T::from(fan_in as f32).unwrap().sqrt();
21    (0..shape.iter().product())
22        .map(|_| T::from(rand::random::<f32>()).unwrap() * (bound + bound) - bound)
23        .collect()
24}

26/// Computes the mean squared error between two tensors
27pub fn mse_loss<T>(output: &ArrayD<T>, target: &ArrayD<T>) -> T
28where
29    T: Float + FromPrimitive + ScalarOperand + std::ops::AddAssign + std::iter::Sum,
30{
31    assert_eq!(
32        output.shape(),
33        target.shape(),
34        "Shapes must match for MSE loss"
35    );
36
37    let diff = output - target;
38    let squared_diff = &diff * &diff;
39    let sum: T = squared_diff.iter().cloned().sum();
40    let count = T::from_usize(squared_diff.len()).expect("Failed to convert length to float");
41    sum / count
42}

44/// Computes the softmax of a 1D tensor
45pub fn softmax<T>(x: &ArrayD<T>) -> ArrayD<T>
46where
47    T: Float + ScalarOperand + std::ops::Sub<Output = T> + std::ops::Div<Output = T>,
48{
49    let max = x.fold(T::neg_infinity(), |a, &b| a.max(b));
50    let exp = x.mapv(|v| (v - max).exp());
51    let sum = exp.sum();
52    exp / sum
53}
