1//! Keras風のSequential API
2//! Keras-like Sequential API
3//!
4//! このモジュールは、レイヤーを順次積み重ねてニューラルネットワークを
5//! 構築するためのシンプルで直感的なAPIを提供します。
6//!
7//! ## 使用例
8//!
9//! ```no_run
10//! use rustorch::models::sequential::Sequential;
11//! use rustorch::nn::{Linear, ReLU, Dropout};
12//! use rustorch::optim::Adam;
13//! use rustorch::nn::loss::CrossEntropyLoss;
14//!
15//! let mut model = Sequential::new()
16//!     .add(Linear::new(784, 128))
17//!     .add(ReLU::new())
18//!     .add(Dropout::new(0.2, false))
19//!     .add(Linear::new(128, 64))
20//!     .add(ReLU::new())
21//!     .add(Linear::new(64, 10));
22//!
23//! // モデルのコンパイル
24//! model.compile(
25//!     Adam::default_params(0.001),
26//!     CrossEntropyLoss::new(),
27//!     vec!["accuracy".to_string()]
28//! );
29//!
30//! // 訓練（実際の実装では DataLoader を使用）
31//! // model.fit(train_data, validation_data, epochs, batch_size, verbose);
32//! ```

43/// Sequential APIの中核となるモデルクラス
44/// Core model class for Sequential API
45pub struct Sequential<T>
46where
47    T: Float
48        + Send
49        + Sync
50        + 'static
51        + Debug
52        + Clone
53        + ndarray::ScalarOperand
54        + num_traits::FromPrimitive,
55{
56    /// レイヤーのリスト
57    layers: Vec<Box<dyn Module<T> + Send + Sync>>,
58    /// コンパイル済みかどうか
59    compiled: bool,
60    /// オプティマイザー（コンパイル後に設定）
61    optimizer: Option<Box<dyn Optimizer + Send + Sync>>,
62    /// 損失関数（コンパイル後に設定）
63    loss_fn: Option<Box<dyn Loss<T> + Send + Sync>>,
64    /// メトリクス（コンパイル後に設定）
65    metrics: Vec<String>,
66    /// 訓練モードかどうか
67    training: bool,
68    /// モデルの名前（オプション）
69    name: Option<String>,
70}

173            return Err(anyhow::anyhow!(
174                "Index {} out of bounds for {} layers",
175                index,
176                self.layers.len()
177            ));
178        }
179        self.layers.remove(index);
180        Ok(())
181    }

195    /// モデルをコンパイル
196    /// Compile the model
197    pub fn compile<O, L>(&mut self, optimizer: O, loss_fn: L, metrics: Vec<String>) -> Result<()>
198    where
199        O: Optimizer + Send + Sync + 'static,
200        L: Loss<T> + Send + Sync + 'static,
201    {
202        if self.layers.is_empty() {
203            return Err(anyhow::anyhow!(
204                "Cannot compile empty model. Add layers first."
205            ));
206        }
207
208        self.optimizer = Some(Box::new(optimizer));
209        self.loss_fn = Some(Box::new(loss_fn));
210        self.metrics = metrics;
211        self.compiled = true;
212
213        Ok(())
214    }

222    /// モデルの概要を表示
223    /// Display model summary
224    pub fn summary(&self) -> String {
225        let mut summary = String::new();
226
227        if let Some(ref name) = self.name {
228            summary.push_str(&format!("Model: \"{}\"\n", name));
229        } else {
230            summary.push_str("Sequential Model\n");
231        }
232
233        summary.push_str("_________________________________________________________________\n");
234        summary.push_str("Layer (type)                 Output Shape              Param #   \n");
235        summary.push_str("=================================================================\n");
236
237        let mut total_params = 0;
238        for (i, layer) in self.layers.iter().enumerate() {
239            let layer_info = format!("Layer_{} (type)", i);
240            summary.push_str(&format!("{:<2} {}\n", i, layer_info));
241            total_params += layer.parameters().len();
242        }
243
244        summary.push_str("=================================================================\n");
245        summary.push_str(&format!("Total params: {}\n", total_params));
246        summary.push_str(&format!("Trainable params: {}\n", total_params)); // 簡略化
247        summary.push_str(&format!("Non-trainable params: 0\n"));
248        summary.push_str("_________________________________________________________________\n");
249
250        if self.compiled {
251            summary.push_str("\nModel compiled with:\n");
252            summary.push_str(&format!(" - Optimizer: {}\n", "Configured")); // 簡略化
253            summary.push_str(&format!(" - Loss: {}\n", "Configured")); // 簡略化
254            summary.push_str(&format!(" - Metrics: {:?}\n", self.metrics));
255        } else {
256            summary.push_str("\nModel not compiled yet. Call compile() before training.\n");
257        }
258
259        summary
260    }

334impl<T> Module<T> for Sequential<T>
335where
336    T: Float
337        + Send
338        + Sync
339        + 'static
340        + Debug
341        + Clone
342        + ndarray::ScalarOperand
343        + num_traits::FromPrimitive,
344{
345    /// 順伝播
346    /// Forward pass
347    fn forward(&self, input: &Variable<T>) -> Variable<T> {
348        let mut output = input.clone();
349
350        for layer in &self.layers {
351            output = layer.forward(&output);
352        }
353
354        output
355    }
356
357    /// 訓練モードに設定
358    /// Set training mode
359    fn train(&mut self) {
360        self.training = true;
361        for layer in &mut self.layers {
362            layer.train();
363        }
364    }
365
366    /// 評価モードに設定
367    /// Set evaluation mode  
368    fn eval(&mut self) {
369        self.training = false;
370        for layer in &mut self.layers {
371            layer.eval();
372        }
373    }
374
375    /// パラメータを取得
376    /// Get parameters
377    fn parameters(&self) -> Vec<Variable<T>> {
378        let mut params = Vec::new();
379        for layer in &self.layers {
380            let mut layer_params = layer.parameters();
381            params.append(&mut layer_params);
382        }
383        params
384    }
385
386    /// Any型への変換
387    /// Convert to Any type
388    fn as_any(&self) -> &dyn std::any::Any {
389        self
390    }
391}
