27/// モデル保存エラー
28/// Model save error
29#[derive(Debug)]
30pub enum SaveError {
31    /// ファイルI/Oエラー
32    IoError(std::io::Error),
33    /// シリアライゼーションエラー
34    SerializationError(String),
35    /// 形式エラー
36    FormatError(String),
37}
38
39impl std::fmt::Display for SaveError {
40    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
41        match self {
42            SaveError::IoError(e) => write!(f, "IO error: {}", e),
43            SaveError::SerializationError(e) => write!(f, "Serialization error: {}", e),
44            SaveError::FormatError(e) => write!(f, "Format error: {}", e),
45        }
46    }
47}
48
49impl std::error::Error for SaveError {
50    fn source(&self) -> Option<&(dyn std::error::Error + 'static)> {
51        match self {
52            SaveError::IoError(e) => Some(e),
53            _ => None,
54        }
55    }
56}
57
58impl From<std::io::Error> for SaveError {
59    fn from(error: std::io::Error) -> Self {
60        SaveError::IoError(error)
61    }
62}
63
64impl From<serde_json::Error> for SaveError {
65    fn from(error: serde_json::Error) -> Self {
66        SaveError::SerializationError(error.to_string())
67    }
68}

116/// モデル状態辞書
117/// Model state dictionary
118#[derive(Debug, Clone, Serialize, Deserialize)]
119pub struct ModelStateDict {
120    /// パラメータ
121    pub parameters: HashMap<String, Vec<f64>>,
122    /// メタデータ
123    pub metadata: HashMap<String, String>,
124    /// モデル設定
125    pub config: HashMap<String, String>,
126    /// バージョン情報
127    pub version: String,
128    /// 作成日時
129    pub created_at: String,
130}
131
132impl ModelStateDict {
133    /// 新しい状態辞書を作成
134    /// Create a new state dictionary
135    pub fn new() -> Self {
136        ModelStateDict {
137            parameters: HashMap::new(),
138            metadata: HashMap::new(),
139            config: HashMap::new(),
140            version: std::env::var("CARGO_PKG_VERSION").unwrap_or_else(|_| "0.2.2".to_string()),
141            created_at: chrono::Utc::now().to_rfc3339(),
142        }
143    }
144
145    /// パラメータを追加
146    /// Add parameter
147    pub fn add_parameter(&mut self, name: String, values: Vec<f64>) {
148        self.parameters.insert(name, values);
149    }
150
151    /// メタデータを追加
152    /// Add metadata
153    pub fn add_metadata(&mut self, key: String, value: String) {
154        self.metadata.insert(key, value);
155    }
156
157    /// 設定を追加
158    /// Add configuration
159    pub fn add_config(&mut self, key: String, value: String) {
160        self.config.insert(key, value);
161    }
162
163    /// パラメータ数を取得
164    /// Get parameter count
165    pub fn parameter_count(&self) -> usize {
166        self.parameters.values().map(|v| v.len()).sum()
167    }
168
169    /// サイズを取得（バイト）
170    /// Get size in bytes
171    pub fn size_bytes(&self) -> usize {
172        // 概算サイズ計算
173        self.parameter_count() * std::mem::size_of::<f64>()
174            + self
175                .metadata
176                .iter()
177                .map(|(k, v)| k.len() + v.len())
178                .sum::<usize>()
179            + self
180                .config
181                .iter()
182                .map(|(k, v)| k.len() + v.len())
183                .sum::<usize>()
184    }
185}
186
187impl Default for ModelStateDict {
188    fn default() -> Self {
189        Self::new()
190    }
191}

193/// モデル保存器
194/// Model saver
195pub struct ModelSaver;
196
197impl ModelSaver {
198    /// モデルを保存
199    /// Save model
200    pub fn save<T, M>(model: &M, path: &Path, format: SerializationFormat) -> Result<(), SaveError>
201    where
202        T: Float
203            + 'static
204            + Send
205            + Sync
206            + Debug
207            + ndarray::ScalarOperand
208            + num_traits::FromPrimitive,
209        M: Model<T>,
210    {
211        let state_dict = Self::extract_state_dict(model);
212
213        match format {
214            SerializationFormat::Json => Self::save_json(&state_dict, path),
215            SerializationFormat::Binary => Self::save_binary(&state_dict, path),
216            SerializationFormat::MessagePack => Self::save_messagepack(&state_dict, path),
217        }
218    }
219
220    /// モデルから状態辞書を抽出
221    /// Extract state dictionary from model
222    fn extract_state_dict<T, M>(model: &M) -> ModelStateDict
223    where
224        T: Float
225            + 'static
226            + Send
227            + Sync
228            + Debug
229            + ndarray::ScalarOperand
230            + num_traits::FromPrimitive,
231        M: Model<T>,
232    {
233        let mut state_dict = ModelStateDict::new();
234
235        // モデル設定を追加
236        for (key, value) in model.config() {
237            state_dict.add_config(key, value);
238        }
239
240        // パラメータを抽出（簡略化実装）
241        let parameters = model.parameters();
242        for (i, param) in parameters.iter().enumerate() {
243            let param_name = format!("param_{}", i);
244            let values = Self::extract_parameter_values(param);
245            state_dict.add_parameter(param_name, values);
246        }
247
248        // メタデータを追加
249        state_dict.add_metadata("model_summary".to_string(), model.summary());
250        state_dict.add_metadata("mode".to_string(), format!("{:?}", model.mode()));
251
252        state_dict
253    }
254
255    /// パラメータ値を抽出
256    /// Extract parameter values
257    fn extract_parameter_values<T>(_param: &Variable<T>) -> Vec<f64>
258    where
259        T: Float
260            + 'static
261            + Send
262            + Sync
263            + Debug
264            + ndarray::ScalarOperand
265            + num_traits::FromPrimitive,
266    {
267        // 実装は簡略化 - 実際にはテンソルから値を抽出
268        vec![0.0; 100] // プレースホルダー
269    }
270
271    /// JSON形式で保存
272    /// Save in JSON format
273    fn save_json(state_dict: &ModelStateDict, path: &Path) -> Result<(), SaveError> {
274        let file = File::create(path)?;
275        let writer = BufWriter::new(file);
276        serde_json::to_writer_pretty(writer, state_dict)?;
277        Ok(())
278    }
279
280    /// バイナリ形式で保存
281    /// Save in binary format
282    fn save_binary(state_dict: &ModelStateDict, path: &Path) -> Result<(), SaveError> {
283        let json_data = serde_json::to_vec(state_dict)?;
284        let mut file = File::create(path)?;
285        file.write_all(&json_data)?;
286        Ok(())
287    }
288
289    /// MessagePack形式で保存
290    /// Save in MessagePack format
291    fn save_messagepack(_state_dict: &ModelStateDict, _path: &Path) -> Result<(), SaveError> {
292        // MessagePack実装は省略（rmp-serdeクレートが必要）
293        Err(SaveError::FormatError(
294            "MessagePack not implemented".to_string(),
295        ))
296    }
297
298    /// チェックポイントを保存
299    /// Save checkpoint
300    pub fn save_checkpoint<T, M>(
301        model: &M,
302        epoch: usize,
303        loss: f64,
304        optimizer_state: Option<&HashMap<String, Vec<f64>>>,
305        path: &Path,
306    ) -> Result<(), SaveError>
307    where
308        T: Float
309            + 'static
310            + Send
311            + Sync
312            + Debug
313            + ndarray::ScalarOperand
314            + num_traits::FromPrimitive,
315        M: Model<T>,
316    {
317        let mut state_dict = Self::extract_state_dict(model);
318
319        // チェックポイント情報を追加
320        state_dict.add_metadata("checkpoint_epoch".to_string(), epoch.to_string());
321        state_dict.add_metadata("checkpoint_loss".to_string(), loss.to_string());
322
323        if let Some(opt_state) = optimizer_state {
324            for (key, values) in opt_state {
325                state_dict.add_parameter(format!("optimizer_{}", key), values.clone());
326            }
327        }
328
329        Self::save_json(&state_dict, path)
330    }
331}

333/// モデル読み込み器
334/// Model loader
335pub struct ModelLoader;
336
337impl ModelLoader {
338    /// モデルを読み込み
339    /// Load model
340    pub fn load(path: &Path, format: SerializationFormat) -> Result<ModelStateDict, LoadError> {
341        match format {
342            SerializationFormat::Json => Self::load_json(path),
343            SerializationFormat::Binary => Self::load_binary(path),
344            SerializationFormat::MessagePack => Self::load_messagepack(path),
345        }
346    }
347
348    /// JSON形式で読み込み
349    /// Load from JSON format
350    fn load_json(path: &Path) -> Result<ModelStateDict, LoadError> {
351        let file = File::open(path)?;
352        let reader = BufReader::new(file);
353        let state_dict: ModelStateDict = serde_json::from_reader(reader)?;
354        Self::validate_state_dict(&state_dict)?;
355        Ok(state_dict)
356    }
357
358    /// バイナリ形式で読み込み
359    /// Load from binary format
360    fn load_binary(path: &Path) -> Result<ModelStateDict, LoadError> {
361        let mut file = File::open(path)?;
362        let mut buffer = Vec::new();
363        file.read_to_end(&mut buffer)?;
364        let state_dict: ModelStateDict = serde_json::from_slice(&buffer)?;
365        Self::validate_state_dict(&state_dict)?;
366        Ok(state_dict)
367    }
368
369    /// MessagePack形式で読み込み
370    /// Load from MessagePack format
371    fn load_messagepack(_path: &Path) -> Result<ModelStateDict, LoadError> {
372        // MessagePack実装は省略
373        Err(LoadError::FormatError(
374            "MessagePack not implemented".to_string(),
375        ))
376    }
377
378    /// 状態辞書を検証
379    /// Validate state dictionary
380    fn validate_state_dict(state_dict: &ModelStateDict) -> Result<(), LoadError> {
381        // バージョン互換性チェック
382        let current_version =
383            std::env::var("CARGO_PKG_VERSION").unwrap_or_else(|_| "0.2.2".to_string());
384        if state_dict.version != current_version {
385            println!(
386                "Warning: Model version {} differs from current version {}",
387                state_dict.version, current_version
388            );
389        }
390
391        // 必須フィールドの存在チェック
392        if state_dict.parameters.is_empty() {
393            return Err(LoadError::DeserializationError(
394                "No parameters found in state dictionary".to_string(),
395            ));
396        }
397
398        Ok(())
399    }
400
401    /// チェックポイントを読み込み
402    /// Load checkpoint
403    pub fn load_checkpoint(path: &Path) -> Result<(ModelStateDict, CheckpointInfo), LoadError> {
404        let state_dict = Self::load_json(path)?;
405
406        let epoch = state_dict
407            .metadata
408            .get("checkpoint_epoch")
409            .and_then(|s| s.parse().ok())
410            .unwrap_or(0);
411
412        let loss = state_dict
413            .metadata
414            .get("checkpoint_loss")
415            .and_then(|s| s.parse().ok())
416            .unwrap_or(0.0);
417
418        let checkpoint_info = CheckpointInfo { epoch, loss };
419
420        Ok((state_dict, checkpoint_info))
421    }
422
423    /// 形式を自動検出
424    /// Auto-detect format
425    pub fn detect_format(path: &Path) -> Result<SerializationFormat, LoadError> {
426        if let Some(extension) = path.extension() {
427            match extension.to_str() {
428                Some("json") => Ok(SerializationFormat::Json),
429                Some("bin" | "dat") => Ok(SerializationFormat::Binary),
430                Some("msgpack" | "mp") => Ok(SerializationFormat::MessagePack),
431                _ => Err(LoadError::FormatError(format!(
432                    "Unknown file extension: {:?}",
433                    extension
434                ))),
435            }
436        } else {
437            Err(LoadError::FormatError(
438                "No file extension found".to_string(),
439            ))
440        }
441    }
442}

454/// モデル変換器
455/// Model converter
456pub struct ModelConverter;
457
458impl ModelConverter {
459    /// 形式を変換
460    /// Convert format
461    pub fn convert(
462        input_path: &Path,
463        output_path: &Path,
464        input_format: SerializationFormat,
465        _output_format: SerializationFormat,
466    ) -> RusTorchResult<()> {
467        let state_dict = ModelLoader::load(input_path, input_format)?;
468        ModelSaver::save_json(&state_dict, output_path)?;
469        Ok(())
470    }
471
472    /// モデルを圧縮
473    /// Compress model
474    pub fn compress(
475        input_path: &Path,
476        output_path: &Path,
477        compression_ratio: f64,
478    ) -> RusTorchResult<()> {
479        let mut state_dict = ModelLoader::load(input_path, SerializationFormat::Json)?;
480
481        // パラメータを量子化（簡略化実装）
482        for (_name, params) in &mut state_dict.parameters {
483            Self::quantize_parameters(params, compression_ratio);
484        }
485
486        state_dict.add_metadata("compressed".to_string(), "true".to_string());
487        state_dict.add_metadata(
488            "compression_ratio".to_string(),
489            compression_ratio.to_string(),
490        );
491
492        ModelSaver::save_json(&state_dict, output_path)?;
493        Ok(())
494    }
495
496    /// パラメータを量子化
497    /// Quantize parameters
498    fn quantize_parameters(params: &mut Vec<f64>, ratio: f64) {
499        // 簡単な量子化実装
500        for param in params.iter_mut() {
501            *param = (*param * ratio).round() / ratio;
502        }
503    }
504}

510impl ModelInfo {
511    /// モデル情報を表示
512    /// Display model information
513    pub fn display(path: &Path) -> Result<(), LoadError> {
514        let format = ModelLoader::detect_format(path)?;
515        let state_dict = ModelLoader::load(path, format)?;
516
517        println!("Model Information:");
518        println!("  Path: {:?}", path);
519        println!("  Format: {:?}", format);
520        println!("  Version: {}", state_dict.version);
521        println!("  Created: {}", state_dict.created_at);
522        println!("  Parameters: {}", state_dict.parameter_count());
523        println!(
524            "  Size: {:.2} MB",
525            state_dict.size_bytes() as f64 / 1024.0 / 1024.0
526        );
527
528        println!("\nConfiguration:");
529        for (key, value) in &state_dict.config {
530            println!("  {}: {}", key, value);
531        }
532
533        println!("\nMetadata:");
534        for (key, value) in &state_dict.metadata {
535            println!("  {}: {}", key, value);
536        }
537
538        Ok(())
539    }

541    /// モデルを比較
542    /// Compare models
543    pub fn compare(path1: &Path, path2: &Path) -> Result<(), LoadError> {
544        let format1 = ModelLoader::detect_format(path1)?;
545        let format2 = ModelLoader::detect_format(path2)?;
546
547        let state_dict1 = ModelLoader::load(path1, format1)?;
548        let state_dict2 = ModelLoader::load(path2, format2)?;
549
550        println!("Model Comparison:");
551        println!("  Model 1: {:?}", path1);
552        println!("  Model 2: {:?}", path2);
553
554        println!("\nParameter Count:");
555        println!("  Model 1: {}", state_dict1.parameter_count());
556        println!("  Model 2: {}", state_dict2.parameter_count());
557
558        println!("\nSize:");
559        println!(
560            "  Model 1: {:.2} MB",
561            state_dict1.size_bytes() as f64 / 1024.0 / 1024.0
562        );
563        println!(
564            "  Model 2: {:.2} MB",
565            state_dict2.size_bytes() as f64 / 1024.0 / 1024.0
566        );
567
568        // 設定の違いをチェック
569        let mut config_diffs = Vec::new();
570        for (key, value1) in &state_dict1.config {
571            if let Some(value2) = state_dict2.config.get(key) {
572                if value1 != value2 {
573                    config_diffs.push((key.clone(), value1.clone(), value2.clone()));
574                }
575            } else {
576                config_diffs.push((key.clone(), value1.clone(), "N/A".to_string()));
577            }
578        }
579
580        if !config_diffs.is_empty() {
581            println!("\nConfiguration Differences:");
582            for (key, val1, val2) in config_diffs {
583                println!("  {}: {} vs {}", key, val1, val2);
584            }
585        }
586
587        Ok(())
588    }
