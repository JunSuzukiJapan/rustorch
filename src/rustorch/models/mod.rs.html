46/// 基本モデルトレイト
47/// Base model trait
48pub trait Model<T>: Module<T>
49where
50    T: Float + 'static + Send + Sync + ndarray::ScalarOperand + num_traits::FromPrimitive,
51{
52    /// モデルを訓練モードに設定
53    /// Set model to training mode
54    fn train(&mut self);
55
56    /// モデルを評価モードに設定
57    /// Set model to evaluation mode
58    fn eval(&mut self);
59
60    /// 現在のモードを取得
61    /// Get current mode
62    fn mode(&self) -> ModelMode;
63
64    /// モデルの設定を取得
65    /// Get model configuration
66    fn config(&self) -> HashMap<String, String>;
67
68    /// モデルの概要を表示
69    /// Display model summary
70    fn summary(&self) -> String;
71}

78{
79    /// The model type that this builder creates

86}

101/// Inference engine for model evaluation
102/// モデル評価用の推論エンジン
103#[derive(Debug)]
104pub struct InferenceEngine<T: Float + Send + Sync + 'static> {
105    _phantom: std::marker::PhantomData<T>,
106}
107
108impl<T: Float + Send + Sync + 'static + ndarray::ScalarOperand + num_traits::FromPrimitive>
109    InferenceEngine<T>
110{
111    /// Create a new inference engine
112    /// 新しい推論エンジンを作成
113    pub fn new() -> Self {
114        Self {
115            _phantom: std::marker::PhantomData,
116        }
117    }
118
119    /// Perform prediction using the given model
120    /// 指定されたモデルを使用して予測を実行
121    pub fn predict<M: Model<T>>(&self, model: &M, input: &Variable<T>) -> Variable<T> {
122        model.forward(input)
123    }
124}

126/// Evaluation metrics for model performance
127/// モデル性能の評価メトリクス
128#[derive(Debug, Clone)]
129pub struct Metrics {
130    /// Model accuracy score
131    /// モデルの精度スコア
132    pub accuracy: f64,
133    /// Model precision score
134    /// モデルの適合率スコア
135    pub precision: f64,
136    /// Model recall score
137    /// モデルの再現率スコア
138    pub recall: f64,
139    /// Model F1 score
140    /// モデルのF1スコア
141    pub f1_score: f64,
142    /// Model loss value
143    /// モデルの損失値
144    pub loss: f64,
145}
146
147impl Metrics {
148    /// Create new metrics with default values
149    /// デフォルト値で新しいメトリクスを作成
150    pub fn new() -> Self {
151        Self {
152            accuracy: 0.0,
153            precision: 0.0,
154            recall: 0.0,
155            f1_score: 0.0,
156            loss: 0.0,
157        }
158    }
159
160    /// Create metrics with specified values
161    /// 指定された値でメトリクスを作成
162    pub fn with_values(
163        accuracy: f64,
164        precision: f64,
165        recall: f64,
166        f1_score: f64,
167        loss: f64,
168    ) -> Self {
169        Self {
170            accuracy,
171            precision,
172            recall,
173            f1_score,
174            loss,
175        }
176    }
