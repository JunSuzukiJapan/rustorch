14/// 訓練設定
15/// Training configuration

54/// 訓練結果
55/// Training result
56#[derive(Debug, Clone)]
57pub struct TrainingResult {
58    /// 訓練損失の履歴
59    pub train_losses: Vec<f64>,
60    /// 検証損失の履歴
61    pub val_losses: Vec<f64>,
62    /// 訓練精度の履歴
63    pub train_accuracies: Vec<f64>,
64    /// 検証精度の履歴
65    pub val_accuracies: Vec<f64>,
66    /// 総訓練時間
67    pub total_training_time: Duration,
68    /// 最良の検証損失
69    pub best_val_loss: f64,
70    /// 最良の検証精度
71    pub best_val_accuracy: f64,
72    /// 早期停止したかどうか
73    pub early_stopped: bool,
74    /// 完了したエポック数
75    pub completed_epochs: usize,
76}

78impl TrainingResult {
79    /// 新しい訓練結果を作成
80    /// Create a new training result
81    pub fn new() -> Self {
82        TrainingResult {
83            train_losses: Vec::new(),
84            val_losses: Vec::new(),
85            train_accuracies: Vec::new(),
86            val_accuracies: Vec::new(),
87            total_training_time: Duration::new(0, 0),
88            best_val_loss: f64::INFINITY,
89            best_val_accuracy: 0.0,
90            early_stopped: false,
91            completed_epochs: 0,
92        }
93    }
94
95    /// 訓練結果のサマリーを表示
96    /// Display training result summary
97    pub fn summary(&self) -> String {
98        format!(
99            "Training Summary:\n\
100             - Completed epochs: {}\n\
101             - Total time: {:.2}s\n\
102             - Best validation loss: {:.4}\n\
103             - Best validation accuracy: {:.4}\n\
104             - Early stopped: {}",
105            self.completed_epochs,
106            self.total_training_time.as_secs_f64(),
107            self.best_val_loss,
108            self.best_val_accuracy,
109            self.early_stopped
110        )
111    }
112}

117    }

120/// モデル訓練器
121/// Model trainer
122pub struct Trainer<T, M, O, L>
123where
124    T: Float + 'static + Send + Sync + Debug + ndarray::ScalarOperand + num_traits::FromPrimitive,
125    M: Model<T>,
126    O: Optimizer,
127    L: Loss<T>,
128{
129    model: M,
130    _optimizer: O,
131    _loss_fn: L,
132    config: TrainingConfig,
133    _phantom: std::marker::PhantomData<T>,
134}
135
136impl<T, M, O, L> Trainer<T, M, O, L>
137where
138    T: Float + 'static + Send + Sync + Debug + ndarray::ScalarOperand + num_traits::FromPrimitive,
139    M: Model<T>,
140    O: Optimizer,
141    L: Loss<T>,
142{
143    /// 新しい訓練器を作成
144    /// Create a new trainer
145    pub fn new(model: M, optimizer: O, loss_fn: L, config: TrainingConfig) -> Self {
146        Self {
147            model,
148            _optimizer: optimizer,
149            _loss_fn: loss_fn,
150            config,
151            _phantom: std::marker::PhantomData,
152        }
153    }
154
155    /// モデルを訓練
156    /// Train the model
157    pub fn train<D>(&mut self, _train_dataset: D, _val_dataset: Option<D>) -> TrainingResult
158    where
159        D: Dataset<(Tensor<T>, Tensor<T>)> + Clone,
160    {
161        let mut result = TrainingResult::new();
162        let start_time = Instant::now();
163
164        // DataLoader の実装は簡略化 - 実際には適切なイテレータが必要
165        // let train_loader = DataLoader::new(train_dataset.clone(), self.config.batch_size, true);
166        // let val_loader = val_dataset.map(|dataset| DataLoader::new(dataset, self.config.batch_size, false));
167
168        let mut best_val_loss = f64::INFINITY;
169        let mut patience_counter = 0;
170
171        for epoch in 0..self.config.epochs {
172            // 訓練フェーズ（簡略化実装）
173            Model::train(&mut self.model);
174            let train_loss = 0.5 - (epoch as f64 * 0.05); // シミュレーション
175            let train_acc = 0.5 + (epoch as f64 * 0.08); // シミュレーション
176            result.train_losses.push(train_loss);
177            result.train_accuracies.push(train_acc);
178
179            // 検証フェーズ（簡略化実装）
180            if epoch % self.config.validation_frequency == 0 {
181                Model::eval(&mut self.model);
182                let val_loss = train_loss + 0.1; // シミュレーション
183                let val_acc = train_acc - 0.05; // シミュレーション
184                result.val_losses.push(val_loss);
185                result.val_accuracies.push(val_acc);
186
187                // 最良モデルの更新
188                if val_loss < best_val_loss {
189                    best_val_loss = val_loss;
190                    result.best_val_loss = val_loss;
191                    result.best_val_accuracy = val_acc;
192                    patience_counter = 0;
193                } else {
194                    patience_counter += 1;
195                }
196
197                // 早期停止チェック
198                if let Some(patience) = self.config.early_stopping_patience {
199                    if patience_counter >= patience {
200                        result.early_stopped = true;
201                        break;
202                    }
203                }
204            }
205
206            // ログ出力
207            if epoch % self.config.log_frequency == 0 {
208                println!(
209                    "Epoch {}/{}: Train Loss: {:.4}, Train Acc: {:.4}",
210                    epoch + 1,
211                    self.config.epochs,
212                    train_loss,
213                    train_acc
214                );
215
216                if let Some(val_loss) = result.val_losses.last() {
217                    if let Some(val_acc) = result.val_accuracies.last() {
218                        println!(
219                            "           Val Loss: {:.4}, Val Acc: {:.4}",
220                            val_loss, val_acc
221                        );
222                    }
223                }
224            }
225
226            result.completed_epochs = epoch + 1;
227        }
228
229        result.total_training_time = start_time.elapsed();
230        result
231    }

234    /// Get reference to the model
235    pub fn model(&self) -> &M {

237    }

240    /// Get mutable reference to the model
241    pub fn model_mut(&mut self) -> &mut M {
242        &mut self.model
243    }
244}

274    /// 単一サンプルの推論
275    /// Inference for a single sample
276    pub fn predict(&self, input: &Variable<T>) -> Variable<T> {
277        self.model.forward(input)
278    }
279
280    /// バッチ推論
281    /// Batch inference
282    pub fn predict_batch(&self, inputs: Vec<Variable<T>>) -> Vec<Variable<T>> {
283        inputs
284            .into_iter()
285            .map(|input| self.predict(&input))
286            .collect()
287    }

297    }

335            _phantom: std::marker::PhantomData,

350        self
351    }

354    /// Set learning rate
355    pub fn learning_rate(mut self, lr: f64) -> Self {
356        self.config.learning_rate = lr;
357        self
358    }

375    /// Set device

377        self.config.device = device;
