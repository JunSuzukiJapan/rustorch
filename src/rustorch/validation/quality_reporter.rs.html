644impl QualityReporter {
645    /// Create new quality reporter
646    /// 新しい品質レポーターを作成
647    pub fn new(config: ReportConfiguration) -> Self {
648        Self {
649            config,
650            validation_history: VecDeque::new(),
651            quality_history: VecDeque::new(),
652            report_stats: ReportStatistics::default(),
653            start_time: SystemTime::now(),
654        }
655    }
656
657    /// Add validation result to history
658    /// 検証結果を履歴に追加
659    pub fn add_validation_result(&mut self, summary: &ValidationSummary) -> RusTorchResult<()> {
660        self.validation_history.push_back(summary.clone());
661        if self.validation_history.len() > self.config.max_history_entries {
662            self.validation_history.pop_front();
663        }
664        Ok(())
665    }
666
667    /// Add quality assessment to history
668    /// 品質評価を履歴に追加
669    pub fn add_quality_assessment(
670        &mut self,
671        assessment: &DataQualityAssessment,
672    ) -> RusTorchResult<()> {
673        self.quality_history.push_back(assessment.clone());
674        if self.quality_history.len() > self.config.max_history_entries {
675            self.quality_history.pop_front();
676        }
677        Ok(())
678    }
679
680    /// Generate comprehensive quality report
681    /// 包括的品質レポートを生成
682    pub fn generate_report(&self, format: ReportFormat) -> RusTorchResult<String> {
683        let start_time = std::time::Instant::now();
684
685        // Build report structure
686        let report = self.build_report(&format)?;
687
688        // Format report based on requested format
689        let formatted_report = match format {
690            ReportFormat::Summary => self.format_summary_report(&report)?,
691            ReportFormat::Detailed => self.format_detailed_report(&report)?,
692            ReportFormat::Executive => self.format_executive_report(&report)?,
693            ReportFormat::Technical => self.format_technical_report(&report)?,
694            ReportFormat::Json => self.format_json_report(&report)?,
695            ReportFormat::Csv => self.format_csv_report(&report)?,
696        };
697
698        println!(
699            "📊 Quality report generated in {:.2}ms",
700            start_time.elapsed().as_secs_f64() * 1000.0
701        );
702
703        Ok(formatted_report)
704    }
705
706    /// Build report structure from historical data
707    /// 履歴データからレポート構造を構築
708    fn build_report(&self, format: &ReportFormat) -> RusTorchResult<QualityReport> {
709        let metadata = self.build_metadata(format)?;
710        let executive_summary = self.build_executive_summary()?;
711        let quality_overview = self.build_quality_overview()?;
712        let trend_analysis = if self.config.generate_trend_analysis {
713            Some(self.build_trend_analysis()?)
714        } else {
715            None
716        };
717        let issue_analysis = self.build_issue_analysis()?;
718        let recommendations = self.build_recommendations()?;
719        let technical_details = if self.config.include_detailed_metrics {
720            Some(self.build_technical_details()?)
721        } else {
722            None
723        };
724
725        Ok(QualityReport {
726            metadata,
727            executive_summary,
728            quality_overview,
729            trend_analysis,
730            issue_analysis,
731            recommendations,
732            technical_details,
733        })
734    }
735
736    /// Build report metadata
737    /// レポートメタデータを構築
738    fn build_metadata(&self, format: &ReportFormat) -> RusTorchResult<ReportMetadata> {
739        let now = SystemTime::now();
740        let start = self
741            .validation_history
742            .front()
743            .map(|v| v.validation_result.validation_time)

988    /// Get average quality score
989    /// 平均品質スコアを取得
990    pub fn get_average_quality_score(&self) -> f64 {
991        if self.quality_history.is_empty() {
992            return 0.0;
993        }
994
995        let sum: f64 = self
996            .quality_history
997            .iter()
998            .map(|assessment| assessment.overall_score)
999            .sum();
1000        sum / self.quality_history.len() as f64
1001    }

1020    /// Count critical issues
1021    /// 重要問題をカウント
1022    fn count_critical_issues(&self) -> usize {
1023        self.validation_history
1024            .iter()
1025            .flat_map(|v| &v.validation_result.issues)
1026            .filter(|issue| {
1027                matches!(
1028                    issue.severity,
1029                    crate::validation::core::IssueSeverity::Critical
1030                )
1031            })
1032            .count()
1033    }
