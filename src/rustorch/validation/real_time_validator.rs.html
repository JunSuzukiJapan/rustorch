188impl ValidationBuffer {
189    /// Create new validation buffer
190    /// 新しい検証バッファを作成
191    pub fn new(max_size: usize) -> Self {
192        Self {
193            results: VecDeque::with_capacity(max_size),
194            max_size,
195        }
196    }
197
198    /// Add result to buffer
199    /// バッファに結果を追加
200    pub fn add_result(&mut self, result: StreamingValidationResult) {
201        if self.results.len() >= self.max_size {
202            self.results.pop_front();
203        }
204        self.results.push_back(result);
205    }
206
207    /// Get recent results
208    /// 最近の結果を取得
209    pub fn get_recent_results(&self, count: usize) -> Vec<&StreamingValidationResult> {
210        self.results.iter().rev().take(count).collect()
211    }
212}

214impl RealTimeValidator {
215    /// Create new real-time validator
216    /// 新しいリアルタイム検証器を作成
217    pub fn new(config: RealTimeConfig) -> RusTorchResult<Self> {
218        let buffer = Arc::new(Mutex::new(ValidationBuffer::new(config.buffer_size)));
219
220        Ok(Self {
221            config,
222            buffer,
223            is_running: Arc::new(Mutex::new(false)),
224            stats: ValidationStats::default(),
225        })
226    }
227
228    /// Start real-time monitoring
229    /// リアルタイム監視を開始
230    pub fn start_monitoring(&mut self) -> RusTorchResult<()> {
231        let mut running = self
232            .is_running
233            .lock()
234            .map_err(|_| RusTorchError::Validation {
235                message: "Failed to acquire running lock".to_string(),
236            })?;
237
238        if *running {
239            return Err(RusTorchError::Validation {
240                message: "Real-time validator already running".to_string(),
241            });
242        }
243
244        *running = true;
245        println!("🔍 Real-time data validation started");
246        Ok(())
247    }
248
249    /// Stop real-time monitoring
250    /// リアルタイム監視を停止
251    pub fn stop_monitoring(&mut self) -> RusTorchResult<()> {
252        let mut running = self
253            .is_running
254            .lock()
255            .map_err(|_| RusTorchError::Validation {
256                message: "Failed to acquire running lock".to_string(),
257            })?;
258
259        *running = false;
260        println!("🔍 Real-time data validation stopped");
261        Ok(())
262    }
263
264    /// Validate data chunk in real-time
265    /// リアルタイムでデータチャンクを検証
266    pub fn validate_realtime<T>(
267        &mut self,
268        tensor: &crate::tensor::Tensor<T>,
269    ) -> RusTorchResult<StreamingValidationResult>
270    where
271        T: num_traits::Float + std::fmt::Debug + Clone + Send + Sync + 'static,
272    {
273        let streaming_validation = StreamingValidation::new(self.config.clone());
274        let result = streaming_validation.validate_chunk(tensor)?;
275
276        // Update statistics
277        self.stats.total_chunks += 1;
278        if result.is_valid {
279            self.stats.valid_chunks += 1;
280        } else {
281            self.stats.invalid_chunks += 1;
282        }
283
284        // Update averages
285        let total = self.stats.total_chunks as f64;
286        self.stats.avg_processing_time = Duration::from_nanos(
287            (self.stats.avg_processing_time.as_nanos() as f64 * (total - 1.0)
288                + result.processing_time.as_nanos() as f64) as u64
289                / total as u64,
290        );
291        self.stats.avg_quality_score =
292            (self.stats.avg_quality_score * (total - 1.0) + result.quality_score) / total;
293
294        // Store result in buffer
295        let mut buffer = self.buffer.lock().map_err(|_| RusTorchError::Validation {
296            message: "Failed to acquire buffer lock".to_string(),
297        })?;
298        buffer.add_result(result.clone());
299
300        Ok(result)
301    }
302
303    /// Get validation statistics
304    /// 検証統計を取得
305    pub fn get_stats(&self) -> &ValidationStats {
306        &self.stats
307    }
