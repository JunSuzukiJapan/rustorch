347    /// Assess data quality for a tensor
348    /// テンソルのデータ品質を評価
349    pub fn assess_quality<T>(
350        &mut self,
351        tensor: &crate::tensor::Tensor<T>,
352    ) -> RusTorchResult<DataQualityAssessment>
353    where
354        T: num_traits::Float + std::fmt::Debug + Clone + Send + Sync + 'static,
355    {
356        let start_time = Instant::now();
357
358        // Collect basic data characteristics
359        let characteristics = self.collect_data_characteristics(tensor);
360
361        // Assess each quality dimension
362        let mut dimensions = HashMap::new();
363
364        // Completeness assessment
365        dimensions.insert(
366            QualityDimension::Completeness,
367            self.assess_completeness(tensor, &characteristics)?,
368        );
369
370        // Accuracy assessment
371        dimensions.insert(
372            QualityDimension::Accuracy,
373            self.assess_accuracy(tensor, &characteristics)?,
374        );
375
376        // Consistency assessment
377        dimensions.insert(
378            QualityDimension::Consistency,
379            self.assess_consistency(tensor, &characteristics)?,
380        );
381
382        // Validity assessment
383        dimensions.insert(
384            QualityDimension::Validity,
385            self.assess_validity(tensor, &characteristics)?,
386        );
387
388        // Uniqueness assessment
389        dimensions.insert(
390            QualityDimension::Uniqueness,
391            self.assess_uniqueness(tensor, &characteristics)?,
392        );
393
394        // Timeliness assessment
395        dimensions.insert(
396            QualityDimension::Timeliness,
397            self.assess_timeliness(&characteristics)?,
398        );
399
400        // Integrity assessment
401        dimensions.insert(
402            QualityDimension::Integrity,
403            self.assess_integrity(tensor, &characteristics)?,
404        );
405
406        // Calculate overall score
407        let overall_score = self.calculate_overall_score(&dimensions);
408
409        // Analyze trends if history exists
410        let trends = if self.history.len() > 2 {
411            Some(self.analyze_trends(&overall_score))
412        } else {
413            None
414        };
415
416        let assessment = DataQualityAssessment {
417            overall_score,
418            dimensions,
419            timestamp: SystemTime::now(),
420            characteristics,
421            trends,
422        };
423
424        // Update history
425        self.history.push_back(assessment.clone());
426        if self.history.len() > self.max_history_size {
427            self.history.pop_front();
428        }
429
430        // Update aggregated statistics
431        self.update_aggregated_stats(&assessment);
432
433        println!(
434            "📊 Quality assessment completed in {:.2}ms, score: {:.3}",
435            start_time.elapsed().as_secs_f64() * 1000.0,
436            overall_score
437        );
438
439        Ok(assessment)
440    }

745    fn update_aggregated_stats(&mut self, assessment: &DataQualityAssessment) {
746        self.aggregated_stats.total_assessments += 1;
747
748        let count = self.aggregated_stats.total_assessments as f64;
749        let old_mean = self.aggregated_stats.average_overall_score;
750        let new_score = assessment.overall_score;
751
752        // Update running average
753        self.aggregated_stats.average_overall_score = old_mean + (new_score - old_mean) / count;
754
755        // Update best/worst scores
756        if self.aggregated_stats.total_assessments == 1 {
757            self.aggregated_stats.best_score = new_score;
758            self.aggregated_stats.worst_score = new_score;
759        } else {
760            self.aggregated_stats.best_score = self.aggregated_stats.best_score.max(new_score);
761            self.aggregated_stats.worst_score = self.aggregated_stats.worst_score.min(new_score);
762        }
763
764        // Update variance (simplified calculation)
765        let variance_delta =
766            (new_score - old_mean) * (new_score - self.aggregated_stats.average_overall_score);
767        self.aggregated_stats.score_variance =
768            (self.aggregated_stats.score_variance * (count - 1.0) + variance_delta) / count;
769
770        // Update stability measure
771        self.aggregated_stats.stability_measure = 1.0 - self.aggregated_stats.score_variance.sqrt();
772    }
