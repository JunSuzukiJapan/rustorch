291    /// Optimize memory allocation based on current strategy
292    /// 現在の戦略に基づいてメモリ割り当てを最適化
293    pub fn optimize_allocation(&self, shape: &[usize]) -> RusTorchResult<ArrayD<T>> {
294        let start_time = Instant::now();
295
296        let total_elements: usize = shape.iter().product();
297        let size_class = self.get_size_class(total_elements);
298
299        // Try smart cache first
300        if let Some(cached) = self.try_cache_retrieval(shape)? {
301            self.update_stats_cache_hit();
302            return Ok(cached);
303        }
304
305        // Try preallocated pools
306        if let Some(prealloc) = self.try_prealloc_retrieval(size_class, shape)? {
307            self.update_stats_zero_copy();
308            return Ok(prealloc);
309        }
310
311        // Predict future usage and preallocate if beneficial
312        if self.config.enable_prediction {
313            if let Some(prediction) = self.predict_memory_usage()? {
314                self.consider_preallocation(&prediction)?;
315            }
316        }
317
318        // Perform allocation with current strategy
319        let array = self.allocate_with_strategy(shape)?;
320
321        // Update optimization statistics
322        self.update_optimization_stats(start_time.elapsed());
323
324        Ok(array)
325    }

327    /// Optimize memory deallocation
328    /// メモリ解放を最適化
329    pub fn optimize_deallocation(&self, array: ArrayD<T>) -> RusTorchResult<()> {
330        let shape = array.shape().to_vec();
331        let total_elements: usize = shape.iter().product();
332        let size_class = self.get_size_class(total_elements);
333
334        // Check if array should be cached
335        if self.should_cache(&array) {
336            self.add_to_cache(array, &shape)?;
337            return Ok(());
338        }
339
340        // Check if array should be added to preallocation pool
341        if self.should_preallocate(size_class) {
342            self.add_to_prealloc_pool(array, size_class)?;
343            return Ok(());
344        }
345
346        // Otherwise, let it be deallocated normally
347        Ok(())
348    }

350    /// Perform memory defragmentation
351    /// メモリデフラグメンテーションを実行
352    pub fn defragment_memory(&self) -> RusTorchResult<usize> {
353        let start_time = Instant::now();
354        let mut memory_reclaimed = 0;
355
356        // Clean up cache based on access patterns
357        {
358            let mut cache = self.smart_cache.write().map_err(|_| {
359                RusTorchError::MemoryError("Failed to acquire cache write lock".to_string())
360            })?;
361
362            let mut to_remove = Vec::new();
363            let now = Instant::now();
364
365            for (key, entry) in cache.iter() {
366                let age = now.duration_since(entry.last_accessed);
367                let access_rate = entry.access_count as f64 / age.as_secs_f64().max(1.0);
368
369                // Remove entries with low access rate and old age
370                if access_rate < 0.1 && age > Duration::from_secs(300) {
371                    to_remove.push(key.clone());
372                }
373            }
374
375            for key in to_remove {
376                if let Some(entry) = cache.remove(&key) {
377                    memory_reclaimed += entry.data.len() * std::mem::size_of::<T>();
378                }
379            }
380        }
381
382        // Clean up preallocation pools
383        {
384            let mut pools = self.prealloc_pools.write().map_err(|_| {
385                RusTorchError::MemoryError("Failed to acquire pools write lock".to_string())
386            })?;
387
388            for (_, pool) in pools.iter_mut() {
389                // Keep only the most recently added arrays
390                let keep_count = (pool.len() / 2).max(1);
391                let remove_count = pool.len() - keep_count;
392
393                for _ in 0..remove_count {
394                    if let Some(array) = pool.pop_front() {
395                        memory_reclaimed += array.len() * std::mem::size_of::<T>();
396                    }
397                }
398            }
399        }
400
401        // Update statistics
402        {
403            let mut stats = self.stats.write().map_err(|_| {
404                RusTorchError::MemoryError("Failed to acquire stats write lock".to_string())
405            })?;
406            stats.defragmentations += 1;
407            stats.memory_saved += memory_reclaimed;
408            stats.optimization_time += start_time.elapsed();
409        }
410
411        Ok(memory_reclaimed)
412    }

414    /// Predict future memory usage
415    /// 将来のメモリ使用量を予測
416    pub fn predict_memory_usage(&self) -> RusTorchResult<Option<MemoryPrediction>> {
417        if !self.config.enable_prediction {
418            return Ok(None);
419        }
420
421        let history = self.usage_history.read().map_err(|_| {
422            RusTorchError::MemoryError("Failed to acquire history read lock".to_string())
423        })?;
424
425        if history.len() < self.config.prediction_window {
426            return Ok(None);
427        }
428
429        // Simple trend analysis for prediction
430        let recent_snapshots: Vec<_> = history
431            .iter()
432            .rev()
433            .take(self.config.prediction_window)
434            .collect();
435
436        let mut total_usage_trend = 0.0;
437
438        for i in 1..recent_snapshots.len() {
439            let curr = recent_snapshots[i - 1];
440            let prev = recent_snapshots[i];
441
442            total_usage_trend += curr.total_usage as f64 - prev.total_usage as f64;
443        }
444
445        total_usage_trend /= (recent_snapshots.len() - 1) as f64;
446
447        let current_usage = recent_snapshots[0].total_usage;
448        let predicted_memory =
449            ((current_usage as f64 + total_usage_trend * 5.0) as usize).max(current_usage);
450
451        let confidence = if total_usage_trend.abs() < current_usage as f64 * 0.1 {
452            0.8 // High confidence for stable trends
453        } else {
454            0.5 // Lower confidence for volatile trends
455        };
456
457        let recommended_prealloc = ((predicted_memory as f64 * self.config.preallocation_factor)
458            as usize)
459            .saturating_sub(current_usage);
460
461        Ok(Some(MemoryPrediction {
462            predicted_memory,
463            confidence,
464            horizon: Duration::from_secs(300), // 5 minutes ahead
465            recommended_prealloc,
466        }))
467    }

523    fn try_cache_retrieval(&self, shape: &[usize]) -> RusTorchResult<Option<ArrayD<T>>> {
524        let cache_key = format!("{:?}", shape);
525
526        let mut cache = self.smart_cache.write().map_err(|_| {
527            RusTorchError::MemoryError("Failed to acquire cache write lock".to_string())
528        })?;
529
530        if let Some(entry) = cache.get_mut(&cache_key) {
531            entry.last_accessed = Instant::now();
532            entry.access_count += 1;
533
534            // Clone the data and clear the original for reuse
535            let mut result = entry.data.clone();
536            result.fill(T::zero());
537            return Ok(Some(result));
538        }
539
540        Ok(None)
541    }

543    fn try_prealloc_retrieval(
544        &self,
545        size_class: usize,
546        shape: &[usize],
547    ) -> RusTorchResult<Option<ArrayD<T>>> {
548        let mut pools = self.prealloc_pools.write().map_err(|_| {
549            RusTorchError::MemoryError("Failed to acquire pools write lock".to_string())
550        })?;
551
552        if let Some(pool) = pools.get_mut(&size_class) {
553            if let Some(array) = pool.pop_back() {
554                // Try to reshape if needed
555                let total_elements: usize = shape.iter().product();
556                if array.len() == total_elements {
557                    let array_clone = array.clone();
558                    if let Ok(reshaped) = array_clone.into_shape_with_order(IxDyn(shape)) {
559                        return Ok(Some(reshaped));
560                    }
561                }
562                // Put back if can't reshape
563                pool.push_back(array);
564            }
565        }
566
567        Ok(None)
568    }

604        let cache_key = format!("{:?}", shape);
605        let entry = CacheEntry {
606            data: array,

614            RusTorchError::MemoryError("Failed to acquire cache write lock".to_string())
615        })?;
616
617        cache.insert(cache_key, entry);

619        Ok(())

677
678        // Analyze recent pressure levels to adjust strategy
679        if let Some(latest) = history.back() {
680            let new_strategy = match latest.pressure_level {
681                PressureLevel::Low => OptimizationStrategy::SpeedFirst,
682                PressureLevel::Medium => OptimizationStrategy::Balanced,
683                PressureLevel::High | PressureLevel::Critical => OptimizationStrategy::MemoryFirst,
684            };
685
686            let mut strategy = self.current_strategy.write().map_err(|_| {
687                RusTorchError::MemoryError("Failed to acquire strategy write lock".to_string())
688            })?;
689            *strategy = new_strategy;
690        }
691
692        Ok(())
693    }
694
695    fn update_stats_cache_hit(&self) {
696        if let Ok(mut stats) = self.stats.write() {
697            stats.cache_hit_ratio = (stats.cache_hit_ratio * 0.9) + 0.1; // Simple moving average
698        }
699    }
700
701    fn update_stats_zero_copy(&self) {
702        if let Ok(mut stats) = self.stats.write() {
703            stats.zero_copy_operations += 1;
704        }
705    }
706
707    fn update_optimization_stats(&self, duration: Duration) {
708        if let Ok(mut stats) = self.stats.write() {
709            stats.total_optimizations += 1;
710            stats.optimization_time += duration;
711        }
712    }
