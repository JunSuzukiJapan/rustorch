47/// Memory allocation patterns
48/// メモリ割り当てパターン
49#[derive(Debug, Clone, PartialEq, Eq, Hash)]
50pub enum AllocationPattern {
51    /// Short-lived allocations (< 1 second)
52    /// 短期割り当て（< 1秒）
53    ShortLived,
54    /// Medium-lived allocations (1 second - 1 minute)
55    /// 中期割り当て（1秒 - 1分）
56    MediumLived,
57    /// Long-lived allocations (> 1 minute)
58    /// 長期割り当て（> 1分）
59    LongLived,
60    /// Leaked allocations (never deallocated)
61    /// リークした割り当て（解放されない）
62    Leaked,
63    /// Cyclic allocations (repetitive pattern)
64    /// 循環割り当て（反復パターン）
65    Cyclic,
66}
67
68impl AllocationPattern {
69    /// Classify allocation pattern based on lifetime
70    /// ライフタイムに基づいて割り当てパターンを分類
71    pub fn classify(lifetime: Option<Duration>, allocated_at: SystemTime) -> Self {
72        match lifetime {
73            Some(duration) => {
74                if duration < Duration::from_secs(1) {
75                    AllocationPattern::ShortLived
76                } else if duration < Duration::from_secs(60) {
77                    AllocationPattern::MediumLived
78                } else {
79                    AllocationPattern::LongLived
80                }
81            }
82            None => {
83                // Check if allocation is old enough to be considered leaked
84                let age = SystemTime::now()
85                    .duration_since(allocated_at)
86                    .unwrap_or(Duration::from_secs(0));
87
88                if age > Duration::from_secs(300) {
89                    // 5 minutes
90                    AllocationPattern::Leaked
91                } else {
92                    AllocationPattern::MediumLived
93                }
94            }
95        }
96    }
97}

323    /// Record memory allocation
324    /// メモリ割り当てを記録
325    pub fn record_allocation(&self, size: usize, source_location: String) -> RusTorchResult<u64> {
326        let start_time = Instant::now();
327
328        let id = {
329            let mut next_id = self
330                .next_id
331                .lock()
332                .map_err(|_| RusTorchError::MemoryError("Failed to acquire ID lock".to_string()))?;
333            let id = *next_id;
334            *next_id += 1;
335            id
336        };
337
338        let mut call_stack = Vec::new();
339        if self.config.enable_stack_trace {
340            // In a real implementation, we would capture the actual call stack
341            // For now, we'll use a placeholder
342            call_stack.push("tensor::core::Tensor::new".to_string());
343            call_stack.push("main".to_string());
344        }
345
346        let record = AllocationRecord {
347            id,
348            size,
349            allocated_at: SystemTime::now(),
350            source_location,
351            call_stack,
352            deallocated_at: None,
353            lifetime: None,
354            pattern: AllocationPattern::MediumLived, // Will be updated on deallocation
355        };
356
357        // Update current usage
358        {
359            let mut current = self.current_usage.write().map_err(|_| {
360                RusTorchError::MemoryError("Failed to acquire usage write lock".to_string())
361            })?;
362            *current += size;
363
364            // Update peak usage
365            let mut peak = self.peak_usage.write().map_err(|_| {
366                RusTorchError::MemoryError("Failed to acquire peak write lock".to_string())
367            })?;
368            if *current > *peak {
369                *peak = *current;
370            }
371        }
372
373        // Store record
374        {
375            let mut records = self.records.write().map_err(|_| {
376                RusTorchError::MemoryError("Failed to acquire records write lock".to_string())
377            })?;
378
379            records.insert(id, record);
380
381            // Clean up old records if necessary
382            if records.len() > self.config.max_records {
383                Self::cleanup_old_records(&mut records, self.config.max_records / 2);
384            }
385        }
386
387        // Update statistics
388        {
389            let mut stats = self.stats.write().map_err(|_| {
390                RusTorchError::MemoryError("Failed to acquire stats write lock".to_string())
391            })?;
392            stats.total_allocations += 1;
393            stats.analysis_overhead += start_time.elapsed();
394        }
395
396        Ok(id)
397    }

399    /// Record memory deallocation
400    /// メモリ解放を記録
401    pub fn record_deallocation(&self, id: u64) -> RusTorchResult<()> {
402        let start_time = Instant::now();
403        let dealloc_time = SystemTime::now();
404
405        let size = {
406            let mut records = self.records.write().map_err(|_| {
407                RusTorchError::MemoryError("Failed to acquire records write lock".to_string())
408            })?;
409
410            if let Some(record) = records.get_mut(&id) {
411                record.deallocated_at = Some(dealloc_time);
412                record.lifetime = dealloc_time.duration_since(record.allocated_at).ok();
413                record.pattern = AllocationPattern::classify(record.lifetime, record.allocated_at);
414                record.size
415            } else {
416                return Err(RusTorchError::MemoryError(format!(
417                    "Allocation ID {} not found",
418                    id
419                )));
420            }
421        };
422
423        // Update current usage
424        {
425            let mut current = self.current_usage.write().map_err(|_| {
426                RusTorchError::MemoryError("Failed to acquire usage write lock".to_string())
427            })?;
428            *current = current.saturating_sub(size);
429        }
430
431        // Update statistics
432        {
433            let mut stats = self.stats.write().map_err(|_| {
434                RusTorchError::MemoryError("Failed to acquire stats write lock".to_string())
435            })?;
436            stats.total_deallocations += 1;
437            stats.analysis_overhead += start_time.elapsed();
438        }
439
440        Ok(())
441    }

443    /// Generate comprehensive memory report
444    /// 包括的なメモリレポートを生成
445    pub fn generate_report(&self) -> RusTorchResult<MemoryReport> {
446        let start_time = Instant::now();
447
448        let records = self.records.read().map_err(|_| {
449            RusTorchError::MemoryError("Failed to acquire records read lock".to_string())
450        })?;
451
452        let current_usage = *self.current_usage.read().map_err(|_| {
453            RusTorchError::MemoryError("Failed to acquire usage read lock".to_string())
454        })?;
455
456        let peak_usage = *self.peak_usage.read().map_err(|_| {
457            RusTorchError::MemoryError("Failed to acquire peak read lock".to_string())
458        })?;
459
460        let stats = self.stats.read().map_err(|_| {
461            RusTorchError::MemoryError("Failed to acquire stats read lock".to_string())
462        })?;
463
464        // Analyze records
465        let mut pattern_distribution = HashMap::new();
466        let mut location_stats = HashMap::new();
467        let mut total_allocated = 0;
468        let mut active_count = 0;
469        let mut leaked_count = 0;
470        let mut leaked_bytes = 0;
471        let mut oldest_leak_age = None;
472
473        for record in records.values() {
474            // Update pattern distribution
475            *pattern_distribution
476                .entry(record.pattern.clone())
477                .or_insert(0) += 1;
478
479            // Update location statistics
480            let location_stat = location_stats
481                .entry(record.source_location.clone())
482                .or_insert((0, 0, 0)); // (count, total_size, leak_count)
483            location_stat.0 += 1;
484            location_stat.1 += record.size;
485
486            total_allocated += record.size;
487
488            if record.deallocated_at.is_none() {
489                active_count += 1;
490
491                // Check for potential leaks
492                let age = SystemTime::now()
493                    .duration_since(record.allocated_at)
494                    .unwrap_or(Duration::from_secs(0));
495
496                if age > self.config.leak_threshold {
497                    leaked_count += 1;
498                    leaked_bytes += record.size;
499                    location_stat.2 += 1;
500
501                    if oldest_leak_age.is_none() || age > oldest_leak_age.unwrap() {
502                        oldest_leak_age = Some(age);
503                    }
504                }
505            }
506        }
507
508        // Generate hotspots
509        let mut hotspots = Vec::new();
510        if self.config.enable_hotspot_analysis {
511            for (location, (count, total_size, leak_count)) in location_stats {
512                if count >= self.config.hotspot_threshold {
513                    hotspots.push(MemoryHotspot {
514                        location,
515                        total_allocations: count,
516                        total_memory: total_size,
517                        avg_size: total_size as f64 / count as f64,
518                        peak_concurrent: count, // Simplified
519                        leak_count,
520                        frequency: count as f64 / 3600.0, // Simplified: allocations per hour
521                    });
522                }
523            }
524
525            // Sort by total memory usage
526            hotspots.sort_by(|a, b| b.total_memory.cmp(&a.total_memory));
527        }
528
529        let leak_stats = LeakStats {
530            potential_leaks: leaked_count,
531            leaked_bytes,
532            oldest_leak_age,
533            leak_rate: leaked_count as f64 / 3600.0, // Simplified
534        };
535
536        let fragmentation_analysis = FragmentationAnalysis {
537            fragmentation_ratio: 0.1,            // Simplified
538            pool_count: 8,                       // Simplified
539            avg_pool_utilization: 0.75,          // Simplified
540            wasted_memory: total_allocated / 20, // Simplified: 5% waste
541        };
542
543        let avg_allocation_size = if stats.total_allocations > 0 {
544            total_allocated as f64 / stats.total_allocations as f64
545        } else {
546            0.0
547        };
548
549        let report = MemoryReport {
550            generated_at: SystemTime::now(),
551            total_allocations: stats.total_allocations,
552            total_deallocations: stats.total_deallocations,
553            active_allocations: active_count,
554            total_allocated_bytes: total_allocated,
555            current_memory_usage: current_usage,
556            peak_memory_usage: peak_usage,
557            avg_allocation_size,
558            leak_stats,
559            pattern_distribution,
560            hotspots,
561            fragmentation_analysis,
562        };
563
564        // Update statistics
565        drop(stats);
566        {
567            let mut stats = self.stats.write().map_err(|_| {
568                RusTorchError::MemoryError("Failed to acquire stats write lock".to_string())
569            })?;
570            stats.reports_generated += 1;
571            stats.last_report_time = Some(SystemTime::now());
572            stats.analysis_overhead += start_time.elapsed();
573        }
574
575        Ok(report)
576    }

620    // Private helper methods
621
622    fn cleanup_old_records(records: &mut HashMap<u64, AllocationRecord>, target_size: usize) {
623        if records.len() <= target_size {
624            return;
625        }
626
627        // Sort by allocation time and remove oldest deallocated records
628        let mut to_remove = Vec::new();
629        let mut deallocated_records: Vec<_> = records
630            .iter()
631            .filter(|(_, record)| record.deallocated_at.is_some())
632            .collect();
633
634        deallocated_records.sort_by_key(|(_, record)| record.allocated_at);
635
636        let remove_count = records.len() - target_size;
637        for (id, _) in deallocated_records.into_iter().take(remove_count) {
638            to_remove.push(*id);
639        }
640
641        for id in to_remove {
642            records.remove(&id);
643        }
644    }
