35/// Legacy memory pool for backward compatibility
36/// 後方互換性のためのレガシーメモリプール
37pub struct MemoryPool<T: Float> {
38    pools: Vec<Arc<Mutex<VecDeque<ArrayD<T>>>>>,
39    max_pool_size: usize,
40}
41
42impl<T: Float + Clone + 'static> MemoryPool<T> {
43    /// Create a new memory pool
44    /// 新しいメモリプールを作成
45    pub fn new(max_pool_size: usize) -> Self {
46        Self {
47            pools: Vec::new(),
48            max_pool_size,
49        }
50    }
51
52    /// Get pool index based on total elements
53    /// 総要素数に基づいてプールインデックスを取得
54    fn get_pool_index(&self, total_elements: usize) -> usize {
55        // Use log2 bucketing for different sizes
56        // サイズ別にlog2バケッティングを使用
57        if total_elements <= 64 {
58            0
59        } else if total_elements <= 256 {
60            1
61        } else if total_elements <= 1024 {
62            2
63        } else if total_elements <= 4096 {
64            3
65        } else if total_elements <= 16384 {
66            4
67        } else if total_elements <= 65536 {
68            5
69        } else {
70            6
71        }
72    }
73
74    /// Ensure pool exists for given index
75    /// 指定されたインデックスのプールが存在することを確認
76    fn ensure_pool(&mut self, index: usize) {
77        while self.pools.len() <= index {
78            self.pools.push(Arc::new(Mutex::new(VecDeque::new())));
79        }
80    }
81
82    /// Allocate a tensor from the pool or create new one
83    /// プールからテンソルを割り当てるか新しく作成
84    pub fn allocate(&mut self, shape: &[usize]) -> ArrayD<T> {
85        let total_elements: usize = shape.iter().product();
86        let pool_index = self.get_pool_index(total_elements);
87
88        self.ensure_pool(pool_index);
89
90        if let Ok(mut pool) = self.pools[pool_index].lock() {
91            if let Some(mut array) = pool.pop_front() {
92                // Reuse existing array if shape matches
93                // 形状が一致する場合は既存の配列を再利用
94                if array.shape() == shape {
95                    array.fill(T::zero());
96                    return array;
97                }
98                // If shape doesn't match, try to reshape
99                // 形状が一致しない場合はリシェイプを試行
100                if array.len() >= total_elements {
101                    // Clone array before attempting reshape to avoid move
102                    // リシェイプ試行前に配列をクローンして移動を回避
103                    let cloned_array = array.clone();
104                    match cloned_array.into_shape_with_order(IxDyn(shape)) {
105                        Ok(reshaped) => return reshaped,
106                        Err(_) => {
107                            // Put back original array if reshape failed
108                            // リシェイプが失敗した場合は元の配列を戻す
109                            pool.push_back(array);
110                        }
111                    }
112                } else {
113                    // Put back if can't reuse
114                    // 再利用できない場合は戻す
115                    pool.push_back(array);
116                }
117            }
118        }
119
120        // Create new array if no suitable one in pool
121        // プールに適切なものがない場合は新しい配列を作成
122        ArrayD::zeros(IxDyn(shape))
123    }
124
125    /// Return a tensor to the pool for reuse
126    /// 再利用のためにテンソルをプールに返却
127    pub fn deallocate(&mut self, array: ArrayD<T>) {
128        let total_elements = array.len();
129        let pool_index = self.get_pool_index(total_elements);
130
131        self.ensure_pool(pool_index);
132
133        if let Ok(mut pool) = self.pools[pool_index].lock() {
134            if pool.len() < self.max_pool_size {
135                pool.push_back(array);
136            }
137            // If pool is full, just drop the array
138            // プールが満杯の場合は配列を破棄
139        }
140    }
141
142    /// Get statistics about pool usage
143    /// プール使用状況の統計を取得
144    pub fn stats(&self) -> PoolStats {
145        let mut total_cached = 0;
146        let mut pool_sizes = Vec::new();
147
148        for pool in &self.pools {
149            if let Ok(pool) = pool.lock() {
150                let size = pool.len();
151                pool_sizes.push(size);
152                total_cached += size;
153            }
154        }
155
156        PoolStats {
157            total_pools: self.pools.len(),
158            total_cached_arrays: total_cached,
159            pool_sizes,
160            max_pool_size: self.max_pool_size,
161        }
162    }
163
164    /// Clear all pools
165    /// 全プールをクリア
166    pub fn clear(&mut self) {
167        for pool in &self.pools {
168            if let Ok(mut pool) = pool.lock() {
169                pool.clear();
170            }
171        }
172    }
173}

405    pub fn optimize_all(&self) -> RusTorchResult<OptimizationSummary> {
406        let start_time = std::time::Instant::now();
407
408        // Perform garbage collection
409        let gc_stats = self.pool.garbage_collect()?;
410
411        // Defragment memory

417        let total_time = start_time.elapsed();
418
419        Ok(OptimizationSummary {

422            compression_memory_saved: compression_saved,

427    /// Get current system health status
428    /// 現在のシステムヘルス状況を取得
429    pub fn get_health_status(&self) -> RusTorchResult<SystemHealthStatus> {
430        let snapshot = self.monitor.get_current_snapshot()?;
431        let trend = self.monitor.analyze_trend()?;
432        let prediction = self.optimizer.predict_memory_usage()?;
433
434        let health_score = self.calculate_health_score(&snapshot, &trend, &prediction)?;
435
436        Ok(SystemHealthStatus {
437            current_snapshot: snapshot,
438            trend_analysis: trend,
439            memory_prediction: prediction,
440            health_score,
441            recommendations: self.generate_recommendations(health_score)?,
442        })
443    }

445    // Private helper methods
446
447    fn calculate_health_score(
448        &self,
449        snapshot: &Option<pressure_monitor::MemorySnapshot>,
450        trend: &Option<pressure_monitor::PressureTrend>,
451        prediction: &Option<MemoryPrediction>,
452    ) -> RusTorchResult<f64> {
453        let mut score = 1.0;
454
455        // Factor in current pressure
456        if let Some(snap) = snapshot {
457            score -= snap.pressure_ratio * 0.4; // Reduce score based on pressure
458        }
459
460        // Factor in trend
461        if let Some(t) = trend {
462            if t.direction > 0.0 {
463                // Increasing pressure trend
464                score -= t.strength * 0.3;
465            }
466        }
467
468        // Factor in prediction confidence
469        if let Some(pred) = prediction {
470            if pred.confidence < 0.5 {
471                score -= 0.1; // Reduce score for low prediction confidence
472            }
473        }
474
475        Ok(score.max(0.0).min(1.0))
476    }

478    fn generate_recommendations(&self, health_score: f64) -> RusTorchResult<Vec<String>> {
479        let mut recommendations = Vec::new();
480
481        if health_score < 0.3 {
482            recommendations.push("Critical: Immediate memory optimization required".to_string());
483            recommendations.push("Consider reducing batch sizes or model complexity".to_string());
484            recommendations.push("Run comprehensive memory cleanup".to_string());
485        } else if health_score < 0.6 {
486            recommendations.push("Warning: High memory pressure detected".to_string());
487            recommendations.push("Consider running memory defragmentation".to_string());
488            recommendations.push("Monitor memory usage closely".to_string());
489        } else if health_score < 0.8 {
490            recommendations.push("Moderate memory pressure".to_string());
491            recommendations.push("Consider periodic cleanup".to_string());
492        } else {
493            recommendations.push("Memory system operating normally".to_string());
494        }
495
496        Ok(recommendations)
497    }
