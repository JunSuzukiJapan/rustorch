1//! フェーズ２リファクタリング効果検証用ベンチマーク実行ツール
2//! Benchmark runner to verify Phase 2 refactoring effectiveness
3
4use rustorch::optim::benchmarks::OptimizerBenchmark;
5
6fn main() {
7    println!("🚀 RusTorch フェーズ２ リファクタリング効果検証");
8    println!("Phase 2 Refactoring Effectiveness Verification");
9    println!("{}", "=".repeat(60));
10
11    let mut benchmark = OptimizerBenchmark::new();
12
13    // 1. Adam系最適化器の比較ベンチマーク実行
14    println!("\n📊 Adam系最適化器比較ベンチマーク実行中...");
15    let results = benchmark.run_adam_comparison();
16
17    // 2. L-BFGS専用ベンチマーク実行
18    println!("\n🔬 L-BFGS専用ベンチマーク実行中...");
19    let lbfgs_results = benchmark.run_lbfgs_benchmark();
20
21    // 3. パフォーマンスレポート生成
22    println!("\n📈 パフォーマンスレポート生成中...");
23    let report = benchmark.generate_report(&results);
24    println!("{}", report);
25
26    // 4. L-BFGS結果追加表示
27    if !lbfgs_results.is_empty() {
28        println!("## 🔬 L-BFGS専用結果\n");
29        for (config_name, result) in lbfgs_results {
30            println!(
31                "**{}**: {:.2}μs/step, {:.1} steps/sec, {}MB memory",
32                config_name,
33                result.avg_step_time_us,
34                result.steps_per_second,
35                result.peak_memory_mb
36            );
37        }
38        println!();
39    }
40
41    println!("\n✅ ベンチマーク完了 - Benchmark Completed");
42    println!("🎉 リファクタリング効果検証完了！");
43}
